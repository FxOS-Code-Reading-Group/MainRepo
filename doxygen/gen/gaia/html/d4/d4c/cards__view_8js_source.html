00380         // Display http:// protocol anyway

00384         native.hostname == current.hostname &&

00388       } else if (current.protocol == 'app:') {
00389         // Avoid displaying app:// protocol

00391       } else {
00392         getOffOrigin.cache[cacheKey] = current.protocol + '//' +
00393           current.hostname;
