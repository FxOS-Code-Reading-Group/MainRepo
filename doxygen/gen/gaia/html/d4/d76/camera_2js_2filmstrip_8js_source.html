00238   function handleSwipe(e) {
00239     // Because the stuff around the media frame does not change position
00240     // when the phone is rotated, we don't alter these directions based
00241     // on orientation. To dismiss the preview, the user always swipes toward
00242     // the filmstrip.

00246       if (e.detail.vy < -1)

00248       break;

00252       break;
00253     case 'right': // go to previous image if fast enough
00254       if (e.detail.vx > 1 && currentItemIndex > 0)
00255         previewItem(currentItemIndex - 1);

00313   function addItem(item) {
00314     // Thumbnails go from most recent to least recent.
00315     items.unshift(item);
00316 
00317     // Create an image element for this new thumbnail and display it

00320     item.element.classList.add('thumbnail');
00321     filmstrip.insertBefore(item.element, filmstrip.firstElementChild);

00323     // If we have too many thumbnails now, remove the oldest one from
00324     // the array, and remove its element from the filmstrip and release
00325     // its blob url
00326     if (items.length > MAX_THUMBNAILS) {
00327       var oldest = items.pop();

00389   // Create a thumbnail size canvas, copy the <img> or <video> into it
00390   // cropping the edges as needed to make it fit, and then extract the
00391   // thumbnail image as a blob and pass it to the callback.
00392   function createThumbnail(imageBlob, video, rotation, mirrored, callback) 
00393   {
00394     offscreenImage.src = URL.createObjectURL(imageBlob);
00395     offscreenImage.onload = function() {
00396       // Create a thumbnail image
00397       var canvas = document.createElement('canvas');
00398       var context = canvas.getContext('2d');
00399       canvas.width = THUMBNAIL_WIDTH;
00400       canvas.height = THUMBNAIL_HEIGHT;
00401       var imgWidth = offscreenImage.width;
00402       var imgHeight = offscreenImage.height;
00403       var scalex = canvas.width / imgWidth;
00404       var scaley = canvas.height / imgHeight;
00405 
00406       // Take the larger of the two scales: we crop the image to the thumbnail
00407       var scale = Math.max(scalex, scaley);
00408 
00409       // Calculate the region of the image that will be copied to the
00410       // canvas to create the thumbnail
00411       var w = Math.round(THUMBNAIL_WIDTH / scale);
00412       var h = Math.round(THUMBNAIL_HEIGHT / scale);
00413       var x = Math.round((imgWidth - w) / 2);
00414       var y = Math.round((imgHeight - h) / 2);
00415 
00416       var centerX = Math.floor(THUMBNAIL_WIDTH / 2);
00417       var centerY = Math.floor(THUMBNAIL_HEIGHT / 2);
00418 
00419       // If a orientation is specified, rotate/mirroring the canvas context.
00420       if (rotation || mirrored) {
00421         context.save();
00422         // All transformation are applied to the center of the thumbnail.
00423         context.translate(centerX, centerY);
00424       }
00425 
00426       if (mirrored) {
00427         context.scale(-1, 1);
00428       }
00429       if (rotation) {
00430         switch (rotation) {
00431         case 90:
00432           context.rotate(Math.PI / 2);
00433           break;
00434         case 180:
00435           context.rotate(Math.PI);
00436           break;
00437         case 270:
00438           context.rotate(-Math.PI / 2);
00439           break;
00440         }
00441       }
00442 
00443       if (rotation || mirrored) {
00444         context.translate(-centerX, -centerY);
00445       }
00446 
00447       // Draw that region of the image into the canvas, scaling it down
00448       context.drawImage(offscreenImage, x, y, w, h,
00449                         0, 0, THUMBNAIL_WIDTH, THUMBNAIL_HEIGHT);
00450 
00451       // Restore the default rotation so the play arrow comes out correctly
00452       if (rotation || mirrored) {
00453         context.restore();
00454       }
00455 
00456       // We're done with the offscreen image now
00457       URL.revokeObjectURL(offscreenImage.src);
00458       offscreenImage.onload = null;
00459       offscreenImage.src = '';
00460 
00461       if (video) {
00462         // Superimpose a translucent play button over
00463         // the thumbnail to distinguish it from a still photo
00464         // thumbnail. First draw a transparent gray circle.
00465         context.fillStyle = 'rgba(0, 0, 0, .3)';
00466         context.beginPath();
00467         context.arc(THUMBNAIL_WIDTH / 2, THUMBNAIL_HEIGHT / 2,
00468                              THUMBNAIL_HEIGHT / 3, 0, 2 * Math.PI, false);
00469         context.fill();
00470 
00471         // Now outline the circle in white
00472         context.strokeStyle = 'rgba(255,255,255,.6)';
00473         context.lineWidth = 2;
00474         context.stroke();
00475 
00476         // And add a white play arrow.
00477         context.beginPath();
00478         context.fillStyle = 'rgba(255,255,255,.6)';
00479         // The height of an equilateral triangle is sqrt(3)/2 times the side
00480         var side = THUMBNAIL_HEIGHT / 3;
00481         var triangle_height = side * Math.sqrt(3) / 2;
00482         context.moveTo(THUMBNAIL_WIDTH / 2 + triangle_height * 2 / 3,
00483                                 THUMBNAIL_HEIGHT / 2);
00484         context.lineTo(THUMBNAIL_WIDTH / 2 - triangle_height / 3,
00485                                 THUMBNAIL_HEIGHT / 2 - side / 2);
00486         context.lineTo(THUMBNAIL_WIDTH / 2 - triangle_height / 3,
00487                                 THUMBNAIL_HEIGHT / 2 + side / 2);
00488         context.closePath();
00489         context.fill();
00490       }
