00051 template<class Alloc, class Copy>
00052 bool nsTArray_base<Alloc, Copy>::UsesAutoArrayBuffer() const {
00053   if (!mHdr->mIsAutoArray) {
00054     return false;
00055   }
00056 
00057   // This is nuts.  If we were sane, we'd pass elemAlign as a parameter to
00058   // this function.  Unfortunately this function is called in nsTArray_base's
00059   // destructor, at which point we don't know elem_type's alignment.
00060   //
00061   // We'll fall on our face and return true when we should say false if
00062   //
00063   //   * we're not using our auto buffer,
00064   //   * elemAlign == 4, and
00065   //   * mHdr == GetAutoArrayBuffer(8).
00066   //
00067   // This could happen if |*this| lives on the heap and malloc allocated our
00068   // buffer on the heap adjacent to |*this|.
00069   //
00070   // However, we can show that this can't happen.  If |this| is an auto array
00071   // (as we ensured at the beginning of the method), GetAutoArrayBuffer(8)
00072   // always points to memory owned by |*this|, because (as we assert below)
00073   //
00074   //   * GetAutoArrayBuffer(8) is at most 4 bytes past GetAutoArrayBuffer(4), and 
00075   //   * sizeof(nsTArrayHeader) > 4.
00076   //
00077   // Since nsAutoTArray always contains an nsTArrayHeader,
00078   // GetAutoArrayBuffer(8) will always point inside the auto array object,
00079   // even if it doesn't point at the beginning of the header.
00080   //
00081   // Note that this means that we can't store elements with alignment 16 in an
00082   // nsTArray, because GetAutoArrayBuffer(16) could lie outside the memory
00083   // owned by this nsAutoTArray.  We statically assert that elem_type's
00084   // alignment is 8 bytes or less in nsAutoArrayBase.
00085 
00086   static_assert(sizeof(nsTArrayHeader) > 4,
00087                 "see comment above");
00088 
00089 #ifdef DEBUG
00090   ptrdiff_t diff = reinterpret_cast<const char*>(GetAutoArrayBuffer(8)) -
00091                    reinterpret_cast<const char*>(GetAutoArrayBuffer(4));
00092   NS_ABORT_IF_FALSE(diff >= 0 && diff <= 4, "GetAutoArrayBuffer doesn't do what we expect.");
00093 #endif
00094 
00095   return mHdr == GetAutoArrayBuffer(4) || mHdr == GetAutoArrayBuffer(8);
00096 }

00100 typename Alloc::ResultTypeProxy
00101 nsTArray_base<Alloc, Copy>::EnsureCapacity(size_type capacity, size_type elemSize) {
00102   // This should be the most common case so test this first
00103   if (capacity <= mHdr->mCapacity)
00104     return Alloc::SuccessResult();
00105 
00106   // If the requested memory allocation exceeds size_type(-1)/2, then
00107   // our doubling algorithm may not be able to allocate it.
00108   // Additionally we couldn't fit in the Header::mCapacity
00109   // member. Just bail out in cases like that.  We don't want to be
00110   // allocating 2 GB+ arrays anyway.
00111   if ((uint64_t)capacity * elemSize > size_type(-1)/2) {
00112     Alloc::SizeTooBig();
00113     return Alloc::FailureResult();
00114   }
00115 
00116   if (mHdr == EmptyHdr()) {
00117     // Malloc() new data
00118     Header *header = static_cast<Header*>
00119                      (Alloc::Malloc(sizeof(Header) + capacity * elemSize));
00120     if (!header)
00121       return Alloc::FailureResult();
00122     header->mLength = 0;
00123     header->mCapacity = capacity;
00124     header->mIsAutoArray = 0;
00125     mHdr = header;
00126 
00127     return Alloc::SuccessResult();
00128   }
00129 
00130   // We increase our capacity so |capacity * elemSize + sizeof(Header)| is the
00131   // next power of two, if this value is less than pageSize bytes, or otherwise
00132   // so it's the next multiple of pageSize.
00133   const uint32_t pageSizeBytes = 12;
00134   const uint32_t pageSize = 1 << pageSizeBytes;
00135 
00136   uint32_t minBytes = capacity * elemSize + sizeof(Header);
00137   uint32_t bytesToAlloc;
00138   if (minBytes >= pageSize) {
00139     // Round up to the next multiple of pageSize.
00140     bytesToAlloc = pageSize * ((minBytes + pageSize - 1) / pageSize);
00141   }
00142   else {
00143     // Round up to the next power of two.  See
00144     // http://graphics.stanford.edu/~seander/bithacks.html
00145     bytesToAlloc = minBytes - 1;
00146     bytesToAlloc |= bytesToAlloc >> 1;
00147     bytesToAlloc |= bytesToAlloc >> 2;
00148     bytesToAlloc |= bytesToAlloc >> 4;
00149     bytesToAlloc |= bytesToAlloc >> 8;
00150     bytesToAlloc |= bytesToAlloc >> 16;
00151     bytesToAlloc++;
00152 
00153     MOZ_ASSERT((bytesToAlloc & (bytesToAlloc - 1)) == 0,
00154                "nsTArray's allocation size should be a power of two!");
00155   }
00156 
00157   Header *header;
00158   if (UsesAutoArrayBuffer() || !Copy::allowRealloc) {
00159     // Malloc() and copy
00160     header = static_cast<Header*>(Alloc::Malloc(bytesToAlloc));
00161     if (!header)
00162       return Alloc::FailureResult();
00163 
00164     Copy::CopyHeaderAndElements(header, mHdr, Length(), elemSize);
00165 
00166     if (!UsesAutoArrayBuffer())
00167       Alloc::Free(mHdr);
00168   } else {
00169     // Realloc() existing data
00170     header = static_cast<Header*>(Alloc::Realloc(mHdr, bytesToAlloc));
00171     if (!header)
00172       return Alloc::FailureResult();
00173   }
00174 
00175   // How many elements can we fit in bytesToAlloc?
00176   uint32_t newCapacity = (bytesToAlloc - sizeof(Header)) / elemSize;
00177   MOZ_ASSERT(newCapacity >= capacity, "Didn't enlarge the array enough!");
00178   header->mCapacity = newCapacity;
00179 
00180   mHdr = header;
00181 
00182   return Alloc::SuccessResult();
00183 }

00187 nsTArray_base<Alloc, Copy>::ShrinkCapacity(size_type elemSize, size_t elemAlign) {
00188   if (mHdr == EmptyHdr() || UsesAutoArrayBuffer())
00189     return;
00190 
00191   if (mHdr->mLength >= mHdr->mCapacity)  // should never be greater than...
00192     return;
00193 
00194   size_type length = Length();
00195 
00196   if (IsAutoArray() && GetAutoArrayBuffer(elemAlign)->mCapacity >= length) {
00197     Header* header = GetAutoArrayBuffer(elemAlign);
00198 
00199     // Copy data, but don't copy the header to avoid overwriting mCapacity
00200     header->mLength = length;
00201     Copy::CopyElements(header + 1, mHdr + 1, length, elemSize);
00202 
00203     Alloc::Free(mHdr);
00204     mHdr = header;
00205     return;
00206   }
00207 
00208   if (length == 0) {
00209     MOZ_ASSERT(!IsAutoArray(), "autoarray should have fit 0 elements");
00210     Alloc::Free(mHdr);
00211     mHdr = EmptyHdr();
00212     return;
00213   }
00214 
00215   size_type size = sizeof(Header) + length * elemSize;
00216   void *ptr = Alloc::Realloc(mHdr, size);
00217   if (!ptr)
00218     return;
00219   mHdr = static_cast<Header*>(ptr);

00226                                 size_type oldLen, size_type newLen,
00227                                 size_type elemSize, size_t elemAlign) {
00228   if (oldLen == newLen)
00229     return;
00230 
00231   // Determine how many elements need to be shifted
00232   size_type num = mHdr->mLength - (start + oldLen);
00233 
00234   // Compute the resulting length of the array
00235   mHdr->mLength += newLen - oldLen;
00236   if (mHdr->mLength == 0) {
00237     ShrinkCapacity(elemSize, elemAlign);
00238   } else {
00239     // Maybe nothing needs to be shifted
00240     if (num == 0)
00241       return;

00243     start *= elemSize;
00244     newLen *= elemSize;

00247     Copy::MoveElements(base + newLen, base + oldLen, num, elemSize);

00307                                               size_type elemSize, size_t elemAlign) {
00308 
00309   // EnsureNotUsingAutoArrayBuffer will set mHdr = sEmptyHdr even if we have an
00310   // auto buffer.  We need to point mHdr back to our auto buffer before we
00311   // return, otherwise we'll forget that we have an auto buffer at all!
00312   // IsAutoArrayRestorer takes care of this for us.
00313 
00314   IsAutoArrayRestorer ourAutoRestorer(*this, elemAlign);
00315   typename nsTArray_base<Allocator, Copy>::IsAutoArrayRestorer otherAutoRestorer(other, elemAlign);
00316 
00317   // If neither array uses an auto buffer which is big enough to store the
00318   // other array's elements, then ensure that both arrays use malloc'ed storage
00319   // and swap their mHdr pointers.
00320   if ((!UsesAutoArrayBuffer() || Capacity() < other.Length()) &&
00321       (!other.UsesAutoArrayBuffer() || other.Capacity() < Length())) {
00322 
00323     if (!EnsureNotUsingAutoArrayBuffer(elemSize) ||
00324         !other.EnsureNotUsingAutoArrayBuffer(elemSize)) {
00325       return Alloc::FailureResult();
00326     }
00327 
00328     Header *temp = mHdr;
00329     mHdr = other.mHdr;
00330     other.mHdr = temp;
00331 
00332     return Alloc::SuccessResult();
00333   }
00334 
00335   // Swap the two arrays by copying, since at least one is using an auto
00336   // buffer which is large enough to hold all of the other's elements.  We'll
00337   // copy the shorter array into temporary storage.
00338   //
00339   // (We could do better than this in some circumstances.  Suppose we're
00340   // swapping arrays X and Y.  X has space for 2 elements in its auto buffer,
00341   // but currently has length 4, so it's using malloc'ed storage.  Y has length
00342   // 2.  When we swap X and Y, we don't need to use a temporary buffer; we can
00343   // write Y straight into X's auto buffer, write X's malloc'ed buffer on top
00344   // of Y, and then switch X to using its auto buffer.)

00346   if (!Alloc::Successful(EnsureCapacity(other.Length(), elemSize)) ||
00347       !Allocator::Successful(other.EnsureCapacity(Length(), elemSize))) {
00348     return Alloc::FailureResult();
00349   }
00350 
00351   // The EnsureCapacity calls above shouldn't have caused *both* arrays to
00352   // switch from their auto buffers to malloc'ed space.
00353   NS_ABORT_IF_FALSE(UsesAutoArrayBuffer() ||
00354                     other.UsesAutoArrayBuffer(),
00355                     "One of the arrays should be using its auto buffer.");

00358   size_type largerLength = XPCOM_MAX(Length(), other.Length());

00360   if (Length() <= other.Length()) {
00361     smallerElements = Hdr() + 1;
00362     largerElements = other.Hdr() + 1;
00363   }
00364   else {
00365     smallerElements = other.Hdr() + 1;
00366     largerElements = Hdr() + 1;
00367   }
00368

00370   // allocate this space on the stack, if it's not too large.  Sounds like a
