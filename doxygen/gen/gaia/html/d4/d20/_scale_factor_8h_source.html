00015  * This class represents a scaling factor between two different pixel unit
00016  * systems. This is effectively a type-safe float, intended to be used in
00017  * combination with the known-type instances of gfx::Point, gfx::Rect, etc.
00018  *
00019  * Note that some parts of the code that pre-date this class used separate
00020  * scaling factors for the x and y axes. However, at runtime these values
00021  * were always expected to be the same, so this class uses only one scale
00022  * factor for both axes. The two constructors that take two-axis scaling
00023  * factors check to ensure that this assertion holds.
00024  */
00025 template<class src, class dst>
00026 struct ScaleFactor {
00027   float scale;
00028 
00029   ScaleFactor() : scale(1.0) {}
00030   ScaleFactor(const ScaleFactor<src, dst>& aCopy) : scale(aCopy.scale) {}
00031   explicit ScaleFactor(float aScale) : scale(aScale) {}
00032 
00033   explicit ScaleFactor(float aX, float aY) : scale(aX) {
00034     MOZ_ASSERT(fabs(aX - aY) < 1e-6);
00035   }
00036 
00037   explicit ScaleFactor(gfxSize aScale) : scale(aScale.width) {
00038     MOZ_ASSERT(fabs(aScale.width - aScale.height) < 1e-6);
00039   }
00040 
00041   ScaleFactor<dst, src> Inverse() {
00042     return ScaleFactor<dst, src>(1 / scale);
00043   }
00044 
00045   bool operator==(const ScaleFactor<src, dst>& aOther) const {
00046     return scale == aOther.scale;
00047   }
00048 
00049   bool operator!=(const ScaleFactor<src, dst>& aOther) const {
00050     return !(*this == aOther);
00051   }
00052 
00053   template<class other>
00054   ScaleFactor<other, dst> operator/(const ScaleFactor<src, other>& aOther) const {
00055     return ScaleFactor<other, dst>(scale / aOther.scale);
00056   }
00057 
00058   template<class other>
