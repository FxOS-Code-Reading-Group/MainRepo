00026 /*
00027 ** PR_Malloc, PR_Calloc, PR_Realloc, and PR_Free have the same signatures
00028 ** as their libc equivalent malloc, calloc, realloc, and free, and have
00029 ** the same semantics.  (Note that the argument type size_t is replaced
00030 ** by PRUint32.)  Memory allocated by PR_Malloc, PR_Calloc, or PR_Realloc
00031 ** must be freed by PR_Free.

00042 /*
00043 ** The following are some convenience macros defined in terms of

00047 /***********************************************************************
00048 ** FUNCTION:    PR_MALLOC()
00049 ** DESCRIPTION:
00050 **   PR_NEW() allocates an untyped item of size _size from the heap.
00051 ** INPUTS:  _size: size in bytes of item to be allocated
00052 ** OUTPUTS: untyped pointer to the node allocated
00053 ** RETURN:  pointer to node or error returned from malloc().
00054 ***********************************************************************/
00055 #define PR_MALLOC(_bytes) (PR_Malloc((_bytes)))
00056 
00057 /***********************************************************************
00058 ** FUNCTION:    PR_NEW()
00059 ** DESCRIPTION:
00060 **   PR_NEW() allocates an item of type _struct from the heap.
00061 ** INPUTS:  _struct: a data type
00062 ** OUTPUTS: pointer to _struct
00063 ** RETURN:  pointer to _struct or error returns from malloc().
00064 ***********************************************************************/
00065 #define PR_NEW(_struct) ((_struct *) PR_MALLOC(sizeof(_struct)))

00104 **   PR_DELETE() unallocates an object previosly allocated via PR_NEW()

00110 #define PR_DELETE(_ptr) { PR_Free(_ptr); (_ptr) = NULL; }

00114 ** DESCRIPTION:
00115 **   PR_FREEIF() conditionally unallocates an object previously allocated

00117 **   equal to zero (0), the object is not released.
00118 ** INPUTS:  pointer to previously allocated object
00119 ** OUTPUTS: the referenced object is conditionally returned to the heap
00120 ** RETURN:  void
00121 ***********************************************************************/
00122 #define PR_FREEIF(_ptr) if (_ptr) PR_DELETE(_ptr)
