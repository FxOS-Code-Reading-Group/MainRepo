03277 inline bool
03278 nsFrameList::ContinueRemoveFrame(nsIFrame* aFrame)
03279 {
03280   MOZ_ASSERT(!aFrame->GetPrevSibling() || !aFrame->GetNextSibling(),
03281              "Forgot to call StartRemoveFrame?");
03282   if (aFrame == mLastChild) {
03283     MOZ_ASSERT(!aFrame->GetNextSibling(), "broken frame list");
03284     nsIFrame* prevSibling = aFrame->GetPrevSibling();
03285     if (!prevSibling) {
03286       MOZ_ASSERT(aFrame == mFirstChild, "broken frame list");
03287       mFirstChild = mLastChild = nullptr;
03288       return true;
03289     }
03290     MOZ_ASSERT(prevSibling->GetNextSibling() == aFrame, "Broken frame linkage");
03291     prevSibling->SetNextSibling(nullptr);
03292     mLastChild = prevSibling;
03293     return true;
03294   }
03295   if (aFrame == mFirstChild) {
03296     MOZ_ASSERT(!aFrame->GetPrevSibling(), "broken frame list");
03297     mFirstChild = aFrame->GetNextSibling();
03298     aFrame->SetNextSibling(nullptr);
03299     MOZ_ASSERT(mFirstChild, "broken frame list");
03300     return true;
03301   }
03302   return false;
03303 }
