00032 // Popup control state enum. The values in this enum must go from most
00033 // permissive to least permissive so that it's safe to push state in
00034 // all situations. Pushing popup state onto the stack never makes the
00035 // current popup state less permissive (see
00036 // nsGlobalWindow::PushPopupControlState()).
00037 enum PopupControlState {
00038   openAllowed = 0,  // open that window without worries
00039   openControlled,   // it's a popup, but allow it
00040   openAbused,       // it's a popup. disallow it, but allow domain override.
00041   openOverridden    // disallow window open
00042 };
