00063 // This was all created at a time in which we were using NSPR for host
00064 // resolution and we were propagating NSPR types like "PRAddrInfo" and
00065 // "PRNetAddr" all over Gecko. This made it hard to use another host
00066 // resolver -- we were locked into NSPR. The goal here is to get away
00067 // from that. We'll translate what we get from NSPR or any other host
00068 // resolution library into the types below and use them in Gecko.

00077 // This struct is similar to operating system structs like "sockaddr", used for
00078 // things like "connect" and "getsockname". When tempted to cast or do dumb
00079 // copies of this struct to another struct, bear compiler-computed padding
00080 // in mind. The size of this struct, and the layout of the data in it, may
00081 // not be what you expect.
00082 union NetAddr {
00083   struct {
00084     uint16_t family;                /* address family (0x00ff maskable) */
00085     char data[14];                  /* raw address data */
00086   } raw;
00087   struct {
00088     uint16_t family;                /* address family (AF_INET) */
00089     uint16_t port;                  /* port number */
00090     uint32_t ip;                    /* The actual 32 bits of address */
00091   } inet;
00092   struct {
00093     uint16_t family;                /* address family (AF_INET6) */
00094     uint16_t port;                  /* port number */
00095     uint32_t flowinfo;              /* routing information */
00096     IPv6Addr ip;                    /* the actual 128 bits of address */
00097     uint32_t scope_id;              /* set of interfaces for a scope */
00098   } inet6;
00099 #if defined(XP_UNIX) || defined(XP_OS2)
00100   struct {                          /* Unix domain socket address */
00101     uint16_t family;                /* address family (AF_UNIX) */
00102 #ifdef XP_OS2
00103     char path[108];                 /* null-terminated pathname */
00104 #else
00105     char path[104];                 /* null-terminated pathname */
00106 #endif
00107   } local;
00108 #endif
00109 };
