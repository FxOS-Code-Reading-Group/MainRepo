00319       return;

00326       case 'touchstart':

00328         this._keyPressStart = Date.now();

00330 
00331         if (key != 'delete') {
00332           if (keypadSoundIsEnabled) {
00333             // We do not support long press if not on a call
00334             TonePlayer.start(gTonesFrequencies[key], !this._onCall);
00335           }
00336         }
00337 
00338         // Manage long press
00339         if ((key == '0' && !this._onCall) || key == 'delete') {
00340           this._holdTimer = setTimeout(function(self) {
