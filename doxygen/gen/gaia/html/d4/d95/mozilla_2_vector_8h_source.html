00039 /*
00040  * Check that the given capacity wastes the minimal amount of space if
00041  * allocated on the heap.  This means that cap*sizeof(T) is as close to a
00042  * power-of-two as possible.  growStorageBy() is responsible for ensuring
00043  * this.
00044  */
00045 template<typename T>
00046 static bool CapacityHasExcessSpace(size_t cap)
00047 {
00048   size_t size = cap * sizeof(T);
00049   return RoundUpPow2(size) - size >= sizeof(T);
00050 }

00052 /*
00053  * This template class provides a default implementation for vector operations
00054  * when the element type is not known to be a POD, as judged by IsPod.
00055  */
00056 template<typename T, size_t N, class AP, class ThisVector, bool IsPod>
00057 struct VectorImpl
00058 {
00059     /* Destroys constructed objects in the range [begin, end). */
00060     static inline void destroy(T* begin, T* end) {
00061       for (T* p = begin; p < end; ++p)
00062         p->~T();
00063     }
00064 
00065     /* Constructs objects in the uninitialized range [begin, end). */
00066     static inline void initialize(T* begin, T* end) {
00067       for (T* p = begin; p < end; ++p)
00068         new(p) T();
00069     }
00070 
00071     /*
00072      * Copy-constructs objects in the uninitialized range
00073      * [dst, dst+(srcend-srcbeg)) from the range [srcbeg, srcend).
00074      */
00075     template<typename U>
00076     static inline void copyConstruct(T* dst, const U* srcbeg, const U* srcend) {
00077       for (const U* p = srcbeg; p < srcend; ++p, ++dst)
00078         new(dst) T(*p);
00079     }
00080 
00081     /*
00082      * Move-constructs objects in the uninitialized range
00083      * [dst, dst+(srcend-srcbeg)) from the range [srcbeg, srcend).
00084      */
00085     template<typename U>
00086     static inline void moveConstruct(T* dst, const U* srcbeg, const U* srcend) {
00087       for (const U* p = srcbeg; p < srcend; ++p, ++dst)
00088         new(dst) T(Move(*p));
00089     }
00090 
00091     /*
00092      * Copy-constructs objects in the uninitialized range [dst, dst+n) from the
00093      * same object u.
00094      */
00095     template<typename U>
00096     static inline void copyConstructN(T* dst, size_t n, const U& u) {
00097       for (T* end = dst + n; dst < end; ++dst)
00098         new(dst) T(u);
00099     }
00100 
00101     /*
00102      * Grows the given buffer to have capacity newCap, preserving the objects
00103      * constructed in the range [begin, end) and updating v. Assumes that (1)
00104      * newCap has not overflowed, and (2) multiplying newCap by sizeof(T) will
00105      * not overflow.
00106      */
00107     static inline bool
00108     growTo(VectorBase<T, N, AP, ThisVector>& v, size_t newCap) {
00109       MOZ_ASSERT(!v.usingInlineStorage());
00110       MOZ_ASSERT(!CapacityHasExcessSpace<T>(newCap));
00111       T* newbuf = reinterpret_cast<T*>(v.malloc_(newCap * sizeof(T)));
00112       if (!newbuf)
00113         return false;
00114       T* dst = newbuf;
00115       T* src = v.beginNoCheck();
00116       for (; src < v.endNoCheck(); ++dst, ++src)
00117         new(dst) T(Move(*src));
00118       VectorImpl::destroy(v.beginNoCheck(), v.endNoCheck());
00119       v.free_(v.mBegin);
00120       v.mBegin = newbuf;
00121       /* v.mLength is unchanged. */
00122       v.mCapacity = newCap;
00123       return true;
00124     }
00125 };

00127 /*
00128  * This partial template specialization provides a default implementation for
00129  * vector operations when the element type is known to be a POD, as judged by
00130  * IsPod.
00131  */
00132 template<typename T, size_t N, class AP, class ThisVector>
00133 struct VectorImpl<T, N, AP, ThisVector, true>
00134 {
00135     static inline void destroy(T*, T*) {}
00136 
00137     static inline void initialize(T* begin, T* end) {
00138       /*
00139        * You would think that memset would be a big win (or even break even)
00140        * when we know T is a POD. But currently it's not. This is probably
00141        * because |append| tends to be given small ranges and memset requires
00142        * a function call that doesn't get inlined.
00143        *
00144        * memset(begin, 0, sizeof(T) * (end-begin));
00145        */
00146       for (T* p = begin; p < end; ++p)
00147         new(p) T();
00148     }
00149 
00150     template<typename U>
00151     static inline void copyConstruct(T* dst, const U* srcbeg, const U* srcend) {
00152       /*
00153        * See above memset comment. Also, notice that copyConstruct is
00154        * currently templated (T != U), so memcpy won't work without
00155        * requiring T == U.
00156        *
00157        * memcpy(dst, srcbeg, sizeof(T) * (srcend - srcbeg));
00158        */
00159       for (const U* p = srcbeg; p < srcend; ++p, ++dst)
00160         *dst = *p;
00161     }
00162 
00163     template<typename U>
00164     static inline void moveConstruct(T* dst, const U* srcbeg, const U* srcend) {
00165       copyConstruct(dst, srcbeg, srcend);
00166     }
00167 
00168     static inline void copyConstructN(T* dst, size_t n, const T& t) {
00169       for (T* end = dst + n; dst < end; ++dst)
00170         *dst = t;
00171     }

00173     static inline bool
00174     growTo(VectorBase<T, N, AP, ThisVector>& v, size_t newCap) {
00175       MOZ_ASSERT(!v.usingInlineStorage());
00176       MOZ_ASSERT(!CapacityHasExcessSpace<T>(newCap));
00177       size_t oldSize = sizeof(T) * v.mCapacity;
00178       size_t newSize = sizeof(T) * newCap;

00180       if (!newbuf)
00181         return false;
00182       v.mBegin = newbuf;
00183       /* v.mLength is unchanged. */
00184       v.mCapacity = newCap;

00637 /*
00638  * This function will create a new heap buffer with capacity newCap,
00639  * move all elements in the inline buffer to this new buffer,
00640  * and fail on OOM.
00641  */
00642 template<typename T, size_t N, class AP, class TV>
00643 inline bool
00644 VectorBase<T, N, AP, TV>::convertToHeapStorage(size_t newCap)
00645 {
00646   MOZ_ASSERT(usingInlineStorage());
00647 
00648   /* Allocate buffer. */
00649   MOZ_ASSERT(!detail::CapacityHasExcessSpace<T>(newCap));
00650   T* newBuf = reinterpret_cast<T*>(this->malloc_(newCap * sizeof(T)));
00651   if (!newBuf)
00652     return false;
00653 
00654   /* Copy inline elements into heap buffer. */
00655   Impl::moveConstruct(newBuf, beginNoCheck(), endNoCheck());
00656   Impl::destroy(beginNoCheck(), endNoCheck());
00657 
00658   /* Switch in heap buffer. */
00659   mBegin = newBuf;
00660   /* mLength is unchanged. */
00661   mCapacity = newCap;
00662   return true;

00665 template<typename T, size_t N, class AP, class TV>
00666 MOZ_NEVER_INLINE bool
00667 VectorBase<T, N, AP, TV>::growStorageBy(size_t incr)
00668 {
00669   MOZ_ASSERT(mLength + incr > mCapacity);
00670   MOZ_ASSERT_IF(!usingInlineStorage(),
00671                 !detail::CapacityHasExcessSpace<T>(mCapacity));
00672 
00673   /*
00674    * When choosing a new capacity, its size should is as close to 2**N bytes
00675    * as possible.  2**N-sized requests are best because they are unlikely to
00676    * be rounded up by the allocator.  Asking for a 2**N number of elements
00677    * isn't as good, because if sizeof(T) is not a power-of-two that would
00678    * result in a non-2**N request size.
00679    */
00680 
00681   size_t newCap;
00682 
00683   if (incr == 1) {
00684     if (usingInlineStorage()) {
00685       /* This case occurs in ~70--80% of the calls to this function. */
00686       size_t newSize =
00687         tl::RoundUpPow2<(sInlineCapacity + 1) * sizeof(T)>::value;
00688       newCap = newSize / sizeof(T);
00689       goto convert;
00690     }
00691 
00692     if (mLength == 0) {
00693       /* This case occurs in ~0--10% of the calls to this function. */
00694       newCap = 1;
00695       goto grow;
00696     }
00697 
00698     /* This case occurs in ~15--20% of the calls to this function. */
00699 
00700     /*
00701      * Will mLength * 4 *sizeof(T) overflow?  This condition limits a vector
00702      * to 1GB of memory on a 32-bit system, which is a reasonable limit.  It
00703      * also ensures that
00704      *
00705      *   static_cast<char*>(end()) - static_cast<char*>(begin())
00706      *
00707      * doesn't overflow ptrdiff_t (see bug 510319).
00708      */
00709     if (mLength & tl::MulOverflowMask<4 * sizeof(T)>::value) {
00710       this->reportAllocOverflow();
00711       return false;
00712     }
00713 
00714     /*
00715      * If we reach here, the existing capacity will have a size that is already
00716      * as close to 2^N as sizeof(T) will allow.  Just double the capacity, and
00717      * then there might be space for one more element.
00718      */
00719     newCap = mLength * 2;
00720     if (detail::CapacityHasExcessSpace<T>(newCap))
00721       newCap += 1;
00722   } else {
00723     /* This case occurs in ~2% of the calls to this function. */
00724     size_t newMinCap = mLength + incr;
00725 
00726     /* Did mLength + incr overflow?  Will newCap * sizeof(T) overflow? */
00727     if (newMinCap < mLength ||
00728         newMinCap & tl::MulOverflowMask<2 * sizeof(T)>::value)
00729     {
00730       this->reportAllocOverflow();
00731       return false;
00732     }
00733 
00734     size_t newMinSize = newMinCap * sizeof(T);
00735     size_t newSize = RoundUpPow2(newMinSize);
00736     newCap = newSize / sizeof(T);
00737   }
00738 
00739   if (usingInlineStorage()) {
00740   convert:

01147 /*
01148  * STL-like container providing a short-lived, dynamic buffer.  Vector calls the
01149  * constructors/destructors of all elements stored in its internal buffer, so
01150  * non-PODs may be safely used.  Additionally, Vector will store the first N
01151  * elements in-place before resorting to dynamic allocation.
01152  *
01153  * T requirements:
01154  *  - default and copy constructible, assignable, destructible
01155  *  - operations do not throw
01156  * N requirements:
01157  *  - any value, however, N is clamped to min/max values
01158  * AllocPolicy:
01159  *  - see "Allocation policies" in AllocPolicy.h (defaults to
01160  *    mozilla::MallocAllocPolicy)
01161  *
01162  * Vector is not reentrant: T member functions called during Vector member
01163  * functions must not call back into the same object!
01164  */

01166          size_t MinInlineCapacity = 0,
01167          class AllocPolicy = MallocAllocPolicy>

01169   : public VectorBase<T,
01170                       MinInlineCapacity,
01171                       AllocPolicy,
01172                       Vector<T, MinInlineCapacity, AllocPolicy> >
01173 {
