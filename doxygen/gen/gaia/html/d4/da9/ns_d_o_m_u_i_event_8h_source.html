00038   CalculateScreenPoint(nsPresContext* aPresContext, nsEvent* aEvent)
00039   {
00040     if (!aEvent ||
00041         (aEvent->eventStructType != NS_MOUSE_EVENT &&
00042          aEvent->eventStructType != NS_MOUSE_SCROLL_EVENT &&
00043          aEvent->eventStructType != NS_WHEEL_EVENT &&
00044          aEvent->eventStructType != NS_DRAG_EVENT &&
00045          aEvent->eventStructType != NS_SIMPLE_GESTURE_EVENT)) {
00046       return nsIntPoint(0, 0);
00047     }
00048 
00049     nsGUIEvent* event = static_cast<nsGUIEvent*>(aEvent);
00050     if (!event->widget) {
00051       return mozilla::LayoutDeviceIntPoint::ToUntyped(aEvent->refPoint);
00052     }
00053 
00054     mozilla::LayoutDeviceIntPoint offset = aEvent->refPoint +
00055       mozilla::LayoutDeviceIntPoint::FromUntyped(event->widget->WidgetToScreenOffset());
00056     nscoord factor = aPresContext->DeviceContext()->UnscaledAppUnitsPerDevPixel();
00057     return nsIntPoint(nsPresContext::AppUnitsToIntCSSPixels(offset.x * factor),
00058                       nsPresContext::AppUnitsToIntCSSPixels(offset.y * factor));
00059   }

00061   static CSSIntPoint CalculateClientPoint(nsPresContext* aPresContext,
00062                                           nsEvent* aEvent,
00063                                           CSSIntPoint* aDefaultClientPoint)
00064   {
00065     if (!aEvent ||
00066         (aEvent->eventStructType != NS_MOUSE_EVENT &&
00067          aEvent->eventStructType != NS_MOUSE_SCROLL_EVENT &&
00068          aEvent->eventStructType != NS_WHEEL_EVENT &&
00069          aEvent->eventStructType != NS_DRAG_EVENT &&
00070          aEvent->eventStructType != NS_SIMPLE_GESTURE_EVENT) ||
00071         !aPresContext ||
00072         !static_cast<nsGUIEvent*>(aEvent)->widget) {
00073       return aDefaultClientPoint
00074              ? *aDefaultClientPoint
00075              : CSSIntPoint(0, 0);
00076     }
00077 
00078     nsIPresShell* shell = aPresContext->GetPresShell();
00079     if (!shell) {
00080       return CSSIntPoint(0, 0);
00081     }
00082     nsIFrame* rootFrame = shell->GetRootFrame();
00083     if (!rootFrame) {
00084       return CSSIntPoint(0, 0);
00085     }
00086     nsPoint pt =
00087       nsLayoutUtils::GetEventCoordinatesRelativeTo(aEvent, rootFrame);
00088 
00089     return CSSIntPoint::FromAppUnitsRounded(pt);
00090   }
