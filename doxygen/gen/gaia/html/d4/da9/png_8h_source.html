00335 /*
00336  * Y2K compliance in libpng:
00337  * =========================
00338  *
00339  *    June 27, 2013
00340  *
00341  *    Since the PNG Development group is an ad-hoc body, we can't make
00342  *    an official declaration.
00343  *
00344  *    This is your unofficial assurance that libpng from version 0.71 and
00345  *    upward through 1.5.17 are Y2K compliant.  It is my belief that
00346  *    earlier versions were also Y2K compliant.
00347  *
00348  *    Libpng only has two year fields.  One is a 2-byte unsigned integer
00349  *    that will hold years up to 65535.  The other holds the date in text
00350  *    format, and will hold years up to 9999.
00351  *
00352  *    The integer is
00353  *        "png_uint_16 year" in png_time_struct.
00354  *
00355  *    The string is
00356  *        "char time_buffer[29]" in png_struct.  This will be no
00357  *    longer used in libpng-1.6.0 and will be removed from libpng-1.7.0.
00358  *
00359  *    There are seven time-related functions:
00360  *        png.c: png_convert_to_rfc_1123() in png.c
00361  *          (formerly png_convert_to_rfc_1152() in error)
00362  *        png_convert_from_struct_tm() in pngwrite.c, called in pngwrite.c

01202 #ifdef PNG_READ_ALPHA_MODE_SUPPORTED
01203 /* How the alpha channel is interpreted - this affects how the color channels of
01204  * a PNG file are returned when an alpha channel, or tRNS chunk in a palette
01205  * file, is present.
01206  *
01207  * This has no effect on the way pixels are written into a PNG output
01208  * datastream. The color samples in a PNG datastream are never premultiplied
01209  * with the alpha samples.
01210  *
01211  * The default is to return data according to the PNG specification: the alpha
01212  * channel is a linear measure of the contribution of the pixel to the
01213  * corresponding composited pixel.  The gamma encoded color channels must be
01214  * scaled according to the contribution and to do this it is necessary to undo
01215  * the encoding, scale the color values, perform the composition and reencode
01216  * the values.  This is the 'PNG' mode.
01217  *
01218  * The alternative is to 'associate' the alpha with the color information by
01219  * storing color channel values that have been scaled by the alpha.  The
01220  * advantage is that the color channels can be resampled (the image can be
01221  * scaled) in this form.  The disadvantage is that normal practice is to store
01222  * linear, not (gamma) encoded, values and this requires 16-bit channels for
01223  * still images rather than the 8-bit channels that are just about sufficient if
01224  * gamma encoding is used.  In addition all non-transparent pixel values,
01225  * including completely opaque ones, must be gamma encoded to produce the final
01226  * image.  This is the 'STANDARD', 'ASSOCIATED' or 'PREMULTIPLIED' mode (the
01227  * latter being the two common names for associated alpha color channels.)
01228  *
01229  * Since it is not necessary to perform arithmetic on opaque color values so
01230  * long as they are not to be resampled and are in the final color space it is
01231  * possible to optimize the handling of alpha by storing the opaque pixels in
01232  * the PNG format (adjusted for the output color space) while storing partially
01233  * opaque pixels in the standard, linear, format.  The accuracy required for
01234  * standard alpha composition is relatively low, because the pixels are
01235  * isolated, therefore typically the accuracy loss in storing 8-bit linear
01236  * values is acceptable.  (This is not true if the alpha channel is used to
01237  * simulate transparency over large areas - use 16 bits or the PNG mode in
01238  * this case!)  This is the 'OPTIMIZED' mode.  For this mode a pixel is
01239  * treated as opaque only if the alpha value is equal to the maximum value.
01240  *
01241  * The final choice is to gamma encode the alpha channel as well.  This is
01242  * broken because, in practice, no implementation that uses this choice
01243  * correctly undoes the encoding before handling alpha composition.  Use this
01244  * choice only if other serious errors in the software or hardware you use
01245  * mandate it; the typical serious error is for dark halos to appear around
01246  * opaque areas of the composited PNG image because of arithmetic overflow.
01247  *
01248  * The API function png_set_alpha_mode specifies which of these choices to use
01249  * with an enumerated 'mode' value and the gamma of the required output:
01250  */
01251 #define PNG_ALPHA_PNG           0 /* according to the PNG standard */
01252 #define PNG_ALPHA_STANDARD      1 /* according to Porter/Duff */
01253 #define PNG_ALPHA_ASSOCIATED    1 /* as above; this is the normal practice */
01254 #define PNG_ALPHA_PREMULTIPLIED 1 /* as above */
01255 #define PNG_ALPHA_OPTIMIZED     2 /* 'PNG' for opaque pixels, else 'STANDARD' */
01256 #define PNG_ALPHA_BROKEN        3 /* the alpha channel is gamma encoded */
01257 
01258 PNG_FP_EXPORT(227, void, png_set_alpha_mode, (png_structp png_ptr, int mode,
01259     double output_gamma))
01260 PNG_FIXED_EXPORT(228, void, png_set_alpha_mode_fixed, (png_structp png_ptr,
01261     int mode, png_fixed_point output_gamma))
01262 #endif

01264 #if defined(PNG_READ_GAMMA_SUPPORTED) || defined(PNG_READ_ALPHA_MODE_SUPPORTED)
01265 /* The output_gamma value is a screen gamma in libpng terminology: it expresses
01266  * how to decode the output values, not how they are encoded.  The values used
01267  * correspond to the normal numbers used to describe the overall gamma of a
01268  * computer display system; for example 2.2 for an sRGB conformant system.  The
01269  * values are scaled by 100000 in the _fixed version of the API (so 220000 for
01270  * sRGB.)
01271  *
01272  * The inverse of the value is always used to provide a default for the PNG file
01273  * encoding if it has no gAMA chunk and if png_set_gamma() has not been called
01274  * to override the PNG gamma information.
01275  *
01276  * When the ALPHA_OPTIMIZED mode is selected the output gamma is used to encode
01277  * opaque pixels however pixels with lower alpha values are not encoded,
01278  * regardless of the output gamma setting.
01279  *
01280  * When the standard Porter Duff handling is requested with mode 1 the output
01281  * encoding is set to be linear and the output_gamma value is only relevant
01282  * as a default for input data that has no gamma information.  The linear output
01283  * encoding will be overridden if png_set_gamma() is called - the results may be
01284  * highly unexpected!
01285  *
01286  * The following numbers are derived from the sRGB standard and the research
01287  * behind it.  sRGB is defined to be approximated by a PNG gAMA chunk value of
01288  * 0.45455 (1/2.2) for PNG.  The value implicitly includes any viewing
01289  * correction required to take account of any differences in the color
01290  * environment of the original scene and the intended display environment; the
01291  * value expresses how to *decode* the image for display, not how the original
01292  * data was *encoded*.
01293  *
01294  * sRGB provides a peg for the PNG standard by defining a viewing environment.
01295  * sRGB itself, and earlier TV standards, actually use a more complex transform
01296  * (a linear portion then a gamma 2.4 power law) than PNG can express.  (PNG is
01297  * limited to simple power laws.)  By saying that an image for direct display on
01298  * an sRGB conformant system should be stored with a gAMA chunk value of 45455
01299  * (11.3.3.2 and 11.3.3.5 of the ISO PNG specification) the PNG specification
01300  * makes it possible to derive values for other display systems and
01301  * environments.
01302  *
01303  * The Mac value is deduced from the sRGB based on an assumption that the actual
01304  * extra viewing correction used in early Mac display systems was implemented as
01305  * a power 1.45 lookup table.
01306  *
01307  * Any system where a programmable lookup table is used or where the behavior of
01308  * the final display device characteristics can be changed requires system
01309  * specific code to obtain the current characteristic.  However this can be
01310  * difficult and most PNG gamma correction only requires an approximate value.
01311  *
01312  * By default, if png_set_alpha_mode() is not called, libpng assumes that all
01313  * values are unencoded, linear, values and that the output device also has a
01314  * linear characteristic.  This is only very rarely correct - it is invariably
01315  * better to call png_set_alpha_mode() with PNG_DEFAULT_sRGB than rely on the
01316  * default if you don't know what the right answer is!
01317  *
01318  * The special value PNG_GAMMA_MAC_18 indicates an older Mac system (pre Mac OS
01319  * 10.6) which used a correction table to implement a somewhat lower gamma on an
01320  * otherwise sRGB system.
01321  *
01322  * Both these values are reserved (not simple gamma values) in order to allow
01323  * more precise correction internally in the future.
01324  *
01325  * NOTE: the following values can be passed to either the fixed or floating
01326  * point APIs, but the floating point API will also accept floating point
01327  * values.
01328  */
01329 #define PNG_DEFAULT_sRGB -1       /* sRGB gamma and color space */
01330 #define PNG_GAMMA_MAC_18 -2       /* Old Mac '1.8' gamma and color space */
01331 #define PNG_GAMMA_sRGB   220000   /* Television standards--matches sRGB gamma */
01332 #define PNG_GAMMA_LINEAR PNG_FP_1 /* Linear */
01333 #endif

01335 /* The following are examples of calls to png_set_alpha_mode to achieve the
01336  * required overall gamma correction and, where necessary, alpha
01337  * premultiplication.
01338  *
01339  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_DEFAULT_sRGB);
01340  *    This is the default libpng handling of the alpha channel - it is not
01341  *    pre-multiplied into the color components.  In addition the call states
01342  *    that the output is for a sRGB system and causes all PNG files without gAMA
01343  *    chunks to be assumed to be encoded using sRGB.
01344  *
01345  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_GAMMA_MAC);
01346  *    In this case the output is assumed to be something like an sRGB conformant
01347  *    display preceeded by a power-law lookup table of power 1.45.  This is how
01348  *    early Mac systems behaved.
01349  *
01350  * png_set_alpha_mode(pp, PNG_ALPHA_STANDARD, PNG_GAMMA_LINEAR);
01351  *    This is the classic Jim Blinn approach and will work in academic
01352  *    environments where everything is done by the book.  It has the shortcoming
01353  *    of assuming that input PNG data with no gamma information is linear - this
01354  *    is unlikely to be correct unless the PNG files where generated locally.
01355  *    Most of the time the output precision will be so low as to show
01356  *    significant banding in dark areas of the image.
01357  *
01358  * png_set_expand_16(pp);
01359  * png_set_alpha_mode(pp, PNG_ALPHA_STANDARD, PNG_DEFAULT_sRGB);
01360  *    This is a somewhat more realistic Jim Blinn inspired approach.  PNG files
01361  *    are assumed to have the sRGB encoding if not marked with a gamma value and
01362  *    the output is always 16 bits per component.  This permits accurate scaling
01363  *    and processing of the data.  If you know that your input PNG files were
01364  *    generated locally you might need to replace PNG_DEFAULT_sRGB with the
01365  *    correct value for your system.
01366  *
01367  * png_set_alpha_mode(pp, PNG_ALPHA_OPTIMIZED, PNG_DEFAULT_sRGB);
01368  *    If you just need to composite the PNG image onto an existing background
01369  *    and if you control the code that does this you can use the optimization
01370  *    setting.  In this case you just copy completely opaque pixels to the
01371  *    output.  For pixels that are not completely transparent (you just skip
01372  *    those) you do the composition math using png_composite or png_composite_16
01373  *    below then encode the resultant 8-bit or 16-bit values to match the output
01374  *    encoding.
01375  *
01376  * Other cases
01377  *    If neither the PNG nor the standard linear encoding work for you because
01378  *    of the software or hardware you use then you have a big problem.  The PNG
01379  *    case will probably result in halos around the image.  The linear encoding
01380  *    will probably result in a washed out, too bright, image (it's actually too
01381  *    contrasty.)  Try the ALPHA_OPTIMIZED mode above - this will probably
01382  *    substantially reduce the halos.  Alternatively try:
01383  *
01384  * png_set_alpha_mode(pp, PNG_ALPHA_BROKEN, PNG_DEFAULT_sRGB);
01385  *    This option will also reduce the halos, but there will be slight dark
01386  *    halos round the opaque parts of the image where the background is light.
01387  *    In the OPTIMIZED mode the halos will be light halos where the background
01388  *    is dark.  Take your pick - the halos are unavoidable unless you can get
01389  *    your hardware/software fixed!  (The OPTIMIZED approach is slightly
01390  *    faster.)
01391  *
01392  * When the default gamma of PNG files doesn't match the output gamma.
01393  *    If you have PNG files with no gamma information png_set_alpha_mode allows
01394  *    you to provide a default gamma, but it also sets the ouput gamma to the
01395  *    matching value.  If you know your PNG files have a gamma that doesn't
01396  *    match the output you can take advantage of the fact that
01397  *    png_set_alpha_mode always sets the output gamma but only sets the PNG
01398  *    default if it is not already set:
01399  *
01400  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_DEFAULT_sRGB);
01401  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_GAMMA_MAC);
01402  *    The first call sets both the default and the output gamma values, the
01403  *    second call overrides the output gamma without changing the default.  This
01404  *    is easier than achieving the same effect with png_set_gamma.  You must use
01405  *    PNG_ALPHA_PNG for the first call - internal checking in png_set_alpha will
01406  *    fire if more than one call to png_set_alpha_mode and png_set_background is
01407  *    made in the same read operation, however multiple calls with PNG_ALPHA_PNG
01408  *    are ignored.
01409  */

02525 /* Two macros to return the first row and first column of the original,
02526  * full, image which appears in a given pass.  'pass' is in the range 0
02527  * to 6 and the result is in the range 0 to 7.
02528  */
02529 #define PNG_PASS_START_ROW(pass) (((1&~(pass))<<(3-((pass)>>1)))&7)
02530 #define PNG_PASS_START_COL(pass) (((1& (pass))<<(3-(((pass)+1)>>1)))&7)
02531 
02532 /* A macro to return the offset between pixels in the output row for a pair of
02533  * pixels in the input - effectively the inverse of the 'COL_SHIFT' macro that
02534  * follows.  Note that ROW_OFFSET is the offset from one row to the next whereas
02535  * COL_OFFSET is from one column to the next, within a row.
02536  */
02537 #define PNG_PASS_ROW_OFFSET(pass) ((pass)>2?(8>>(((pass)-1)>>1)):8)
02538 #define PNG_PASS_COL_OFFSET(pass) (1<<((7-(pass))>>1))
02539 
02540 /* Two macros to help evaluate the number of rows or columns in each
02541  * pass.  This is expressed as a shift - effectively log2 of the number or
02542  * rows or columns in each 8x8 tile of the original image.
02543  */
02544 #define PNG_PASS_ROW_SHIFT(pass) ((pass)>2?(8-(pass))>>1:3)
02545 #define PNG_PASS_COL_SHIFT(pass) ((pass)>1?(7-(pass))>>1:3)

02548  * pass of an image given its height or width.  In fact these macros may
02549  * return non-zero even though the sub-image is empty, because the other
02550  * dimension may be empty for a small image.
02551  */
02552 #define PNG_PASS_ROWS(height, pass) (((height)+(((1<<PNG_PASS_ROW_SHIFT(pass))\
02553    -1)-PNG_PASS_START_ROW(pass)))>>PNG_PASS_ROW_SHIFT(pass))
02554 #define PNG_PASS_COLS(width, pass) (((width)+(((1<<PNG_PASS_COL_SHIFT(pass))\
