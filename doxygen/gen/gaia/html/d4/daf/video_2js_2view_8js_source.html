00357     if (controlShowing) {

00368     // Since we don't always get reliable 'ended' events, see if
00369     // we've reached the end this way.
00370     // See: https://bugzilla.mozilla.org/show_bug.cgi?id=783512
00371     // If we're within 1 second of the end of the video, register
00372     // a timeout a half a second after we'd expect an ended event.
00373     if (!endedTimer) {
00374       if (!dragging && dom.player.currentTime >= dom.player.duration - 1) {
00375         var timeUntilEnd = (dom.player.duration - dom.player.currentTime + .5);
00376         endedTimer = setTimeout(playerEnded, timeUntilEnd * 1000);
00377       }
00378     } else if (dragging && dom.player.currentTime < dom.player.duration - 1) {
00379       // If there is a timer set and we drag away from the end, cancel the timer
00380       clearTimeout(endedTimer);
00381       endedTimer = null;
