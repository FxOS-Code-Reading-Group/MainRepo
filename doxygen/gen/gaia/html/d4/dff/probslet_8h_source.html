00045 /*
00046 *************************************************************************
00047 ** FUNCTION:    PR_Select
00048 ** DESCRIPTION:
00049 **
00050 ** The call returns as soon as I/O is ready on one or more of the underlying
00051 ** file/socket descriptors or an exceptional condition is pending. A count of the 
00052 ** number of ready descriptors is returned unless a timeout occurs in which case 
00053 ** zero is returned.  On return, PR_Select replaces the given descriptor sets with 
00054 ** subsets consisting of those descriptors that are ready for the requested condition.
00055 ** The total number of ready descriptors in all the sets is the return value.
00056 **
00057 ** INPUTS:
00058 **   PRInt32 num             
00059 **       This argument is unused but is provided for select(unix) interface
00060 **       compatability.  All input PR_fd_set arguments are self-describing
00061 **       with its own maximum number of elements in the set.
00062 **                               
00063 **   PR_fd_set *readfds
00064 **       A set describing the io descriptors for which ready for reading
00065 **       condition is of interest.  
00066 **                               
00067 **   PR_fd_set *writefds
00068 **       A set describing the io descriptors for which ready for writing
00069 **       condition is of interest.  
00070 **                               
00071 **   PR_fd_set *exceptfds
00072 **       A set describing the io descriptors for which exception pending
00073 **       condition is of interest.  
00074 **
00075 **   Any of the above readfds, writefds or exceptfds may be given as NULL 
00076 **   pointers if no descriptors are of interest for that particular condition.                          
00077 **   
00078 **   PRIntervalTime timeout  
00079 **       Amount of time the call will block waiting for I/O to become ready. 
00080 **       If this time expires without any I/O becoming ready, the result will
00081 **       be zero.
00082 **
00083 ** OUTPUTS:    
00084 **   PR_fd_set *readfds
00085 **       A set describing the io descriptors which are ready for reading.
00086 **                               
00087 **   PR_fd_set *writefds
00088 **       A set describing the io descriptors which are ready for writing.
00089 **                               
00090 **   PR_fd_set *exceptfds
00091 **       A set describing the io descriptors which have pending exception.
00092 **
00093 ** RETURN:PRInt32
00094 **   Number of io descriptors with asked for conditions or zero if the function
00095 **   timed out or -1 on failure.  The reason for the failure is obtained by 
00096 **   calling PR_GetError().
00097 ** XXX can we implement this on windoze and mac?
00098 **************************************************************************
00099 */
00100 NSPR_API(PRInt32) PR_Select(
00101     PRInt32 num, PR_fd_set *readfds, PR_fd_set *writefds,
00102     PR_fd_set *exceptfds, PRIntervalTime timeout);

00104 /* 
00105 ** The following are not thread safe for two threads operating on them at the
00106 ** same time.
00107 **
00108 ** The following routines are provided for manipulating io descriptor sets.
00109 ** PR_FD_ZERO(&fdset) initializes a descriptor set fdset to the null set.
00110 ** PR_FD_SET(fd, &fdset) includes a particular file descriptor fd in fdset.
00111 ** PR_FD_CLR(fd, &fdset) removes a file descriptor fd from fdset.  
00112 ** PR_FD_ISSET(fd, &fdset) is nonzero if file descriptor fd is a member of 
00113 ** fdset, zero otherwise.
00114 **
00115 ** PR_FD_NSET(osfd, &fdset) includes a particular native file descriptor osfd
00116 ** in fdset.
00117 ** PR_FD_NCLR(osfd, &fdset) removes a native file descriptor osfd from fdset.  
00118 ** PR_FD_NISSET(osfd, &fdset) is nonzero if native file descriptor osfd is a member of 
00119 ** fdset, zero otherwise.
00120 */
