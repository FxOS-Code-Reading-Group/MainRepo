00136     // Primitive types cannot have properties

00147     if (keys.length === 0) {

00149         var name = value.name ? ': ' + value.name : '';

00270   function formatProperty(ctx, value, recurseTimes, visibleKeys, key, array) {
00271     var name, str, desc;
00272     desc = Object.getOwnPropertyDescriptor(value, key) || { value: value[key] };
00273     if (desc.get) {
00274       if (desc.set) {
00275         str = ctx.stylize('[Getter/Setter]', 'special');
00276       } else {
00277         str = ctx.stylize('[Getter]', 'special');
00278       }
00279     } else {
00280       if (desc.set) {
00281         str = ctx.stylize('[Setter]', 'special');
00282       }
00283     }
00284     if (visibleKeys.indexOf(key) < 0) {
00285       name = '[' + key + ']';
00286     }
00287     if (!str) {
00288       if (ctx.seen.indexOf(desc.value) < 0) {
00289         if (recurseTimes === null) {
00290           str = formatValue(ctx, desc.value, null);
00291         } else {
00292           str = formatValue(ctx, desc.value, recurseTimes - 1);
00293         }
00294         if (str.indexOf('\n') > -1) {
00295           if (array) {
00296             str = str.split('\n').map(function(line) {
00297               return ' ' + line;
00298             }).join('\n').substr(2);
00299           } else {
00300             str = '\n' + str.split('\n').map(function(line) {
00301               return ' ' + line;
00302             }).join('\n');
00303           }
00304         }
00305       } else {
00306         str = ctx.stylize('[Circular]', 'special');
00307       }
00308     }
00309     if (typeof name === 'undefined') {
00310       if (array && key.match(/^\d+$/)) {
00311         return str;
00312       }
00313       name = JSON.stringify('' + key);
00314       if (name.match(/^"([a-zA-Z_][a-zA-Z_0-9]*)"$/)) {
00315         name = name.substr(1, name.length - 2);
00316         name = ctx.stylize(name, 'name');
00317       } else {
00318         name = name.replace(/'/g, "\\'")
00319                    .replace(/\\"/g, '"')
00320                    .replace(/(^"|"$)/g, "'");
00321         name = ctx.stylize(name, 'string');
00322       }
00323     }
00324 
00325     return name + ': ' + str;
