00013 /*
00014  * JS::PerfMeasurement is a generic way to access detailed performance
00015  * measurement APIs provided by your operating system.  The details of
00016  * exactly how this works and what can be measured are highly
00017  * system-specific, but this interface is (one hopes) implementable
00018  * on top of all of them.
00019  *
00020  * To use this API, create a PerfMeasurement object, passing its
00021  * constructor a bitmask indicating which events you are interested
00022  * in.  Thereafter, Start() zeroes all counters and starts timing;
00023  * Stop() stops timing again; and the counters for the events you
00024  * requested are available as data values after calling Stop().  The
00025  * object may be reused for many measurements.
00026  */
00027 class JS_FRIEND_API(PerfMeasurement)
00028 {
00029   protected:
00030     // Implementation-specific data, if any.
00031     void* impl;
00032 
00033   public:
00034     /*
00035      * Events that may be measured.  Taken directly from the list of
00036      * "generalized hardware performance event types" in the Linux
00037      * perf_event API, plus some of the "software events".
00038      */
00039     enum EventMask {
00040         CPU_CYCLES          = 0x00000001,
00041         INSTRUCTIONS        = 0x00000002,
00042         CACHE_REFERENCES    = 0x00000004,
00043         CACHE_MISSES        = 0x00000008,
00044         BRANCH_INSTRUCTIONS = 0x00000010,
00045         BRANCH_MISSES       = 0x00000020,
00046         BUS_CYCLES          = 0x00000040,
00047         PAGE_FAULTS         = 0x00000080,
00048         MAJOR_PAGE_FAULTS   = 0x00000100,
00049         CONTEXT_SWITCHES    = 0x00000200,
00050         CPU_MIGRATIONS      = 0x00000400,
00051 
00052         ALL                 = 0x000007ff,
00053         NUM_MEASURABLE_EVENTS  = 11
00054     };
00055 
00056     /*
00057      * Bitmask of events that will be measured when this object is
00058      * active (between Start() and Stop()).  This may differ from the
00059      * bitmask passed to the constructor if the platform does not
00060      * support measuring all of the requested events.
00061      */
00062     const EventMask eventsMeasured;
00063 
00064     /*
00065      * Counters for each measurable event.
00066      * Immediately after one of these objects is created, all of the
00067      * counters for enabled events will be zero, and all of the
00068      * counters for disabled events will be uint64_t(-1).
00069      */
00070     uint64_t cpu_cycles;
00071     uint64_t instructions;
00072     uint64_t cache_references;
00073     uint64_t cache_misses;
00074     uint64_t branch_instructions;
00075     uint64_t branch_misses;
00076     uint64_t bus_cycles;
00077     uint64_t page_faults;
00078     uint64_t major_page_faults;
00079     uint64_t context_switches;
00080     uint64_t cpu_migrations;
00081 
00082     /*
00083      * Prepare to measure the indicated set of events.  If not all of
00084      * the requested events can be measured on the current platform,
00085      * then the eventsMeasured bitmask will only include the subset of
00086      * |toMeasure| corresponding to the events that can be measured.
00087      */
00088     PerfMeasurement(EventMask toMeasure);
00089 
00090     /* Done with this set of measurements, tear down OS-level state. */
00091     ~PerfMeasurement();
00092 
00093     /* Start a measurement cycle. */
00094     void start();
00095 
00096     /*
00097      * End a measurement cycle, and for each enabled counter, add the
00098      * number of measured events of that type to the appropriate
00099      * visible variable.
00100      */
00101     void stop();
00102 
00103     /* Reset all enabled counters to zero. */
00104     void reset();
00105 
00106     /*
00107      * True if this platform supports measuring _something_, i.e. it's

00109      */
00110     static bool canMeasureSomething();
