00053 def parse_ini(filename):
00054     log = logging.getLogger(__name__)
00055     with open(filename) as f:
00056         data = f.readlines()
00057     section = 'default'
00058     imports = { section: [] }
00059     for line in data:
00060         if line.strip() == "" or line.startswith('!') or line.startswith('#'):
00061             continue
00062         elif line.strip().startswith('['): # Section header
00063             section = section_line.search(line).group('section')
00064             imports[section] = []
00065         elif '@import' in line: # Import lines
00066             property_file = import_line.search(line).group('filename')
00067             imports[section].append(property_file)
00068         else:
00069             log.warn('parse_ini - found a line with contents '
00070                      'unaccounted for "%s"', line.strip())
00071     return imports
