00533     static bool IsInverseFillType(FillType fill) {
00534         SK_COMPILE_ASSERT(0 == kWinding_FillType, fill_type_mismatch);
00535         SK_COMPILE_ASSERT(1 == kEvenOdd_FillType, fill_type_mismatch);
00536         SK_COMPILE_ASSERT(2 == kInverseWinding_FillType, fill_type_mismatch);
00537         SK_COMPILE_ASSERT(3 == kInverseEvenOdd_FillType, fill_type_mismatch);
00538         return (fill & 2) != 0;
00539     }
