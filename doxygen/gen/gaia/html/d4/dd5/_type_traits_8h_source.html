00307 // The trickery used to implement IsBaseOf here makes it possible to use it for
00308 // the cases of private and multiple inheritance.  This code was inspired by the
00309 // sample code here:
00310 //
00311 // http://stackoverflow.com/questions/2910979/how-is-base-of-works
00312 template<class Base, class Derived>
00313 struct BaseOfHelper
00314 {
00315   public:
00316     operator Base*() const;
00317     operator Derived*();
00318 };
00319 
00320 template<class Base, class Derived>
00321 struct BaseOfTester
00322 {
00323   private:
00324     template<class T>
00325     static char test(Derived*, T);
00326     static int test(Base*, int);
00327 
00328   public:
00329     static const bool value =
00330       sizeof(test(BaseOfHelper<Base, Derived>(), int())) == sizeof(char);
00331 };
00332 
00333 template<class Base, class Derived>
00334 struct BaseOfTester<Base, const Derived>
00335 {
00336   private:
00337     template<class T>
00338     static char test(Derived*, T);
00339     static int test(Base*, int);
00340 
00341   public:
00342     static const bool value =
00343       sizeof(test(BaseOfHelper<Base, Derived>(), int())) == sizeof(char);
00344 };

00357 /*
00358  * IsBaseOf allows to know whether a given class is derived from another.
00359  *
00360  * Consider the following class definitions:
00361  *
00362  *   class A {};
00363  *   class B : public A {};
00364  *   class C {};
00365  *
00366  * mozilla::IsBaseOf<A, B>::value is true;
00367  * mozilla::IsBaseOf<A, C>::value is false;
00368  */

00370 struct IsBaseOf
00371   : IntegralConstant<bool, detail::BaseOfTester<Base, Derived>::value>
