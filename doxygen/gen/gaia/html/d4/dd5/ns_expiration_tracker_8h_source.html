00100     nsresult AddObject(T* aObj) {
00101       nsExpirationState* state = aObj->GetExpirationState();
00102       NS_ASSERTION(!state->IsTracked(), "Tried to add an object that's already tracked");
00103       nsTArray<T*>& generation = mGenerations[mNewestGeneration];
00104       uint32_t index = generation.Length();
00105       if (index > nsExpirationState::MAX_INDEX_IN_GENERATION) {
00106         NS_WARNING("More than 256M elements tracked, this is probably a problem");

00125     void RemoveObject(T* aObj) {
00126       nsExpirationState* state = aObj->GetExpirationState();
00127       NS_ASSERTION(state->IsTracked(), "Tried to remove an object that's not tracked");
00128       nsTArray<T*>& generation = mGenerations[state->mGeneration];
00129       uint32_t index = state->mIndexInGeneration;
00130       NS_ASSERTION(generation.Length() > index &&
00131                    generation[index] == aObj, "Object is lying about its index");
00132       // Move the last object to fill the hole created by removing aObj
00133       uint32_t last = generation.Length() - 1;
00134       T* lastObj = generation[last];
00135       generation[index] = lastObj;
00136       lastObj->GetExpirationState()->mIndexInGeneration = index;
00137       generation.RemoveElementAt(last);
00138       state->mGeneration = nsExpirationState::NOT_TRACKED;
00139       // We do not check whether we need to stop the timer here. The timer
00140       // will check that itself next time it fires. Checking here would not
00141       // be efficient since we'd need to track all generations. Also we could
00142       // thrash by incessantly creating and destroying timers if someone
00143       // kept adding and removing an object from the tracker.

00162     void AgeOneGeneration() {
00163       if (mInAgeOneGeneration) {
00164         NS_WARNING("Can't reenter AgeOneGeneration from NotifyExpired");
00165         return;
00166       }
00167       
00168       mInAgeOneGeneration = true;
00169       uint32_t reapGeneration = 
00170         mNewestGeneration > 0 ? mNewestGeneration - 1 : K - 1;
00171       nsTArray<T*>& generation = mGenerations[reapGeneration];
00172       // The following is rather tricky. We have to cope with objects being
00173       // removed from this generation either because of a call to RemoveObject
00174       // (or indirectly via MarkUsed) inside NotifyExpired. Fortunately no
00175       // objects can be added to this generation because it's not the newest
00176       // generation. We depend on the fact that RemoveObject can only cause
00177       // the indexes of objects in this generation to *decrease*, not increase.
00178       // So if we start from the end and work our way backwards we are guaranteed
00179       // to see each object at least once.
00180       uint32_t index = generation.Length();
00181       for (;;) {
00182         // Objects could have been removed so index could be outside
00183         // the array
00184         index = XPCOM_MIN(index, generation.Length());
00185         if (index == 0)
00186           break;
00187         --index;
00188         NotifyExpired(generation[index]);
00189       }
00190       // Any leftover objects from reapGeneration just end up in the new
00191       // newest-generation. This is bad form, though, so warn if there are any.
00192       if (!generation.IsEmpty()) {
00193         NS_WARNING("Expired objects were not removed or marked used");
00194       }
00195       // Free excess memory used by the generation array, since we probably
00196       // just removed most or all of its elements.
00197       generation.Compact();
00198       mNewestGeneration = reapGeneration;
00199       mInAgeOneGeneration = false;
00200     }
