00074 NS_ReadLine (StreamType* aStream, nsLineBuffer<CharT> * aBuffer,
00075              StringType & aLine, bool *more)
00076 {
00077   CharT eolchar = 0; // the first eol char or 1 after \r\n or \n\r is found
00078 
00079   aLine.Truncate();
00080 
00081   while (1) { // will be returning out of this loop on eol or eof
00082     if (aBuffer->start == aBuffer->end) { // buffer is empty.  Read into it.
00083       uint32_t bytesRead;
00084       nsresult rv = aStream->Read(aBuffer->buf, kLineBufferSize, &bytesRead);
00085       if (NS_FAILED(rv) || MOZ_UNLIKELY(bytesRead == 0)) {
00086         *more = false;
00087         return rv;
00088       }
00089       aBuffer->start = aBuffer->buf;
00090       aBuffer->end = aBuffer->buf + bytesRead;
00091       *(aBuffer->end) = '\0';
00092     }
00093 
00094     /*
00095      * Walk the buffer looking for an end-of-line.
00096      * There are 3 cases to consider:
00097      *  1. the eol char is the last char in the buffer
00098      *  2. the eol char + one more char at the end of the buffer
00099      *  3. the eol char + two or more chars at the end of the buffer
00100      * we need at least one char after the first eol char to determine if
00101      * it's a \r\n or \n\r sequence (and skip over it), and we need one
00102      * more char after the end-of-line to set |more| correctly.
00103      */
00104     CharT* current = aBuffer->start;
00105     if (MOZ_LIKELY(eolchar == 0)) {
00106       for ( ; current < aBuffer->end; ++current) {
00107         if (*current == '\n' || *current == '\r') {
00108           eolchar = *current;
00109           *current++ = '\0';
00110           aLine.Append(aBuffer->start);
00111           break;
00112         }
00113       }
00114     }
00115     if (MOZ_LIKELY(eolchar != 0)) {
00116       for ( ; current < aBuffer->end; ++current) {
00117         if ((eolchar == '\r' && *current == '\n') ||
00118             (eolchar == '\n' && *current == '\r')) {
00119           eolchar = 1;
00120           continue;
00121         }
00122         aBuffer->start = current;
00123         *more = true;
00124         return NS_OK;
00125       }
00126     }
00127 
00128     if (eolchar == 0)
00129       aLine.Append(aBuffer->start);
00130     aBuffer->start = aBuffer->end; // mark the buffer empty
00131   }
00132 }
