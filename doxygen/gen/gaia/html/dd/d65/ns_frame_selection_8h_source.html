00098   // mDirection: eDirPrevious or eDirNext.
00099   //             * Note for visual bidi movement:
00100   //             eDirPrevious means 'left-then-up' if the containing block is LTR, 
00101   //             'right-then-up' if it is RTL.
00102   //             eDirNext means 'right-then-down' if the containing block is LTR, 
00103   //             'left-then-down' if it is RTL.
00104   //             Between paragraphs, eDirPrevious means "go to the visual end of the 
00105   //             previous paragraph", and eDirNext means "go to the visual beginning
00106   //             of the next paragraph".
00107   //             Used with: eSelectCharacter, eSelectWord, eSelectLine, eSelectParagraph.
00108   nsDirection mDirection;

00118   // mWordMovementType: An enum that determines whether to prefer the start or end of a word
