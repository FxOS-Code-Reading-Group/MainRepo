00071 // A stack entry exists to allow the JS engine to inform SPS of the current
00072 // backtrace, but also to instrument particular points in C++ in case stack
00073 // walking is not available on the platform we are running on.
00074 //
00075 // Each entry has a descriptive string, a relevant stack address, and some extra
00076 // information the JS engine might want to inform SPS of. This class inherits
00077 // from the JS engine's version of the entry to ensure that the size and layout
00078 // of the two representations are consistent.
00079 class StackEntry : public js::ProfileEntry
00080 {
00081 public:
00082 
00083   bool isCopyLabel() volatile {
00084     return !((uintptr_t)stackAddress() & 0x1);
00085   }
00086 
00087   void setStackAddressCopy(void *sparg, bool copy) volatile {
00088     // Tagged pointer. Less significant bit used to track if mLabel needs a
00089     // copy. Note that we don't need the last bit of the stack address for
00090     // proper ordering. This is optimized for encoding within the JS engine's
00091     // instrumentation, so we do the extra work here of encoding a bit.
00092     // Last bit 1 = Don't copy, Last bit 0 = Copy.
00093     if (copy) {
00094       setStackAddress(reinterpret_cast<void*>(
00095                         reinterpret_cast<uintptr_t>(sparg) & ~0x1));
00096     } else {
00097       setStackAddress(reinterpret_cast<void*>(
00098                         reinterpret_cast<uintptr_t>(sparg) | 0x1));
00099     }
00100   }
00101 };

00118   ~PseudoStack() {
00119     clearMarkers();
00120     if (mStackPointer != 0 || mSignalLock != false ||

00122       // We're releasing the pseudostack while it's still in use.
00123       // The label macros keep a non ref counted reference to the

00130   void addMarker(const char *aMarker)
00131   {
00132     char* markerCopy = strdup(aMarker);
00133     mSignalLock = true;
00134     STORE_SEQUENCER();
00135 
00136     if (mQueueClearMarker) {
00137       clearMarkers();
00138     }
00139     if (!aMarker) {
00140       return; //discard
00141     }
00142     if (size_t(mMarkerPointer) == mozilla::ArrayLength(mMarkers)) {
00143       return; //array full, silently drop
00144     }
00145     mMarkers[mMarkerPointer] = markerCopy;
00146     mMarkerPointer++;
00147 
00148     mSignalLock = false;
00149     STORE_SEQUENCER();
00150   }

00152   // called within signal. Function must be reentrant
00153   const char* getMarker(int aMarkerId)
00154   {
00155     // if mSignalLock then the stack is inconsistent because it's being
00156     // modified by the profiled thread. Post pone these markers
00157     // for the next sample. The odds of a livelock are nearly impossible
00158     // and would show up in a profile as many sample in 'addMarker' thus
00159     // we ignore this scenario.
00160     // if mQueueClearMarker then we've the sampler thread has already
00161     // thread the markers then they are pending deletion.
00162     if (mSignalLock || mQueueClearMarker || aMarkerId < 0 ||
00163       static_cast<mozilla::sig_safe_t>(aMarkerId) >= mMarkerPointer) {
00164       return nullptr;
00165     }
00166     return mMarkers[aMarkerId];
00167   }
