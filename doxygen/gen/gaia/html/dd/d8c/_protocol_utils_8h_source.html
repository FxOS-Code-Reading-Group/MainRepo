00054 // Used internally to represent a "trigger" that might cause a state
00055 // transition.  Triggers are normalized across parent+child to Send
00056 // and Recv (instead of child-in, child-out, parent-in, parent-out) so
00057 // that they can share the same state machine implementation.  To
00058 // further normalize, |Send| is used for 'call', |Recv| for 'answer'.
00059 struct Trigger
00060 {
00061     enum Action { Send, Recv };
00062 
00063     Trigger(Action action, int32_t msg) :
00064         mAction(action),
00065         mMsg(msg)
00066     {}
00067 
00068     Action mAction;
00069     int32_t mMsg;
00070 };
