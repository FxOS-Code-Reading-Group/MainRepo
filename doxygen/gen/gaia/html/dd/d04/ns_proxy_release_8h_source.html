00104 template<class T>
00105 class nsMainThreadPtrHolder MOZ_FINAL
00106 {
00107 public:
00108   // We can only acquire a pointer on the main thread. We to fail fast for
00109   // threading bugs, so by default we assert if our pointer is used or acquired
00110   // off-main-thread. But some consumers need to use the same pointer for
00111   // multiple classes, some of which are main-thread-only and some of which
00112   // aren't. So we allow them to explicitly disable this strict checking.
00113   nsMainThreadPtrHolder(T* ptr, bool strict = true) : mRawPtr(NULL), mStrict(strict) {
00114     // We can only AddRef our pointer on the main thread, which means that the
00115     // holder must be constructed on the main thread.
00116     MOZ_ASSERT(!mStrict || NS_IsMainThread());
00117     NS_IF_ADDREF(mRawPtr = ptr);
00118   }
00119 
00120   // We can be released on any thread.
00121   ~nsMainThreadPtrHolder() {
00122     if (NS_IsMainThread()) {
00123       NS_IF_RELEASE(mRawPtr);
00124     } else if (mRawPtr) {
00125       nsCOMPtr<nsIThread> mainThread = do_GetMainThread();
00126       if (!mainThread) {
00127         NS_WARNING("Couldn't get main thread! Leaking pointer.");
00128         return;
00129       }
00130       NS_ProxyRelease(mainThread, mRawPtr);
00131     }
00132   }
00133 
00134   T* get() {
00135     // Nobody should be touching the raw pointer off-main-thread.
00136     if (mStrict && MOZ_UNLIKELY(!NS_IsMainThread())) {
00137       NS_ERROR("Can't dereference nsMainThreadPtrHolder off main thread");
00138       MOZ_CRASH();
00139     }
00140     return mRawPtr;
00141   }
00142 
00143   bool operator==(const nsMainThreadPtrHolder<T>& aOther) const { return mRawPtr == aOther.mRawPtr; }
00144 
00145   NS_INLINE_DECL_THREADSAFE_REFCOUNTING(nsMainThreadPtrHolder<T>)
00146 
00147 private:
00148   // Our wrapped pointer.
00149   T* mRawPtr;
00150 
00151   // Whether to strictly enforce thread invariants in this class.
00152   bool mStrict;
00153 
00154   // Copy constructor and operator= not implemented. Once constructed, the
00155   // holder is immutable.
00156   T& operator=(nsMainThreadPtrHolder& other);
