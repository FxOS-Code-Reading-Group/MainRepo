00099 nsresult
00100 JSValToDashArray(JSContext* cx, const JS::Value& patternArray,
00101                  FallibleTArray<T>& dashes)
00102 {
00103     // The cap is pretty arbitrary.  16k should be enough for
00104     // anybody...
00105     static const uint32_t MAX_NUM_DASHES = 1 << 14;
00106 
00107     if (!JSVAL_IS_PRIMITIVE(patternArray)) {
00108         JS::Rooted<JSObject*> obj(cx, JSVAL_TO_OBJECT(patternArray));
00109         uint32_t length;
00110         if (!JS_GetArrayLength(cx, obj, &length)) {
00111             // Not an array-like thing
00112             return NS_ERROR_INVALID_ARG;
00113         } else if (length > MAX_NUM_DASHES) {
00114             // Too many dashes in the pattern
00115             return NS_ERROR_ILLEGAL_VALUE;
00116         }
00117 
00118         bool haveNonzeroElement = false;
00119         for (uint32_t i = 0; i < length; ++i) {
00120             JS::Rooted<JS::Value> elt(cx);
00121             double d;
00122             if (!JS_GetElement(cx, obj, i, &elt)) {
00123                 return NS_ERROR_FAILURE;
00124             }
00125             if (!(CoerceDouble(elt, &d) &&
00126                   FloatValidate(d) &&
00127                   d >= 0.0)) {
00128                 // Pattern elements must be finite "numbers" >= 0.
00129                 return NS_ERROR_INVALID_ARG;
00130             } else if (d > 0.0) {
00131                 haveNonzeroElement = true;
00132             }
00133             if (!dashes.AppendElement(d)) {
00134                 return NS_ERROR_OUT_OF_MEMORY;
00135             }
00136         }
00137 
00138         if (dashes.Length() > 0 && !haveNonzeroElement) {
00139             // An all-zero pattern makes no sense.
00140             return NS_ERROR_ILLEGAL_VALUE;
00141         }
00142     } else if (!(JSVAL_IS_VOID(patternArray) || JSVAL_IS_NULL(patternArray))) {
00143         // undefined and null mean "reset to no dash".  Any other
00144         // random garbage is a type error.
00145         return NS_ERROR_INVALID_ARG;
00146     }
00147 
00148     return NS_OK;
00149 }
