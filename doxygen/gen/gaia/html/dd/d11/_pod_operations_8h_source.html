00028 PodZero(T* t)

00030   memset(t, 0, sizeof(T));

00034 template<typename T>
00035 static void
00036 PodZero(T* t, size_t nelem)
00037 {
00038   /*
00039    * This function is often called with 'nelem' small; we use an inline loop
00040    * instead of calling 'memset' with a non-constant length.  The compiler
00041    * should inline the memset call with constant size, though.
00042    */
00043   for (T* end = t + nelem; t < end; t++)
00044     memset(t, 0, sizeof(T));
00045 }

00047 /*
00048  * Arrays implicitly convert to pointers to their first element, which is
00049  * dangerous when combined with the above PodZero definitions.  Adding an
00050  * overload for arrays is ambiguous, so we need another identifier.  The
00051  * ambiguous overload is left to catch mistaken uses of PodZero; if you get a
00052  * compile error involving PodZero and array types, use PodArrayZero instead.
00053  */

00055 static void PodZero(T (&t)[N]) MOZ_DELETE;

00085 template<typename T>
00086 MOZ_ALWAYS_INLINE static void
00087 PodCopy(T* dst, const T* src, size_t nelem)
00088 {
00089   MOZ_ASSERT(dst != src);
00090   MOZ_ASSERT_IF(src < dst, PointerRangeSize(src, static_cast<const T*>(dst)) >= nelem);
00091   MOZ_ASSERT_IF(dst < src, PointerRangeSize(static_cast<const T*>(dst), src) >= nelem);
00092 
00093   if (nelem < 128) {
00094     /*
00095      * Avoid using operator= in this loop, as it may have been
00096      * intentionally deleted by the POD type.
00097      */
00098     for (const T* srcend = src + nelem; src < srcend; src++, dst++)
00099       PodAssign(dst, src);
00100   } else {
00101     memcpy(dst, src, nelem * sizeof(T));
00102   }
00103 }

00105 template<typename T>
00106 MOZ_ALWAYS_INLINE static void
00107 PodCopy(volatile T* dst, const volatile T* src, size_t nelem)
00108 {
00109   MOZ_ASSERT(dst != src);
00110   MOZ_ASSERT_IF(src < dst,
00111                 PointerRangeSize(src, static_cast<const volatile T*>(dst)) >= nelem);
00112   MOZ_ASSERT_IF(dst < src,
00113                 PointerRangeSize(static_cast<const volatile T*>(dst), src) >= nelem);
00114 
00115   /*
00116    * Volatile |dst| requires extra work, because it's undefined behavior to
00117    * modify volatile objects using the mem* functions.  Just write out the
00118    * loops manually, using operator= rather than memcpy for the same reason,
00119    * and let the compiler optimize to the extent it can.
00120    */
00121   for (const volatile T* srcend = src + nelem; src < srcend; src++, dst++)
00122     *dst = *src;
00123 }
