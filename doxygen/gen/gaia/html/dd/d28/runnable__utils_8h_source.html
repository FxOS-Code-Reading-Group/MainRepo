00024 // The generated file contains four major function templates
00025 // (in variants for arbitrary numbers of arguments up to 10,
00026 // which is why it is machine generated). The four templates
00027 // are:
00028 //
00029 // WrapRunnable(o, m, ...) -- wraps a member function m of an object ptr o
00030 // WrapRunnableRet(o, m, ..., r) -- wraps a member function m of an object ptr o
00031 //                                  the function returns something that can
00032 //                                  be assigned to *r
00033 // WrapRunnableNM(f, ...) -- wraps a function f
00034 // WrapRunnableNMRet(f, ..., r) -- wraps a function f that returns something
00035 //                                 that can be assigned to *r

00037 // All of these template functions return a Runnable* which can be passed

00042 static inline nsresult RUN_ON_THREAD(nsIEventTarget *thread, nsIRunnable *runnable, uint32_t flags) {
00043   RefPtr<nsIRunnable> runnable_ref(runnable);
00044   if (thread) {
00045     bool on;
00046     nsresult rv;
00047     rv = thread->IsOnCurrentThread(&on);
00048 
00049     // If the target thread has already shut down, we don't want to assert.
00050     if (rv != NS_ERROR_NOT_INITIALIZED) {
00051       MOZ_ASSERT(NS_SUCCEEDED(rv));
00052     }
00053 
00054     NS_ENSURE_SUCCESS(rv, rv);
00055     if(!on) {
00056       return thread->Dispatch(runnable_ref, flags);
00057     }
00058   }
00059   return runnable_ref->Run();
00060 }

00062 static inline nsresult RUN_ON_THREAD(nsIEventTarget *thread, runnable_args_base *runnable, uint32_t flags) {
00063   // Detect attempts to return a value when in async mode, since this
00064   // most likely means someone is trying to assign to a heap variable
00065   // which is now out of scope.
00066   MOZ_ASSERT((!(runnable->returns_value()) || (flags != NS_DISPATCH_NORMAL)));
00067 
00068   return RUN_ON_THREAD(thread, static_cast<nsIRunnable *>(runnable), flags);
