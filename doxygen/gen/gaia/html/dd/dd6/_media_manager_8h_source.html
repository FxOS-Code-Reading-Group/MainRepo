00215     NS_IMETHOD
00216     Run() MOZ_OVERRIDE
00217     {
00218       NS_ASSERTION(NS_IsMainThread(), "Only call on main thread");
00219       // Make sure mStream is cleared and our reference to the DOMMediaStream
00220       // is dropped on the main thread, no matter what happens in this method.
00221       // Otherwise this object might be destroyed off the main thread,
00222       // releasing DOMMediaStream off the main thread, which is not allowed.
00223       nsRefPtr<DOMMediaStream> stream = mStream.forget();
00224 
00225       nsCOMPtr<nsIObserverService> obs = mozilla::services::GetObserverService();
00226       if (!obs) {
00227         NS_WARNING("Could not get the Observer service for GetUserMedia recording notification.");
00228         return NS_ERROR_FAILURE;
00229       }
00230       nsString msg;
00231       switch (mStatus) {
00232         case STARTING:
00233           msg = NS_LITERAL_STRING("starting");
00234           stream->OnTracksAvailable(mOnTracksAvailableCallback.forget());
00235           break;
00236         case STOPPING:
00237           msg = NS_LITERAL_STRING("shutdown");
00238           if (mListener) {
00239             mListener->SetStopped();
00240           }
00241           break;
00242       }
00243       obs->NotifyObservers(nullptr,
00244                            "recording-device-events",
00245                            msg.get());
00246       // Forward recording events to parent process.
00247       // The events are gathered in chrome process and used for recording indicator
00248       if (XRE_GetProcessType() != GeckoProcessType_Default) {
00249         unused << mozilla::dom::ContentChild::GetSingleton()->SendRecordingDeviceEvents(msg);
00250       }
00251       return NS_OK;
00252     }

00254   protected:
00255     nsRefPtr<GetUserMediaCallbackMediaStreamListener> mListener; // threadsafe
00256     nsRefPtr<DOMMediaStream> mStream;
00257     nsAutoPtr<DOMMediaStream::OnTracksAvailableCallback> mOnTracksAvailableCallback;
00258     GetUserMediaStatus mStatus;

00269 class MediaOperationRunnable : public nsRunnable

00271 public:
00272   // so we can send Stop without AddRef()ing from the MSG thread
00273   MediaOperationRunnable(MediaOperation aType,

00278     MediaEngineSource* aVideoSource,
00279     bool aNeedsFinish)
00280     : mType(aType)
00281     , mStream(aStream)

00285     , mListener(aListener)
00286     , mFinish(aNeedsFinish)

00290   {

00294   NS_IMETHOD
00295   Run() MOZ_OVERRIDE
00296   {
00297     SourceMediaStream *source = mListener->GetSourceStream();
00298     // No locking between these is required as all the callbacks for the
00299     // same MediaStream will occur on the same thread.
00300     if (!source) // means the stream was never Activated()
00301       return NS_OK;
00302 
00303     switch (mType) {
00304       case MEDIA_START:
00305         {
00306           NS_ASSERTION(!NS_IsMainThread(), "Never call on main thread");
00307           nsresult rv;
00308 
00309           source->SetPullEnabled(true);
00310 
00311           DOMMediaStream::TrackTypeHints expectedTracks = 0;
00312           if (mAudioSource) {
00313             rv = mAudioSource->Start(source, kAudioTrack);
00314             if (NS_SUCCEEDED(rv)) {
00315               expectedTracks |= DOMMediaStream::HINT_CONTENTS_AUDIO;
00316             } else {
00317               MM_LOG(("Starting audio failed, rv=%d",rv));
00318             }
00319           }
00320           if (mVideoSource) {
00321             rv = mVideoSource->Start(source, kVideoTrack);
00322             if (NS_SUCCEEDED(rv)) {
00323               expectedTracks |= DOMMediaStream::HINT_CONTENTS_VIDEO;
00324             } else {
00325               MM_LOG(("Starting video failed, rv=%d",rv));
00326             }
00327           }
00328 
00329           mOnTracksAvailableCallback->SetExpectedTracks(expectedTracks);
00330 
00331           MM_LOG(("started all sources"));
00332           // Forward mOnTracksAvailableCallback to GetUserMediaNotificationEvent,
00333           // because mOnTracksAvailableCallback needs to be added to mStream
00334           // on the main thread.
00335           nsRefPtr<GetUserMediaNotificationEvent> event =
00336             new GetUserMediaNotificationEvent(GetUserMediaNotificationEvent::STARTING,
00337                                               mStream.forget(),
00338                                               mOnTracksAvailableCallback.forget());
00339           NS_DispatchToMainThread(event, NS_DISPATCH_NORMAL);
00340         }
00341         break;
00342 
00343       case MEDIA_STOP:
00344         {
00345           NS_ASSERTION(!NS_IsMainThread(), "Never call on main thread");
00346           if (mAudioSource) {
00347             mAudioSource->Stop(source, kAudioTrack);
00348             mAudioSource->Deallocate();
00349           }
00350           if (mVideoSource) {
00351             mVideoSource->Stop(source, kVideoTrack);
00352             mVideoSource->Deallocate();
00353           }
00354           // Do this after stopping all tracks with EndTrack()
00355           if (mFinish) {
00356             source->Finish();
00357           }
00358           nsRefPtr<GetUserMediaNotificationEvent> event =
00359             new GetUserMediaNotificationEvent(mListener, GetUserMediaNotificationEvent::STOPPING);
00360 
00361           NS_DispatchToMainThread(event, NS_DISPATCH_NORMAL);
00362         }
00363         break;
00364 
00365       default:
00366         MOZ_ASSERT(false,"invalid MediaManager operation");
00367         break;
00368     }
00369     return NS_OK;
00370   }
00371 
00372 private:
00373   MediaOperation mType;
00374   nsRefPtr<DOMMediaStream> mStream;
00375   nsAutoPtr<DOMMediaStream::OnTracksAvailableCallback> mOnTracksAvailableCallback;
00376   nsRefPtr<MediaEngineSource> mAudioSource; // threadsafe
00377   nsRefPtr<MediaEngineSource> mVideoSource; // threadsafe
00378   nsRefPtr<GetUserMediaCallbackMediaStreamListener> mListener; // threadsafe
00379   bool mFinish;
