00093     static inline uint32_t
00094     RecoverPixel(uint32_t black, uint32_t white)
00095     {
00096         const uint32_t GREEN_MASK = 0x0000FF00;
00097         const uint32_t ALPHA_MASK = 0xFF000000;
00098 
00099         /* |diff| here is larger when the source image pixel is more transparent.
00100            If both renderings are from the same source image composited with OVER,
00101            then the color values on white will always be greater than those on
00102            black, so |diff| would not overflow.  However, overflow may happen, for
00103            example, when a plugin plays a video and the image is rapidly changing.
00104            If there is overflow, then behave as if we limit to the difference to
00105            >= 0, which will make the rendering opaque.  (Without this overflow
00106            will make the rendering transparent.) */
00107         uint32_t diff = (white & GREEN_MASK) - (black & GREEN_MASK);
00108         /* |diff| is 0xFFFFxx00 on overflow and 0x0000xx00 otherwise, so use this
00109             to limit the transparency. */
00110         uint32_t limit = diff & ALPHA_MASK;
00111         /* The alpha bits of the result */
00112         uint32_t alpha = (ALPHA_MASK - (diff << 16)) | limit;
00113 
00114         return alpha | (black & ~ALPHA_MASK);
00115     }
