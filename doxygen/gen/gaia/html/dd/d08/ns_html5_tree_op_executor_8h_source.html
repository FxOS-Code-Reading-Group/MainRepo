00250     void PostPendingAppendNotification(nsIContent* aParent, nsIContent* aChild) {
00251       bool newParent = true;
00252       const nsIContentPtr* first = mElementsSeenInThisAppendBatch.Elements();
00253       const nsIContentPtr* last = first + mElementsSeenInThisAppendBatch.Length() - 1;
00254       for (const nsIContentPtr* iter = last; iter >= first; --iter) {
00255 #ifdef DEBUG_NS_HTML5_TREE_OP_EXECUTOR_FLUSH
00256         sAppendBatchSlotsExamined++;
00257 #endif
00258         if (*iter == aParent) {
00259           newParent = false;
00260           break;
00261         }
00262       }
00263       if (aChild->IsElement()) {
00264         mElementsSeenInThisAppendBatch.AppendElement(aChild);
00265       }
00266       mElementsSeenInThisAppendBatch.AppendElement(aParent);
00267       if (newParent) {
00268         mPendingNotifications.AppendElement(aParent);
00269       }
00270 #ifdef DEBUG_NS_HTML5_TREE_OP_EXECUTOR_FLUSH
00271       sAppendBatchExaminations++;

00275     void FlushPendingAppendNotifications() {
00276       NS_PRECONDITION(mFlushState == eInDocUpdate, "Notifications flushed outside update");
00277       mFlushState = eNotifying;
00278       const nsHtml5PendingNotification* start = mPendingNotifications.Elements();
00279       const nsHtml5PendingNotification* end = start + mPendingNotifications.Length();
00280       for (nsHtml5PendingNotification* iter = (nsHtml5PendingNotification*)start; iter < end; ++iter) {
00281         iter->Fire();
00282       }
00283       mPendingNotifications.Clear();
00284 #ifdef DEBUG_NS_HTML5_TREE_OP_EXECUTOR_FLUSH
00285       if (mElementsSeenInThisAppendBatch.Length() > sAppendBatchMaxSize) {
00286         sAppendBatchMaxSize = mElementsSeenInThisAppendBatch.Length();
00287       }
00288 #endif
00289       mElementsSeenInThisAppendBatch.Clear();
00290       NS_ASSERTION(mFlushState == eNotifying, "mFlushState out of sync");
00291       mFlushState = eInDocUpdate;

00294     inline bool HaveNotified(nsIContent* aNode) {
00295       NS_PRECONDITION(aNode, "HaveNotified called with null argument.");
00296       const nsHtml5PendingNotification* start = mPendingNotifications.Elements();
00297       const nsHtml5PendingNotification* end = start + mPendingNotifications.Length();
00298       for (;;) {
00299         nsIContent* parent = aNode->GetParent();
00300         if (!parent) {
00301           return true;
00302         }
00303         for (nsHtml5PendingNotification* iter = (nsHtml5PendingNotification*)start; iter < end; ++iter) {
00304           if (iter->Contains(parent)) {
00305             return iter->HaveNotifiedIndex(parent->IndexOf(aNode));
00306           }
00307         }
00308         aNode = parent;
00309       }
