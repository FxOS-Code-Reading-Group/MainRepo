00206 inline int32_t
00207 nsAttrValue::GetIntInternal() const
00208 {
00209   NS_ASSERTION(BaseType() == eIntegerBase,
00210                "getting integer from non-integer");
00211   // Make sure we get a signed value.
00212   // Lets hope the optimizer optimizes this into a shift. Unfortunatly signed
00213   // bitshift right is implementaion dependant.
00214   return static_cast<int32_t>(mBits & ~NS_ATTRVALUE_INTEGERTYPE_MASK) /
00215          NS_ATTRVALUE_INTEGERTYPE_MULTIPLIER;
