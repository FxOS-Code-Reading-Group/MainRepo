00056   bestMatch: function spm_bestMatchIndex(variants, matches) {
00057     var bestMatchIndex = 0;
00058     var bestLocalIndex = 0;
00059     var bestMatchLength = 0;
00060     var allMatches = [];
00061     var matchNum = 0;
00062     matches.forEach(function(match, matchIndex) {
00063 
00064       var indexes = [];
00065       match.forEach(function(number, localIndex) {
00066         var sanitizedNumber = this.sanitizedNumber(number);
00067 
00068         variants.forEach(function match(variant) {
00069           if (variant.indexOf(sanitizedNumber) !== -1 ||
00070               sanitizedNumber.indexOf(variant) !== -1) {
00071             var length = sanitizedNumber.length;
00072 
00073             if (indexes.indexOf(localIndex) === -1) {
00074               indexes.push(localIndex);
00075               matchNum += 1;
00076             }
00077             if (length > bestMatchLength) {
00078               bestMatchLength = length;
00079               bestMatchIndex = matchIndex;
00080               bestLocalIndex = localIndex;
00081             }
00082           }
00083         });
00084       }, this);
00085       // use first phone number if no match result
00086       if (indexes.length) {
00087         allMatches.push(indexes);
00088       } else {
00089         matchNum += 1;
00090         allMatches.push([0]);
00091       }
00092 
00093     }, this);
00094 
00095     return {
00096       totalMatchNum: matchNum,
00097       allMatches: allMatches,
00098       bestMatchIndex: bestMatchIndex,
00099       localIndex: bestLocalIndex
00100     };

00412   _trunkPrefixes: function spm_trunkPrefixes(number) {
00413     var variants = [];

00421         prefixesWithNoTrunk.push(prefix.p);

00433       }
00434
