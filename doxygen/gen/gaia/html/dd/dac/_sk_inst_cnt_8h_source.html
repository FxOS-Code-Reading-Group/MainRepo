00012 /*
00013  * The instance counting system consists of three macros that create the
00014  * instance counting machinery. A class is added to the system by adding:
00015  *   SK_DECLARE_INST_COUNT at the top of its declaration for derived classes
00016  *   SK_DECLARE_INST_COUNT_ROOT at the top of its declaration for a root class
00017  *   SK_DEFINE_INST_COUNT at the top of its .cpp file (for both kinds).
00018  * At the end of an application a call to all the "root" objects'
00019  * CheckInstanceCount methods should be made
00020  */

00056         }                                                                   \
00057                                                                             \

00059             sk_atomic_inc(&gInstanceCount);                                 \
00060         }                                                                   \
00061                                                                             \

00063             sk_atomic_dec(&gInstanceCount);                                 \
00064         }                                                                   \
00065                                                                             \

00069     } fInstanceCountHelper;                                                 \
00070                                                                             \

00079     static int CheckInstanceCount(int level = 0, bool cleanUp = false) {    \
00080         if (gPrintInstCount && 0 != SkInstanceCountHelper::gInstanceCount) {\
00081             SkDebugf("%*c Leaked %s: %d\n",                                 \
00082                      4*level, ' ', #className,                              \
00083                      SkInstanceCountHelper::gInstanceCount);                \
00084         }                                                                   \
00085         if (NULL == SkInstanceCountHelper::gChildren) {                     \
00086             return SkInstanceCountHelper::gInstanceCount;                   \
00087         }                                                                   \
00088         int childCount = SkInstanceCountHelper::gChildren->count();         \
00089         int count = SkInstanceCountHelper::gInstanceCount;                  \
00090         for (int i = 0; i < childCount; ++i) {                              \
00091             count -= (*(*SkInstanceCountHelper::gChildren)[i])(level+1, cleanUp); \
00092         }                                                                   \
00093         SkASSERT(count >= 0);                                               \
00094         if (gPrintInstCount && childCount > 0 && count > 0) {               \
00095             SkDebugf("%*c Leaked ???: %d\n", 4*(level + 1), ' ', count);    \
00096         }                                                                   \
00097         if (cleanUp) {                                                      \
00098             delete SkInstanceCountHelper::gChildren;                        \
00099             SkInstanceCountHelper::gChildren = NULL;                        \
00100         }                                                                   \
00101         return SkInstanceCountHelper::gInstanceCount;                       \
00102     }                                                                       \
00103                                                                             \
00104     static void AddInstChild(templateType SkInstanceCountHelper::PFCheckInstCnt \
00105                                                        childCheckInstCnt) { \
00106         if (CheckInstanceCount != childCheckInstCnt &&                      \
00107             NULL != SkInstanceCountHelper::gChildren) {                     \
00108             SkInstanceCountHelper::gChildren->push_back(childCheckInstCnt); \
00109         }                                                                   \
00110     }
