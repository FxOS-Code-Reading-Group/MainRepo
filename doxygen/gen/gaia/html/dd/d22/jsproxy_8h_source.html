00017 /*
00018  * A proxy is a JSObject that implements generic behavior by providing custom
00019  * implementations for each object trap. The implementation for each trap is
00020  * provided by a C++ object stored on the proxy, known as its handler.
00021  *
00022  * A major use case for proxies is to forward each trap to another object,
00023  * known as its target. The target can be an arbitrary C++ object. Not every
00024  * proxy has the notion of a target, however.
00025  *
00026  * Proxy traps are grouped into fundamental and derived traps. Every proxy has
00027  * to at least provide implementations for the fundamental traps, but the
00028  * derived traps can be implemented in terms of the fundamental ones
00029  * BaseProxyHandler provides implementations of the derived traps in terms of
00030  * the (pure virtual) fundamental traps.
00031  *
00032  * To minimize code duplication, a set of abstract proxy handler classes is
00033  * provided, from which other handlers may inherit. These abstract classes
00034  * are organized in the following hierarchy:
00035  *
00036  * BaseProxyHandler
00037  * |
00038  * DirectProxyHandler
00039  * |
00040  * Wrapper
00041  */
00042 
00043 /*
00044  * BaseProxyHandler is the most generic kind of proxy handler. It does not make
00045  * any assumptions about the target. Consequently, it does not provide any
00046  * default implementation for the fundamental traps. It does, however, implement
00047  * the derived traps in terms of the fundamental ones. This allows consumers of
00048  * this class to define any custom behavior they want.
00049  */

00357 class JS_FRIEND_API(AutoEnterPolicy)
00358 {
00359   public:
00360     typedef BaseProxyHandler::Action Action;
00361     AutoEnterPolicy(JSContext *cx, BaseProxyHandler *handler,
00362                     HandleObject wrapper, HandleId id, Action act, bool mayThrow)
00363 #ifdef DEBUG
00364         : context(NULL)
00365 #endif
00366     {
00367         allow = handler->hasPolicy() ? handler->enter(cx, wrapper, id, act, &rv)
00368                                      : true;
00369         recordEnter(cx, wrapper, id);
00370         // We want to throw an exception if all of the following are true:
00371         // * The policy disallowed access.
00372         // * The policy set rv to false, indicating that we should throw.
00373         // * The caller did not instruct us to ignore exceptions.
00374         // * The policy did not throw itself.
00375         if (!allow && !rv && mayThrow && !JS_IsExceptionPending(cx))
00376             reportError(cx, id);
00377     }
00378 
00379     virtual ~AutoEnterPolicy() { recordLeave(); }
00380     inline bool allowed() { return allow; }
00381     inline bool returnValue() { JS_ASSERT(!allowed()); return rv; }
00382 
00383   protected:
00384     // no-op constructor for subclass
00385     AutoEnterPolicy()
00386 #ifdef DEBUG
00387         : context(NULL)
00388 #endif
00389         {};
00390     void reportError(JSContext *cx, jsid id);
00391     bool allow;
00392     bool rv;
00393 
00394 #ifdef DEBUG
00395     JSContext *context;
00396     mozilla::Maybe<HandleObject> enteredProxy;
00397     mozilla::Maybe<HandleId> enteredId;
00398     // NB: We explicitly don't track the entered action here, because sometimes
00399     // SET traps do an implicit GET during their implementation, leading to
00400     // spurious assertions.
00401     AutoEnterPolicy *prev;
00402     void recordEnter(JSContext *cx, HandleObject proxy, HandleId id);
00403     void recordLeave();
00404 
00405     friend JS_FRIEND_API(void) assertEnteredPolicy(JSContext *cx, JSObject *proxy, jsid id);
00406 #else
00407     inline void recordEnter(JSContext *cx, JSObject *proxy, jsid id) {}
00408     inline void recordLeave() {}
00409 #endif
