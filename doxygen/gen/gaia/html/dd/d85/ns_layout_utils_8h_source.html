01003   /*
01004    * Likewise, but for 'height', 'min-height', or 'max-height'.
01005    */
01006   static nscoord ComputeHeightValue(nscoord aContainingBlockHeight,
01007                                     nscoord aContentEdgeToBoxSizingBoxEdge,
01008                                     const nsStyleCoord& aCoord)
01009   {
01010     MOZ_ASSERT(aContainingBlockHeight != NS_AUTOHEIGHT || !aCoord.HasPercent(),
01011                "caller must deal with %% of unconstrained height");
01012     MOZ_ASSERT(aCoord.IsCoordPercentCalcUnit());
01013 
01014     nscoord result =
01015       nsRuleNode::ComputeCoordPercentCalc(aCoord, aContainingBlockHeight);
01016     // Clamp calc(), and the subtraction for box-sizing.
01017     return std::max(0, result - aContentEdgeToBoxSizingBoxEdge);
01018   }

01028   static bool IsPaddingZero(const nsStyleCoord &aCoord)
01029   {
01030     return (aCoord.GetUnit() == eStyleUnit_Coord &&
01031             aCoord.GetCoordValue() == 0) ||
01032            (aCoord.GetUnit() == eStyleUnit_Percent &&
01033             aCoord.GetPercentValue() == 0.0f) ||
01034            (aCoord.IsCalcUnit() &&
01035             // clamp negative calc() to 0

01037             nsRuleNode::ComputeCoordPercentCalc(aCoord, 0) <= 0);

01890 template<bool IsLessThanOrEqual(nsIFrame*, nsIFrame*)>
01891 /* static */ nsIFrame*
01892 nsLayoutUtils::SortedMerge(nsIFrame *aLeft, nsIFrame *aRight)
01893 {
01894   NS_PRECONDITION(aLeft && aRight, "SortedMerge must have non-empty lists");
01895 
01896   nsIFrame *result;
01897   // Unroll first iteration to avoid null-check 'result' inside the loop.
01898   if (IsLessThanOrEqual(aLeft, aRight)) {
01899     result = aLeft;
01900     aLeft = aLeft->GetNextSibling();
01901     if (!aLeft) {
01902       result->SetNextSibling(aRight);
01903       return result;
01904     }
01905   }
01906   else {
01907     result = aRight;
01908     aRight = aRight->GetNextSibling();
01909     if (!aRight) {
01910       result->SetNextSibling(aLeft);
01911       return result;
01912     }
01913   }
01914 
01915   nsIFrame *last = result;
01916   for (;;) {
01917     if (IsLessThanOrEqual(aLeft, aRight)) {
01918       last->SetNextSibling(aLeft);
01919       last = aLeft;
01920       aLeft = aLeft->GetNextSibling();
01921       if (!aLeft) {
01922         last->SetNextSibling(aRight);
01923         return result;
01924       }
01925     }
01926     else {
01927       last->SetNextSibling(aRight);
01928       last = aRight;
01929       aRight = aRight->GetNextSibling();
01930       if (!aRight) {
01931         last->SetNextSibling(aLeft);
01932         return result;
01933       }
01934     }
01935   }
01936 }

01938 template<bool IsLessThanOrEqual(nsIFrame*, nsIFrame*)>
01939 /* static */ nsIFrame*
01940 nsLayoutUtils::MergeSort(nsIFrame *aSource)
01941 {
01942   NS_PRECONDITION(aSource, "MergeSort null arg");
01943 
01944   nsIFrame *sorted[32] = { nullptr };
01945   nsIFrame **fill = &sorted[0];
01946   nsIFrame **left;
01947   nsIFrame *rest = aSource;
01948 
01949   do {
01950     nsIFrame *current = rest;
01951     rest = rest->GetNextSibling();
01952     current->SetNextSibling(nullptr);
01953 
01954     // Merge it with sorted[0] if present; then merge the result with sorted[1] etc.
01955     // sorted[0] is a list of length 1 (or nullptr).
01956     // sorted[1] is a list of length 2 (or nullptr).
01957     // sorted[2] is a list of length 4 (or nullptr). etc.
01958     for (left = &sorted[0]; left != fill && *left; ++left) {
01959       current = SortedMerge<IsLessThanOrEqual>(*left, current);
01960       *left = nullptr;
01961     }
01962 
01963     // Fill the empty slot that we couldn't merge with the last result.
01964     *left = current;
01965 
01966     if (left == fill)
01967       ++fill;
01968   } while (rest);
01969 
01970   // Collect and merge the results.
01971   nsIFrame *result = nullptr;
01972   for (left = &sorted[0]; left != fill; ++left) {
01973     if (*left) {
01974       result = result ? SortedMerge<IsLessThanOrEqual>(*left, result) : *left;
01975     }
01976   }
01977   return result;
01978 }

01990 template<bool IsLessThanOrEqual(nsIFrame*, nsIFrame*)>
01991 /* static */ bool
01992 nsLayoutUtils::IsFrameListSorted(nsFrameList& aFrameList)
01993 {
01994   if (aFrameList.IsEmpty()) {
01995     // empty lists are trivially sorted.
01996     return true;
01997   }
01998 
01999   // We'll walk through the list with two iterators, one trailing behind the
02000   // other. The list is sorted IFF trailingIter <= iter, across the whole list.
02001   nsFrameList::Enumerator trailingIter(aFrameList);
02002   nsFrameList::Enumerator iter(aFrameList);
02003   iter.Next(); // Skip |iter| past first frame. (List is nonempty, so we can.)
02004 
02005   // Now, advance the iterators in parallel, comparing each adjacent pair.
02006   while (!iter.AtEnd()) {
02007     MOZ_ASSERT(!trailingIter.AtEnd(), "trailing iter shouldn't finish first");
02008     if (!IsLessThanOrEqual(trailingIter.get(), iter.get())) {
02009       return false;
02010     }
02011     trailingIter.Next();
02012     iter.Next();
02013   }
02014 
02015   // We made it to the end without returning early, so the list is sorted.
02016   return true;
02017 }

02019 template<typename PointType, typename RectType, typename CoordType>
02020 /* static */ bool
02021 nsLayoutUtils::PointIsCloserToRect(PointType aPoint, const RectType& aRect,
02022                                    CoordType& aClosestXDistance,
02023                                    CoordType& aClosestYDistance)
02024 {
02025   CoordType fromLeft = aPoint.x - aRect.x;
02026   CoordType fromRight = aPoint.x - aRect.XMost();
02027 
02028   CoordType xDistance;
02029   if (fromLeft >= 0 && fromRight <= 0) {
02030     xDistance = 0;
02031   } else {
02032     xDistance = std::min(abs(fromLeft), abs(fromRight));
02033   }
02034 
02035   if (xDistance <= aClosestXDistance) {
02036     if (xDistance < aClosestXDistance) {
02037       aClosestYDistance = std::numeric_limits<CoordType>::max();
02038     }
02039 
02040     CoordType fromTop = aPoint.y - aRect.y;
02041     CoordType fromBottom = aPoint.y - aRect.YMost();
02042 
02043     CoordType yDistance;
02044     if (fromTop >= 0 && fromBottom <= 0) {
02045       yDistance = 0;
02046     } else {
02047       yDistance = std::min(abs(fromTop), abs(fromBottom));
02048     }
02049 
02050     if (yDistance < aClosestYDistance) {
02051       aClosestXDistance = xDistance;
02052       aClosestYDistance = yDistance;
02053       return true;
02054     }
02055   }
02056 
02057   return false;
02058 }
