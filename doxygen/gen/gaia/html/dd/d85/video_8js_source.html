00412 function share(blobs) {
00413   if (blobs.length === 0)
00414     return;
00415 
00416   var names = [], types = [], fullpaths = [];
00417 
00418   // Get the file name (minus path) and type of each blob
00419   blobs.forEach(function(blob) {
00420     // Discard the path, we just want the base name
00421     var name = blob.name;
00422     // We try to fix Bug 814323 by using
00423     // current workaround of bluetooth transfer
00424     // so we will pass both filenames and fullpaths
00425     // The fullpaths can be removed after Bug 811615 is fixed
00426     fullpaths.push(name);
00427     name = name.substring(name.lastIndexOf('/') + 1);
00428     names.push(name);
00429 
00430     // And we just want the first component of the type "image" or "video"
00431     var type = blob.type;
00432     if (type)
00433       type = type.substring(0, type.indexOf('/'));
00434     types.push(type);
00435   });
00436 
00437   // If there is just one type, or if all types are the same, then use
00438   // that type plus '/*'. Otherwise, use 'multipart/mixed'
00439   // If all the blobs are image we use 'image/*'. If all are videos
00440   // we use 'video/*'. Otherwise, 'multipart/mixed'.
00441   var type;
00442   if (types.length === 1 || types.every(function(t) { return t === types[0]; }))
00443     type = types[0] + '/*';
00444   else
00445     type = 'multipart/mixed';
00446 
00447   var a = new MozActivity({
00448     name: 'share',
00449     data: {
00450       type: type,
00451       number: blobs.length,
00452       blobs: blobs,
00453       filenames: names,
00454       filepaths: fullpaths
00455     }
00456   });
00457 
00458   a.onerror = function(e) {
00459     if (a.error.name === 'NO_PROVIDER') {
00460       var msg = navigator.mozL10n.get('share-noprovider');
00461       alert(msg);
00462     } else {
00463       console.warn('share activity error:', a.error.name);
00464     }
00465   };
00466 }

00584 // Align vertically fullscreen view
00585 function setPlayerSize() {
00586   var containerWidth = window.innerWidth;
00587   var containerHeight = window.innerHeight;
00588 
00589   // Don't do anything if we don't know our size.
00590   // This could happen if we get a resize event before our metadata loads
00591   if (!dom.player.videoWidth || !dom.player.videoHeight)
00592     return;
00593 
00594   var width, height; // The size the video will appear, after rotation
00595   var rotation = 'metadata' in currentVideo ?
00596     currentVideo.metadata.rotation : 0;
00597 
00598   switch (rotation) {
00599   case 0:
00600   case 180:
00601     width = dom.player.videoWidth;
00602     height = dom.player.videoHeight;
00603     break;
00604   case 90:
00605   case 270:
00606     width = dom.player.videoHeight;
00607     height = dom.player.videoWidth;
00608   }
00609 
00610   var xscale = containerWidth / width;
00611   var yscale = containerHeight / height;
00612   var scale = Math.min(xscale, yscale);
00613 
00614   // scale large videos down and scale small videos up
00615   // this might result in lower image quality for small videos
00616   width *= scale;
00617   height *= scale;
00618 
00619   var left = ((containerWidth - width) / 2);
00620   var top = ((containerHeight - height) / 2);
00621 
00622   var transform;
00623   switch (rotation) {
00624   case 0:
00625     transform = 'translate(' + left + 'px,' + top + 'px)';
00626     break;
00627   case 90:
00628     transform =
00629       'translate(' + (left + width) + 'px,' + top + 'px) ' +
00630       'rotate(90deg)';
00631     break;
00632   case 180:
00633     transform =
00634       'translate(' + (left + width) + 'px,' + (top + height) + 'px) ' +
00635       'rotate(180deg)';
00636     break;
00637   case 270:
00638     transform =
00639       'translate(' + left + 'px,' + (top + height) + 'px) ' +
00640       'rotate(270deg)';
00641     break;
00642   }
00643 
00644   transform += ' scale(' + scale + ')';
00645 
00646   dom.player.style.transform = transform;
00647 }

00844 // Update the progress bar and play head as the video plays
00845 function timeUpdated() {
00846   if (controlShowing) {
00847     // We can't update a progress bar if we don't know how long
00848     // the video is. It is kind of a bug that the <video> element
00849     // can't figure this out for ogv videos.
00850     if (dom.player.duration === Infinity || dom.player.duration === 0) {
00851       return;
00852     }
00853 
00854     var percent = (dom.player.currentTime / dom.player.duration) * 100 + '%';
00855 
00856     dom.elapsedText.textContent = MediaUtils.formatDuration(
00857       dom.player.currentTime);
00858     dom.elapsedTime.style.width = percent;
00859     // Don't move the play head if the user is dragging it.
00860     if (!dragging)
00861       dom.playHead.style.left = percent;
00862   }
00863 
00864   // Since we don't always get reliable 'ended' events, see if
00865   // we've reached the end this way.
00866   // See: https://bugzilla.mozilla.org/show_bug.cgi?id=783512
00867   // If we're within 1 second of the end of the video, register
00868   // a timeout a half a second after we'd expect an ended event.
00869   if (!endedTimer) {
00870     if (!dragging && dom.player.currentTime >= dom.player.duration - 1) {
00871       var timeUntilEnd = (dom.player.duration - dom.player.currentTime + .5);
00872       endedTimer = setTimeout(playerEnded, timeUntilEnd * 1000);
00873     }
00874   } else if (dragging && dom.player.currentTime < dom.player.duration - 1) {
00875     // If there is a timer set and we drag away from the end, cancel the timer
00876     clearTimeout(endedTimer);
00877     endedTimer = null;
00878   }
00879 }
