00107 /*
00108  * This utility pales in comparison to Boost's aligned_storage. The utility
00109  * simply assumes that uint64_t is enough alignment for anyone. This may need
00110  * to be extended one day...
00111  *
00112  * As an important side effect, pulling the storage into this template is
00113  * enough obfuscation to confuse gcc's strict-aliasing analysis into not giving
00114  * false negatives when we cast from the char buffer to whatever type we've
00115  * constructed using the bytes.
00116  */
00117 template<size_t nbytes>
00118 struct AlignedStorage
00119 {
00120     union U {
00121       char bytes[nbytes];
00122       uint64_t _;
00123     } u;
00124 
00125     const void* addr() const { return u.bytes; }
00126     void* addr() { return u.bytes; }
00127 };
00128 
00129 template<class T>
00130 struct AlignedStorage2
00131 {
00132     union U {
00133       char bytes[sizeof(T)];
00134       uint64_t _;
00135     } u;
00136 
00137     const T* addr() const { return reinterpret_cast<const T*>(u.bytes); }
00138     T* addr() { return static_cast<T*>(static_cast<void*>(u.bytes)); }
00139 };

00282 /*
00283  * Safely subtract two pointers when it is known that end >= begin.  This avoids
00284  * the common compiler bug that if (size_t(end) - size_t(begin)) has the MSB
00285  * set, the unsigned subtraction followed by right shift will produce -1, or
00286  * size_t(-1), instead of the real difference.
00287  */
00288 template<class T>
00289 MOZ_ALWAYS_INLINE size_t
00290 PointerRangeSize(T* begin, T* end)
00291 {
00292   MOZ_ASSERT(end >= begin);
00293   return (size_t(end) - size_t(begin)) / sizeof(T);
00294 }
