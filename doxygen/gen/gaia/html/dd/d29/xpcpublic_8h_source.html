00281 bool NonVoidStringToJsval(JSContext* cx, mozilla::dom::DOMString& str,
00282                           JS::Value *rval)
00283 {
00284     if (!str.HasStringBuffer()) {
00285         // It's an actual XPCOM string
00286         return NonVoidStringToJsval(cx, str.AsAString(), rval);
00287     }
00288 
00289     uint32_t length = str.StringBufferLength();
00290     if (length == 0) {
00291         *rval = JS_GetEmptyStringValue(cx);
00292         return true;
00293     }
00294 
00295     nsStringBuffer* buf = str.StringBuffer();
00296     bool shared;
00297     if (!XPCStringConvert::StringBufferToJSVal(cx, buf, length, rval,
00298                                                &shared)) {
00299         return false;
00300     }
00301     if (shared) {
00302         // JS now needs to hold a reference to the buffer
00303         buf->AddRef();
00304     }
00305     return true;
00306 }
