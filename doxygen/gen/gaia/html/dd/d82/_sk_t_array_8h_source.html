00370     inline void checkRealloc(int delta) {
00371         SkASSERT(fCount >= 0);
00372         SkASSERT(fAllocCount >= 0);
00373 
00374         SkASSERT(-delta <= fCount);
00375 
00376         int newCount = fCount + delta;
00377         int newAllocCount = fAllocCount;
00378 
00379         if (newCount > fAllocCount || newCount < (fAllocCount / 3)) {
00380             // whether we're growing or shrinking, we leave at least 50% extra space for future
00381             // growth (clamped to the reserve count).
00382             newAllocCount = SkMax32(newCount + ((newCount + 1) >> 1), fReserveCount);
00383         }
00384         if (newAllocCount != fAllocCount) {
00385 
00386             fAllocCount = newAllocCount;
00387             char* newMemArray;
00388 
00389             if (fAllocCount == fReserveCount && NULL != fPreAllocMemArray) {
00390                 newMemArray = (char*) fPreAllocMemArray;
00391             } else {
00392                 newMemArray = (char*) sk_malloc_throw(fAllocCount*sizeof(T));
00393             }
00394 
00395             SkTArrayExt::copyAndDelete<T>(this, newMemArray);
00396 
00397             if (fMemArray != fPreAllocMemArray) {
00398                 sk_free(fMemArray);
00399             }
