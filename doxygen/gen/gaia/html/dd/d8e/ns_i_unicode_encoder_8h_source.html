00051 //
00052 //    p = the buffer pointer   (char*)

00059 //
00060 #define ENCODER_BUFFER_ALLOC_IF_NEEDED(p,e,s,l,sb,sbl,al) \
00061   PR_BEGIN_MACRO                                          \
00062     if (e                                                 \
00063         && NS_SUCCEEDED((e)->GetMaxLength((s), (l), &(al)))\
00064         && ((al) > (int32_t)(sbl))                        \
00065         && (nullptr!=((p)=(char*)nsMemory::Alloc((al)+1))) \
00066         ) {                                               \
00067     }                                                     \
00068     else {                                                \
00069       (p) = (char*)(sb);                                  \
00070       (al) = (sbl);                                       \
00071     }                                                     \
