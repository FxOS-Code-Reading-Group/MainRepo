00760     // We handle UTF-8 strings, numbers, and blobs
00761     function getMetadataValue(data, end, tagtype) {
00762       // Loop until we find a data atom
00763       while (data.index < end) {
00764         var size = data.readUnsignedInt();
00765         var type = data.readASCIIText(4);
00766         if (type !== 'data') {
00767           data.advance(size - 8);
00768           continue;
00769         }
00770 
00771         // We've found the data atom.
00772         // Return its (first) value or throw an error.
00773         var datatype = data.readUnsignedInt() & 0xFFFFFF;
00774         data.advance(4); // Ignore locale
00775 
00776         var datasize = size - 16; // the rest of the atom is the value
00777 
00778         // Special case for track number
00779         if (tagtype === 'trkn') {
00780           data.advance(2);
00781           return data.readUnsignedShort();
00782         }

00786           return data.readUTF8Text(datasize);

00790             end: data.sliceOffset + data.viewOffset + data.index + datasize,

00797             type: 'image/png'

00919     function getImage(file) {
00920       // Get the embedded image from the music file
00921       var embedded = file.slice(metadata.picture.start,
00922                                 metadata.picture.end,
00923                                 metadata.picture.type);
00924       // Convert to a blob url
00925       var embeddedURL = URL.createObjectURL(embedded);
00926       // Load it into an image element
00927       offscreenImage.src = embeddedURL;
00928       offscreenImage.onerror = function() {
00929         URL.revokeObjectURL(embeddedURL);
00930         offscreenImage.removeAttribute('src');
00931         // Something went wrong reading the embedded image.
00932         // Return a default one instead
00933         console.warn('Album cover art failed to load', file.name);
00934         callback(null);
00935       };
00936       offscreenImage.onload = function() {
00937         // We've loaded the image, now copy it to a canvas
00938         var canvas = document.createElement('canvas');
00939         canvas.width = THUMBNAIL_WIDTH;
00940         canvas.height = THUMBNAIL_HEIGHT;
00941         var context = canvas.getContext('2d');
00942         var scalex = canvas.width / offscreenImage.width;
00943         var scaley = canvas.height / offscreenImage.height;
00944 
00945         // Take the larger of the two scales: we crop the image to the thumbnail
00946         var scale = Math.max(scalex, scaley);
00947 
00948         // If the image was already thumbnail size, it is its own thumbnail
00949         if (scale >= 1) {
00950           offscreenImage.removeAttribute('src');
00951           cacheThumbnail(key, embedded, embeddedURL);
00952           callback(embeddedURL);
00953           return;
00954         }
00955 
00956         // Calculate the region of the image that will be copied to the
00957         // canvas to create the thumbnail
00958         var w = Math.round(THUMBNAIL_WIDTH / scale);
00959         var h = Math.round(THUMBNAIL_HEIGHT / scale);
00960         var x = Math.round((offscreenImage.width - w) / 2);
00961         var y = Math.round((offscreenImage.height - h) / 2);
00962 
00963         // Draw that region of the image into the canvas, scaling it down
00964         context.drawImage(offscreenImage, x, y, w, h,
00965                           0, 0, THUMBNAIL_WIDTH, THUMBNAIL_HEIGHT);
00966 
00967         // We're done with the image now
00968         offscreenImage.removeAttribute('src');
00969         URL.revokeObjectURL(embeddedURL);
00970 
00971         canvas.toBlob(function(blob) {
00972           var url = URL.createObjectURL(blob);
00973           cacheThumbnail(key, blob, url);
00974           callback(url);
00975         }, 'image/jpeg');
00976       };
00977     }
