00014 /*
00015  * This header exports one public method in the mozilla namespace:
00016  *
00017  *   template<class SmartPtr>
00018  *   void ClearOnShutdown(SmartPtr *aPtr)
00019  *
00020  * This function takes a pointer to a smart pointer and nulls the smart pointer
00021  * on shutdown.
00022  *
00023  * This is useful if you have a global smart pointer object which you don't
00024  * want to "leak" on shutdown.
00025  *
00026  * Although ClearOnShutdown will work with any smart pointer (i.e., nsCOMPtr,
00027  * nsRefPtr, nsAutoPtr, StaticRefPtr, and StaticAutoPtr), you probably want to
00028  * use it only with StaticRefPtr and StaticAutoPtr.  There is no way to undo a
00029  * call to ClearOnShutdown, so you can call it only on smart pointers which you
00030  * know will live until the program shuts down.  In practice, these are likely
00031  * global variables, which should be Static{Ref,Auto}Ptr.
00032  *
00033  * ClearOnShutdown is currently main-thread only because we don't want to
00034  * accidentally free an object from a different thread than the one it was
00035  * created on.
00036  */
00037 
00038 namespace mozilla {
00039 namespace ClearOnShutdown_Internal {
00040 
00041 class ShutdownObserver : public LinkedListElement<ShutdownObserver>
00042 {
00043 public:
00044   virtual void Shutdown() = 0;
00045   virtual ~ShutdownObserver() {}
00046 };
00047 
00048 template<class SmartPtr>
00049 class PointerClearer : public ShutdownObserver
00050 {
00051 public:
00052   PointerClearer(SmartPtr *aPtr)
00053     : mPtr(aPtr)
00054   {}
00055 
00056   virtual void Shutdown()
00057   {
00058     if (mPtr) {
00059       *mPtr = NULL;
00060     }
00061   }
00062 
00063 private:
00064   SmartPtr *mPtr;
00065 };
00066 
00067 extern bool sHasShutDown;
00068 extern StaticAutoPtr<LinkedList<ShutdownObserver> > sShutdownObservers;
00069 
00070 } // namespace ClearOnShutdown_Internal
00071 
00072 template<class SmartPtr>
00073 inline void ClearOnShutdown(SmartPtr *aPtr)
00074 {
00075   using namespace ClearOnShutdown_Internal;
00076 
00077   MOZ_ASSERT(NS_IsMainThread());
00078   MOZ_ASSERT(!sHasShutDown);
00079 
00080   if (!sShutdownObservers) {
00081     sShutdownObservers = new LinkedList<ShutdownObserver>();
00082   }
00083   sShutdownObservers->insertBack(new PointerClearer<SmartPtr>(aPtr));
00084 }
00085 
00086 // Called when XPCOM is shutting down, after all shutdown notifications have
00087 // been sent and after all threads' event loops have been purged.
00088 inline void KillClearOnShutdown()
00089 {
00090   using namespace ClearOnShutdown_Internal;
00091 
00092   MOZ_ASSERT(NS_IsMainThread());
00093 
00094   if (sShutdownObservers) {
00095     ShutdownObserver *observer;
00096     while ((observer = sShutdownObservers->popFirst())) {
00097       observer->Shutdown();
00098       delete observer;
00099     }
00100   }
00101 
00102   sShutdownObservers = nullptr;
00103   sHasShutDown = true;
00104 }
