00053 static inline ShaderProgramType
00054 ShaderProgramFromSurfaceFormat(gfx::SurfaceFormat aFormat)
00055 {
00056   switch (aFormat) {
00057     case gfx::FORMAT_B8G8R8A8:
00058       return BGRALayerProgramType;
00059     case gfx::FORMAT_B8G8R8X8:
00060       return BGRXLayerProgramType;
00061     case gfx::FORMAT_R8G8B8A8:
00062       return RGBALayerProgramType;
00063     case gfx::FORMAT_R8G8B8X8:
00064     case gfx::FORMAT_R5G6B5:
00065       return RGBXLayerProgramType;
00066     case gfx::FORMAT_A8:
00067       // We don't have a specific luminance shader
00068       break;
00069     default:
00070       NS_ASSERTION(false, "Unhandled surface format!");
00071   }
00072   return ShaderProgramType(0);
00073 }

00075 static inline ShaderProgramType
00076 ShaderProgramFromTargetAndFormat(GLenum aTarget,
00077                                  gfx::SurfaceFormat aFormat)
00078 {
00079   switch(aTarget) {
00080     case LOCAL_GL_TEXTURE_EXTERNAL:
00081       MOZ_ASSERT(aFormat == gfx::FORMAT_R8G8B8A8);
00082       return RGBAExternalLayerProgramType;
00083     case LOCAL_GL_TEXTURE_RECTANGLE_ARB:
00084       MOZ_ASSERT(aFormat == gfx::FORMAT_R8G8B8A8 ||
00085                  aFormat == gfx::FORMAT_R8G8B8X8);
00086       if (aFormat == gfx::FORMAT_R8G8B8A8)
00087         return RGBARectLayerProgramType;
00088       else
00089         return RGBXRectLayerProgramType;
00090     default:
00091       return ShaderProgramFromSurfaceFormat(aFormat);
00092   }
00093 }

00122   {
00123     if (aType < 0 ||

00137         aType == BGRARectLayerProgramType)
00138       return false;

00141            aType == RGBARectLayerProgramType ||
00142            aType == RGBXRectLayerProgramType ||
00143            aType == RGBALayerProgramType;
