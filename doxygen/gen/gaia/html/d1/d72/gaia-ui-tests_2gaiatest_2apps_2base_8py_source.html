00086     def wait_for_condition(self, method, timeout=_default_timeout, message="Condition timed out"):
00087         """Calls the method provided with the driver as an argument until the return value is not False."""
00088         end_time = time.time() + timeout
00089         while time.time() < end_time:
00090             try:
00091                 value = method(self.marionette)

00093                     return value

00095                 pass
