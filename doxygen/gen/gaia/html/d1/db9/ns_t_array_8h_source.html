00031 //
00032 // nsTArray is a resizable array class, like std::vector.
00033 //
00034 // Unlike std::vector, which follows C++'s construction/destruction rules,
00035 // nsTArray assumes that your "T" can be memmoved()'ed safely.
00036 //
00037 // The public classes defined in this header are
00038 //
00039 //   nsTArray<T>,
00040 //   FallibleTArray<T>,
00041 //   nsAutoTArray<T, N>, and
00042 //   AutoFallibleTArray<T, N>.
00043 //
00044 // nsTArray and nsAutoTArray are infallible; if one tries to make an allocation
00045 // which fails, it crashes the program.  In contrast, FallibleTArray and
00046 // AutoFallibleTArray are fallible; if you use one of these classes, you must
00047 // check the return values of methods such as Append() which may allocate.  If
00048 // in doubt, choose an infallible type.
00049 //
00050 // InfallibleTArray and AutoInfallibleTArray are aliases for nsTArray and
00051 // nsAutoTArray.
00052 //
00053 // If you just want to declare the nsTArray types (e.g., if you're in a header
00054 // file and don't need the full nsTArray definitions) consider including
00055 // nsTArrayForwardDeclare.h instead of nsTArray.h.
00056 //
00057 // The template parameter (i.e., T in nsTArray<T>) specifies the type of the
00058 // elements and has the following requirements:
00059 //
00060 //   T MUST be safely memmove()'able.
00061 //   T MUST define a copy-constructor.
00062 //   T MAY define operator< for sorting.
00063 //   T MAY define operator== for searching.
00064 //
00065 // (Note that the memmove requirement may be relaxed for certain types - see
00066 // nsTArray_CopyElements below.)
00067 //
00068 // For methods taking a Comparator instance, the Comparator must be a class
00069 // defining the following methods:
00070 //
00071 //   class Comparator {
00072 //     public:
00073 //       /** @return True if the elements are equals; false otherwise. */
00074 //       bool Equals(const elem_type& a, const Item& b) const;
00075 //
00076 //       /** @return True if (a < b); false otherwise. */
00077 //       bool LessThan(const elem_type& a, const Item& b) const;
00078 //   };
00079 //
00080 // The Equals method is used for searching, and the LessThan method is used for
00081 // searching and sorting.  The |Item| type above can be arbitrary, but must
00082 // match the Item type passed to the sort or search function.
00083 //

01133   // This method searches for the smallest index of an element that is strictly
01134   // greater than |item|.  If |item| is inserted at this index, the array will
01135   // remain sorted and |item| would come after all elements that are equal to
01136   // it.  If |item| is greater than or equal to all elements in the array, the
01137   // array length is returned.
01138   //
01139   // Note that consumers who want to know whether there are existing items equal
01140   // to |item| in the array can just check that the return value here is > 0 and
01141   // indexing into the previous slot gives something equal to |item|.
01142   //
01143   //
01144   // @param item   The item to search for.
01145   // @param comp   The Comparator used.
01146   // @return        The index of greatest element <= to |item|
01147   // @precondition The array is sorted
01148   template<class Item, class Comparator>
01149   index_type
01150   IndexOfFirstElementGt(const Item& item,
01151                         const Comparator& comp) const {
01152     // invariant: low <= [idx] <= high
01153     index_type low = 0, high = Length();
01154     while (high > low) {
01155       index_type mid = (high + low) >> 1;
01156       // Comparators are not required to provide a LessThan(Item&, elem_type),
01157       // so we can't do comp.LessThan(item, ElementAt(mid)).
01158       if (comp.LessThan(ElementAt(mid), item) ||
01159           comp.Equals(ElementAt(mid), item)) {
01160         // item >= ElementAt(mid), so our desired index is at least mid+1.
01161         low = mid + 1;
01162       } else {
01163         // item < ElementAt(mid).  Our desired index is therefore at most mid.
01164         high = mid;
01165       }
01166     }
01167     MOZ_ASSERT(high == low);
01168     return low;
01169   }

01572     index_type end = Length() - 1;
01573     while ((index * 2) < end) {
01574       const index_type left = (index * 2) + 1;
01575       const index_type right = (index * 2) + 2;

01577       if (comp.LessThan(item, elem[left])) {
01578         if (left < end &&

01580           index = right;
01581         } else {

01583         }
01584       } else if (left < end &&
01585                  comp.LessThan(item, elem[right])) {
01586         index = right;
01587       } else {

01589       }
