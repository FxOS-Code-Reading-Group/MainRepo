00094   // There's a possibility that we already have an actor that wraps the same
00095   // string or int because we do all this identifier construction
00096   // asynchronously. In this case we need to hand out the canonical version
00097   // created by the child side.
00098   //
00099   // In order to deal with temporary identifiers which appear on the stack,
00100   // identifiers use the following state invariants:
00101   //
00102   // * mCanonicalIdentifier is non-NULL: this is a duplicate identifier, no
00103   //   further information is necessary.
00104   // * mHashed is false: this identifier is a newborn, non-permanent identifier
00105   // * mHashed is true, mTemporaryRefs is 0: this identifier is permanent
00106   // * mHashed is true, mTemporaryRefs is non-0: this identifier is temporary;
00107   //   if NPN_GetFooIdentifier is called for it, we need to retain it. If
00108   //   all stack references are lost, unhash it because it will soon be 
00109   //   deleted.
00110

00112   bool mHashed;
00113   unsigned int mTemporaryRefs;
00114   bool mIsString;
