00035 /* NS_roundup30 rounds towards infinity for positive and       */
00036 /* negative numbers.                                           */
00037 
00038 #if defined(XP_WIN32) && defined(_M_IX86) && !defined(__GNUC__)
00039 inline NS_HIDDEN_(int32_t) NS_lroundup30(float x)
00040 {
00041     /* Code derived from Laurent de Soras' paper at             */
00042     /* http://ldesoras.free.fr/doc/articles/rounding_en.pdf     */
00043 
00044     /* Rounding up on Windows is expensive using the float to   */
00045     /* int conversion and the floor function. A faster          */
00046     /* approach is to use f87 rounding while assuming the       */
00047     /* default rounding mode of rounding to the nearest         */
00048     /* integer. This rounding mode, however, actually rounds    */
00049     /* to the nearest integer so we add the floating point      */
00050     /* number to itself and add our rounding factor before      */
00051     /* doing the conversion to an integer. We then do a right   */
00052     /* shift of one bit on the integer to divide by two.        */
00053 
00054     /* This routine doesn't handle numbers larger in magnitude  */
00055     /* than 2^30 but this is fine for NSToCoordRound because    */
00056     /* Coords are limited to 2^30 in magnitude.                 */
00057 
00058     static const double round_to_nearest = 0.5f;
00059     int i;
00060 
00061     __asm {
00062       fld     x                   ; load fp argument
00063       fadd    st, st(0)           ; double it
00064       fadd    round_to_nearest    ; add the rounding factor
00065       fistp   dword ptr i         ; convert the result to int
00066     }
00067     return i >> 1;                /* divide by 2 */
00068 }
00069 #endif /* XP_WIN32 && _M_IX86 && !__GNUC__ */
