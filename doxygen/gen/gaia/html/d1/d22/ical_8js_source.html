02638     var type;
02639     switch (letter) {
02640       case 'P':

02650         break;

02652         type = 'weeks';
02653         break;
02654       case 'H':
02655         type = 'hours';
02656         break;
02657       case 'M':
02658         type = 'minutes';
02659         break;
02660       case 'S':
02661         type = 'seconds';
02662         break;
02663       default:
02664         // Not a valid chunk
02665         return 0;

02669       if (!number && number !== 0) {
02670         throw new Error(

02676         throw new Error(
02677           'invalid duration value: Invalid number "' + number + '" before "' + letter + '"'
02678         );
02679       }
02680       object[type] = num;
02681     }
