00020 // Fired after Places inited.
00021 #define TOPIC_PLACES_INIT_COMPLETE "places-init-complete"
00022 // Fired when initialization fails due to a locked database.
00023 #define TOPIC_DATABASE_LOCKED "places-database-locked"
00024 // This topic is received when the profile is about to be lost.  Places does
00025 // initial shutdown work and notifies TOPIC_PLACES_SHUTDOWN to all listeners.
00026 // Any shutdown work that requires the Places APIs should happen here.
00027 #define TOPIC_PROFILE_CHANGE_TEARDOWN "profile-change-teardown"
00028 // This topic is received just before the profile is lost.  Places begins
00029 // shutting down the connection and notifies TOPIC_PLACES_WILL_CLOSE_CONNECTION
00030 // to all listeners.  Only critical database cleanups should happen here,
00031 // some APIs may bail out already.
00032 #define TOPIC_PROFILE_BEFORE_CHANGE "profile-before-change"
00033 // Fired when Places is shutting down.  Any code should stop accessing Places
00034 // APIs after this notification.  If you need to listen for Places shutdown
00035 // you should only use this notification, next ones are intended only for
00036 // internal Places use.
00037 #define TOPIC_PLACES_SHUTDOWN "places-shutdown"
00038 // For Internal use only.  Fired when connection is about to be closed, only
00039 // cleanup tasks should run at this stage, nothing should be added to the
00040 // database, nor APIs should be called.
00041 #define TOPIC_PLACES_WILL_CLOSE_CONNECTION "places-will-close-connection"
00042 // Fired when the connection has gone, nothing will work from now on.
