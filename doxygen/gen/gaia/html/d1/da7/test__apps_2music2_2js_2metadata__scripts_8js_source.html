00627   // Metadata includes title, artist, album and possibly the track number.
00628   // Year, comment and genre are ignored.
00629   //
00630   // Format information:
00631   //   http://www.id3.org/ID3v1
00632   //   http://en.wikipedia.org/wiki/ID3

00634   function parseID3v1Metadata(footer) {
00635     var title = footer.getASCIIText(3, 30);
00636     var artist = footer.getASCIIText(33, 30);
00637     var album = footer.getASCIIText(63, 30);
00638     var p = title.indexOf('\0');
00639     if (p !== -1)
00640       title = title.substring(0, p);
00641     p = artist.indexOf('\0');
00642     if (p !== -1)
00643       artist = artist.substring(0, p);
00644     p = album.indexOf('\0');
00645     if (p !== -1)
00646       album = album.substring(0, p);
00647 
00648     metadata[TITLE] = title || undefined;
00649     metadata[ARTIST] = artist || undefined;
00650     metadata[ALBUM] = album || undefined;
00651     var b1 = footer.getUint8(125);
00652     var b2 = footer.getUint8(126);
00653     if (b1 === 0 && b2 !== 0)
00654       metadata[TRACKNUM] = b2;
00655     var genre = footer.getUint8(127);
00656     metadata[GENRE] = genre || undefined;
00657     metadataCallback(metadata);

00699     function parseID3(id3) {
00700       // skip the extended header, if there is one
00701       if (has_extended_header) {
00702         id3.advance(id3.readUnsignedInt());
00703       }
00704 
00705       // Now we have a series of frames, each of which is one ID3 tag
00706       while (id3.index < id3.byteLength) {
00707         var tagid, tagsize, tagflags;
00708 
00709         // If there is a null byte here, then we've found padding
00710         // and we're done
00711         if (id3.getUint8(id3.index) === 0)
00712           break;
00713 
00714         switch (id3version) {
00715         case 2:
00716           tagid = id3.readASCIIText(3);
00717           tagsize = id3.readUint24();
00718           tagflags = 0;
00719           break;
00720         case 3:
00721           tagid = id3.readASCIIText(4);
00722           tagsize = id3.readUnsignedInt();
00723           tagflags = id3.readUnsignedShort();
00724           break;
00725         case 4:
00726           tagid = id3.readASCIIText(4);
00727           tagsize = id3.readID3Uint28BE();
00728           tagflags = id3.readUnsignedShort();
00729           break;
00730         }
00731 
00732         var nexttag = id3.index + tagsize;
00733         var tagname = ID3V2TAGS[tagid];
00734 
00735         // Skip tags we don't care about
00736         if (!tagname) {
00737           id3.index = nexttag;
00738           continue;
00739         }
00740 
00741         // Skip compressed, encrypted, grouped, or synchronized tags that
00742         // we can't decode
00743         if ((tagflags & 0xFF) !== 0) {
00744           console.warn('Skipping', tagid, 'tag with flags', tagflags);
00745           id3.index = nexttag;
00746           continue;
00747         }
00748 
00749         // Wrap it in try so we don't crash the whole thing on one bad tag
00750         try {
00751           // Now get the tag value
00752           var tagvalue = null;
00753 
00754           switch (tagid) {
00755           case 'TIT2':
00756           case 'TT2':
00757           case 'TPE1':
00758           case 'TP1':
00759           case 'TALB':
00760           case 'TCON':
00761           case 'TAL':
00762             tagvalue = readText(id3, tagsize);
00763             break;
00764           case 'TRCK':
00765           case 'TRK':
00766             tagvalue = parseInt(readText(id3, tagsize));
00767             break;
00768           case 'APIC':
00769           case 'PIC':
00770             tagvalue = readPic(id3, tagsize, tagid);
00771             break;
00772           }
00773 
00774           if (tagvalue !== null)
00775             metadata[tagname] = tagvalue;
00776         }
00777         catch (e) {
00778           console.warn('Error parsing mp3 metadata tag', tagid, ':', e);
00779         }
00780 
00781         // Make sure we're at the start of the next tag before continuing
00782         id3.index = nexttag;
00783       }
00784 
00785       handleCoverArt(metadata);
00786     }

00852     // the first packet to find the start of the second.

00854     var p1_segment_lengths = header.getUnsignedByteArray(27, p1_num_segments);
00855     var p1_length = Array.reduce(p1_segment_lengths, sum, 0);
00856 
00857     var p2_header = 27 + p1_num_segments + p1_length;
00858     var p2_num_segments = header.getUint8(p2_header + 26);
00859     var p2_segment_lengths = header.getUnsignedByteArray(p2_header + 27,
00860                                                          p2_num_segments);
00861     var p2_length = Array.reduce(p2_segment_lengths, sum, 0);
00862     var p2_offset = p2_header + 27 + p2_num_segments;
00863 
00864     // Now go fetch page 2
00865     header.getMore(p2_offset, p2_length, function(page, error) {
00866       if (error) {
00867         errorCallback(error);
00868         return;
00869       }
00870 
00871       // Look for a comment packet from a supported codec
00872       var first_byte = page.readByte();
00873       var valid = false;
00874       switch (first_byte) {
00875         case 3:
00876           valid = page.readASCIIText(6) === 'vorbis';
00877           break;
00878         case 79:
00879           valid = page.readASCIIText(7) === 'pusTags';

00892         var comment = page.readUTF8Text(comment_length);

00904             }

00908           // http://wiki.xiph.org/VorbisComment
00909           // http://flac.sourceforge.net/format.html#metadata_block_picture
00910         }
00911       }

00938     function findMoovAtom(atom) {

00941         var size = atom.readUnsignedInt();

00943

00988     function parseMoovAtom(data, end) {
00989       data.advance(8); // skip the size and type of this atom
00990 
00991       // Find the udta and trak atoms within the moov atom
00992       // There will only be one udta atom, but there may be multiple trak
00993       // atoms. In that case, this is probably a movie file and we'll reject
00994       // it when we find a track that is not an mp4 audio codec.
00995       while (data.index < end) {
00996         var size = data.readUnsignedInt();
00997         var type = data.readASCIIText(4);
00998         var nextindex = data.index + size - 8;
00999         if (type === 'udta') {       // Metadata is inside here
01000           parseUdtaAtom(data, end);
01001           data.index = nextindex;
01002         }
01003         else if (type === 'trak') {  // We find the audio format inside here
01004           data.advance(-8); // skip back to beginning

01010               if (stbl) {

01012                 if (stsd) {

01015                   if (codec !== 'mp4a') {
01016                     throw 'Unsupported format in MP4 container: ' + codec;
01017                   }
01018                 }
01019               }
01020             }
01021           }
01022           data.index = nextindex;
01023         }
01024         else {
01025           data.advance(size - 8);
01026         }
01027       }

01107     // Find the data atom and return its value or throw an error
01108     // We handle UTF-8 strings, numbers, and blobs
01109     function getMetadataValue(data, end, tagtype) {
01110       // Loop until we find a data atom
01111       while (data.index < end) {
