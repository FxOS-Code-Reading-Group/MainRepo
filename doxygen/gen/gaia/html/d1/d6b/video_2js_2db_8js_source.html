00059 function enumerateDB() {
00060   if (enumerated)
00061     return;

00063 
00064   var batch = [];
00065   var batchSize = 4;
00066 
00067   videodb.enumerate('date', null, 'prev', function(videoinfo) {
00068     // When we're done with the enumeration flush any batched files
00069     if (videoinfo === null) {
00070       flush();
00071       return;
00072     }
00073 
00074     var isVideo = videoinfo.metadata.isVideo;
00075 
00076     // If we know this is not a video, ignore it
00077     if (isVideo === false) {
00078       return;
00079     }
00080 
00081     // If we don't have metadata for this video yet, add it to the
00082     // metadata queue to get processed. Once the metadata is
00083     // available, it will be passed to addVideo()
00084     if (isVideo === undefined) {
00085       addToMetadataQueue(videoinfo);
00086       return;
00087     }
00088 
00089     // If we've parsed the metadata and know this is a video, display it.
00090     if (isVideo === true) {
00091       batch.push(videoinfo);
00092       if (batch.length >= batchSize) {
00093         flush();
00094         batchSize *= 2;
00095       }
00096     }
00097   });
