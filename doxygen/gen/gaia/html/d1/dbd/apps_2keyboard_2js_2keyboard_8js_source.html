00004 /*
00005  * keyboard.js:
00006  *
00007  * This is the main module of the Gaia keyboard app. It does these things:
00008  *
00009  * - Hides and shows the keyboard in response to focuschange events from
00010  *   navigator.mozKeyboard
00011  *
00012  * - Loads the input method (from imes/) module required by a keyboard
00013  *
00014  * - Handles settings changes for the user's language, keyboard layouts, and
00015  *   preferences for word suggestions, and updates the keyboard accordingly.
00016  *
00017  * - Handles resize events (caused by orientation changes) and updates
00018  *   the keyboard accordingly.
00019  *
00020  * - Handles (using code that was formerly in a separate controller.js module)
00021  *   mouse events over the keyboard and passes them to the input method.
00022  *
00023  * - When the input method sends value back to the keyboard, it turns
00024  *   them into synthetic key events using the navigator.mozKeyboard API.
00025  *
00026  * This module includes code that was formerly in the controller.js and
00027  * feedback.js modules. Other modules handle other parts of the keyboard:
00028  *
00029  *   layout.js: defines data structures that represent keyboard layouts
00030  *   render.js: creates the on-screen keyboard with HTML and CSS
00031  *
00032  * Input methods are in subdirectories of imes/.  The latin input method
00033  * in imes/latin/ provides word suggestions, auto capitalization, and
00034  * punctuation assistance.
00035  *
00036  * Each input method implements the following interface which the keyboard
00037  * uses to communicate with it. init() and click() are the only two required
00038  * methods; the keyboard checks that other methods are defined before
00039  * invoking them:
00040  *
00041  *    init(keyboard)
00042  *      keyboard is the object that the IM uses to communicate with the keyboard
00043  *
00044  *    activate(language, suggestionsEnabled, inputstate)
00045  *      the keyboard calls this method when it becomes active.
00046  *      language is the current language.  suggestionsEnabled
00047  *      specifies whether the user wants word suggestions inputstate
00048  *      is an object that holds the state of the input field or
00049  *      textarea being typed into.  it includes content, cursor
00050  *      position and type and inputmode attributes.
00051  *
00052  *    deactivate()
00053  *      called when the keyboard is hidden.
00054  *
00055  *    empty:
00056  *      clear any currently displayed candidates/suggestions.
00057  *      The latin input method does not use this, and it is not clear
00058  *      to me whether the Asian IMs need it either.
00059  *
00060  *    click(keycode, x, y):
00061  *      This is the main method: the keyboard calls this each time the
00062  *      user taps a key. The keyboard does not actually generate any
00063  *      key events until the input method tells it to. The x and y coordinate
00064  *      arguments can be used to improve the IM's word suggestions, in
00065  *      conjunction with the layout data from setLayoutParams().
00066  *      The coordinates aren't passed for the Backspace key, however.
00067  *
00068  *    select(word, data):
00069  *      Called when the user selects a displayed candidate or word suggestion.
00070  *
00071  *    setLayoutParams(params):
00072  *      Gives the IM information about the onscreen coordinates of
00073  *      each key. Used with latin IM only.  Can be used with click
00074  *      coordinates to improve predictions, but it may not currently
00075  *      be used.
00076  *
00077  *    getMoreCandidates(indicator, maxCount, callback):
00078  *      (optional) Called when the render needs more candidates to show on the
00079  *      candidate panel.
00080  *
00081  * The init method of each IM is passed an object that it uses to
00082  * communicate with the keyboard. That interface object defines the following
00083  * properties and methods:
00084  *
00085  *    path:
00086  *      a url that the IM can use to load dictionaries or other resources
00087  *
00088  *    sendCandidates:
00089  *      A method that makes the keyboard display candidates or suggestions
00090  *
00091  *    sendPendingSymbols:
00092  *      like sendCandidates, but used by Asian IMs.
00093  *
00094  *    sendKey(keycode):
00095  *      Generate output. Typically the keyboard will just pass this
00096  *      keycode to mozKeyboard.sendKey(). The IM could call
00097  *      mozKeyboard.sendKey() directly, but doing it this way allows
00098  *      us to chain IMs, I think.
00099  *
00100  *    sendString:
00101  *      Outputs a string of text by repeated calls to sendKey().
00102  *
00103  *    alterKeyboard(layout): allows the IM to modify the keyboard layout
00104  *      by specifying a new layout name. Only used by asian ims currently.
00105  *
00106  *    setLayoutPage(): allows the IM to switch between default and symbol
00107  *      layouts on the keyboard. Used by the latin IM.
00108  *
00109  *    setUpperCase(upperCase, upperCaseLocked): allows the IM to switch between
00110  *      uppercase and lowercase layout on the keyboard. Used by the latin IM.
00111  *      - upperCase: to enable the upper case or not.
00112  *      - upperCaseLocked: to change the caps lock state.
00113  *
00114  *    resetUpperCase(): allows the IM to reset the upperCase to lowerCase
00115  *      without knowing the internal states like caps lock and current layout
00116  *      page while keeping setUpperCase simple as it is.
00117  *
00118  *    getNumberOfCandidatesPerRow(): allow the IM to know how many candidates
00119  *      the Render need in one row so that IM can reduce search time and run the
00120  *      remaining process when "getMoreCandidates" is called.
00121  *
00122  */

00598 //
00599 // This function takes a keyboard layout, makes a copy of that layout
00600 // and then modifies it to add meta keys for switching languages and
00601 // switching to numbers and symbols. It may also add keys (like a
00602 // ".com" and '@') that are specific to input of currentInputType.
00603 //
00604 // The return value is the modified layout data structure.
00605 //
00606 // Normally, this function modifies the base layout for the specified
00607 // keyboard name, but it may use a different starting layout depending on
00608 // currentInputType and layoutPage.
00609 //
00610 function modifyLayout(keyboardName) {
00611   // One level copy
00612   function copy(obj) {
00613     var newObj = {};
00614     for (var prop in obj) if (obj.hasOwnProperty(prop)) {
00615       newObj[prop] = obj[prop];
00616     }
00617     return newObj;
00618   }
00619 
00620   var altLayoutName;
00621 
00622   switch (currentInputType) {
00623     case 'tel':
00624       altLayoutName = 'telLayout';
00625       break;
00626     case 'number':
00627       altLayoutName = currentInputMode === 'digit' ?
00628                                            'pinLayout' : 'numberLayout';
00629       break;
00630     // The matches when type="password", "text", or "search",
00631     // see mapInputType() for details
00632     case 'text':
00633       if (currentInputMode === 'digit') {
00634         altLayoutName = 'pinLayout';
00635       } else if (currentInputMode === 'numeric') {
00636         altLayoutName = 'numberLayout';
00637       }
00638       break;
00639   }
00640 
00641   if (layoutPage === LAYOUT_PAGE_SYMBOLS_I) {

01902 //
01903 // getSettings: Query the value of multiple settings at once.
01904 //
01905 // settings is an object whose property names are the settings to query
01906 // and whose property values are the default values to use if no such
01907 // setting is found.  This function will create a setting lock and
01908 // request the value of each of the specified settings.  Once it receives
01909 // a response to all of the queries, it passes all the settings values to
01910 // the specified callback function.  The argument to the callback function
01911 // is an object just like the settings object, where the property name is
01912 // the setting name and the property value is the setting value (or the
01913 // default value if the setting was not found).
01914 //
01915 function getSettings(settings, callback) {
01916   var results = {};
01917   try {
01918     var lock = navigator.mozSettings.createLock();
01919   }
01920   catch (e) {
01921     // If settings is broken, just return the default values
01922     console.warn('Exception in mozSettings.createLock():', e,
01923                  '\nUsing default values');
01924     for (var p in settings)
01925       results[p] = settings[p];
01926     callback(results);
01927   }
01928   var settingNames = Object.keys(settings);
01929   var numSettings = settingNames.length;
01930   var numResults = 0;
01931 
01932   for (var i = 0; i < numSettings; i++) {
01933     requestSetting(settingNames[i]);
01934   }

01936   function requestSetting(name) {
01937     try {
01938       var request = lock.get(name);
01939     }
01940     catch (e) {
01941       console.warn('Exception querying setting', name, ':', e,
01942                    '\nUsing default value');
01943       recordResult(name, settings[name]);
01944       return;
01945     }
01946     request.onsuccess = function() {
01947       var value = request.result[name];
01948       if (value === undefined)
01949         value = settings[name]; // Use the default value
01950       recordResult(name, value);
01951     };
01952     request.onerror = function(evt) {
01953       console.warn('Error querying setting', name, ':', evt.error);
01954       recordResult(name, settings[name]);
01955     };
01956   }
01957 
01958   function recordResult(name, value) {
01959     results[name] = value;
01960     numResults++;
01961     if (numResults === numSettings) {
01962       callback(results);
01963     }
01964   }
