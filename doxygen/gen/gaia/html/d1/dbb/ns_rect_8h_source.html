00060 #ifdef NS_COORD_IS_FLOAT

00065     if (MOZ_UNLIKELY(w > nscoord_MAX)) {
00066       NS_WARNING("Overflowed nscoord_MAX in conversion to nscoord width");
00067       int64_t xdiff = nx - nscoord_MIN / 2;
00068       if (xdiff < 0) {

00070         nx = nscoord_MIN / 2;

00253 nsRect::ConvertAppUnitsRoundOut(int32_t aFromAPP, int32_t aToAPP) const
00254 {
00255   if (aFromAPP == aToAPP) {
00256     return *this;
00257   }
00258 
00259   nsRect rect;
00260   nscoord right = NSToCoordCeil(NSCoordScale(XMost(), aFromAPP, aToAPP));
00261   nscoord bottom = NSToCoordCeil(NSCoordScale(YMost(), aFromAPP, aToAPP));
00262   rect.x = NSToCoordFloor(NSCoordScale(x, aFromAPP, aToAPP));
00263   rect.y = NSToCoordFloor(NSCoordScale(y, aFromAPP, aToAPP));
00264   rect.width = (right - rect.x);
00265   rect.height = (bottom - rect.y);
00266 
00267   return rect;
00268 }
