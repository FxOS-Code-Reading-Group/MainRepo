00106 /*
00107  * wait for a specific slot event.
00108  * event is a specific event to wait for. Currently only 
00109  *    PK11TokenChangeOrRemovalEvent and PK11TokenPresentEvents are defined.
00110  * timeout can be an interval time to wait, PR_INTERVAL_NO_WAIT (meaning only
00111  * poll once), or PR_INTERVAL_NO_TIMEOUT (meaning block until a change).
00112  * pollInterval is a suggested pulling interval value. '0' means use the 
00113  *  default. Future implementations that don't poll may ignore this value.
00114  * series is the current series for the last slot. This should be the series 
00115  *  value for the slot the last time you read persistant information from the
00116  *  slot. For instance, if you publish a cert from the slot, you should obtain
00117  *  the slot series at that time. Then PK11_WaitForTokenEvent can detect a 
00118  *  a change in the slot between the time you publish and the time 
00119  *  PK11_WaitForTokenEvent is called, elliminating potential race conditions.
00120  *
00121  * The current status that is returned is:
00122  *   PK11TokenNotRemovable - always returned for any non-removable token.
00123  *   PK11TokenPresent - returned when the token is present and we are waiting
00124  *     on a PK11TokenPresentEvent. Then next event to look for is a 
00125  *     PK11TokenChangeOrRemovalEvent.
00126  *   PK11TokenChanged - returned when the old token has been removed and a new
00127  *     token ad been inserted, and we are waiting for a 
00128  *     PK11TokenChangeOrRemovalEvent. The next event to look for is another
00129  *     PK11TokenChangeOrRemovalEvent.
00130  *   PK11TokenRemoved - returned when the token is not present and we are 
00131  *     waiting for a PK11TokenChangeOrRemovalEvent. The next event to look for 
00132  *     is a PK11TokenPresentEvent.
00133  */
00134 PK11TokenStatus PK11_WaitForTokenEvent(PK11SlotInfo *slot, PK11TokenEvent event,
00135     PRIntervalTime timeout, PRIntervalTime pollInterval, int series);

00161 /*
00162  * Open a new database using the softoken. The caller is responsible for making
00163  * sure the module spec is correct and usable. The caller should ask for one
00164  * new database per call if the caller wants to get meaningful information
00165  * about the new database.
00166  *
00167  * moduleSpec is the same data that you would pass to softoken at
00168  * initialization time under the 'tokens' options. For example, if you were
00169  * to specify tokens=<0x4=[configdir='./mybackup' tokenDescription='Backup']>
00170  * You would specify "configdir='./mybackup' tokenDescription='Backup'" as your
00171  * module spec here. The slot ID will be calculated for you by
00172  * SECMOD_OpenUserDB().
00173  *
00174  * Typical parameters here are configdir, tokenDescription and flags.
00175  *
00176  * a Full list is below:
00177  *
00178  *
00179  *  configDir - The location of the databases for this token. If configDir is
00180  *         not specified, and noCertDB and noKeyDB is not specified, the load
00181  *         will fail.
00182  *   certPrefix - Cert prefix for this token.
00183  *   keyPrefix - Prefix for the key database for this token. (if not specified,
00184  *         certPrefix will be used).
00185  *   tokenDescription - The label value for this token returned in the
00186  *         CK_TOKEN_INFO structure with an internationalize string (UTF8).
00187  *         This value will be truncated at 32 bytes (no NULL, partial UTF8
00188  *         characters dropped). You should specify a user friendly name here
00189  *         as this is the value the token will be referred to in most
00190  *         application UI's. You should make sure tokenDescription is unique.
00191  *   slotDescription - The slotDescription value for this token returned
00192  *         in the CK_SLOT_INFO structure with an internationalize string
00193  *         (UTF8). This value will be truncated at 64 bytes (no NULL, partial
00194  *         UTF8 characters dropped). This name will not change after the
00195  *         database is closed. It should have some number to make this unique.
00196  *   minPWLen - minimum password length for this token.
00197  *   flags - comma separated list of flag values, parsed case-insensitive.
00198  *         Valid flags are:
00199  *              readOnly - Databases should be opened read only.
00200  *              noCertDB - Don't try to open a certificate database.
00201  *              noKeyDB - Don't try to open a key database.
00202  *              forceOpen - Don't fail to initialize the token if the
00203  *                databases could not be opened.
00204  *              passwordRequired - zero length passwords are not acceptable
00205  *                (valid only if there is a keyDB).
00206  *              optimizeSpace - allocate smaller hash tables and lock tables.
00207  *                When this flag is not specified, Softoken will allocate
00208  *                large tables to prevent lock contention.
00209  */
00210 PK11SlotInfo *SECMOD_OpenUserDB(const char *moduleSpec);
00211 SECStatus SECMOD_CloseUserDB(PK11SlotInfo *slot);
