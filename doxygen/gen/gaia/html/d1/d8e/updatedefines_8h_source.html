00048 // On Windows, _snprintf and _snwprintf don't guarantee null termination. These
00049 // macros always leave room in the buffer for null termination and set the end
00050 // of the buffer to null in case the string is larger than the buffer. Having
00051 // multiple nulls in a string is fine and this approach is simpler (possibly
00052 // faster) than calculating the string length to place the null terminator and
00053 // truncates the string as _snprintf and _snwprintf do on other platforms.
00054 static int mysnprintf(char* dest, size_t count, const char* fmt, ...)
00055 {
00056   size_t _count = count - 1;
00057   va_list varargs;
00058   va_start(varargs, fmt);
00059   int result = _vsnprintf(dest, count - 1, fmt, varargs);
00060   va_end(varargs);
00061   dest[_count] = '\0';
00062   return result;
00063 }
00064 #define snprintf mysnprintf
00065 static int mywcsprintf(WCHAR* dest, size_t count, const WCHAR* fmt, ...)
00066 {
00067   size_t _count = count - 1;
00068   va_list varargs;
00069   va_start(varargs, fmt);
00070   int result = _vsnwprintf(dest, count - 1, fmt, varargs);
00071   va_end(varargs);
00072   dest[_count] = L'\0';
00073   return result;
00074 }
