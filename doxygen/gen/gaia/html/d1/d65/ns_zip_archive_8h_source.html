00361   T* Forget() {
00362     if (!mReturnBuf)
00363       return NULL;
00364     // In uncompressed mmap case, give up buffer
00365     if (mAutoBuf.get() == mReturnBuf) {
00366       mReturnBuf = NULL;
00367       return (T*) mAutoBuf.forget();
00368     }
00369     T *ret = (T*) malloc(Length());
00370     memcpy(ret, mReturnBuf, Length());
00371     mReturnBuf = NULL;
00372     return ret;
00373   }
