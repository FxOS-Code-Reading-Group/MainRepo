00210 template<typename IntegerType>
00211 struct MinValue
00212 {
00213   private:
00214     typedef typename UnsignedType<IntegerType>::Type UnsignedIntegerType;
00215     static const size_t PosOfSignBit = PositionOfSignBit<IntegerType>::value;
00216 
00217   public:
00218     // Bitwise ops may return a larger type, that's why we cast explicitly.
00219     // In C++, left bit shifts on signed values is undefined by the standard
00220     // unless the shifted value is representable.
00221     // Notice that signed-to-unsigned conversions are always well-defined in
00222     // the standard as the value congruent to 2**n, as expected. By contrast,
00223     // unsigned-to-signed is only well-defined if the value is representable.
00224     static const IntegerType value =
00225         IsSigned<IntegerType>::value
00226         ? IntegerType(UnsignedIntegerType(1) << PosOfSignBit)
00227         : IntegerType(0);
00228 };

00353 template<typename T>
00354 inline bool
00355 IsAddValid(T x, T y)
00356 {
00357   // Addition is valid if the sign of x+y is equal to either that of x or that
00358   // of y. Since the value of x+y is undefined if we have a signed type, we
00359   // compute it using the unsigned type of the same size.
00360   // Beware! These bitwise operations can return a larger integer type,
00361   // if T was a small type like int8_t, so we explicitly cast to T.
00362 
00363   typename UnsignedType<T>::Type ux = x;
00364   typename UnsignedType<T>::Type uy = y;
00365   typename UnsignedType<T>::Type result = ux + uy;
00366   return IsSigned<T>::value
00367          ? HasSignBit(BinaryComplement(T((result ^ x) & (result ^ y))))
00368          : BinaryComplement(x) >= y;
00369 }

00371 template<typename T>
00372 inline bool
00373 IsSubValid(T x, T y)
00374 {
00375   // Subtraction is valid if either x and y have same sign, or x-y and x have
00376   // same sign. Since the value of x-y is undefined if we have a signed type,
00377   // we compute it using the unsigned type of the same size.
00378   typename UnsignedType<T>::Type ux = x;
00379   typename UnsignedType<T>::Type uy = y;
00380   typename UnsignedType<T>::Type result = ux - uy;
00381 
00382   return IsSigned<T>::value
00383          ? HasSignBit(BinaryComplement(T((result ^ x) & (x ^ y))))
00384          : x >= y;
00385 }
