00013 /*
00014 ** prlog.h -- Declare interfaces to NSPR's Logging service
00015 **
00016 ** NSPR provides a logging service that is used by NSPR itself and is
00017 ** available to client programs.
00018 **
00019 ** To use the service from a client program, you should create a
00020 ** PRLogModuleInfo structure by calling PR_NewLogModule(). After
00021 ** creating the LogModule, you can write to the log using the PR_LOG()
00022 ** macro.
00023 **
00024 ** Initialization of the log service is handled by NSPR initialization.
00025 **
00026 ** At execution time, you must enable the log service. To enable the
00027 ** log service, set the environment variable: NSPR_LOG_MODULES
00028 ** variable.
00029 **
00030 ** NSPR_LOG_MODULES variable has the form:
00031 **
00032 **     <moduleName>:<value>[, <moduleName>:<value>]*
00033 **
00034 ** Where:
00035 **  <moduleName> is the name passed to PR_NewLogModule().
00036 **  <value> is a numeric constant, e.g. 5. This value is the maximum
00037 ** value of a log event, enumerated by PRLogModuleLevel, that you want
00038 ** written to the log.
00039 ** 
00040 ** For example: to record all events of greater value than or equal to
00041 ** PR_LOG_ERROR for a LogModule names "gizmo", say:
00042 ** 
00043 ** set NSPR_LOG_MODULES=gizmo:2
00044 ** 
00045 ** Note that you must specify the numeric value of PR_LOG_ERROR.
00046 ** 
00047 ** Special LogModule names are provided for controlling NSPR's log
00048 ** service at execution time. These controls should be set in the
00049 ** NSPR_LOG_MODULES environment variable at execution time to affect
00050 ** NSPR's log service for your application.
00051 ** 
00052 ** The special LogModule "all" enables all LogModules. To enable all
00053 ** LogModule calls to PR_LOG(), say:
00054 ** 
00055 ** set NSPR_LOG_MODULES=all:5
00056 ** 
00057 ** The special LogModule name "sync" tells the NSPR log service to do
00058 ** unbuffered logging.
00059 ** 
00060 ** The special LogModule name "bufsize:<size>" tells NSPR to set the
00061 ** log buffer to <size>.
00062 **
00063 ** The environment variable NSPR_LOG_FILE specifies the log file to use
00064 ** unless the default of "stderr" is acceptable. For MS Windows
00065 ** systems, NSPR_LOG_FILE can be set to a special value: "WinDebug"
00066 ** (case sensitive). This value causes PR_LOG() output to be written
00067 ** using the Windows API OutputDebugString(). OutputDebugString()
00068 ** writes to the debugger window; some people find this helpful.
00069 ** 
00070 **
00071 ** To put log messages in your programs, use the PR_LOG macro:
00072 **
00073 **     PR_LOG(<module>, <level>, (<printfString>, <args>*));
00074 **
00075 ** Where <module> is the address of a PRLogModuleInfo structure, and
00076 ** <level> is one of the levels defined by the enumeration:
00077 ** PRLogModuleLevel. <args> is a printf() style of argument list. That
00078 ** is: (fmtstring, ...).
00079 **
00080 ** Example:
00081 ** 
00082 ** main() {
00083 **    PRIntn one = 1;
00084 **    PRLogModuleInfo * myLm = PR_NewLogModule("gizmo");
00085 **    PR_LOG( myLm, PR_LOG_ALWAYS, ("Log this! %d\n", one)); 
00086 **    return; 
00087 ** }
00088 ** 
00089 ** Note the use of printf() style arguments as the third agrument(s) to
00090 ** PR_LOG().
00091 ** 
00092 ** After compiling and linking you application, set the environment:
00093 ** 
00094 ** set NSPR_LOG_MODULES=gizmo:5
00095 ** set NSPR_LOG_FILE=logfile.txt
00096 ** 
00097 ** When you execute your application, the string "Log this! 1" will be
00098 ** written to the file "logfile.txt".
00099 ** 
00100 ** Note to NSPR engineers: a number of PRLogModuleInfo structures are
00101 ** defined and initialized in prinit.c. See this module for ideas on
00102 ** what to log where.
00103 ** 
00104 */

00106 typedef enum PRLogModuleLevel {
00107     PR_LOG_NONE = 0,                /* nothing */
00108     PR_LOG_ALWAYS = 1,              /* always printed */
00109     PR_LOG_ERROR = 2,               /* error messages */
00110     PR_LOG_WARNING = 3,             /* warning messages */
00111     PR_LOG_DEBUG = 4,               /* debug messages */
00112 
00113     PR_LOG_NOTICE = PR_LOG_DEBUG,   /* notice messages */

00115     PR_LOG_MIN = PR_LOG_DEBUG,      /* minimal debugging messages */
00116     PR_LOG_MAX = PR_LOG_DEBUG       /* maximal debugging messages */
