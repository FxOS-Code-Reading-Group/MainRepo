00028 // Workaround for Necko re-entrancy dangers. We buffer IPDL messages in a
00029 // queue if still dispatching previous one(s) to listeners/observers.
00030 // Otherwise synchronous XMLHttpRequests and/or other code that spins the
00031 // event loop (ex: IPDL rpc) could cause listener->OnDataAvailable (for
00032 // instance) to be dispatched and called before mListener->OnStartRequest has
00033 // completed.
00034 
00035 class AutoEventEnqueuerBase;

00037 class ChannelEventQueue
00038 {
00039   NS_INLINE_DECL_REFCOUNTING(ChannelEventQueue)
00040 
00041  public:
00042   ChannelEventQueue(nsISupports *owner)
00043     : mSuspendCount(0)
00044     , mSuspended(false)
00045     , mForced(false)
00046     , mFlushing(false)
00047     , mOwner(owner) {}
00048 
00049   ~ChannelEventQueue() {}
00050 
00051   // Checks to determine if an IPDL-generated channel event can be processed
00052   // immediately, or needs to be queued using Enqueue().
00053   inline bool ShouldEnqueue();
00054 
00055   // Puts IPDL-generated channel event into queue, to be run later
00056   // automatically when EndForcedQueueing and/or Resume is called.
00057   inline void Enqueue(ChannelEvent* callback);
00058 
00059   // After StartForcedQueueing is called, ShouldEnqueue() will return true and
00060   // no events will be run/flushed until EndForcedQueueing is called.
00061   // - Note: queueing may still be required after EndForcedQueueing() (if the
00062   //   queue is suspended, etc):  always call ShouldEnqueue() to determine
00063   //   whether queueing is needed.
00064   inline void StartForcedQueueing();
00065   inline void EndForcedQueueing();
00066 
00067   // Suspend/resume event queue.  ShouldEnqueue() will return true and no events
00068   // will be run/flushed until resume is called.  These should be called when
00069   // the channel owning the event queue is suspended/resumed.
00070   inline void Suspend();
00071   // Resume flushes the queue asynchronously, i.e. items in queue will be
00072   // dispatched in a new event on the current thread.
00073   inline void Resume();
00074 
00075  private:
00076   inline void MaybeFlushQueue();
00077   void FlushQueue();
00078   inline void CompleteResume();
00079 
00080   nsTArray<nsAutoPtr<ChannelEvent> > mEventQueue;
00081 
00082   uint32_t mSuspendCount;
00083   bool     mSuspended;
00084   bool mForced;
00085   bool mFlushing;
00086 
00087   // Keep ptr to avoid refcount cycle: only grab ref during flushing.
00088   nsISupports *mOwner;

00130 inline void
00131 ChannelEventQueue::CompleteResume()
00132 {
00133   // channel may have been suspended again since Resume fired event to call this.
00134   if (!mSuspendCount) {
00135     // we need to remain logically suspended (for purposes of queuing incoming
00136     // messages) until this point, else new incoming messages could run before
00137     // queued ones.
00138     mSuspended = false;
00139     MaybeFlushQueue();
00140   }
00141 }

00143 inline void
00144 ChannelEventQueue::Resume()
00145 {
00146   // Resuming w/o suspend: error in debug mode, ignore in build
00147   MOZ_ASSERT(mSuspendCount > 0);
00148   if (mSuspendCount <= 0) {
00149     return;
00150   }
00151 
00152   if (!--mSuspendCount) {
00153     nsRefPtr<nsRunnableMethod<ChannelEventQueue> > event =
00154       NS_NewRunnableMethod(this, &ChannelEventQueue::CompleteResume);
00155     NS_DispatchToCurrentThread(event);
00156   }
00157 }

00159 inline void
00160 ChannelEventQueue::MaybeFlushQueue()
00161 {
00162   // Don't flush if forced queuing on, we're already being flushed, or
00163   // suspended, or there's nothing to flush
00164   if (!mForced && !mFlushing && !mSuspended && !mEventQueue.IsEmpty())
00165     FlushQueue();
00166 }
