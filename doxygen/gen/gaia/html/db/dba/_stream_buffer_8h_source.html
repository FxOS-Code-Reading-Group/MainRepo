00038 inline TrackTicks TimeToTicksRoundUp(TrackRate aRate, StreamTime aTime)
00039 {
00040   NS_ASSERTION(0 < aRate && aRate <= TRACK_RATE_MAX, "Bad rate");
00041   NS_ASSERTION(0 <= aTime && aTime <= STREAM_TIME_MAX, "Bad time");
00042   return (aTime*aRate + (1 << MEDIA_TIME_FRAC_BITS) - 1) >> MEDIA_TIME_FRAC_BITS;
00043 }

00045 inline TrackTicks TimeToTicksRoundDown(TrackRate aRate, StreamTime aTime)

00047   NS_ASSERTION(0 < aRate && aRate <= TRACK_RATE_MAX, "Bad rate");
00048   NS_ASSERTION(0 <= aTime && aTime <= STREAM_TIME_MAX, "Bad time");
00049   return (aTime*aRate) >> MEDIA_TIME_FRAC_BITS;

00052 inline StreamTime TicksToTimeRoundUp(TrackRate aRate, TrackTicks aTicks)
00053 {
00054   NS_ASSERTION(0 < aRate && aRate <= TRACK_RATE_MAX, "Bad rate");
00055   NS_ASSERTION(0 <= aTicks && aTicks <= TRACK_TICKS_MAX, "Bad samples");
00056   return ((aTicks << MEDIA_TIME_FRAC_BITS) + aRate - 1)/aRate;
00057 }

00059 inline StreamTime TicksToTimeRound(TrackRate aRate, TrackTicks aTicks)
00060 {
00061   NS_ASSERTION(0 < aRate && aRate <= TRACK_RATE_MAX, "Bad rate");
00062   NS_ASSERTION(0 <= aTicks && aTicks <= TRACK_TICKS_MAX, "Bad samples");
00063   return ((aTicks << MEDIA_TIME_FRAC_BITS) + aRate/2)/aRate;
00064 }
