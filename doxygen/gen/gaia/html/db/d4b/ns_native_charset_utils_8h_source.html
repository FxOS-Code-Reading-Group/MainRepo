00027 /* 
00028  * This function indicates whether the character encoding used in the file
00029  * system (more exactly what's used for |GetNativeFoo| and |SetNativeFoo|
00030  * of |nsIFile|) is UTF-8 or not. Knowing that helps us avoid an
00031  * unncessary encoding conversion in some cases. For instance, to get the leaf
00032  * name in UTF-8 out of nsIFile, we can just use |GetNativeLeafName| rather
00033  * than using |GetLeafName| and converting the result to UTF-8 if the file 
00034  * system  encoding is UTF-8.
00035  * On Unix (but not on Mac OS X), it depends on the locale and is not known
00036  * in advance (at the compilation time) so that this function needs to be 
00037  * a real function. On Mac OS X it's always UTF-8 while on Windows 
00038  * and other platforms (e.g. OS2), it's never UTF-8.  
00039  */
00040 #if defined(XP_UNIX) && !defined(XP_MACOSX) && !defined(ANDROID)
00041 bool NS_IsNativeUTF8();
00042 #else
00043 inline bool NS_IsNativeUTF8()
00044 {
00045 #if defined(XP_MACOSX) || defined(ANDROID)
00046     return true;
00047 #else
00048     return false;
00049 #endif
00050 }
00051 #endif
