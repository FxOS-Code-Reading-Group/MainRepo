00103 interface nsIDOMSimpleGestureEvent : nsIDOMMouseEvent
00104 {
00105   /* Swipe direction constants */
00106   const unsigned long DIRECTION_UP = 1;
00107   const unsigned long DIRECTION_DOWN = 2;
00108   const unsigned long DIRECTION_LEFT = 4;
00109   const unsigned long DIRECTION_RIGHT = 8;
00110 
00111   /* Rotational direction constants */

00114 
00115   /* Read-write value for swipe events.
00116    *
00117    * Reports the directions that can be swiped to; multiple directions
00118    * should be OR'ed together.
00119    *
00120    * The allowedDirections field is designed to be set on SwipeGestureStart
00121    * events by event listeners.  Its value after event dispatch determines
00122    * the behavior of the swipe animation that is about to begin.
00123    * Specifically, if the user swipes in a direction that can't be swiped
00124    * to, the animation will have a bounce effect.
00125    * Future SwipeGestureUpdate, SwipeGesture and SwipeGestureEnd events
00126    * will carry the allowDirections value that was set on the SwipeStart
00127    * event.  Changing this field on non-SwipeGestureStart events doesn't
00128    * have any effect.
00129    */
00130   attribute unsigned long allowedDirections;
00131 
00132   /* Direction of a gesture. Diagonals are indicated by OR'ing the
00133    * applicable constants together.
00134    *
00135    * Swipes gestures may occur in any direction.
00136    *
00137    * Magnify gestures do not have a direction.
00138    *
00139    * Rotation gestures will be either ROTATION_COUNTERCLOCKWISE or
00140    * ROTATION_CLOCKWISE.
00141    */
00142   readonly attribute unsigned long direction;
00143 
00144   /* Delta value for magnify, rotate and swipe gestures.
00145    *
00146    * For rotation, the value is in degrees and is positive for
00147    * clockwise rotation and negative for counterclockwise
00148    * rotation.
00149    *
00150    * For magnification, the value will be positive for a "zoom in"
00151    * (i.e, increased magnification) and negative for a "zoom out"
00152    * (i.e., decreased magnification).  The particular units
00153    * represented by the "delta" are currently implementation specific.
00154    *
00155    * XXX - The units for measuring magnification are currently
00156    * unspecified because the units used by Mac OS X are currently
00157    * undocumented.  The values are typically in the range of 0.0 to
00158    * 100.0, but it is only safe currently to rely on the delta being
00159    * positive or negative.
00160    *
00161    * For swipe start, update and end events, the value is a fraction
00162    * of one "page".  If the resulting swipe will have DIRECTION_LEFT, the
00163    * delta value will be positive; for DIRECTION_RIGHT, delta is negative.
00164    * If this seems backwards to you, look at it this way:  If the current
00165    * page is pushed to the right during the animation (positive delta),
00166    * the page left to the current page will be visible after the swipe
00167    * (DIRECTION_LEFT).
00168    *
00169    * Units on Windows represent the difference between the initial
00170    * and current/final width between the two touch points on the input
00171    * device and are measured in pixels.
00172    */
00173   readonly attribute double delta;
