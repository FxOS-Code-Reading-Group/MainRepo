00038  *
00039  * The double buffer renderer begins with no initialize surfaces.

00041  * Swapping still occurs regardless if the buffers are initialized.
00042  */

00046   // Returns width in "display pixels".  A "display pixel" is the smallest
00047   // fully addressable part of a display.  But in HiDPI modes each "display
00048   // pixel" corresponds to more than one device pixel.  Multiply display pixels
00049   // by mContentsScaleFactor to get device pixels.
00050   size_t GetFrontSurfaceWidth();

00052   // mContentsScaleFactor to get device pixels.
00053   size_t GetFrontSurfaceHeight();
00054   double GetFrontSurfaceContentsScaleFactor();

00056   // mContentsScaleFactor to get device pixels.
00057   size_t GetBackSurfaceWidth();
00058   // Returns height in "display pixels".  Multiply by
00059   // mContentsScaleFactor to get device pixels.
00060   size_t GetBackSurfaceHeight();
00061   double GetBackSurfaceContentsScaleFactor();

00064   bool HasBackSurface();

00072                         double aContentsScaleFactor,
