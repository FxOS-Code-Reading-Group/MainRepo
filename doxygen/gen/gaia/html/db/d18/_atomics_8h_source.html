00095 enum MemoryOrdering {
00096   /*
00097    * Relaxed ordering is the simplest memory ordering: none at all.
00098    * When the result of a write is observed, nothing may be inferred
00099    * about other memory.  Writes ostensibly performed "before" on the
00100    * writing thread may not yet be visible.  Writes performed "after" on
00101    * the writing thread may already be visible, if the compiler or CPU
00102    * reordered them.  (The latter can happen if reads and/or writes get
00103    * held up in per-processor caches.)  Relaxed ordering means
00104    * operations can always use cached values (as long as the actual
00105    * updates to atomic values actually occur, correctly, eventually), so
00106    * it's usually the fastest sort of atomic access.  For this reason,
00107    * *it's also the most dangerous kind of access*.
00108    *
00109    * Relaxed ordering is good for things like process-wide statistics
00110    * counters that don't need to be consistent with anything else, so
00111    * long as updates themselves are atomic.  (And so long as any
00112    * observations of that value can tolerate being out-of-date -- if you
00113    * need some sort of up-to-date value, you need some sort of other
00114    * synchronizing operation.)  It's *not* good for locks, mutexes,
00115    * reference counts, etc. that mediate access to other memory, or must
00116    * be observably consistent with other memory.
00117    *
00118    * x86 architectures don't take advantage of the optimization
00119    * opportunities that relaxed ordering permits.  Thus it's possible
00120    * that using relaxed ordering will "work" on x86 but fail elsewhere
00121    * (ARM, say, which *does* implement non-sequentially-consistent
00122    * relaxed ordering semantics).  Be extra-careful using relaxed
00123    * ordering if you can't easily test non-x86 architectures!
00124    */
00125   Relaxed,
00126   /*
00127    * When an atomic value is updated with ReleaseAcquire ordering, and
00128    * that new value is observed with ReleaseAcquire ordering, prior
00129    * writes (atomic or not) are also observable.  What ReleaseAcquire
00130    * *doesn't* give you is any observable ordering guarantees for
00131    * ReleaseAcquire-ordered operations on different objects.  For
00132    * example, if there are two cores that each perform ReleaseAcquire
00133    * operations on separate objects, each core may or may not observe
00134    * the operations made by the other core.  The only way the cores can
00135    * be synchronized with ReleaseAcquire is if they both
00136    * ReleaseAcquire-access the same object.  This implies that you can't
00137    * necessarily describe some global total ordering of ReleaseAcquire
00138    * operations.
00139    *
00140    * ReleaseAcquire ordering is good for (as the name implies) atomic
00141    * operations on values controlling ownership of things: reference
00142    * counts, mutexes, and the like.  However, if you are thinking about
00143    * using these to implement your own locks or mutexes, you should take
00144    * a good, hard look at actual lock or mutex primitives first.
00145    */
00146   ReleaseAcquire,
00147   /*
00148    * When an atomic value is updated with SequentiallyConsistent
00149    * ordering, all writes observable when the update is observed, just
00150    * as with ReleaseAcquire ordering.  But, furthermore, a global total
00151    * ordering of SequentiallyConsistent operations *can* be described.
00152    * For example, if two cores perform SequentiallyConsistent operations
00153    * on separate objects, one core will observably perform its update
00154    * (and all previous operations will have completed), then the other
00155    * core will observably perform its update (and all previous
00156    * operations will have completed).  (Although those previous
00157    * operations aren't themselves ordered -- they could be intermixed,
00158    * or ordered if they occur on atomic values with ordering
00159    * requirements.)  SequentiallyConsistent is the *simplest and safest*
00160    * ordering of atomic operations -- it's always as if one operation
00161    * happens, then another, then another, in some order -- and every
00162    * core observes updates to happen in that single order.  Because it
00163    * has the most synchronization requirements, operations ordered this
00164    * way also tend to be slowest.
00165    *
00166    * SequentiallyConsistent ordering can be desirable when multiple
00167    * threads observe objects, and they all have to agree on the
00168    * observable order of changes to them.  People expect
00169    * SequentiallyConsistent ordering, even if they shouldn't, when
00170    * writing code, atomic or otherwise.  SequentiallyConsistent is also
00171    * the ordering of choice when designing lockless data structures.  If
00172    * you don't know what order to use, use this one.
00173    */
00174   SequentiallyConsistent,
00175 };

00334  *

00388 template<typename T, MemoryOrdering Order>
00389 struct IntrinsicMemoryOps
00390 {
00391     static T load(const T& ptr) {
00392       Barrier<Order>::beforeLoad();
00393       T val = ptr;
00394       Barrier<Order>::afterLoad();
00395       return val;
00396     }
00397     static void store(T& ptr, T val) {
00398       Barrier<Order>::beforeStore();
00399       ptr = val;
00400       Barrier<Order>::afterStore();
00401     }
00402     static T exchange(T& ptr, T val) {
00403       // __sync_lock_test_and_set is only an acquire barrier; loads and stores
00404       // can't be moved up from after to before it, but they can be moved down
00405       // from before to after it.  We may want a stricter ordering, so we need
00406       // an explicit barrier.
00407 
00408       Barrier<Order>::beforeStore();
00409       return __sync_lock_test_and_set(&ptr, val);
00410     }
00411     static bool compareExchange(T& ptr, T oldVal, T newVal) {
00412       return __sync_bool_compare_and_swap(&ptr, oldVal, newVal);
00413     }
00414 };

00522 /*
00523  * The PrimitiveIntrinsics template should define |Type|, the datatype of size
00524  * DataSize upon which we operate, and the following eight functions.
00525  *
00526  * static Type add(Type* ptr, Type val);
00527  * static Type sub(Type* ptr, Type val);
00528  * static Type or_(Type* ptr, Type val);
00529  * static Type xor_(Type* ptr, Type val);
00530  * static Type and_(Type* ptr, Type val);
00531  *
00532  *   These functions perform the obvious operation on the value contained in
00533  *   |*ptr| combined with |val| and return the value previously stored in
00534  *   |*ptr|.
00535  *
00536  * static void store(Type* ptr, Type val);
00537  *
00538  *   This function atomically stores |val| into |*ptr| and must provide a full
00539  *   memory fence after the store to prevent compiler and hardware instruction
00540  *   reordering.  It should also act as a compiler barrier to prevent reads and
00541  *   writes from moving to after the store.
00542  *
00543  * static Type exchange(Type* ptr, Type val);
00544  *
00545  *   This function atomically stores |val| into |*ptr| and returns the previous
00546  *   contents of *ptr;
00547  *
00548  * static bool compareExchange(Type* ptr, Type oldVal, Type newVal);
00549  *
00550  *   This function atomically performs the following operation:
00551  *
00552  *     if (*ptr == oldVal) {
00553  *       *ptr = newVal;
00554  *       return true;
00555  *     } else {
00556  *       return false;
00557  *     }
00558  *
00559  */

00721 {
00722     static ValueType load(const ValueType& ptr) {
00723       Barrier<Order>::beforeLoad();
00724       ValueType val = ptr;
00725       Barrier<Order>::afterLoad();
00726       return val;
00727     }
00728     static void store(ValueType& ptr, ValueType val) {
00729       // For SequentiallyConsistent, Primitives::store() will generate the
00730       // proper memory fence.  Everything else just needs a barrier before
00731       // the store.
00732       if (Order == SequentiallyConsistent) {
00733         Primitives::store(reinterpret_cast<PrimType*>(&ptr),
00734                           Cast::toPrimType(val));
00735       } else {
00736         Barrier<Order>::beforeStore();
00737         ptr = val;
00738       }
00739     }

00741       PrimType oldval =

00743                              Cast::toPrimType(val));
00744       return Cast::fromPrimType(oldval);
00745     }
00746     static bool compareExchange(ValueType& ptr, ValueType oldVal, ValueType newVal) {
00747       return Primitives::compareExchange(reinterpret_cast<PrimType*>(&ptr),
00748                                          Cast::toPrimType(oldVal),
