00407  *
00408  * Display an edited version of the specified image in a <canvas> element
00409  * inside the specified container element. The image (or cropped region of
00410  * the image) will be displayed as large as possible within the container's

00412  * The edits object may include these properties:
00413  *
00414  *  gamma: a float specifying gamma correction
00415  *  matrix: a 4x4 matrix that represents a transformation of each rgba pixel.
00416  *    this can be used to convert to bw or sepia, for example.
00417  *  borderWidth: the size of the border as a fraction of the image width
00418  *  borderColor: a [r, g, b, a] array specifying border color

00421  * getFullSizeBlob() function that creates a full-size version of the
00422  * edited image.
00423  *
00424  * This class also handles cropping.  See:
00425  *
00426  *   showCropOverlay()
00427  *   hideCropOverlay()
00428  *   cropImage()

00431  * This code expects WebGL GLSL shader programs in scripts with ids
00432  * edit-vertex-shader and edit-fragment-shader. It dynamically creates

00807 // Called when the first pan event comes in on the crop canvas
00808 ImageEditor.prototype.cropStart = function(ev) {
00809   var self = this;
00810   var region = this.cropRegion;
00811   var dest = this.dest;
00812   var rect = this.previewCanvas.getBoundingClientRect();
00813   var x0 = ev.detail.position.screenX - rect.left - dest.x;
00814   var y0 = ev.detail.position.screenY - rect.top - dest.y;
00815   var left = region.left;
00816   var top = region.top;
00817   var right = region.right;
00818   var bottom = region.bottom;
00819   var aspectRatio = this.cropAspectWidth ?
00820     this.cropAspectWidth / this.cropAspectHeight :
00821     0;
00822   var centerX = (region.left + region.right) / 2;
00823   var centerY = (region.top + region.bottom) / 2;
00824 
00825   // Return true if (x0,y0) is within 25 pixels of (x,y)
00826   function hit(x, y) {
00827     return (x0 > x - 25 && x0 < x + 25 &&
00828             y0 > y - 25 && y0 < y + 25);
00829   }
00830 
00831   if (hit((left + right) / 2, top))
00832     drag('n');
00833   else if (hit(right, (top + bottom) / 2))
00834     drag('e');
00835   else if (hit((left + right) / 2, bottom))
00836     drag('s');
00837   else if (hit(left, (top + bottom) / 2))
00838     drag('w');
00839   else if (!aspectRatio) {
00840     if (hit(right, top))
00841       drag('ne');
00842     else if (hit(right, bottom))
00843       drag('se');
00844     else if (hit(left, bottom))
00845       drag('sw');
00846     else if (hit(left, top))
00847       drag('nw');
00848     else
00849       drag(); // with no argument, do a pan instead of a drag
00850   }
00851   else
00852     drag(); // pan
00853 
00854   function drag(handle) {
00855     window.addEventListener('pan', move, true);
00856     window.addEventListener('swipe', up, true);
00857 
00858     self.drawCropControls(handle); // highlight drag handle
00859 
00860     function move(e) {
00861       var dx = e.detail.absolute.dx;
00862       var dy = e.detail.absolute.dy;
00863 
00864       var newleft = region.left;
00865       var newright = region.right;
00866       var newtop = region.top;
00867       var newbottom = region.bottom;
00868 
00869       if (!handle) {
00870         pan(dx, dy);
00871         return;
00872       }
00873 
00874       switch (handle) {
00875       case 'n':
00876         newtop = top + dy;
00877         break;
00878       case 'ne':
00879         newright = right + dx;
00880         newtop = top + dy;
00881         break;
00882       case 'e':
00883         newright = right + dx;
00884         break;
00885       case 'se':
00886         newright = right + dx;
00887         newbottom = bottom + dy;
00888         break;
00889       case 's':
00890         newbottom = bottom + dy;
00891         break;
00892       case 'sw':
00893         newleft = left + dx;
00894         newbottom = bottom + dy;
00895         break;
00896       case 'w':
00897         newleft = left + dx;
00898         break;
00899       case 'nw':
00900         newleft = left + dx;
00901         newtop = top + dy;
00902         break;
00903       }
00904 
00905       // If there is an aspect ratio, make sure we maintain it.
00906       // Note that if there is an aspect ratio we won't display
00907       // the corner drag handles, so we don't have to handle those.
00908       if (aspectRatio) {
00909         var width, height;
00910         switch (handle) {
00911         case 'n':
00912         case 's':
00913           // change width to match the new height, keeping the center still
00914           height = newbottom - newtop;
00915           width = height * aspectRatio;
00916           newleft = Math.floor(centerX - Math.floor(width / 2));
00917           newright = Math.ceil(centerX + Math.ceil(width / 2));
00918           break;
00919         case 'e':
00920         case 'w':
00921           // Change height to match new width, keeping center still
00922           width = newright - newleft;
00923           height = width / aspectRatio;
00924           newtop = Math.floor(centerY - Math.floor(height / 2));
00925           newbottom = Math.ceil(centerY + Math.ceil(height / 2));
00926           break;
00927         }
00928       }
00929 
00930       // Now if the new region is out of bounds then bail out without
00931       // changing the region at all and ignore this move event
00932       if (newtop < 0 || newleft < 0 ||
00933           newright > dest.w || newbottom > dest.h)
00934         return;
00935 
00936       // Don't let the crop region become smaller than 100x100. If it does
00937       // then the sensitive regions of the crop handles start to intersect.
00938       // If there is a cropping aspect ratio, then the minimum size in
00939       // one dimension will be 100 and will be larger in the other.
00940       var minWidth = 100, minHeight = 100;
00941       if (aspectRatio) {
00942         if (aspectRatio > 1)
00943           minWidth = Math.round(minWidth * aspectRatio);
00944         else if (aspectRatio < 1)
00945           minHeight = Math.round(minHeight / aspectRatio);
00946       }
00947 
00948       // if the width is less than the minimum allowed (due to orientation
00949       // change), only allow the crop region to get bigger
00950       var newWidth = newright - newleft;
00951       if ((newWidth < (region.right - region.left)) && (newWidth < minWidth))
00952         return;
00953       var newHeight = newbottom - newtop;
00954       if ((newHeight < (region.bottom - region.top)) && (newHeight < minHeight))
00955         return;
00956 
00957       // Otherwise, all is well, so update the crop region and redraw
00958       region.left = newleft;
00959       region.right = newright;
00960       region.top = newtop;
00961       region.bottom = newbottom;
00962 
00963       self.drawCropControls(handle);
00964     }
00965 
00966     function pan(dx, dy) {
00967       if (dx > 0)
00968         dx = Math.min(dx, dest.w - right);
00969       if (dx < 0)
00970         dx = Math.max(dx, -left);
00971       if (dy > 0)
00972         dy = Math.min(dy, dest.h - bottom);
00973       if (dy < 0)
00974         dy = Math.max(dy, -top);
00975 
00976       region.left = left + dx;
00977       region.right = right + dx;
00978       region.top = top + dy;
00979       region.bottom = bottom + dy;
00980 
00981       self.drawCropControls();
00982     }
00983 
00984     function up(e) {
00985       window.removeEventListener('pan', move, true);
00986       window.removeEventListener('swipe', up, true);
00987       self.drawCropControls(); // erase drag handle highlight
00988 
00989       e.preventDefault();
00990     }
00991   }
00992 
00993 };

01142 //
01143 // Create a new ImageProcessor object for the specified canvas to do
01144 // webgl transformations on an image.  Expects its shader programs to be in
01145 // <script> elements with ids 'edit-vertex-shader' and 'edit-fragment-shader'.
01146 //
01147 function ImageProcessor(canvas) {
01148   // WebGL context for the canvas
01149   this.canvas = canvas;
01150   var gl = this.context = canvas.getContext('webgl') ||
01151     canvas.getContext('experimental-webgl');
01152 
01153   // Define our shader programs
01154   var vshader = this.vshader = gl.createShader(gl.VERTEX_SHADER);
01155   gl.shaderSource(vshader, ImageProcessor.vertexShader);
01156   gl.compileShader(vshader);
01157   if (!gl.getShaderParameter(vshader, gl.COMPILE_STATUS)) {
01158     var error = new Error('Error compiling vertex shader:' +
01159                           gl.getShaderInfoLog(vshader));
01160     gl.deleteShader(vshader);
01161     throw error;
01162   }
01163 
01164   var fshader = this.fshader = gl.createShader(gl.FRAGMENT_SHADER);
01165   gl.shaderSource(fshader, ImageProcessor.fragmentShader);
01166   gl.compileShader(fshader);
01167   if (!gl.getShaderParameter(fshader, gl.COMPILE_STATUS)) {
01168     var error = new Error('Error compiling fragment shader:' +
01169                           gl.getShaderInfoLog(fshader));
01170     gl.deleteShader(fshader);
01171     throw error;
01172   }
01173 
01174   var program = this.program = gl.createProgram();
01175   gl.attachShader(program, vshader);
01176   gl.attachShader(program, fshader);
01177   gl.linkProgram(program);
01178   if (!gl.getProgramParameter(program, gl.LINK_STATUS)) {
01179     var error = new Error('Error linking GLSL program:' +
01180                           gl.getProgramInfoLog(program));
01181     gl.deleteProgram(program);
01182     throw error;
01183   }
01184   gl.useProgram(program);
01185 
01186   // Create a texture to hold the source image once we have one
01187   this.sourceTexture = gl.createTexture();
01188   gl.bindTexture(gl.TEXTURE_2D, this.sourceTexture);
01189   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
01190   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
01191   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
01192   gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);

01200   this.srcPixelAddress = gl.getAttribLocation(program, 'src_pixel');
01201   this.destPixelAddress = gl.getAttribLocation(program, 'dest_pixel');
01202   this.canvasSizeAddress = gl.getUniformLocation(program, 'canvas_size');
01203   this.imageSizeAddress = gl.getUniformLocation(program, 'image_size');
01204   this.destSizeAddress = gl.getUniformLocation(program, 'dest_size');
01205   this.destOriginAddress = gl.getUniformLocation(program, 'dest_origin');
01206   this.matrixAddress = gl.getUniformLocation(program, 'matrix');
01207   this.gammaAddress = gl.getUniformLocation(program, 'gamma');

01224 ImageProcessor.prototype.draw = function(image,
01225                                          sx, sy, sw, sh,
01226                                          dx, dy, dw, dh,
01227                                          options)
01228 {
01229   var gl = this.context;
01230   gl.viewport(0, 0, gl.drawingBufferWidth, gl.drawingBufferHeight);
01231 
01232   // Set the canvas size and image size
01233   gl.uniform2f(this.canvasSizeAddress, this.canvas.width, this.canvas.height);
01234   gl.uniform2f(this.imageSizeAddress, image.width, image.height);
01235   gl.uniform2f(this.destOriginAddress, dx, dy);
01236   gl.uniform2f(this.destSizeAddress, dw, dh);
01237 
01238   // Set the gamma correction
01239   var gammaArray;
01240   if (options.gamma)
01241     gl.uniform4f(this.gammaAddress,
01242                  options.gamma, options.gamma, options.gamma, options.gamma);
01243   else
01244     gl.uniform4f(this.gammaAddress, 1, 1, 1, 1);
01245 
01246   // Set the color transformation
01247   gl.uniformMatrix4fv(this.matrixAddress, false,
01248                       options.matrix || ImageProcessor.IDENTITY_MATRIX);
01249 
01250   // Set border size and color
01251   if (options.borderWidth)
01252     gl.uniform1f(this.borderWidthAddress, Math.ceil(dw * options.borderWidth));
01253   else
01254     gl.uniform1f(this.borderWidthAddress, 0);

01256   gl.uniform4fv(this.borderColorAddress, options.borderColor || [0, 0, 0, 0]);
01257 
01258   // Define the source rectangle
