00183         // BufferState represents the different states in which a buffer slot
00184         // can be.
00185         enum BufferState {
00186             // FREE indicates that the buffer is not currently being used and
00187             // will not be used in the future until it gets dequeued and
00188             // subsequently queued by the client.
00189             FREE = 0,
00190 
00191             // DEQUEUED indicates that the buffer has been dequeued by the
00192             // client, but has not yet been queued or canceled. The buffer is
00193             // considered 'owned' by the client, and the server should not use
00194             // it for anything.
00195             //
00196             // Note that when in synchronous-mode (mSynchronousMode == true),
00197             // the buffer that's currently attached to the texture may be
00198             // dequeued by the client.  That means that the current buffer can
00199             // be in either the DEQUEUED or QUEUED state.  In asynchronous mode,
00200             // however, the current buffer is always in the QUEUED state.
00201             DEQUEUED = 1,
00202 
00203             // QUEUED indicates that the buffer has been queued by the client,
00204             // and has not since been made available for the client to dequeue.
00205             // Attaching the buffer to the texture does NOT transition the
00206             // buffer away from the QUEUED state. However, in Synchronous mode
00207             // the current buffer may be dequeued by the client under some
00208             // circumstances. See the note about the current buffer in the
00209             // documentation for DEQUEUED.
00210             QUEUED = 2,
00211 
00212             // RENDERING indicates that the buffer has been sent to
00213             // the compositor, and has not yet available for the
00214             // client to dequeue. When the compositor has finished its
00215             // job, the buffer will be returned to FREE state.
00216             RENDERING = 3,
00217         };

00324     // Unlock either returns the buffer to the native window or
00325     // destroys the buffer if the window is already released.
00326     virtual void Unlock() MOZ_OVERRIDE
00327     {
00328         if (mLocked) {
00329             // The window might have been destroyed. The buffer is no longer
00330             // valid at that point.
00331             sp<GonkNativeWindow> window = mNativeWindow.promote();
00332             if (window.get() && window->returnBuffer(mIndex, mGeneration)) {
00333                 mLocked = false;
00334             } else {
00335                 // If the window doesn't exist any more, release the buffer
00336                 // directly.
00337                 ImageBridgeChild *ibc = ImageBridgeChild::GetSingleton();
00338                 ibc->DeallocSurfaceDescriptorGralloc(mSurfaceDescriptor);
00339             }
00340         }
