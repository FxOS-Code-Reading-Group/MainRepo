00001 /*
00002  * This is an implementation of the OMA Download specification:
00003  * http://technical.openmobilealliance.org/Technical/release_program/docs/Download/V1_0-20040625-A/OMA-Download-OTA-V1_0-20040625-A.pdf
00004  *
00005  * It downloads and installs locked wallpaper, ringtones and songs
00006  * roughly following these steps:
00007  *
00008  * 1) Download a download descriptor from the specified URL
00009  *   - onerror: display a message to the user and abort the activity
00010  *
00011  * 2) Validate and parse the download descriptor
00012  *   - extract all the required and optional tags from the descriptor document
00013  *   - if it doesn't contain the tags we expect, display an error message
00014  *     and abort the activity.
00015  *   - if the version number is wrong, display an error, report the failure,
00016  *     and abort the activity.
00017  *
00018  * 3) Check types
00019  *   - If we can't handle any of the types specified in the descriptor,
00020  *     report a failure to the server, display an error to the user and abort.
00021  *   - For audio files, ask the user if they want to use them as a ringtone
00022  *     or as a song
00023  *
00024  * 4) Confirm download and use of media
00025  *   - Display name, vendor, type, size, etc. to the user, remind them
00026  *     that the content will be locked (and that they may be charged for
00027  *     it) and ask them to confirm that they want to download.
00028  *   - If the user cancels, report that to the server and abort the activity.
00029  *
00030  * 5) Download content
00031  *   - Display a progress meter and cancel button
00032  *   - If the user clicks cancel, report to server and abort the activity.
00033  *
00034  * 6) Extract the content, if necessary
00035  *   - If the content type is a DRM message, then it is in multi-part
00036  *     MIME format and we have to extract the first part, using the boundary
00037  *     string from the response headers.
00038  *   - If extracting the content fails, notify the server and user and abort
00039  *
00040  * 7) Allow the user to preview the downloaded files
00041  *   - if the file cannot be loaded or played, abort with an error
00042  *   - otherwise, show a small image, or play part of the audio
00043  *     to the user and allow them to click Cancel or Install.
00044  *
00045  * 8) Install the content
00046  *   - For wallpapers and ringtones, save to the database
00047  *   - For music, encrypt the file and save to device storage
00048  *   - If this fails, report errors and abort
00049  *   - Report success to the server
00050  *   - For ringtones and wallpaper, set them as the default
00051  */

00232     var types = descriptor.types;
00233 
00234     for (var i = 0; i < types.length; i++) {
00235       var type = types[i];
00236 
00237       if (type === 'application/vnd.oma.drm.message') {
00238         // This type just indicates that the download will be a multi-part
00239         // MIME message body rather than the content directly.
00240         // We support it.
00241         continue;
00242       }
00243 
00244       if (type in SupportedImageTypes) {
00245         mimeType = type;
00246         isImage = true;
00247         continue;
00248       }
00249 
00250       if (type in SupportedAudioTypes) {
00251         mimeType = type;
00252         isAudio = true;
00253         continue;

00257                   OMADownloadStatus.NON_ACCEPTABLE_CONTENT,

00396   // Step 6: extract the downloaded content from the array buffer
00397   function extractContent(type, buffer) {
00398     debug('Step 6a: extractContent', type, buffer.byteLength);
00399 
00400     if (type !== 'application/vnd.oma.drm.message') {
00401       // In this case, we already have the content and don't need to
00402       // extract anything. Move on to step 7
00403       previewContent(type, buffer);
00404       return;
00405     }
00406 
00407     // View the array buffer as if it was a binary string
00408     var dataString = new BinaryStringView(buffer);
00409 
00410     // If we're here, then dataString is a multipart message with a
00411     // boundary string at the start and end. There should be only one part
00412     // but we still must strip off the separators and extract the real
00413     // content type.  The type argument should specify the boundary string
00414     // but in practice it seems like it doesn't so we find it by examining
00415     // the end of the string.
00416 
00417     // Make sure there are no trailing newlines
00418     dataString = dataString.trim();

00420     function malformed(why) {
00421       if (why)
00422         debug('malformed drm message:', why);
00423       reportError(DOWNLOAD_ERROR, ERR_BAD_DRM_MESSAGE,
00424                   OMADownloadStatus.ATTRIBUTE_MISMATCH);
00425     }
00426 
00427     // The last two characters should be "--". If they are not, then
00428     // the content is malformed
00429     if (dataString.slice(-2).toString() !== '--')
00430       return malformed('missing trailing --');
00431 
00432     // Find the last newline.
00433     var end = dataString.lastIndexOf('\r\n');
00434     if (end === -1)
00435       return malformed('missing final \r\n');
00436 
00437     // The boundary string begins right after that newline
00438     var boundaryString = dataString.slice(end + 2, -2).toString() + '\r\n';
00439     var dataString = dataString.slice(0, end);
00440 
00441     debug('Boundary string is', boundaryString);
00442 
00443     // The first occurance of the boundary string is the start of
00444     // the content headers
00445     var headerStart =
00446       dataString.indexOf(boundaryString) + boundaryString.length;
00447     if (headerStart === -1)
00448       return malformed('missing boundary string at start');
00449 
00450     // If there are any more occurrances of the boundary string then
00451     // this is probably a .dm object for "combined delivery" and we
00452     // must reject it since we support only forward lock.
00453     if (dataString.indexOf(boundaryString, headerStart) !== -1)
00454       return malformed('more than one part');
00455 
00456     var contentStart = dataString.indexOf('\r\n\r\n', headerStart);
00457     if (contentStart === -1)
00458       return malformed('missing header separator');
00459 
00460     var headers = dataString.slice(headerStart,
00461                                    contentStart).toString().split('\r\n');
00462     var content = dataString.slice(contentStart + 4);
00463 
00464     debug('got first part headers:', JSON.stringify(headers));
00465 
00466     // find the content-type header for the content
00467     var contentType;
00468     for (var i = 0; i < headers.length; i++) {
00469       var header = headers[i].toLowerCase();
00470       if (header.startsWith('content-type:')) {
00471         contentType = header.substring(13).trim();
00472       }
00473     }
00474 
00475     if (!contentType)
00476       return malformed('no content-type header found in first part');
00477 
00478     debug('Found content in drm message file', contentType, content.length);
00479     previewContent(contentType, content.toArrayBuffer());
00480   }
