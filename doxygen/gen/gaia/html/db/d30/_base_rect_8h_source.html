00085   // Returns the rectangle containing the intersection of the points
00086   // (including edges) of *this and aRect. If there are no points in that
00087   // intersection, returns an empty rectangle with x/y set to the std::max of the x/y
00088   // of *this and aRect.
00089   Sub Intersect(const Sub& aRect) const
00090   {
00091     Sub result;
00092     result.x = std::max<T>(x, aRect.x);
00093     result.y = std::max<T>(y, aRect.y);
00094     result.width = std::min<T>(XMost(), aRect.XMost()) - result.x;
00095     result.height = std::min<T>(YMost(), aRect.YMost()) - result.y;
00096     if (result.width < 0 || result.height < 0) {
00097       result.SizeTo(0, 0);
00098     }
00099     return result;
00100   }

00289   // Round the rectangle edges to integer coordinates, such that the rounded
00290   // rectangle has the same set of pixel centers as the original rectangle.
00291   // Edges at offset 0.5 round up.
00292   // Suitable for most places where integral device coordinates
00293   // are needed, but note that any translation should be applied first to
00294   // avoid pixel rounding errors.
00295   // Note that this is *not* rounding to nearest integer if the values are negative.
00296   // They are always rounding as floor(n + 0.5).
00297   // See https://bugzilla.mozilla.org/show_bug.cgi?id=410748#c14
00298   // If you need similar method which is using NS_round(), you should create
00299   // new |RoundAwayFromZero()| method.
00300   void Round()
00301   {
00302     T x0 = static_cast<T>(floor(T(X()) + 0.5));
00303     T y0 = static_cast<T>(floor(T(Y()) + 0.5));
00304     T x1 = static_cast<T>(floor(T(XMost()) + 0.5));
00305     T y1 = static_cast<T>(floor(T(YMost()) + 0.5));
00306 
00307     x = x0;
00308     y = y0;
00309 
00310     width = x1 - x0;
00311     height = y1 - y0;
00312   }
