00221 inline TemporaryRef<TexturedEffect>
00222 CreateTexturedEffect(gfx::SurfaceFormat aFormat,
00223                      TextureSource* aSource,
00224                      const gfx::Filter& aFilter)
00225 {
00226   MOZ_ASSERT(aSource);
00227   RefPtr<TexturedEffect> result;
00228   switch (aFormat) {
00229   case gfx::FORMAT_B8G8R8A8:
00230     result = new EffectBGRA(aSource, true, aFilter);
00231     break;
00232   case gfx::FORMAT_B8G8R8X8:
00233     result = new EffectBGRX(aSource, true, aFilter);
00234     break;
00235   case gfx::FORMAT_R8G8B8X8:
00236     result = new EffectRGBX(aSource, true, aFilter);
00237     break;
00238   case gfx::FORMAT_R5G6B5:
00239     result = new EffectRGBX(aSource, true, aFilter);
00240     break;
00241   case gfx::FORMAT_R8G8B8A8:
00242     result = new EffectRGBA(aSource, true, aFilter);
00243     break;
00244   case gfx::FORMAT_YUV:
00245     result = new EffectYCbCr(aSource, aFilter);
00246     break;
00247   default:
00248     MOZ_CRASH("unhandled program type");
00249   }
00250 
00251   return result;
00252 }

00260 inline TemporaryRef<TexturedEffect>
00261 CreateTexturedEffect(TextureSource* aSource,
00262                      TextureSource* aSourceOnWhite,
00263                      const gfx::Filter& aFilter)
00264 {
00265   MOZ_ASSERT(aSource);
00266   if (aSourceOnWhite) {
00267     MOZ_ASSERT(aSource->GetFormat() == gfx::FORMAT_R8G8B8X8 ||
00268                aSourceOnWhite->GetFormat() == gfx::FORMAT_B8G8R8X8);
00269     return new EffectComponentAlpha(aSource, aSourceOnWhite, aFilter);
00270   }
00271 
00272   return CreateTexturedEffect(aSource->GetFormat(), aSource, aFilter);
00273 }
