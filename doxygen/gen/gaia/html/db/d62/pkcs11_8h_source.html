00020 /* Before including this file (pkcs11.h) (or pkcs11t.h by
00021  * itself), 6 platform-specific macros must be defined.  These
00022  * macros are described below, and typical definitions for them
00023  * are also given.  Be advised that these definitions can depend
00024  * on both the platform and the compiler used (and possibly also
00025  * on whether a PKCS #11 library is linked statically or
00026  * dynamically).
00027  *
00028  * In addition to defining these 6 macros, the packing convention
00029  * for PKCS #11 structures should be set.  The PKCS #11
00030  * convention on packing is that structures should be 1-byte
00031  * aligned.
00032  *
00033  * In a Win32 environment, this might be done by using the
00034  * following preprocessor directive before including pkcs11.h
00035  * or pkcs11t.h:
00036  *
00037  * #pragma pack(push, cryptoki, 1)
00038  *
00039  * and using the following preprocessor directive after including
00040  * pkcs11.h or pkcs11t.h:
00041  *
00042  * #pragma pack(pop, cryptoki)
00043  *
00044  * In a UNIX environment, you're on your own here.  You might
00045  * not need to do anything.
00046  *
00047  *
00048  * Now for the macros:
00049  *
00050  *
00051  * 1. CK_PTR: The indirection string for making a pointer to an
00052  * object.  It can be used like this:
00053  *
00054  * typedef CK_BYTE CK_PTR CK_BYTE_PTR;
00055  *
00056  * In a Win32 environment, it might be defined by
00057  *
00058  * #define CK_PTR *
00059  *
00060  * In a UNIX environment, it might be defined by
00061  *
00062  * #define CK_PTR *
00063  *
00064  *
00065  * 2. CK_DEFINE_FUNCTION(returnType, name): A macro which makes
00066  * an exportable PKCS #11 library function definition out of a
00067  * return type and a function name.  It should be used in the
00068  * following fashion to define the exposed PKCS #11 functions in
00069  * a PKCS #11 library:
00070  *
00071  * CK_DEFINE_FUNCTION(CK_RV, C_Initialize)(
00072  *   CK_VOID_PTR pReserved
00073  * )
00074  * {
00075  *   ...
00076  * }
00077  *
00078  * For defining a function in a Win32 PKCS #11 .dll, it might be
00079  * defined by
00080  *
00081  * #define CK_DEFINE_FUNCTION(returnType, name) \
00082  *   returnType __declspec(dllexport) name
00083  *
00084  * In a UNIX environment, it might be defined by
00085  *
00086  * #define CK_DEFINE_FUNCTION(returnType, name) \
00087  *   returnType name
00088  *
00089  *
00090  * 3. CK_DECLARE_FUNCTION(returnType, name): A macro which makes
00091  * an importable PKCS #11 library function declaration out of a
00092  * return type and a function name.  It should be used in the
00093  * following fashion:
00094  *
00095  * extern CK_DECLARE_FUNCTION(CK_RV, C_Initialize)(
00096  *   CK_VOID_PTR pReserved
00097  * );
00098  *
00099  * For declaring a function in a Win32 PKCS #11 .dll, it might
00100  * be defined by
00101  *
00102  * #define CK_DECLARE_FUNCTION(returnType, name) \
00103  *   returnType __declspec(dllimport) name
00104  *
00105  * In a UNIX environment, it might be defined by
00106  *
00107  * #define CK_DECLARE_FUNCTION(returnType, name) \
00108  *   returnType name
00109  *
00110  *
00111  * 4. CK_DECLARE_FUNCTION_POINTER(returnType, name): A macro
00112  * which makes a PKCS #11 API function pointer declaration or
00113  * function pointer type declaration out of a return type and a
00114  * function name.  It should be used in the following fashion:
00115  *
00116  * // Define funcPtr to be a pointer to a PKCS #11 API function
00117  * // taking arguments args and returning CK_RV.
00118  * CK_DECLARE_FUNCTION_POINTER(CK_RV, funcPtr)(args);
00119  *
00120  * or
00121  *
00122  * // Define funcPtrType to be the type of a pointer to a
00123  * // PKCS #11 API function taking arguments args and returning
00124  * // CK_RV, and then define funcPtr to be a variable of type
00125  * // funcPtrType.
00126  * typedef CK_DECLARE_FUNCTION_POINTER(CK_RV, funcPtrType)(args);
00127  * funcPtrType funcPtr;
00128  *
00129  * For accessing functions in a Win32 PKCS #11 .dll, in might be
00130  * defined by
00131  *
00132  * #define CK_DECLARE_FUNCTION_POINTER(returnType, name) \
00133  *   returnType __declspec(dllimport) (* name)
00134  *
00135  * In a UNIX environment, it might be defined by
00136  *
00137  * #define CK_DECLARE_FUNCTION_POINTER(returnType, name) \
00138  *   returnType (* name)
00139  *
00140  *
00141  * 5. CK_CALLBACK_FUNCTION(returnType, name): A macro which makes
00142  * a function pointer type for an application callback out of
00143  * a return type for the callback and a name for the callback.
00144  * It should be used in the following fashion:
00145  *
00146  * CK_CALLBACK_FUNCTION(CK_RV, myCallback)(args);
00147  *
00148  * to declare a function pointer, myCallback, to a callback
00149  * which takes arguments args and returns a CK_RV.  It can also
00150  * be used like this:
00151  *
00152  * typedef CK_CALLBACK_FUNCTION(CK_RV, myCallbackType)(args);
00153  * myCallbackType myCallback;
00154  *
00155  * In a Win32 environment, it might be defined by
00156  *
00157  * #define CK_CALLBACK_FUNCTION(returnType, name) \
00158  *   returnType (* name)
00159  *
00160  * In a UNIX environment, it might be defined by
00161  *
00162  * #define CK_CALLBACK_FUNCTION(returnType, name) \
00163  *   returnType (* name)
00164  *
00165  *
00166  * 6. NULL_PTR: This macro is the value of a NULL pointer.
00167  *
00168  * In any ANSI/ISO C environment (and in many others as well),
00169  * this should be defined by
00170  *
00171  * #ifndef NULL_PTR
00172  * #define NULL_PTR 0
00173  * #endif
00174  */
