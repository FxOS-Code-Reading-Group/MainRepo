00085 inline bool SkUnichar_IsVariationSelector(SkUnichar uni) {
00086 /*  The 'true' ranges are:
00087  *      0x180B  <= uni <=  0x180D
00088  *      0xFE00  <= uni <=  0xFE0F
00089  *      0xE0100 <= uni <= 0xE01EF
00090  */
00091     if (uni < 0x180B || uni > 0xE01EF) {
00092         return false;
00093     }
00094     if ((uni > 0x180D && uni < 0xFE00) || (uni > 0xFE0F && uni < 0xE0100)) {
00095         return false;
00096     }
00097     return true;
