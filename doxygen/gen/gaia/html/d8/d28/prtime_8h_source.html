00033 /*
00034  * PRTime --
00035  *
00036  *     NSPR represents basic time as 64-bit signed integers relative
00037  *     to midnight (00:00:00), January 1, 1970 Greenwich Mean Time (GMT).
00038  *     (GMT is also known as Coordinated Universal Time, UTC.)
00039  *     The units of time are in microseconds. Negative times are allowed
00040  *     to represent times prior to the January 1970 epoch. Such values are
00041  *     intended to be exported to other systems or converted to human
00042  *     readable form.
00043  *
00044  *     Notes on porting: PRTime corresponds to time_t in ANSI C.  NSPR 1.0
00045  *     simply uses PRInt64.
00046  */
00047 
00048 typedef PRInt64 PRTime;

00060 /*
00061  * PRExplodedTime --
00062  *
00063  *     Time broken down into human-readable components such as year, month,
00064  *     day, hour, minute, second, and microsecond.  Time zone and daylight
00065  *     saving time corrections may be applied.  If they are applied, the
00066  *     offsets from the GMT must be saved in the 'tm_params' field so that
00067  *     all the information is available to reconstruct GMT.
00068  *
00069  *     Notes on porting: PRExplodedTime corrresponds to struct tm in
00070  *     ANSI C, with the following differences:
00071  *       - an additional field tm_usec;
00072  *       - replacing tm_isdst by tm_params;
00073  *       - the month field is spelled tm_month, not tm_mon;
00074  *       - we use absolute year, AD, not the year since 1900.
00075  *     The corresponding type in NSPR 1.0 is called PRTime.  Below is
00076  *     a table of date/time type correspondence in the three APIs:
00077  *         API          time since epoch          time in components
00078  *       ANSI C             time_t                  struct tm
00079  *       NSPR 1.0           PRInt64                   PRTime
00080  *       NSPR 2.0           PRTime                  PRExplodedTime
00081  */
00082 
00083 typedef struct PRExplodedTime {
00084     PRInt32 tm_usec;            /* microseconds past tm_sec (0-99999)  */
00085     PRInt32 tm_sec;             /* seconds past tm_min (0-61, accomodating
00086                                    up to two leap seconds) */
00087     PRInt32 tm_min;             /* minutes past tm_hour (0-59) */
00088     PRInt32 tm_hour;            /* hours past tm_day (0-23) */
00089     PRInt32 tm_mday;            /* days past tm_mon (1-31, note that it
00090                                 starts from 1) */
00091     PRInt32 tm_month;           /* months past tm_year (0-11, Jan = 0) */
00092     PRInt16 tm_year;            /* absolute year, AD (note that we do not
00093                                 count from 1900) */
00094 
00095     PRInt8 tm_wday;             /* calculated day of the week
00096                                 (0-6, Sun = 0) */
00097     PRInt16 tm_yday;            /* calculated day of the year
00098                                 (0-365, Jan 1 = 0) */
00099 
00100     PRTimeParameters tm_params;  /* time parameters used by conversion */
00101 } PRExplodedTime;

00103 /*
00104  * PRTimeParamFn --
00105  *
00106  *     A function of PRTimeParamFn type returns the time zone and
00107  *     daylight saving time corrections for some geographic location,
00108  *     given the current time in GMT.  The input argument gmt should
00109  *     point to a PRExplodedTime that is in GMT, i.e., whose
00110  *     tm_params contains all 0's.
00111  *
00112  *     For any time zone other than GMT, the computation is intended to
00113  *     consist of two steps:
00114  *       - Figure out the time zone correction, tp_gmt_offset.  This number
00115  *         usually depends on the geographic location only.  But it may
00116  *         also depend on the current time.  For example, all of China
00117  *         is one time zone right now.  But this situation may change
00118  *         in the future.
00119  *       - Figure out the daylight saving time correction, tp_dst_offset.
00120  *         This number depends on both the geographic location and the
00121  *         current time.  Most of the DST rules are expressed in local
00122  *         current time.  If so, one should apply the time zone correction
00123  *         to GMT before applying the DST rules.
00124  */
00125 
00126 typedef PRTimeParameters (PR_CALLBACK *PRTimeParamFn)(const PRExplodedTime *gmt);

00132 /*
00133  * The PR_Now routine returns the current time relative to the
00134  * epoch, midnight, January 1, 1970 UTC. The units of the returned
00135  * value are microseconds since the epoch.
00136  *
00137  * The values returned are not guaranteed to advance in a linear fashion
00138  * due to the application of time correction protocols which synchronize
00139  * computer clocks to some external time source. Consequently it should
00140  * not be depended on for interval timing.
00141  *
00142  * The implementation is machine dependent.
00143  * Cf. time_t time(time_t *tp) in ANSI C.
00144  */

00196 /*
00197  * This parses a time/date string into a PRExplodedTime
00198  * struct. It populates all fields but it can't split
00199  * the offset from UTC into tp_gmt_offset and tp_dst_offset in
00200  * most cases (exceptions: PST/PDT, MST/MDT, CST/CDT, EST/EDT, GMT/BST).
00201  * In those cases tp_gmt_offset will be the sum of these two and
00202  * tp_dst_offset will be 0.
00203  * It returns PR_SUCCESS on success, and PR_FAILURE
00204  * if the time/date string can't be parsed.
00205  *
00206  * Many formats are handled, including:
00207  *
00208  *   14 Apr 89 03:20:12
00209  *   14 Apr 89 03:20 GMT
00210  *   Fri, 17 Mar 89 4:01:33
00211  *   Fri, 17 Mar 89 4:01 GMT
00212  *   Mon Jan 16 16:12 PDT 1989
00213  *   Mon Jan 16 16:12 +0130 1989
00214  *   6 May 1992 16:41-JST (Wednesday)
00215  *   22-AUG-1993 10:59:12.82
00216  *   22-AUG-1993 10:59pm
00217  *   22-AUG-1993 12:59am
00218  *   22-AUG-1993 12:59 PM
00219  *   Friday, August 04, 1995 3:54 PM
00220  *   06/21/95 04:24:34 PM
00221  *   20/06/95 21:07
00222  *   95-06-08 19:32:48 EDT
00223  *
00224  * If the input string doesn't contain a description of the timezone,
00225  * we consult the `default_to_gmt' to decide whether the string should
00226  * be interpreted relative to the local time zone (PR_FALSE) or GMT (PR_TRUE).
00227  * The correct value for this argument depends on what standard specified
00228  * the time string which you are parsing.
00229  */
00230 
00231 NSPR_API(PRStatus) PR_ParseTimeStringToExplodedTime (
00232         const char *string,
00233         PRBool default_to_gmt,
00234         PRExplodedTime *result);
