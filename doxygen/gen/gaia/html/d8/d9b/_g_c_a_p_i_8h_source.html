00261 /*
00262  * This should be called when an object that is marked gray is exposed to the JS
00263  * engine (by handing it to running JS code or writing it into live JS
00264  * data). During incremental GC, since the gray bits haven't been computed yet,
00265  * we conservatively mark the object black.
00266  */
00267 static JS_ALWAYS_INLINE void
00268 ExposeGCThingToActiveJS(void *thing, JSGCTraceKind kind)
00269 {
00270     JS_ASSERT(kind != JSTRACE_SHAPE);
00271 
00272     shadow::Runtime *rt = js::gc::GetGCThingRuntime(thing);
00273 #ifdef JSGC_GENERATIONAL
00274     /*
00275      * GC things residing in the nursery cannot be gray: they have no mark bits.
00276      * All live objects in the nursery are moved to tenured at the beginning of
00277      * each GC slice, so the gray marker never sees nursery things.
00278      */
00279     if (uintptr_t(thing) >= rt->gcNurseryStart_ && uintptr_t(thing) < rt->gcNurseryEnd_)
00280         return;
00281 #endif
00282     if (IsIncrementalBarrierNeededOnGCThing(rt, thing, kind))
00283         IncrementalReferenceBarrier(thing, kind);
00284     else if (GCThingIsMarkedGray(thing))
00285         UnmarkGrayGCThingRecursively(thing, kind);
00286 }
