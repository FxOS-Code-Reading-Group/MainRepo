00031 /* Generate PQGParams and PQGVerify structs.
00032  * Length of P specified by L.
00033  *   if L is greater than 1024 then the resulting verify parameters will be
00034  *   DSA2.
00035  * Length of Q specified by N. If zero, The PKCS #11 module will
00036  *   pick an appropriately sized Q for L. If N is specified and L = 1024, then
00037  *   the resulting verify parameters will be DSA2, Otherwise DSA1 parameters 
00038  *   will be returned.
00039  * Length of SEED in bytes specified in seedBytes.
00040  *
00041  * The underlying PKCS #11 module will check the values for L, N, 
00042  * and seedBytes. The rules for softoken are:
00043  * 
00044  * If L <= 1024, then L must be between 512 and 1024 in increments of 64 bits.
00045  * If L <= 1024, then N must be 0 or 160.
00046  * If L >= 1024, then L and N must match the following table:
00047  *   L=1024   N=0 or 160
00048  *   L=2048   N=0 or 224
00049  *   L=2048   N=256
00050  *   L=3072   N=0 or 256
00051  * if L <= 1024
00052  *   seedBbytes must be in the range [20..256].
00053  * if L >= 1024
00054  *   seedBbytes must be in the range [20..L/16].
00055  */
00056 extern SECStatus
00057 PK11_PQG_ParamGenV2(unsigned int L, unsigned int N, unsigned int seedBytes,
00058             PQGParams **pParams, PQGVerify **pVfy);
