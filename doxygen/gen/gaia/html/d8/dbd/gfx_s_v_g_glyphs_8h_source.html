00214                                                 const gfxMatrix& aCTM)
00215     {
00216         if (!aPattern) {
00217             return gfxMatrix();
00218         }
00219         gfxMatrix deviceToUser = aCTM;
00220         deviceToUser.Invert();
00221         return deviceToUser * aPattern->GetMatrix();
00222     }

00224     SimpleTextObjectPaint(gfxPattern *aFillPattern, gfxPattern *aStrokePattern,
00225                           const gfxMatrix& aCTM) :
00226         mFillPattern(aFillPattern ? aFillPattern : new gfxPattern(sZero)),
00227         mStrokePattern(aStrokePattern ? aStrokePattern : new gfxPattern(sZero))
00228     {
00229         mFillMatrix = SetupDeviceToPatternMatrix(aFillPattern, aCTM);
00230         mStrokeMatrix = SetupDeviceToPatternMatrix(aStrokePattern, aCTM);
00231     }
00232 
00233     already_AddRefed<gfxPattern> GetFillPattern(float aOpacity,
00234                                                 const gfxMatrix& aCTM) {

00236             mFillPattern->SetMatrix(aCTM * mFillMatrix);
00237         }
00238         nsRefPtr<gfxPattern> fillPattern = mFillPattern;
