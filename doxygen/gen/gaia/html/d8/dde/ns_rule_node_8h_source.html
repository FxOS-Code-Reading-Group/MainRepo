00134   void* NS_FASTCALL GetStyleData(const nsStyleStructID aSID) {
00135     if (IsReset(aSID)) {
00136       if (mResetData) {
00137         return mResetData->mStyleStructs[aSID];
00138       }
00139     } else {
00140       if (mInheritedData) {

00148                                 nsPresContext *aPresContext, void *aData) {
00149     if (IsReset(aSID)) {
00150       if (!mResetData) {
00151         mResetData = new (aPresContext) nsResetStyleData;
00152       }
00153       mResetData->mStyleStructs[aSID] = aData;
00154     } else {
00155       if (!mInheritedData) {
00156         mInheritedData = new (aPresContext) nsInheritedStyleData;
00157       }
00158       mInheritedData->mStyleStructs[aSID] = aData;
00159     }

00371   uint32_t mNoneBits; // Used to cache the fact that the branch to this
00372                       // node specifies no non-inherited data for a
00373                       // given struct type.  (This usually implies that
00374                       // the entire branch specifies no non-inherited
00375                       // data, although not necessarily, if a
00376                       // non-inherited value is overridden by an
00377                       // explicit 'inherit' value.)  For example, if an
00378                       // entire rule branch specifies no color
00379                       // information, then a bit will be set along every
00380                       // rule node on that branch, so that you can break
00381                       // out of the rule tree early and just inherit
00382                       // from the parent style context.  The presence of
00383                       // this bit means we should just get inherited
00384                       // data from the parent style context, and it is
00385                       // never used for reset structs since their
00386                       // Compute*Data functions don't initialize from
00387                       // inherited data.

00389   // Reference count.  This just counts the style contexts that reference this
00390   // rulenode.  And children the rulenode has had.  When this goes to 0 or
00391   // stops being 0, we notify the style set.
00392   // Note, in particular, that when a child is removed mRefCnt is NOT
00393   // decremented.  This is on purpose; the notifications to the style set are
00394   // only used to determine when it's worth running GC on the ruletree, and
00395   // this setup makes it so we only count unused ruletree leaves for purposes
00396   // of deciding when to GC.  We could more accurately count unused rulenodes
00397   // by releasing/addrefing our parent when our refcount transitions to or from
00398   // 0, but it doesn't seem worth it to do that.
00399   uint32_t mRefCnt;
