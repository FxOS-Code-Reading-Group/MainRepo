00228       // token expressions
00229       var reBlank = /^\s*|\s*$/;
00230       var reComment = /^\s*#|^\s*$/;
00231       var reSection = /^\s*\[(.*)\]\s*$/;
00232       var reImport = /^\s*@import\s+url\((.*)\)\s*$/i;
00233       var reSplit = /^([^=\s]*)\s*=\s*(.+)$/;

00237       // parse the *.properties file into an associative array
00238       function parseRawLines(rawText, extendedSyntax) {
00239         var entries = rawText.replace(reBlank, '').split(/[\r\n]+/);
00240         var currentLang = '*';
00241         var genericLang = lang.replace(/-[a-z]+$/i, '');
00242         var skipLang = false;
00243         var match = '';
00244 
00245         for (var i = 0; i < entries.length; i++) {
00246           var line = entries[i];
00247 
00248           // comment or blank line?
00249           if (reComment.test(line)) {
00250             continue;
00251           }
00252 
00253           // multi-line?
00254           while (reMultiline.test(line) && i < entries.length) {
00255             line = line.slice(0, line.length - 1) +
00256               entries[++i].replace(reBlank, '');
00257           }
00258 
00259           // the extended syntax supports [lang] sections and @import rules
00260           if (extendedSyntax) {
00261             if (reSection.test(line)) { // section start?
00262               match = reSection.exec(line);
00263               currentLang = match[1];
00264               skipLang = (currentLang !== '*') &&
00265                   (currentLang !== lang) && (currentLang !== genericLang);
00266               continue;
00267             } else if (skipLang) {
00268               continue;
00269             }
00270             if (reImport.test(line)) { // @import rule?
00271               match = reImport.exec(line);
00272               loadImport(baseURL + match[1]); // load the resource synchronously
00273             }
00274           }
00275 
00276           // key-value pair
00277           var tmp = line.match(reSplit);
00278           if (tmp && tmp.length == 3) {
00279             // unescape unicode char codes if needed (e.g. '\u00a0')
00280             var val = tmp[2].replace(reUnicode, function(match, token) {
00281               return unescape('%u' + '0000'.slice(token.length) + token);
00282             });
00283             dictionary[tmp[1]] = evalString(val);
00284           }
00285         }
00286       }
