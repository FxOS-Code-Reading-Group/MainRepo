00282 static inline SkPMColor SkFastFourByteInterp(SkPMColor src,
00283                                              SkPMColor dst,
00284                                              U8CPU srcWeight) {
00285     SkASSERT(srcWeight < 256);
00286 
00287     // Reorders ARGB to AG-RB in order to reduce the number of operations.
00288     const uint32_t mask = 0xFF00FF;
00289     uint32_t src_rb = src & mask;
00290     uint32_t src_ag = (src >> 8) & mask;
00291     uint32_t dst_rb = dst & mask;
00292     uint32_t dst_ag = (dst >> 8) & mask;
00293 
00294     // scale = srcWeight + (srcWeight >> 7) is more accurate than
00295     // scale = srcWeight + 1, but 7% slower
00296     int scale = srcWeight + (srcWeight >> 7);
00297 
00298     uint32_t ret_rb = src_rb * scale + (256 - scale) * dst_rb;
00299     uint32_t ret_ag = src_ag * scale + (256 - scale) * dst_ag;
00300 
00301     return (ret_ag & ~mask) | ((ret_rb & ~mask) >> 8);
00302 }

00414 /*  Return c in expanded_rgb_16 format, but also scaled up by 32 (5 bits)
00415     It is now suitable for combining with a scaled expanded_rgb_16 color
00416     as in SkSrcOver32To16().
00417     We must do this 565 high-bit replication, in order for the subsequent add
00418     to saturate properly (and not overflow). If we take the 8 bits as is, it is
00419     possible to overflow.
00420 */
00421 static inline uint32_t SkPMColorToExpanded16x5(SkPMColor c) {
00422     unsigned sr = SkPacked32ToR16(c);
00423     unsigned sg = SkPacked32ToG16(c);
00424     unsigned sb = SkPacked32ToB16(c);
00425 
00426     sr = (sr << 5) | sr;
00427     sg = (sg << 5) | (sg >> 1);
00428     sb = (sb << 5) | sb;
00429     return (sr << 11) | (sg << 21) | (sb << 0);
00430 }

00695 static inline SkPMColor16 SkDitherARGB32To4444(U8CPU a, U8CPU r,
00696                                                U8CPU g, U8CPU b) {
00697     // to ensure that we stay a legal premultiplied color, we take the max()
00698     // of the truncated and dithered alpha values. If we didn't, cases like
00699     // SkDitherARGB32To4444(0x31, 0x2E, ...) would generate SkPackARGB4444(2, 3, ...)
00700     // which is not legal premultiplied, since a < color
00701     unsigned dithered_a = ((a << 1) - ((a >> 4 << 4) | (a >> 4))) >> 4;
00702     a = SkMax32(a >> 4, dithered_a);
00703     // these we just dither in place
00704     r = ((r << 1) - ((r >> 4 << 4) | (r >> 4))) >> 4;
00705     g = ((g << 1) - ((g >> 4 << 4) | (g >> 4))) >> 4;
00706     b = ((b << 1) - ((b >> 4 << 4) | (b >> 4))) >> 4;
00707 
00708     return SkPackARGB4444(a, r, g, b);
00709 }
