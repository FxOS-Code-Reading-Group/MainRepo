00049 /*
00050  * parameters used to initialize softoken. Mostly strings used to 
00051  * internationalize softoken. Memory for the strings are owned by the caller,
00052  * who is free to free them once NSS_ContextInit returns. If the string 
00053  * parameter is NULL (as opposed to empty, zero length), then the softoken
00054  * default is used. These are equivalent to the parameters for 
00055  * PK11_ConfigurePKCS11().
00056  *
00057  * field names match their equivalent parameter names for softoken strings 
00058  * documented at https://developer.mozilla.org/en/PKCS11_Module_Specs.
00059  * 
00060  * minPWLen 
00061  *     Minimum password length in bytes. 
00062  * manufacturerID 
00063  *     Override the default manufactureID value for the module returned in 
00064  *     the CK_INFO, CK_SLOT_INFO, and CK_TOKEN_INFO structures with an 
00065  *     internationalize string (UTF8). This value will be truncated at 32 
00066  *     bytes (not including the trailing NULL, partial UTF8 characters will be
00067  *     dropped). 
00068  * libraryDescription 
00069  *     Override the default libraryDescription value for the module returned in
00070  *     the CK_INFO structure with an internationalize string (UTF8). This value
00071  *     will be truncated at 32 bytes(not including the trailing NULL, partial 
00072  *     UTF8 characters will be dropped). 
00073  * cryptoTokenDescription 
00074  *     Override the default label value for the internal crypto token returned
00075  *     in the CK_TOKEN_INFO structure with an internationalize string (UTF8).
00076  *     This value will be truncated at 32 bytes (not including the trailing
00077  *     NULL, partial UTF8 characters will be dropped). 
00078  * dbTokenDescription 
00079  *     Override the default label value for the internal DB token returned in 
00080  *     the CK_TOKEN_INFO structure with an internationalize string (UTF8). This
00081  *     value will be truncated at 32 bytes (not including the trailing NULL,
00082  *     partial UTF8 characters will be dropped). 
00083  * FIPSTokenDescription 
00084  *     Override the default label value for the internal FIPS token returned in
00085  *     the CK_TOKEN_INFO structure with an internationalize string (UTF8). This
00086  *     value will be truncated at 32 bytes (not including the trailing NULL,
00087  *     partial UTF8 characters will be dropped). 
00088  * cryptoSlotDescription 
00089  *     Override the default slotDescription value for the internal crypto token
00090  *     returned in the CK_SLOT_INFO structure with an internationalize string
00091  *     (UTF8). This value will be truncated at 64 bytes (not including the
00092  *     trailing NULL, partial UTF8 characters will be dropped). 
00093  * dbSlotDescription 
00094  *     Override the default slotDescription value for the internal DB token 
00095  *     returned in the CK_SLOT_INFO structure with an internationalize string 
00096  *     (UTF8). This value will be truncated at 64 bytes (not including the
00097  *     trailing NULL, partial UTF8 characters will be dropped). 
00098  * FIPSSlotDescription 
00099  *     Override the default slotDecription value for the internal FIPS token
00100  *     returned in the CK_SLOT_INFO structure with an internationalize string
00101  *     (UTF8). This value will be truncated at 64 bytes (not including the
00102  *     trailing NULL, partial UTF8 characters will be dropped). 
00103  *
00104  */

00160 /*
00161  * Open the Cert, Key, and Security Module databases, read/write.
00162  * Initialize the Random Number Generator.
00163  * Does not initialize the cipher policies or enables.
00164  * Default policy settings disallow all ciphers.
00165  *
00166  * This allows using application defined prefixes for the cert and key db's
00167  * and an alternate name for the secmod database. NOTE: In future releases,
00168  * the database prefixes my not necessarily map to database names.
00169  *
00170  * configdir - base directory where all the cert, key, and module datbases live.
00171  * certPrefix - prefix added to the beginning of the cert database example: "
00172  *          "https-server1-"
00173  * keyPrefix - prefix added to the beginning of the key database example: "
00174  *          "https-server1-"
00175  * secmodName - name of the security module database (usually "secmod.db").
00176  * flags - change the open options of NSS_Initialize as follows:
00177  *  NSS_INIT_READONLY - Open the databases read only.
00178  *  NSS_INIT_NOCERTDB - Don't open the cert DB and key DB's, just 
00179  *          initialize the volatile certdb.
00180  *  NSS_INIT_NOMODDB  - Don't open the security module DB, just 
00181  *          initialize the  PKCS #11 module.
00182  *      NSS_INIT_FORCEOPEN - Continue to force initializations even if the 
00183  *          databases cannot be opened.
00184  *      NSS_INIT_NOROOTINIT - Don't try to look for the root certs module
00185  *          automatically.
00186  *      NSS_INIT_OPTIMIZESPACE - Use smaller tables and caches.
00187  *      NSS_INIT_PK11THREADSAFE - only load PKCS#11 modules that are
00188  *                      thread-safe, ie. that support locking - either OS
00189  *                      locking or NSS-provided locks . If a PKCS#11
00190  *                      module isn't thread-safe, don't serialize its
00191  *                      calls; just don't load it instead. This is necessary
00192  *                      if another piece of code is using the same PKCS#11
00193  *                      modules that NSS is accessing without going through
00194  *                      NSS, for example the Java SunPKCS11 provider.
00195  *      NSS_INIT_PK11RELOAD - ignore the CKR_CRYPTOKI_ALREADY_INITIALIZED
00196  *                      error when loading PKCS#11 modules. This is necessary
00197  *                      if another piece of code is using the same PKCS#11
00198  *                      modules that NSS is accessing without going through
00199  *                      NSS, for example Java SunPKCS11 provider.
00200  *      NSS_INIT_NOPK11FINALIZE - never call C_Finalize on any
00201  *                      PKCS#11 module. This may be necessary in order to
00202  *                      ensure continuous operation and proper shutdown
00203  *                      sequence if another piece of code is using the same
00204  *                      PKCS#11 modules that NSS is accessing without going
00205  *                      through NSS, for example Java SunPKCS11 provider.
00206  *                      The following limitation applies when this is set :
00207  *                      SECMOD_WaitForAnyTokenEvent will not use
00208  *                      C_WaitForSlotEvent, in order to prevent the need for
00209  *                      C_Finalize. This call will be emulated instead.
00210  *      NSS_INIT_RESERVED - Currently has no effect, but may be used in the
00211  *                      future to trigger better cooperation between PKCS#11
00212  *                      modules used by both NSS and the Java SunPKCS11
00213  *                      provider. This should occur after a new flag is defined
00214  *                      for C_Initialize by the PKCS#11 working group.
00215  *      NSS_INIT_COOPERATE - Sets 4 recommended options for applications that
00216  *                      use both NSS and the Java SunPKCS11 provider.
00217  *
00218  * Also NOTE: This is not the recommended method for initializing NSS. 
00219  * The preferred method is NSS_init().
00220  */

00222 #define NSS_INIT_NOCERTDB   0x2
00223 #define NSS_INIT_NOMODDB    0x4
