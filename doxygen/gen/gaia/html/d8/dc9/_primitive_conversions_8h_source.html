00100 struct PrimitiveConversionTraits_smallInt {
00101   // The output of JS::ToInt32 is determined as follows:
00102   //   1) The value is converted to a double
00103   //   2) Anything that's not a finite double returns 0
00104   //   3) The double is rounded towards zero to the nearest integer
00105   //   4) The resulting integer is reduced mod 2^32.  The output of this
00106   //      operation is an integer in the range [0, 2^32).
00107   //   5) If the resulting number is >= 2^31, 2^32 is subtracted from it.
00108   //
00109   // The result of all this is a number in the range [-2^31, 2^31)
00110   //
00111   // WebIDL conversions for the 8-bit, 16-bit, and 32-bit integer types
00112   // are defined in the same way, except that step 4 uses reduction mod
00113   // 2^8 and 2^16 for the 8-bit and 16-bit types respectively, and step 5
00114   // is only done for the signed types.
00115   //
00116   // C/C++ define integer conversion semantics to unsigned types as taking
00117   // your input integer mod (1 + largest value representable in the
00118   // unsigned type).  Since 2^32 is zero mod 2^8, 2^16, and 2^32,
00119   // converting to the unsigned int of the relevant width will correctly
00120   // perform step 4; in particular, the 2^32 possibly subtracted in step 5
00121   // will become 0.
00122   //
00123   // Once we have step 4 done, we're just going to assume 2s-complement
00124   // representation and cast directly to the type we really want.
00125   //
00126   // So we can cast directly for all unsigned types and for int32_t; for
00127   // the smaller-width signed types we need to cast through the
00128   // corresponding unsigned type.
00129   typedef int32_t jstype;
00130   typedef int32_t intermediateType;
00131   static inline bool converter(JSContext* cx, JS::Handle<JS::Value> v,
00132                                jstype* retval) {
00133     return JS::ToInt32(cx, v, retval);
00134   }
00135 };

00226 {
00227   static_assert(std::numeric_limits<T>::is_integer,
00228                 "This can only be applied to integers!");
00229 
00230   if (!mozilla::IsFinite(d)) {
00231     return ThrowErrorMessage(cx, MSG_ENFORCE_RANGE_NON_FINITE, TypeName<T>::value());
00232   }
00233 
00234   bool neg = (d < 0);
00235   double rounded = floor(neg ? -d : d);

00239     return ThrowErrorMessage(cx, MSG_ENFORCE_RANGE_OUT_OF_RANGE, TypeName<T>::value());
00240   }

00243   return true;

00251 template<typename T>
00252 inline bool
00253 PrimitiveConversionTraits_Clamp(JSContext* cx, const double& d, T* retval)
00254 {
00255   static_assert(std::numeric_limits<T>::is_integer,
00256                 "This can only be applied to integers!");
00257 
00258   if (mozilla::IsNaN(d)) {
00259     *retval = 0;
00260     return true;
00261   }
00262   if (d >= PrimitiveConversionTraits_Limits<T>::max()) {
00263     *retval = PrimitiveConversionTraits_Limits<T>::max();
00264     return true;
00265   }
00266   if (d <= PrimitiveConversionTraits_Limits<T>::min()) {
00267     *retval = PrimitiveConversionTraits_Limits<T>::min();
00268     return true;
00269   }
00270 
00271   MOZ_ASSERT(mozilla::IsFinite(d));
00272 
00273   // Banker's rounding (round ties towards even).
00274   // We move away from 0 by 0.5f and then truncate.  That gets us the right
00275   // answer for any starting value except plus or minus N.5.  With a starting
00276   // value of that form, we now have plus or minus N+1.  If N is odd, this is
00277   // the correct result.  If N is even, plus or minus N is the correct result.
00278   double toTruncate = (d < 0) ? d - 0.5 : d + 0.5;
00279 
00280   T truncated = static_cast<T>(toTruncate);
00281 
00282   if (truncated == toTruncate) {
00283     /*
00284      * It was a tie (since moving away from 0 by 0.5 gave us the exact integer
00285      * we want). Since we rounded away from 0, we either already have an even
00286      * number or we have an odd number but the number we want is one closer to
00287      * 0. So just unconditionally masking out the ones bit should do the trick
00288      * to get us the value we want.
00289      */
00290     truncated &= ~1;
00291   }
00292 
00293   *retval = truncated;
00294   return true;
00295 }
