00003 //
00004 // Given an MP4/Quicktime based video file as a blob, read through its
00005 // atoms to find the track header "tkhd" atom and extract the rotation
00006 // matrix from it. Convert the matrix value to rotation in degrees and
00007 // pass that number to the specified callback function. If no value is
00008 // found but the video file is valid, pass null to the callback. If
00009 // any errors occur, pass an error message (a string) callback.
00010 //
00011 // See also:
00012 // http://androidxref.com/4.0.4/xref/frameworks/base/media/libstagefright/MPEG4Writer.cpp
00013 // https://developer.apple.com/library/mac/#documentation/QuickTime/QTFF/QTFFChap2/qtff2.html
00014 //
00015 function getVideoRotation(blob, rotationCallback) {
00016 
00017   // A utility for traversing the tree of atoms in an MP4 file
00018   function MP4Parser(blob, handlers) {
00019     // Start off with a 1024 chunk from the start of the blob.
00020     BlobView.get(blob, 0, Math.min(1024, blob.size), function(data, error) {
00021       // Make sure that the blob is, in fact, some kind of MP4 file
00022       if (data.byteLength <= 8 || data.getASCIIText(4, 4) !== 'ftyp') {
00023         handlers.errorHandler('not an MP4 file');
00024         return;
00025       }
00026       parseAtom(data);
00027     });
00028 
00029     // Call this with a BlobView object that includes the first 16 bytes of
00030     // an atom. It doesn't matter whether the body of the atom is included.
00031     function parseAtom(data) {
00032       var offset = data.sliceOffset + data.viewOffset; // atom position in blob
00033       var size = data.readUnsignedInt();               // atom length
00034       var type = data.readASCIIText(4);                // atom type
00035       var contentOffset = 8;                           // position of content
00036 
00037       if (size === 0) {
00038         // Zero size means the rest of the file
00039         size = blob.size - offset;
00040       }
00041       else if (size === 1) {
00042         // A size of 1 means the size is in bytes 8-15
00043         size = data.readUnsignedInt() * 4294967296 + data.readUnsignedInt();
00044         contentOffset = 16;
00045       }
00046 
00047       var handler = handlers[type] || handlers.defaultHandler;
00048       if (typeof handler === 'function') {
00049         // If the handler is a function, pass that function a
00050         // DataView object that contains the entire atom
00051         // including size and type.  Then use the return value
00052         // of the function as instructions on what to do next.
00053         data.getMore(data.sliceOffset + data.viewOffset, size, function(atom) {
00054           // Pass the entire atom to the handler function
00055           var rv = handler(atom);
00056 
00057           // If the return value is 'done', stop parsing.
00058           // Otherwise, continue with the next atom.
00059           // XXX: For more general parsing we need a way to pop some
00060           // stack levels.  A return value that is an atom name should mean
00061           // pop back up to this atom type and go on to the next atom
00062           // after that.
00063           if (rv !== 'done') {
00064             parseAtomAt(data, offset + size);
00065           }
00066         });
00067       }
00068       else if (handler === 'children') {
00069         // If the handler is this string, then assume that the atom is
00070         // a container atom and do its next child atom next
00071         var skip = (type === 'meta') ? 4 : 0; // special case for meta atoms
00072         parseAtomAt(data, offset + contentOffset + skip);
00073       }
00074       else if (handler === 'skip' || !handler) {
00075         // Skip the atom entirely and go on to the next one.
00076         // If there is no next one, call the eofHandler or just return
00077         parseAtomAt(data, offset + size);
00078       }
00079       else if (handler === 'done') {
00080         // Stop parsing
00081         return;
00082       }
00083     }
00084 
00085     function parseAtomAt(data, offset) {
00086       if (offset >= blob.size) {
00087         if (handlers.eofHandler)
00088           handlers.eofHandler();
00089         return;
00090       }
00091       else {
00092         data.getMore(offset, 16, parseAtom);
00093       }
00094     }
00095   }
00096 
00097   // We want to loop through the top-level atoms until we find the 'moov'
00098   // atom. Then, within this atom, there are one or more 'trak' atoms.
00099   // Each 'trak' should begin with a 'tkhd' atom. The tkhd atom has
00100   // a transformation matrix at byte 48.  The matrix is 9 32 bit integers.
00101   // We'll interpret those numbers as a rotation of 0, 90, 180 or 270.
00102   // If the video has more than one track, we expect all of them to have
00103   // the same rotation, so we'll only look at the first 'trak' atom that
00104   // we find.
00105   MP4Parser(blob, {
00106     errorHandler: function(msg) { rotationCallback(msg); },
00107     eofHandler: function() { rotationCallback(null); },
00108     defaultHandler: 'skip',  // Skip all atoms other than those listed below
00109     moov: 'children',        // Enumerate children of the moov atom
00110     trak: 'children',        // Enumerate children of the trak atom
00111     tkhd: function(data) {   // Pass the tkhd atom to this function
00112       // The matrix begins at byte 48
00113       data.advance(48);
00114 
00115       var a = data.readUnsignedInt();
00116       var b = data.readUnsignedInt();
00117       data.advance(4); // we don't care about this number
00118       var c = data.readUnsignedInt();
00119       var d = data.readUnsignedInt();
00120 
00121       if (a === 0 && d === 0) { // 90 or 270 degrees
00122         if (b === 0x00010000 && c === 0xFFFF0000)
00123           rotationCallback(90);
00124         else if (b === 0xFFFF0000 && c === 0x00010000)
00125           rotationCallback(270);
00126         else
00127           rotationCallback('unexpected rotation matrix');
00128       }
00129       else if (b === 0 && c === 0) { // 0 or 180 degrees
00130         if (a === 0x00010000 && d === 0x00010000)
00131           rotationCallback(0);
00132         else if (a === 0xFFFF0000 && d === 0xFFFF0000)
00133           rotationCallback(180);
00134         else
00135           rotationCallback('unexpected rotation matrix');
00136       }
00137       else {
00138         rotationCallback('unexpected rotation matrix');
00139       }
00140       return 'done';
00141     }
00142   });
00143 }
