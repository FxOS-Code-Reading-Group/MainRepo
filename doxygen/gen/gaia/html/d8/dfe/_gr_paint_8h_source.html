00237     bool localCoordChangeInverse(const SkMatrix& newToOld) {
00238         SkMatrix oldToNew;
00239         bool computed = false;
00240         for (int i = 0; i < kMaxColorStages; ++i) {
00241             if (this->isColorStageEnabled(i)) {
00242                 if (!computed && !newToOld.invert(&oldToNew)) {
00243                     return false;
00244                 } else {
00245                     computed = true;
00246                 }
00247                 fColorStages[i].localCoordChange(oldToNew);
00248             }
00249         }
00250         for (int i = 0; i < kMaxCoverageStages; ++i) {
00251             if (this->isCoverageStageEnabled(i)) {
00252                 if (!computed && !newToOld.invert(&oldToNew)) {
00253                     return false;
00254                 } else {
00255                     computed = true;
00256                 }
00257                 fCoverageStages[i].localCoordChange(oldToNew);
00258             }
00259         }
