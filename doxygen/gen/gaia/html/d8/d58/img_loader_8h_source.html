00274   // Returns true if we should prefer evicting cache entry |two| over cache
00275   // entry |one|.
00276   // This mixes units in the worst way, but provides reasonable results.
00277   inline static bool CompareCacheEntries(const nsRefPtr<imgCacheEntry> &one,
00278                                          const nsRefPtr<imgCacheEntry> &two)
00279   {
00280     if (!one)
00281       return false;
00282     if (!two)
00283       return true;
00284 
00285     const double sizeweight = 1.0 - sCacheTimeWeight;
00286 
00287     // We want large, old images to be evicted first (depending on their
00288     // relative weights). Since a larger time is actually newer, we subtract
00289     // time's weight, so an older image has a larger weight.
00290     double oneweight = double(one->GetDataSize()) * sizeweight -
00291                        double(one->GetTouchedTime()) * sCacheTimeWeight;
00292     double twoweight = double(two->GetDataSize()) * sizeweight -
00293                        double(two->GetTouchedTime()) * sCacheTimeWeight;
00294 
00295     return oneweight < twoweight;
00296   }
