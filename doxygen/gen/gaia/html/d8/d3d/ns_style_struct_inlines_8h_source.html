00181 uint8_t
00182 nsStyleVisibility::GetEffectivePointerEvents(nsIFrame* aFrame) const
00183 {
00184   if (aFrame->GetContent() && !aFrame->GetContent()->GetParent()) {
00185     // The root element has a cluster of frames associated with it
00186     // (root scroll frame, canvas frame, the actual primary frame). Make
00187     // those take their pointer-events value from the root element's primary
00188     // frame.
00189     nsIFrame* f = aFrame->GetContent()->GetPrimaryFrame();
00190     if (f) {
00191       return f->StyleVisibility()->mPointerEvents;
00192     }
00193   }
00194   return mPointerEvents;
