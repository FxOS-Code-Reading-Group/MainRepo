00225 // Create a shadow layer (PLayerChild) for aLayer, if we're forwarding
00226 // our layer tree to a parent process.  Record the new layer creation
00227 // in the current open transaction as a side effect.
00228 template<typename CreatedMethod> void
00229 CreateShadowFor(ClientLayer* aLayer,
00230                 ClientLayerManager* aMgr,
00231                 CreatedMethod aMethod)
00232 {
00233   PLayerChild* shadow = aMgr->ConstructShadowFor(aLayer);
00234   // XXX error handling
00235   NS_ABORT_IF_FALSE(shadow, "failed to create shadow");
