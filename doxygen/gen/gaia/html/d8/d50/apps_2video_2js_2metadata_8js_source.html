00001 // This file is part of the Gaia Video app.
00002 //
00003 // It includes functions to obtain metadata for video files. The most
00004 // important parts of this metadata are obtained by loading the video into
00005 // an offscreen <video> element.
00006 //
00007 // The phones we want to run on have only a single hardware h.264 video
00008 // decoder, and gecko is not yet smart enough to share the hardware among
00009 // all the <video> elements that want to use it. So all apps (camera,
00010 // gallery and video) that have permission to use the video hardware must be
00011 // careful to only use it while they are in the foreground and to relinquish
00012 // it when they go to the background.
00013 //
00014 // The video app wants to use <video> elements for metadata parsing and for
00015 // playing videos for the user. It can't do both of these things at the same
00016 // time. The MediaDB metadata parsing architecture assumes that it can,
00017 // however. So the Video app does not pass a metadata parser function to
00018 // MediaDB. MediaDB notifies the app about new video files but does not
00019 // include any metadata with those notifications. Instead, we use the
00020 // queueing and metadata parsing functions in this file to handle metadata
00021 // parsing in an interruptible way.
00022 //
00023 // When we have a new video file that needs metadata, we pass it to
00024 // addToMetadataQueue(). When we need to stop parsing metadata (because we
00025 // want to play a video or because we went to the background) we call
00026 // stopParsingMetadata(). And when we're able to parse metadata again we
00027 // call startParsingMetadata(). When a queued video is processed and its
00028 // metadata is ready, it is saved to MediaDB, and then addVideo() is invoked
00029 // to display the video title and thumbnail.
00030 //
00031 
00032 var metadataQueue = [];
00033 var processingQueue = false;
00034 var stopParsingMetadataCallback = null;

00036 // This function queues a fileinfo object with no metadata. When the app is
00037 // able it will obtain metadata for the video and pass the updated fileinfo
00038 // to addVideo() so that it becomes visible to the user.
00039 function addToMetadataQueue(fileinfo) {
00040   metadataQueue.push(fileinfo);
00041   startParsingMetadata();
00042 }

00044 // Start or resume metadata parsing, if conditions are right
00045 function startParsingMetadata() {
00046   // If there is no work queued, or if we're already working, return right away
00047   if (processingQueue || metadataQueue.length === 0)
00048     return;
00049 
00050   // Don't parse metadata if we are not the foreground app. When we're
00051   // in the background we need to allow the foreground app to use the
00052   // video hardware. Also, if the video player is showing then we
00053   // can't parse metadata because we're already using the video hardware.
00054   if (document.hidden || playerShowing) {
00055     return;
00056   }
00057 
00058   // Start processing the queue
00059   processingQueue = true;
00060   showThrobber();
00061   processFirstQueuedItem();
00062 }
00063 
00064 // Stop parsing metadata. If a callback is specified, call it when stopped.
00065 function stopParsingMetadata(callback) {
00066   // If we're not processing metadata, just call the callback right away
00067   if (!processingQueue) {
00068     if (callback)
00069       callback();
00070     return;
00071   }
00072 
00073   // Otherwise, request a stop.
00074   // This variable acts as a flag to prevent processFirstQueuedItem from
00075   // creating any more metadatas.
00076   stopParsingMetadataCallback = callback || true;
00077 }
00078 
00079 // If we've been asked to stop processing, then run the callback, and stop.
00080 // Otherwise, if there is nothing in the queue, do nothing.  Otherwise, take
00081 // the first item off the queue, obtain metadata for it, save the metadata
00082 // to the database, and then call addVideo() to display the newly processed
00083 // video to the user. Finally, use setTimeout() to schedule a new call to
00084 // this same function.
00085 //
00086 // XXX: are there any race conditions here?
00087 // Are we guaranteed to always stop and to always call the callback?
00088 //
00089 function processFirstQueuedItem() {
00090   // If the stop flag is set, call the callback and stop
00091   if (stopParsingMetadataCallback) {
00092     var callback = stopParsingMetadataCallback;
00093     stopParsingMetadataCallback = null;
00094     processingQueue = false;
00095     hideThrobber();
00096 
00097     if (callback !== true)
00098       callback();  // Okay, we've stopped.
00099     return;
00100   }
00101 
00102   // If there is no work queued, update dialog in case there are no playable
00103   // videos; then return right away
00104   if (metadataQueue.length === 0) {
00105     processingQueue = false;
00106     hideThrobber();
00107     updateDialog();
00108     return;
00109   }
00110 
00111   // Otherwise, take a fileinfo object from the front of the queue and
00112   // process it with a <video> tag to obtain size, duration and a poster
00113   // image. This happens asynchronously. When it is done, we call
00114   // processFirstQueuedItem() again to re-check the stop flag and process
00115   // the next item on the queue.
00116   var fileinfo = metadataQueue.shift();
00117   videodb.getFile(fileinfo.name, function(file) {
00118     getMetadata(file, function(metadata) {
00119       // Associate the metadata with this fileinfo object
00120       fileinfo.metadata = metadata;
00121 
00122       // Save it to the database
00123       videodb.updateMetadata(fileinfo.name, metadata);
00124 
00125       // Create and insert a thumbnail for the video
00126       if (metadata.isVideo)
00127         addVideo(fileinfo);
00128 
00129       // And process the next video in the queue
00130       setTimeout(processFirstQueuedItem);
00131     });
00132   }, function(err) {
00133     console.error('getFile error: ', fileinfo.name, err);
00134     processFirstQueuedItem();
00135   });
00136 }

00211   function createThumbnail() {
00212     // Videos often begin with a black screen, so skip ahead 5 seconds
00213     // or 1/10th of the video, whichever is shorter in the hope that we'll
00214     // get a more interesting thumbnail that way.
00215     // Because of bug 874692, corrupt video files may not be seekable,
00216     // and may not fire an error event, so if we aren't able to seek
00217     // after a certain amount of time, we'll abort and assume that the
00218     // video is invalid.
00219     offscreenVideo.currentTime = Math.min(5, offscreenVideo.duration / 10);
00220 
00221     var failed = false;                      // Did seeking fail?
00222     var timeout = setTimeout(fail, 10000);   // Fail after 10 seconds
00223     offscreenVideo.onerror = fail;           // Or if we get an error event
00224     function fail() {                        // Seeking failure case
00225       console.warn('Seek failed while creating thumbnail for', videofile.name,
00226                    '. Ignoring corrupt file.');
00227       failed = true;
00228       clearTimeout(timeout);
00229       offscreenVideo.onerror = null;
00230       metadata.isVideo = false;
00231       unload();
00232       callback(metadata);
00233     }
00234     offscreenVideo.onseeked = function() {   // Seeking success case
00235       if (failed) // Avoid race condition: if we already failed, stop now
00236         return;
00237       clearTimeout(timeout);

00239         if (poster === null) {
00240           // If something goes wrong in captureFrame, it probably means that

00262   // Derive the video title from its filename.
00263   function fileNameToVideoName(filename) {
00264     filename = filename.split('/').pop()
00265       .replace(/\.(webm|ogv|ogg|mp4|3gp)$/, '')

00271 function captureFrame(player, metadata, callback) {
00272   try {
00273     var canvas = document.createElement('canvas');
00274     var ctx = canvas.getContext('2d');
00275     canvas.width = THUMBNAIL_WIDTH;
00276     canvas.height = THUMBNAIL_HEIGHT;
00277 
00278     var vw = player.videoWidth, vh = player.videoHeight;
00279     var tw, th;
00280 
00281     // If a rotation is specified, rotate the canvas context
00282     switch (metadata.rotation) {
00283     case 90:
00284       ctx.translate(THUMBNAIL_WIDTH, 0);
00285       ctx.rotate(Math.PI / 2);
00286       tw = THUMBNAIL_HEIGHT;
00287       th = THUMBNAIL_WIDTH;
00288       break;
00289     case 180:
00290       ctx.translate(THUMBNAIL_WIDTH, THUMBNAIL_HEIGHT);
00291       ctx.rotate(Math.PI);
00292       tw = THUMBNAIL_WIDTH;
00293       th = THUMBNAIL_HEIGHT;
00294       break;
00295     case 270:
00296       ctx.translate(0, THUMBNAIL_HEIGHT);
00297       ctx.rotate(-Math.PI / 2);
00298       tw = THUMBNAIL_HEIGHT;
00299       th = THUMBNAIL_WIDTH;
00300       break;
00301     default:
00302       tw = THUMBNAIL_WIDTH;
00303       th = THUMBNAIL_HEIGHT;
00304       break;
00305     }
00306 
00307     // Figure out what portion of the video we want to draw into the thumbnail
00308     var scale = Math.min(vw / tw, vh / th);
00309     var w = tw * scale, h = th * scale;
00310     var x = (vw - w) / 2, y = (vh - h) / 2;
00311 
00312     // Draw the current video frame into the image
00313     ctx.drawImage(player, x, y, w, h, 0, 0, tw, th);
00314 
00315     // Convert it to an image file and pass to the callback.
00316     canvas.toBlob(callback, 'image/jpeg');
00317   }
00318   catch (e) {
00319     console.error('Exception in captureFrame:', e, e.stack);
00320     callback(null);
00321   }
