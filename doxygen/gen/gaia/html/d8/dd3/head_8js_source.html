00044 // Poll every 50ms for the condition function to return a truthy value
00045 // and then call the callback function when it does.
00046 // This is designed to be used with yield, nextStep, and shortcut functions
00047 // in an idiom like this:
00048 //
00049 //   yield until(function() document.getElementById('foo'), nextStep);
00050 //
00051 // TODO: create a version of this function that waits for a given event
00052 // on a given object instead of polling for a condition
00053 //
00054 // TODO: add an optional timeout argument

00056 function until(condition, callback) {
00057   var timer = setInterval(check, 50);
00058 
00059   function check() {
00060     if (!condition())
00061       return;
00062     clearInterval(timer);
00063     callback();
00064   }

00074 //
00075 // testApp():
00076 //
00077 // This function does the following:
00078 // - unlocks the lock screen
00079 // - launches the app specified by url
00080 // - waits for the app to load
00081 // - passes the app's window and document to testfunc()
00082 // - brings up the app switcher and kills the app
00083 // - locks the lockscreen again
00084 //
00085 // Write tests like this:
00086 //
00087 //  function generatorTest() {
00088 //    yield testApp("http://dialer.gaiamobile.org/", dialerTest)
00089 //  }
00090 //
00091 // function dialerTest(window, document, nextStep) {...}
00092 //
00093 // Note that tests that use testApp() have to be generatorTest() tests.
00094 // This won't work with plain test() tests.  And the function you pass
00095 // to testApp() can itself be a generator. The outer test generator yields
00096 // while the inner app test generator runs.  The third argument to the
00097 // app test generator function will be the equivalent of nextStep()
00098 //
00099 // Note that this testing app depends on details of the the homescreen
00100 // document structure and is therefore somewhat brittle.  If the homescreen
00101 // changes then this test framework may have to change to match.
00102 //
00103 //

00106   g.next();
