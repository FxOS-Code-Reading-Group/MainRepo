02046     static const char* jsTypes[] = {
02047       "text/javascript",
02048       "text/ecmascript",
02049       "application/javascript",
02050       "application/ecmascript",
02051       "application/x-javascript",
02052       "application/x-ecmascript",

02066     for (uint32_t i = 0; jsTypes[i]; ++i) {
