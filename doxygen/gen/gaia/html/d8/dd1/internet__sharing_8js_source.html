00006 /*
00007  * Internet Sharing module responsible for saving and restoring the internet
00008  * sharing state, including Wifi hotspot and USB tethering, based on sim card.
00009  *
00010  * The sharing state is linked with sim card id, iccid. When a user changes sim
00011  * card, this module restore the last state of that sim card.
00012  *
00013  * If the there is no sim card or sim card is not ready, this module use the
00014  * settings of no sim card in which Wifi hotspot will always be disabled.
00015  */
00016 
00017 var InternetSharing = (function() {
00018 
00019   var settings;
00020   // null or unknown state will change to one of the following state
00021   var validCardState = ['absent',
00022                         'pinRequired',
00023                         'pukRequired',
00024                         'networkLocked',
00025                         'corporateLocked',
00026                         'serviceProviderLocked',
00027                         'ready'];
00028 
00029   var observerHooked = false;
00030 
00031   var cardState;
00032 
00033   function addObservers() {
00034     if (observerHooked) {
00035       return;
00036     }
00037     observerHooked = true;
00038     // listen changes after value is restored.
00039     ['usb', 'wifi'].forEach(function(type) {
00040       settings.addObserver('tethering.' + type + '.enabled',
00041                           internetSharingSettingsChangeHanlder);
00042     });
00043   }
00044 
00045   function checkCardAndInternetSharing() {
00046     cardState = IccHelper.cardState;
00047     if (validCardState.indexOf(cardState) > -1) {
00048       // if it is known cardState, we need to load internet sharing state from
00049       // settings
00050       addObservers();
00051       restoreInternetSharingState(cardState);
00052     }
00053   }
00054 
00055   function restoreInternetSharingState(cardState) {
00056     // the function restores settings based on type, cardId.
00057     function doRestore(type, cardId, forceDisabled) {
00058       // build the key for asyncStorage.
00059       var key = 'tethering.' + type + '.simstate.card-' + cardId;
00060       asyncStorage.getItem(key, function callback(value) {
00061         // if forceDisable is true, we need to disable it always.
00062         var simState = forceDisabled ? false : (value || false);
00063         // update value for type
00064         var cset = {};
00065         cset['tethering.' + type + '.enabled'] = simState;
00066         settings.createLock().set(cset);
00067       });
00068     }
00069     // the internet sharing is linked with iccid, if cardState is not ready, we
00070     // just view it as no sim.
00071     // note if it is under pinRequired or pukRequired, it may turned into ready
00072     // state after user typed pin or puk.
00073     if (cardState === 'ready') {
00074       // once cardState is ready, we need to read iccid
00075       // if iccInfo.iccid is not ready, we need to listen change of iccinfo
00076       // change, until iccid is ready
00077       if (!IccHelper.iccInfo || !IccHelper.iccInfo.iccid) {
00078         IccHelper.oniccinfochange = function handler() {
00079           // wait for iccid is filled.
00080           if (IccHelper.iccInfo && IccHelper.iccInfo.iccid) {
00081             IccHelper.oniccinfochange = null;
00082             doRestore('usb', IccHelper.iccInfo.iccid);
00083             doRestore('wifi', IccHelper.iccInfo.iccid);
00084           }
00085         };
00086       } else {
00087         // iccInfo is ready, just use it.
00088         doRestore('usb', IccHelper.iccInfo.iccid);
00089         doRestore('wifi', IccHelper.iccInfo.iccid);
00090       }
00091     } else {
00092       // card is not ready, just use absent to restore the value.
00093       doRestore('usb', 'absent');
00094       // card is not ready, force wifi hotspot disabled.
00095       doRestore('wifi', 'absent', true);
00096     }
00097   }
00098 
00099   function internetSharingSettingsChangeHanlder(evt) {
00100     if (validCardState.indexOf(cardState) === -1) {
00101       return;
00102     }
00103     // link the iccid with current internet state for future restoring.
00104     var type = (evt.settingName.indexOf('wifi') > -1) ? 'wifi' : 'usb';
00105     var cardId = IccHelper.iccInfo.iccid || 'absent';
00106     // wifi hotspot cannot be enabled without sim
00107     if ('wifi' === type && 'absent' === cardId && true === evt.settingValue) {
00108       settings.createLock().set({'tethering.wifi.enabled': false});
00109       return;
00110     }
00111     asyncStorage.setItem('tethering.' + type + '.simstate.card-' + cardId,
00112                          evt.settingValue);
00113   }
00114 
00115   function _init() {
00116     settings = window.navigator.mozSettings;
00117     if (!settings) {
00118       return;
00119     }
00120     if (!IccHelper.enabled) {
00121       return;
00122     }
00123     observerHooked = false;
00124     checkCardAndInternetSharing();
00125     // listen cardstatechange event for ready, pin, puk, or network unlocking.
00126     IccHelper.addEventListener('cardstatechange', checkCardAndInternetSharing);
00127   }
00128   return {init: _init};
00129 })();
