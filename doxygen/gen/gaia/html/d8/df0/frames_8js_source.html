00168 // Pan the item sideways when the user moves their finger across the screen
00169 function panHandler(event) {
00170   if (transitioning)
00171     return;
00172 
00173   var dx = event.detail.relative.dx;
00174   var dy = event.detail.relative.dy;
00175   var oldFrameOffset = frameOffset;
00176 
00177   // If the frames are already being shifted in the same direction as
00178   // dx then this just continues the shift.  Otherwise, dx might shift
00179   // them back toward the center. If the frames are unshifted to begin
00180   // with or become unshifted after applying dx, then we have got to
00181   // pass dx to the pan() method of the frame, because it might pan
00182   // the image within the frame. But that method returns any dx it
00183   // can't use, and we apply that to shifting the frames.
00184 
00185   // If the frames are already shifted and dx is in the same direction, or
00186   // if dx is in the opposite direction but isn't big enough to bring
00187   // the frames back to the center, just adjust the frame positions.
00188   // There is no need to pan the content of the frame in this case.
00189   if ((frameOffset > 0 && dx > 0) ||
00190       (frameOffset < 0 && dx < 0) ||
00191       (frameOffset !== 0 && frameOffset > -dx)) {
00192     frameOffset += dx;
00193   }
00194   else {
00195     // If the frame is shifted, this dx brings it back to center
00196     if (frameOffset !== 0) {
00197       dx += frameOffset;
00198       frameOffset = 0;
00199     }
00200 
00201     // Now let the frame pan its content, and add any dx that it doesn't use
00202     // to the frame offset
00203     frameOffset += currentFrame.pan(dx, dy);
00204   }
00205 
00206   // Don't swipe past the end of the last item or past the start of the first
00207   if ((currentFileIndex === 0 && frameOffset > 0) ||
00208       (currentFileIndex === files.length - 1 && frameOffset < 0)) {
00209     frameOffset = 0;
00210   }
00211 
00212   // If the frameOffset has changed since we started, reposition the frames
00213   if (frameOffset !== oldFrameOffset)
00214     setFramesPosition();
00215 }

00217 // When the user lifts their finger after panning we get this event
00218 function swipeHandler(event) {
00219   // If we just panned within a zoomed-in photo, and the frames are not
00220   // shifted at all, then we don't have to do anything here.
00221   if (frameOffset === 0)
00222     return;
00223 
00224   // 1 means we're going to the next item -1 means the previous
00225   var direction = (frameOffset < 0) ? 1 : -1;
00226 
00227   // If we're in a right-to-left locale, reverse those directions
00228   if (languageDirection === 'rtl')
00229     direction *= -1;
00230 
00231   // Did we pan far enough or swipe fast enough to transition to
00232   // a different item?
00233   var farenough =
00234     Math.abs(frameOffset) > window.innerWidth * TRANSITION_FRACTION;
00235   var velocity = event.detail.vx;
00236   var fastenough = Math.abs(velocity) > TRANSITION_SPEED;
00237 
00238   // Make sure that that the speed and pan amount are in the same direction
00239   var samedirection = velocity === 0 || frameOffset / velocity >= 0;
00240 
00241   // Is there a next or previous item to transition to?
00242   var fileexists =
00243     (direction === 1 && currentFileIndex + 1 < files.length) ||
00244     (direction === -1 && currentFileIndex > 0);
00245 
00246   // If all of these conditions hold, then we'll transition to the
00247   // next photo or the previous photo
00248   if (direction !== 0 && (farenough || fastenough) &&
00249       samedirection && fileexists) {
00250 
00251     // Compute how long the transition should take based on the velocity
00252     var speed = Math.max(Math.abs(velocity), TRANSITION_SPEED);
00253     var time = (window.innerWidth - Math.abs(frameOffset)) / speed;
00254 
00255     // Transition frames in the appropriate direction
00256     if (direction === 1)
00257       nextFile(time);
00258     else
00259       previousFile(time);
00260   }
00261   else if (frameOffset !== 0) {
00262     // Otherwise, just restore the current item by undoing
00263     // the translations we added during panning
00264     var time = Math.abs(frameOffset) / TRANSITION_SPEED;
00265 
00266     currentFrame.container.style.transition =
00267       nextFrame.container.style.transition =
00268       previousFrame.container.style.transition =
00269       'transform ' + time + 'ms ease';
00270 
00271     resetFramesPosition();
00272 
00273     // Ignore  pan and zoom gestures while the transition happens
00274     transitioning = true;
00275     setTimeout(function() { transitioning = false; }, time);
00276   }
00277 }

00381 // Transition to the next file, animating it over the specified time (ms).
00382 // This is used when the user pans.
00383 function nextFile(time) {
00384   // If already displaying the last one, do nothing.
00385   if (currentFileIndex === files.length - 1)
00386     return;
00387 
00388   // If the current frame is using a <video> element instead of just
00389   // displaying a poster image, reset it back to just the image
00390   if (currentFrame.displayingVideo && currentFrame.video.playerShowing)
00391     currentFrame.video.init();
00392 
00393   // Set a flag to ignore pan and zoom gestures during the transition.
00394   transitioning = true;
00395   setTimeout(function() { transitioning = false; }, time);
00396 
00397   // Set transitions for the visible frames
00398   var transition = 'transform ' + time + 'ms ease';
00399   currentFrame.container.style.transition = transition;
00400   nextFrame.container.style.transition = transition;
00401 
00402   // Cycle the three frames so next becomes current,
00403   // current becomes previous, and previous becomes next.
00404   var tmp = previousFrame;
00405   previousFrame = currentFrame;
00406   currentFrame = nextFrame;
00407   nextFrame = tmp;
00408   currentFileIndex++;
00409 
00410   // Move (transition) the frames to their new position
00411   resetFramesPosition();
00412 
00413   // Update the frame for the new next item
00414   setupFrameContent(currentFileIndex + 1, nextFrame);
00415 
00416   // When the transition is done, cleanup
00417   currentFrame.container.addEventListener('transitionend', function done(e) {
00418     this.removeEventListener('transitionend', done);
00419 
00420     // Reposition the item that just transitioned off the screen
00421     // to reset any zooming and panning
00422     previousFrame.reset();
00423   });
00424 
00425   // Disable the edit button if we're now viewing a video, and enable otherwise
00426   if (currentFrame.displayingVideo)
00427     $('fullscreen-edit-button').classList.add('disabled');
00428   else
