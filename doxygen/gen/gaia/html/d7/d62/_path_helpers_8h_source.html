00016 void ArcToBezier(T* aSink, const Point &aOrigin, float aRadius, float aStartAngle,
00017                  float aEndAngle, bool aAntiClockwise)
00018 {
00019   Point startPoint(aOrigin.x + cos(aStartAngle) * aRadius,
00020                    aOrigin.y + sin(aStartAngle) * aRadius);
00021 
00022   aSink->LineTo(startPoint);
00023 
00024   // Clockwise we always sweep from the smaller to the larger angle, ccw
00025   // it's vice versa.
00026   if (!aAntiClockwise && (aEndAngle < aStartAngle)) {
00027     Float correction = Float(ceil((aStartAngle - aEndAngle) / (2.0f * M_PI)));
00028     aEndAngle += float(correction * 2.0f * M_PI);
00029   } else if (aAntiClockwise && (aStartAngle < aEndAngle)) {
00030     Float correction = (Float)ceil((aEndAngle - aStartAngle) / (2.0f * M_PI));
00031     aStartAngle += float(correction * 2.0f * M_PI);
00032   }
00033 
00034   // Sweeping more than 2 * pi is a full circle.
00035   if (!aAntiClockwise && (aEndAngle - aStartAngle > 2 * M_PI)) {
00036     aEndAngle = float(aStartAngle + 2.0f * M_PI);
00037   } else if (aAntiClockwise && (aStartAngle - aEndAngle > 2.0f * M_PI)) {
00038     aEndAngle = float(aStartAngle - 2.0f * M_PI);
00039   }
00040 
00041   // Calculate the total arc we're going to sweep.
00042   Float arcSweepLeft = fabs(aEndAngle - aStartAngle);
00043 
00044   Float sweepDirection = aAntiClockwise ? -1.0f : 1.0f;
00045 
00046   Float currentStartAngle = aStartAngle;
00047 
00048   while (arcSweepLeft > 0) {
00049     // We guarantee here the current point is the start point of the next
00050     // curve segment.
00051     Float currentEndAngle;
00052 
00053     if (arcSweepLeft > M_PI / 2.0f) {
00054       currentEndAngle = Float(currentStartAngle + M_PI / 2.0f * sweepDirection);
00055     } else {
00056       currentEndAngle = currentStartAngle + arcSweepLeft * sweepDirection;
00057     }
00058 
00059     Point currentStartPoint(aOrigin.x + cos(currentStartAngle) * aRadius,
00060                             aOrigin.y + sin(currentStartAngle) * aRadius);
00061     Point currentEndPoint(aOrigin.x + cos(currentEndAngle) * aRadius,
00062                           aOrigin.y + sin(currentEndAngle) * aRadius);
00063 
00064     // Calculate kappa constant for partial curve. The sign of angle in the
00065     // tangent will actually ensure this is negative for a counter clockwise
00066     // sweep, so changing signs later isn't needed.
00067     Float kappa = (4.0f / 3.0f) * tan((currentEndAngle - currentStartAngle) / 4.0f) * aRadius;
00068 
00069     Point tangentStart(-sin(currentStartAngle), cos(currentStartAngle));
00070     Point cp1 = currentStartPoint;
00071     cp1 += tangentStart * kappa;
00072 
00073     Point revTangentEnd(sin(currentEndAngle), -cos(currentEndAngle));
00074     Point cp2 = currentEndPoint;
00075     cp2 += revTangentEnd * kappa;
00076 
00077     aSink->BezierTo(cp1, cp2, currentEndPoint);
00078 
00079     arcSweepLeft -= Float(M_PI / 2.0f);
00080     currentStartAngle = currentEndAngle;
00081   }
00082 }
