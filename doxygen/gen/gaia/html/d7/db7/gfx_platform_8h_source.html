00122 inline const char*
00123 GetBackendName(mozilla::gfx::BackendType aBackend)
00124 {
00125   switch (aBackend) {
00126       case mozilla::gfx::BACKEND_DIRECT2D:
00127         return "direct2d";
00128       case mozilla::gfx::BACKEND_COREGRAPHICS_ACCELERATED:
00129         return "quartz accelerated";
00130       case mozilla::gfx::BACKEND_COREGRAPHICS:
00131         return "quartz";
00132       case mozilla::gfx::BACKEND_CAIRO:
00133         return "cairo";
00134       case mozilla::gfx::BACKEND_SKIA:
00135         return "skia";
00136       case mozilla::gfx::BACKEND_RECORDING:
00137         return "recording";
00138       case mozilla::gfx::BACKEND_DIRECT2D1_1:
00139         return "direct2d 1.1";
00140       case mozilla::gfx::BACKEND_NONE:
00141         return "none";
00142   }
00143   MOZ_CRASH("Incomplete switch");
00144 }

00206     /*
00207      * Cairo doesn't give us a way to create a surface pointing to a context
00208      * without marking it as copy on write. For canvas we want to create
00209      * a surface that points to what is currently being drawn by a canvas
00210      * without a copy thus we need to create a special case. This works on
00211      * most platforms with GetThebesSurfaceForDrawTarget but fails on Mac
00212      * because when we create the surface we vm_copy the memory and never
00213      * notify the context that the canvas has drawn to it thus we end up
00214      * with a static snapshot.
00215      *
00216      * This function guarantes that the gfxASurface reflects the DrawTarget.
00217      */
00218     virtual already_AddRefed<gfxASurface>
00219       CreateThebesSurfaceAliasForDrawTarget_hack(mozilla::gfx::DrawTarget *aTarget) {
00220       // Overwrite me on platform where GetThebesSurfaceForDrawTarget returns
00221       // a snapshot of the draw target.
00222       return GetThebesSurfaceForDrawTarget(aTarget);
00223     }
