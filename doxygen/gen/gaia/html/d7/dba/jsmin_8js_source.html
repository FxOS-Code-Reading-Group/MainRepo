00033 function JSMin(input) {
00034   const EOF = -1;
00035 
00036   var stdin = {
00037     'index': 0,
00038     'end': input.length,
00039     'read': function (len) {
00040       // For now only handle len = 1
00041       if (len !== 1) {
00042         throw new Error('You can only read one character from input at a time.');
00043       }
00044 
00045       // If we are at the end, return EOF (-1)
00046       var index = stdin.index;
00047       if (index === stdin.end) {
00048         return EOF;
00049       }
00050 
00051       // Read the input at our index
00052       var char = input[index];
00053 
00054       // Increment our index
00055       stdin.index = index + 1;
00056 
00057       // Return char
00058       return char;
00059     },
00060     'getIndex': function () {
00061       return stdin.index;
00062     },
00063     // Helper for verifying during dev
00064     'charAt': function (index) {
00065       return (index >= stdin.end) ? EOF : input.charAt(index);
00066     }
00067   };
00068 
00069   var map = {};
00070   var output = '';
00071   options = {
00072     'error': '',
00073     'stdout': {
00074       'write': function (str) {
00075         // Add the string to output
00076         output += str;
00077       },
00078       'writeFromIndex': function (index) {
00079         var char = '\n';
00080 
00081         // If we are at a positive index
00082         if (index >= 0) {
00083           // Grab the char from its index
00084           char = input.charAt(index);
00085 
00086           // Get the length of the current output
00087           var len = output.length;
00088 
00089           // Map the sourceIndex to the destIndex
00090           map[index] = len;
00091         }
00092 
00093         // Output the char to output
00094         output += char;
00095       }
00096     },
00097     'stderr': function (err) {
00098       // Add the error to output
00099       options.error += err;
00100     },
00101     'exit': function (code) {
00102       // Throw the collective error
00103       throw new Error(options.error);
00104     }
00105   };
00106 
00107   // Run jsminc
00108   jsminc(stdin, options);
00109 
00110   // Return the output and map
00111   var retObj = {
00112     'code': output,
00113     'codeMap': map
00114   };
00115   return retObj;
