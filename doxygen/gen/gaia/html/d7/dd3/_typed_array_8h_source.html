00100   static inline JSObject*
00101   Create(JSContext* cx, nsWrapperCache* creator, uint32_t length,
00102          const T* data = NULL) {
00103     JS::Rooted<JSObject*> creatorWrapper(cx);
00104     Maybe<JSAutoCompartment> ac;
00105     if (creator && (creatorWrapper = creator->GetWrapperPreserveColor())) {
00106       ac.construct(cx, creatorWrapper);
00107     }
00108     JSObject* obj = CreateNew(cx, length);
00109     if (!obj) {
00110       return NULL;
00111     }
00112     if (data) {
00113       T* buf = static_cast<T*>(GetData(obj));
00114       memcpy(buf, data, length*sizeof(T));
00115     }
