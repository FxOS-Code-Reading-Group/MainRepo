00001 /*
00002  * visibility_monitor.js
00003  *
00004  * Given a scrolling container element (with overflow-y: scroll set,
00005  * e.g.), monitorChildVisibility() listens for scroll events in order to
00006  * determine which child elements are visible within the element and
00007  * which are not (assuming that the element itself is visible).
00008  *
00009  * When a child scrolls onscreen, it is passed to the onscreen callback.
00010  *
00011  * When a child scrolls offscreen, it is passed to the offscreen callback.
00012  *
00013  * This class also listens for DOM modification events so that it can handle
00014  * children being added to or removed from the scrolling element. It also
00015  * handles resize events.
00016  *
00017  * Note that this class only pays attention to the direct children of
00018  * the container element, not all ancestors.
00019  *
00020  * When you insert a new child into the container, you should create it in
00021  * its offscreen state. If it is inserted offscreen nothing will happen.
00022  * If you insert it onscreen, it will immediately be passed to the onscreen
00023  * callback function
00024  *
00025  * The scrollmargin argument specifies a number of pixels. Elements
00026  * that are within this many pixels of being onscreen are considered
00027  * onscreen.
00028  *
00029  * The scrolldelta parameter is also a number of pixels.  The user
00030  * must scroll this distance before any visibility recomputation is
00031  * done by this code.  This parameter can be used to "batch" up work
00032  * into larger chunks.
00033  *
00034  * By specifing proper onscreen and offscreen functions you can use this
00035  * class to (for example) remove the background-image style of elements
00036  * that are not visible, allowing gecko to free up image memory.
00037  * In that sense, this class can be used to workaround
00038  * https://bugzilla.mozilla.org/show_bug.cgi?id=689623
00039  *
00040  * The return value of this function is an object that has a stop() method.
00041  * calling the stop method stops visiblity monitoring. If you want to restart
00042  * call monitorChildVisiblity() again.
00043  *
00044  * monitorChildVisiblity() makes the following assumptions. If your program
00045  * violates them, the function may not work correctly:
00046  *
00047  *  Child elements of the container element flow left to right and
00048  *  top to bottom. I.e. the nextSibling of a child element never has a
00049  *  smaller clientTop value. They are not absolutely positioned and don't
00050  *  move on their own.
00051  *
00052  *  The children of the container element are themselves all elements;
00053  *  there are no text nodes or comments cluttering things up.
00054  *
00055  *  Children don't change size, either spontaneously or in response to
00056  *  onscreen and offscreen callbacks. Don't set display:none on an element
00057  *  when it goes offscreen, for example.
00058  *
00059  *  Children aren't added or removed to the container while the container
00060  *  or any of its ancestors is hidden with display:none or is removed from
00061  *  the tree. The mutation observer that responds to additions and deletions
00062  *  needs the container and its children to have valid layout data in order
00063  *  to figure out what is onscreen and what is offscreen. Use visiblity:hidden
00064  *  instead of display:none if you need to add or remove children while
00065  *  the container is hidden.
00066  *
00067  *  DocumentFragments are not used to add multiple children at once to
00068  *  the container, and multiple children are not deleted at once by
00069  *  setting innerHTML or innerText to ''.
00070  *
00071  *  The container element only changes size when there is a resize event
00072  *  on the window.
00073  */

00222       // We know that there is at least one element left, so one of these

00226       }

00241   // Adjust the onscreen element range and asynchronously call
00242   // onscreen and offscreen callbacks as needed. We do this
00243   // asynchronously so that if we get lots of scroll events in
00244   // rapid succession and can't keep up, we can skip some of

00290   // This function recomputes the range of onscreen elements. Normally it
00291   // just needs to do small amounts of nextElementSibling
00292   // or previousElementSibling iteration to find the range. But it can also
00293   // start from an unknown state and search the entire container to find
00294   // the range of child elements that are onscreen.
00295   function adjustBounds() {
00296     // If the container has no children, the bounds are null
00297     if (container.firstElementChild === null) {
00298       firstOnscreen = lastOnscreen = null;
00299       return;
00300     }
00301 
00302     // Compute the visible region of the screen, including scroll margin
00303     var scrollTop = container.scrollTop;
00304     var screenTop = scrollTop - scrollmargin;
00305     var screenBottom = scrollTop + container.clientHeight + scrollmargin;
00306 
00307     // This utility function returns ON if the child is onscreen,
00308     // BEFORE if it offscreen before the visible elements and AFTER if
00309     // it is offscreen aafter the visible elements
00310     var BEFORE = -1, ON = 0, AFTER = 1;
00311     function position(child) {
00312       var childTop = child.offsetTop;
00313       var childBottom = childTop + child.offsetHeight;
00314       if (childBottom < screenTop)
00315         return BEFORE;
00316       if (childTop > screenBottom)
00317         return AFTER;
00318       return ON;
00319     }
00320 
00321     // If we don't have a first onscreen element yet, start with the first.
00322     if (!firstOnscreen)
00323       firstOnscreen = container.firstElementChild;
00324 
00325     // Check the position of the top
00326     var toppos = position(firstOnscreen);
00327 
00328     // If the first element is onscreen, see if there are earlier ones
00329     if (toppos === ON) {
00330       var prev = firstOnscreen.previousElementSibling;
00331       while (prev && position(prev) === ON) {
00332         firstOnscreen = prev;
00333         prev = prev.previousElementSibling;
00334       }
00335     }
00336     else if (toppos === BEFORE) {
00337       // The screen is below us, so find the next element that is visible.
00338       var e = firstOnscreen.nextElementSibling;
00339       while (e && position(e) !== ON) {
00340         e = e.nextElementSibling;
00341       }
00342       firstOnscreen = e;
00343     }
00344     else {
00345       // We've scrolled a lot or things have moved so much that the
00346       // entire visible region is now above the first element.
00347       // So scan backwards to find the new lastOnscreen and firstOnscreen
00348       // elements.  Note that if we get here, we can return since we
00349       // will have updated both bounds
00350 
00351       // Loop until we find an onscreen element
00352       lastOnscreen = firstOnscreen.previousElementSibling;
00353       while (lastOnscreen && position(lastOnscreen) !== ON)
00354         lastOnscreen = lastOnscreen.previousElementSibling;
00355 
00356       // Now loop from there to find the first onscreen element
00357       firstOnscreen = lastOnscreen;
00358       prev = firstOnscreen.previousElementSibling;
00359       while (prev && position(prev) === ON) {
00360         firstOnscreen = prev;
00361         prev = prev.previousElementSibling;
00362       }
00363       return;
00364     }
00365 
00366     // Now make the same adjustment on the bottom of the onscreen region
00367     // If we don't have a lastOnscreen value to start with, use the newly
00368     // computed firstOnscreen value.
00369     if (lastOnscreen === null)
00370       lastOnscreen = firstOnscreen;
00371 
00372     var bottompos = position(lastOnscreen);
00373     if (bottompos === ON) {
00374       // If the last element is onscreen, see if there are more below it.
00375       var next = lastOnscreen.nextElementSibling;
00376       while (next && position(next) === ON) {
00377         lastOnscreen = next;
00378         next = next.nextElementSibling;
00379       }
00380     }
00381     else if (bottompos === AFTER) {
00382       // the last element is now below the visible part of the screen
00383       lastOnscreen = lastOnscreen.previousElementSibling;
00384       while (position(lastOnscreen) !== ON)
00385         lastOnscreen = lastOnscreen.previousElementSibling;
00386     }
00387     else {
00388       // First and last are now both above the visible portion of the screen
00389       // So loop down to find their new positions
00390       firstOnscreen = lastOnscreen.nextElementSibling;
00391       while (firstOnscreen && position(firstOnscreen) !== ON) {
00392         firstOnscreen = firstOnscreen.nextElementSibling;
00393       }
00394

00396       var next = lastOnscreen.nextElementSibling;
00397       while (next && position(next) === ON) {
00398         lastOnscreen = next;
00399         next = next.nextElementSibling;

00405   // We use this for asynchronous notification after scroll events.

00407     if (pendingCallbacks) {
00408       // XXX: or we could just return here, which would defer for less time.
00409       clearTimeout(pendingCallbacks);
00410     }
00411     pendingCallbacks = setTimeout(callCallbacks, 0);

00416   // callbacks for elements that have come onscreen or gone offscreen since

00427     // up to, but not including to.

00429       var e = from;

00433         }

00490     // If new elements are visible at the bottom, call those.
00491     // If elements have gone offscreen at the top, call those callbacks
00492     // If elements have gone offscreen at the bottom, call those.
00493     else {
00494       // Are there new onscreen elements at the top?
00495       if (before(firstOnscreen, firstNotifiedOnscreen)) {
00496         onscreen(firstOnscreen, firstNotifiedOnscreen);
00497       }
00498 
00499       // Are there new onscreen elements at the bottom?
00500       if (after(lastOnscreen, lastNotifiedOnscreen)) {
00501         onscreen(lastNotifiedOnscreen.nextElementSibling,
00502                  lastOnscreen.nextElementSibling);
00503       }
00504 
00505       // Have elements gone offscreen at the top?
00506       if (after(firstOnscreen, firstNotifiedOnscreen)) {
00507         offscreen(firstNotifiedOnscreen, firstOnscreen);
00508       }
00509 
00510       // Have elements gone offscreen at the bottom?
00511       if (before(lastOnscreen, lastNotifiedOnscreen)) {
00512         offscreen(lastOnscreen.nextElementSibling,
00513                   lastNotifiedOnscreen.nextElementSibling);
00514       }
00515     }
