00053 typedef enum {

00056 
00057   // The CSS Shaders spec is a subset of the WebGL spec.
00058   //
00059   // In both CSS vertex and fragment shaders, ANGLE:
00060   // (1) Reserves the "css_" prefix.
00061   // (2) Renames the main function to css_main.
00062   // (3) Disables the gl_MaxDrawBuffers built-in.
00063   //
00064   // In CSS fragment shaders, ANGLE:
00065   // (1) Disables the gl_FragColor built-in.
00066   // (2) Disables the gl_FragData built-in.
00067   // (3) Enables the css_MixColor built-in.
00068   // (4) Enables the css_ColorMatrix built-in.
00069   //
00070   // After passing a CSS shader through ANGLE, the browser is expected to append
00071   // a new main function to it.
00072   // This new main function will call the css_main function.
00073   // It may also perform additional operations like varying assignment, texture
00074   // access, and gl_FragColor assignment in order to implement the CSS Shaders
00075   // blend modes.
00076   //
00077   SH_CSS_SHADERS_SPEC = 0x8B42
00078 } ShShaderSpec;

00280 //                 Can be queried by calling ShGetObjectCode().
