00035  * This interface allows us to be notified when a piece of SVG content is
00036  * re-rendered.
00037  *
00038  * Concrete implementations of this interface need to implement
00039  * "GetTarget()" to specify the piece of SVG content that they'd like to
00040  * monitor, and they need to implement "DoUpdate" to specify how we'll react
00041  * when that content gets re-rendered. They also need to implement a
00042  * constructor and destructor, which should call StartListening and

00044  */

00100 /*
00101  * SVG elements reference supporting resources by element ID. We need to
00102  * track when those resources change and when the DOM changes in ways
00103  * that affect which element is referenced by a given ID (e.g., when
00104  * element IDs change). The code here is responsible for that.
00105  * 
00106  * When a frame references a supporting resource, we create a property
00107  * object derived from nsSVGIDRenderingObserver to manage the relationship. The
00108  * property object is attached to the referencing frame.
00109  */
00110 class nsSVGIDRenderingObserver : public nsSVGRenderingObserver {
00111 public:
00112   typedef mozilla::dom::Element Element;
00113   nsSVGIDRenderingObserver(nsIURI* aURI, nsIFrame *aFrame,
00114                          bool aReferenceImage);
00115   virtual ~nsSVGIDRenderingObserver();
00116 
00117 protected:
00118   Element* GetTarget() MOZ_OVERRIDE { return mElement.get(); }
00119 
00120   // This is called when the referenced resource changes.
00121   virtual void DoUpdate() MOZ_OVERRIDE;
00122 
00123   class SourceReference : public nsReferencedElement {
00124   public:
00125     SourceReference(nsSVGIDRenderingObserver* aContainer) : mContainer(aContainer) {}
00126   protected:
00127     virtual void ElementChanged(Element* aFrom, Element* aTo) MOZ_OVERRIDE {
00128       mContainer->StopListening();
00129       nsReferencedElement::ElementChanged(aFrom, aTo);
00130       mContainer->StartListening();
00131       mContainer->DoUpdate();
00132     }
00137     virtual bool IsPersistent() MOZ_OVERRIDE { return true; }
00138   private:
00139     nsSVGIDRenderingObserver* mContainer;
00140   };
00141   
00142   SourceReference mElement;
00143   // The frame that this property is attached to
00144   nsIFrame *mFrame;
00145   // When a presshell is torn down, we don't delete the properties for
00146   // each frame until after the frames are destroyed. So here we remember
00147   // the presshell for the frames we care about and, before we use the frame,
00148   // we test the presshell to see if it's destroying itself. If it is,
00149   // then the frame pointer is not valid and we know the frame has gone away.
00150   nsIPresShell *mFramePresShell;
00151 };
