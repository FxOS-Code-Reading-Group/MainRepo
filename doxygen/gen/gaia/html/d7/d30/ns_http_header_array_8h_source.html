00145 inline void
00146 nsHttpHeaderArray::MergeHeader(nsHttpAtom header,
00147                                nsEntry *entry,
00148                                const nsACString &value)
00149 {
00150     if (value.IsEmpty())
00151         return;   // merge of empty header = no-op
00152 
00153     // Append the new value to the existing value
00154     if (header == nsHttp::Set_Cookie ||
00155         header == nsHttp::WWW_Authenticate ||
00156         header == nsHttp::Proxy_Authenticate)
00157     {
00158         // Special case these headers and use a newline delimiter to
00159         // delimit the values from one another as commas may appear
00160         // in the values of these headers contrary to what the spec says.
00161         entry->value.Append('\n');
00162     } else {
00163         // Delimit each value from the others using a comma (per HTTP spec)
00164         entry->value.AppendLiteral(", ");
00165     }
00166     entry->value.Append(value);
00167 }
