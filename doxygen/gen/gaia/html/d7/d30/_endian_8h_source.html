00007 /* Functions for reading and writing integers in various endiannesses. */
00008 
00009 /*
00010  * The classes LittleEndian and BigEndian expose static methods for
00011  * reading and writing 16-, 32-, and 64-bit signed and unsigned integers
00012  * in their respective endianness.  The naming scheme is:
00013  *
00014  * {Little,Big}Endian::{read,write}{Uint,Int}<bitsize>
00015  *
00016  * For instance, LittleEndian::readInt32 will read a 32-bit signed
00017  * integer from memory in little endian format.  Similarly,
00018  * BigEndian::writeUint16 will write a 16-bit unsigned integer to memory
00019  * in big-endian format.
00020  *
00021  * The class NativeEndian exposes methods for conversion of existing
00022  * data to and from the native endianness.  These methods are intended
00023  * for cases where data needs to be transferred, serialized, etc.
00024  * swap{To,From}{Little,Big}Endian byteswap a single value if necessary.
00025  * Bulk conversion functions are also provided which optimize the
00026  * no-conversion-needed case:
00027  *
00028  * - copyAndSwap{To,From}{Little,Big}Endian;
00029  * - swap{To,From}{Little,Big}EndianInPlace.
00030  *
00031  * The *From* variants are intended to be used for reading data and the
00032  * *To* variants for writing data.
00033  *
00034  * Methods on NativeEndian work with integer data of any type.
00035  * Floating-point data is not supported.
00036  *
00037  * For clarity in networking code, "Network" may be used as a synonym
00038  * for "Big" in any of the above methods or class names.
00039  *
00040  * As an example, reading a file format header whose fields are stored
00041  * in big-endian format might look like:
00042  *
00043  * class ExampleHeader
00044  * {
00045  *   private:
00046  *     uint32_t magic;
00047  *     uint32_t length;
00048  *     uint32_t totalRecords;
00049  *     uint64_t checksum;
00050  *
00051  *   public:
00052  *     ExampleHeader(const void* data) {
00053  *       const uint8_t* ptr = static_cast<const uint8_t*>(data);
00054  *       magic = BigEndian::readUint32(ptr); ptr += sizeof(uint32_t);
00055  *       length = BigEndian::readUint32(ptr); ptr += sizeof(uint32_t);
00056  *       totalRecords = BigEndian::readUint32(ptr); ptr += sizeof(uint32_t);
00057  *       checksum = BigEndian::readUint64(ptr);
00058  *     }
00059  *     ...
00060  * };
00061  */

00288     static void copyAndSwapTo(void* dest, const T* src, size_t count)
00289     {
00290       assertNoOverlap(dest, src, count * sizeof(T));
00291       assertAligned(src);
00292 
00293       if (SourceEndian == DestEndian) {
00294         memcpy(dest, src, count * sizeof(T));

00297 
00298       uint8_t* byteDestPtr = static_cast<uint8_t*>(dest);
00299       for (size_t i = 0; i < count; ++i) {
00300         union {
00301           T val;

00303         } u;
00304         u.val = maybeSwap<SourceEndian, DestEndian>(src[i]);
00305         memcpy(byteDestPtr, u.buffer, sizeof(T));
00306         byteDestPtr += sizeof(T);
00307       }
