00904     const SkRect& computeFastBounds(const SkRect& orig, SkRect* storage) const {
00905         SkPaint::Style style = this->getStyle();
00906         // ultra fast-case: filling with no effects that affect geometry
00907         if (kFill_Style == style) {
00908             uintptr_t effects = reinterpret_cast<uintptr_t>(this->getLooper());
00909             effects |= reinterpret_cast<uintptr_t>(this->getMaskFilter());
00910             effects |= reinterpret_cast<uintptr_t>(this->getPathEffect());
00911             if (!effects) {
00912                 return orig;
00913             }
00914         }
00915 
00916         return this->doComputeFastBounds(orig, storage, style);
00917     }
