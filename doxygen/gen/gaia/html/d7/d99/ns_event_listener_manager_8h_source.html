00193   MOZ_ALWAYS_INLINE bool IsListening(const nsEvent* aEvent) const
00194   {
00195     if (mFlags.mInSystemGroup != aEvent->mFlags.mInSystemGroup) {
00196       return false;
00197     }
00198     // FIXME Should check !mFlags.mCapture when the event is in target
00199     //       phase because capture phase event listeners should not be fired.
00200     //       But it breaks at least <xul:dialog>'s buttons. Bug 235441.
00201     return ((mFlags.mCapture && aEvent->mFlags.mInCapturePhase) ||
00202             (!mFlags.mCapture && aEvent->mFlags.mInBubblingPhase));
00203   }
