01311   /*
01312     If we perform a selection over the list of contacts
01313     and we don't have all the info yet, we send a promise
01314     to the select action.
01315   */
01316   var createSelectPromise = function createSelectPromise() {
01317     var promise = {
01318       canceled: false,
01319       _selected: [],
01320       resolved: false,
01321       successCb: null,
01322       errorCb: null,
01323       resolve: function resolve(values) {
01324         var self = this;
01325         setTimeout(function onResolve() {
01326           // If we have the values parameter we can directly
01327           // resolve this promise
01328           if (values) {
01329             exitSelectMode();
01330             self._selected = values;
01331             self.resolved = true;
01332             if (self.successCb) {
01333               self.successCb(values);
01334             }
01335             return;
01336           }
01337 
01338           // We don't know if we render all the contacts and their checks,
01339           // so fetch ALL the contacts and remove those one we un selected
01340           // remember this is a promise, so is already async.
01341           var notChecked = document.querySelectorAll(
01342             '[name="selectIds[]"]:not(:checked)');
01343           var notSelectedIds = {};
01344           for (var i = 0; i < notChecked.length; i++) {
01345             notSelectedIds[notChecked[i].value] = true;
01346           }
01347           var notSelectedCount = notChecked.length;
01348           var request = navigator.mozContacts.find({});
01349           request.onsuccess = function onAllContacts() {
01350             request.result.forEach(function onContact(contact) {
01351               if (notSelectedCount == 0 ||
01352                 notSelectedIds[contact.id] == undefined) {
01353                 self._selected.push(contact.id);
01354               }
01355             });
01356 
01357             self.resolved = true;
01358             exitSelectMode();
01359             if (self.successCb) {
01360               self.successCb(self._selected);
01361             }
01362           };
01363           request.onerror = function onError() {
01364             exitSelectMode();
01365             self.reject();
01366           };
01367         }, 0);
01368       },
01369       reject: function reject() {
01370         this.canceled = true;
01371         exitSelectMode();
01372 
01373         if (this.errorCb) {
01374           this.errorCb();
01375         }
01376       },
01377       set onsuccess(callback) {
01378         if (this.resolved) {
01379           callback(this._selected);
01380         } else {
01381           this.successCb = callback;
01382         }
01383       },
01384       set onerror(callback) {
01385         if (this.canceled) {
01386           callback();
01387         } else {
01388           this.errorCb = callback;
01389         }
01390       }
01391     };
01392 
01393     return promise;
