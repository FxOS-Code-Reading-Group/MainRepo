00020 /*
00021  * It's reasonable to ask why we have this header at all.  Don't isnan,
00022  * copysign, the built-in comparison operators, and the like solve these
00023  * problems?  Unfortunately, they don't.  We've found that various compilers
00024  * (MSVC, MSVC when compiling with PGO, and GCC on OS X, at least) miscompile
00025  * the standard methods in various situations, so we can't use them.  Some of
00026  * these compilers even have problems compiling seemingly reasonable bitwise
00027  * algorithms!  But with some care we've found algorithms that seem to not
00028  * trigger those compiler bugs.
00029  *
00030  * For the aforementioned reasons, be very wary of making changes to any of
00031  * these algorithms.  If you must make changes, keep a careful eye out for
00032  * compiler bustage, particularly PGO-specific bustage.
00033  */

00035 /*
00036  * These implementations all assume |double| is a 64-bit double format number
00037  * type, compatible with the IEEE-754 standard.  C/C++ don't require this to be
00038  * the case.  But we required this in implementations of these algorithms that
00039  * preceded this header, so we shouldn't break anything if we continue doing so.
00040  */
00041 static_assert(sizeof(double) == sizeof(uint64_t), "double must be 64 bits");

00043 const unsigned DoubleExponentBias = 1023;
00044 const unsigned DoubleExponentShift = 52;
00045 
00046 const uint64_t DoubleSignBit         = 0x8000000000000000ULL;
00047 const uint64_t DoubleExponentBits    = 0x7ff0000000000000ULL;
00048 const uint64_t DoubleSignificandBits = 0x000fffffffffffffULL;
00049 
00050 static_assert((DoubleSignBit & DoubleExponentBits) == 0,
00051               "sign bit doesn't overlap exponent bits");
00052 static_assert((DoubleSignBit & DoubleSignificandBits) == 0,
00053               "sign bit doesn't overlap significand bits");
00054 static_assert((DoubleExponentBits & DoubleSignificandBits) == 0,
00055               "exponent bits don't overlap significand bits");
00056 
00057 static_assert((DoubleSignBit | DoubleExponentBits | DoubleSignificandBits) ==
00058               ~uint64_t(0),
00059               "all bits accounted for");

00062 static MOZ_ALWAYS_INLINE bool
00063 IsNaN(double d)
00064 {
00065   /*
00066    * A double is NaN if all exponent bits are 1 and the significand contains at
00067    * least one non-zero bit.
00068    */
00069   uint64_t bits = BitwiseCast<uint64_t>(d);
00070   return (bits & DoubleExponentBits) == DoubleExponentBits &&
00071          (bits & DoubleSignificandBits) != 0;
00072 }
00073 
00075 static MOZ_ALWAYS_INLINE bool
00076 IsInfinite(double d)
00077 {
00078   /* Infinities have all exponent bits set to 1 and an all-0 significand. */
00079   uint64_t bits = BitwiseCast<uint64_t>(d);
00080   return (bits & ~DoubleSignBit) == DoubleExponentBits;
00081 }
00082 
00084 static MOZ_ALWAYS_INLINE bool
00085 IsFinite(double d)
00086 {
00087   /*
00088    * NaN and Infinities are the only non-finite doubles, and both have all
00089    * exponent bits set to 1.
00090    */
00091   uint64_t bits = BitwiseCast<uint64_t>(d);
00092   return (bits & DoubleExponentBits) != DoubleExponentBits;
00093 }

00099 static MOZ_ALWAYS_INLINE bool
00100 IsNegative(double d)
00101 {
00102   MOZ_ASSERT(!IsNaN(d), "NaN does not have a sign");
00103 
00104   /* The sign bit is set if the double is negative. */
00105   uint64_t bits = BitwiseCast<uint64_t>(d);
00106   return (bits & DoubleSignBit) != 0;
00107 }

00110 static MOZ_ALWAYS_INLINE bool
00111 IsNegativeZero(double d)
00112 {
00113   /* Only the sign bit is set if the double is -0. */
00114   uint64_t bits = BitwiseCast<uint64_t>(d);
00115   return bits == DoubleSignBit;
00116 }

00119 static MOZ_ALWAYS_INLINE int_fast16_t
00120 ExponentComponent(double d)
00121 {
00122   /*
00123    * The exponent component of a double is an unsigned number, biased from its
00124    * actual value.  Subtract the bias to retrieve the actual exponent.
00125    */
00126   uint64_t bits = BitwiseCast<uint64_t>(d);
00127   return int_fast16_t((bits & DoubleExponentBits) >> DoubleExponentShift) -
00128          int_fast16_t(DoubleExponentBias);
00129 }

00154 static MOZ_ALWAYS_INLINE double
00155 SpecificNaN(int signbit, uint64_t significand)
00156 {
00157   MOZ_ASSERT(signbit == 0 || signbit == 1);
00158   MOZ_ASSERT((significand & ~DoubleSignificandBits) == 0);
00159   MOZ_ASSERT(significand & DoubleSignificandBits);
00160 
00161   double d = BitwiseCast<double>((signbit ? DoubleSignBit : 0) |
00162                                  DoubleExponentBits |
00163                                  significand);
00164   MOZ_ASSERT(IsNaN(d));
00165   return d;
