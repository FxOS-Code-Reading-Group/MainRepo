00021 /*
00022 ** Allocate an item.  If "arena" is not NULL, then allocate from there,
00023 ** otherwise allocate from the heap.  If "item" is not NULL, allocate
00024 ** only the data buffer for the item, not the item itself.  If "len" is
00025 ** 0, do not allocate the data buffer for the item; simply set the data
00026 ** field to NULL and the len field to 0.  The item structure is allocated
00027 ** zero-filled; the data buffer is not zeroed.  The caller is responsible
00028 ** for initializing the type field of the item.
00029 **
00030 ** The resulting item is returned; NULL if any error occurs.
00031 **
00032 ** XXX This probably should take a SECItemType, but since that is mostly
00033 ** unused and our improved APIs (aka Stan) are looming, I left it out.
00034 */
00035 extern SECItem *SECITEM_AllocItem(PLArenaPool *arena, SECItem *item,
00036                   unsigned int len);
