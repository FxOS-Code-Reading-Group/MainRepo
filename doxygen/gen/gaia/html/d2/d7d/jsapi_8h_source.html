00663 /*
00664  * Methods usually act upon |this| objects only from a single global object and
00665  * compartment.  Sometimes, however, a method must act upon |this| values from
00666  * multiple global objects or compartments.  In such cases the |this| value a
00667  * method might see will be wrapped, such that various access to the object --
00668  * to its class, its private data, its reserved slots, and so on -- will not
00669  * work properly without entering that object's compartment.  This method
00670  * implements a solution to this problem.
00671  *
00672  * To implement a method that accepts |this| values from multiple compartments,
00673  * define two functions.  The first function matches the IsAcceptableThis type
00674  * and indicates whether the provided value is an acceptable |this| for the
00675  * method; it must be a pure function only of its argument.
00676  *
00677  *   static JSClass AnswerClass = { ... };
00678  *
00679  *   static bool
00680  *   IsAnswerObject(const Value &v)
00681  *   {
00682  *       if (!v.isObject())
00683  *           return false;
00684  *       return JS_GetClass(&v.toObject()) == &AnswerClass;
00685  *   }
00686  *
00687  * The second function implements the NativeImpl signature and defines the
00688  * behavior of the method when it is provided an acceptable |this| value.
00689  * Aside from some typing niceties -- see the CallArgs interface for details --
00690  * its interface is the same as that of JSNative.
00691  *
00692  *   static bool
00693  *   answer_getAnswer_impl(JSContext *cx, JS::CallArgs args)
00694  *   {
00695  *       args.rval().setInt32(42);
00696  *       return true;
00697  *   }
00698  *
00699  * The implementation function is guaranteed to be called *only* with a |this|
00700  * value which is considered acceptable.
00701  *
00702  * Now to implement the actual method, write a JSNative that calls the method
00703  * declared below, passing the appropriate template and runtime arguments.
00704  *
00705  *   static JSBool
00706  *   answer_getAnswer(JSContext *cx, unsigned argc, JS::Value *vp)
00707  *   {
00708  *       JS::CallArgs args = JS::CallArgsFromVp(argc, vp);
00709  *       return JS::CallNonGenericMethod<IsAnswerObject, answer_getAnswer_impl>(cx, args);
00710  *   }
00711  *
00712  * Note that, because they are used as template arguments, the predicate
00713  * and implementation functions must have external linkage. (This is
00714  * unfortunate, but GCC wasn't inlining things as one would hope when we
00715  * passed them as function arguments.)
00716  *
00717  * JS::CallNonGenericMethod will test whether |args.thisv()| is acceptable.  If
00718  * it is, it will call the provided implementation function, which will return
00719  * a value and indicate success.  If it is not, it will attempt to unwrap
00720  * |this| and call the implementation function on the unwrapped |this|.  If
00721  * that succeeds, all well and good.  If it doesn't succeed, a TypeError will
00722  * be thrown.
00723  *
00724  * Note: JS::CallNonGenericMethod will only work correctly if it's called in
00725  *       tail position in a JSNative.  Do not call it from any other place.
00726  */
00727 template<IsAcceptableThis Test, NativeImpl Impl>
00728 JS_ALWAYS_INLINE bool
00729 CallNonGenericMethod(JSContext *cx, CallArgs args)
00730 {
00731     const Value &thisv = args.thisv();
00732     if (Test(thisv))
00733         return Impl(cx, args);
00734 
00735     return detail::CallMethodIfWrapped(cx, Test, Impl, args);
00736 }
00737 
00738 JS_ALWAYS_INLINE bool
00739 CallNonGenericMethod(JSContext *cx, IsAcceptableThis Test, NativeImpl Impl, CallArgs args)
00740 {
00741     const Value &thisv = args.thisv();
00742     if (Test(thisv))
00743         return Impl(cx, args);
00744 
00745     return detail::CallMethodIfWrapped(cx, Test, Impl, args);
00746 }

02088  * Compartments are described in:
02089  *
02090  *   developer.mozilla.org/en-US/docs/SpiderMonkey/SpiderMonkey_compartments
02091  *
02092  * The current compartment of a context may be changed. The preferred way to do
02093  * this is with JSAutoCompartment:

03242 // During global creation, we fire notifications to callbacks registered
03243 // via the Debugger API. These callbacks are arbitrary script, and can touch
03244 // the global in arbitrary ways. When that happens, the global should not be
03245 // in a half-baked state. But this creates a problem for consumers that need
03246 // to set slots on the global to put it in a consistent state.
03247 //
03248 // This API provides a way for consumers to set slots atomically (immediately
03249 // after the global is created), before any debugger hooks are fired. It's
03250 // unfortunately on the clunky side, but that's the way the cookie crumbles.
03251 //
03252 // If callers have no additional state on the global to set up, they may pass
03253 // |FireOnNewGlobalHook| to JS_NewGlobalObject, which causes that function to
03254 // fire the hook as its final act before returning. Otherwise, callers should
03255 // pass |DontFireOnNewGlobalHook|, which means that they are responsible for
03256 // invoking JS_FireOnNewGlobalObject upon successfully creating the global. If
03257 // an error occurs and the operation aborts, callers should skip firing the
03258 // hook. But otherwise, callers must take care to fire the hook exactly once
03259 // before compiling any script in the global's scope (we have assertions in
03260 // place to enforce this). This lets us be sure that debugger clients never miss
03261 // breakpoints.
03262 enum OnNewGlobalHookOption {
03263     FireOnNewGlobalHook,
03264     DontFireOnNewGlobalHook
03265 };

04117 /*
04118  * NB: JS_ExecuteScript and the JS_Evaluate*Script* quadruplets use the obj
04119  * parameter as the initial scope chain header, the 'this' keyword value, and
04120  * the variables object (ECMA parlance for where 'var' and 'function' bind
04121  * names) of the execution context for script.
04122  *
04123  * Using obj as the variables object is problematic if obj's parent (which is
04124  * the scope chain link; see JS_SetParent and JS_NewObject) is not null: in
04125  * this case, variables created by 'var x = 0', e.g., go in obj, but variables
04126  * created by assignment to an unbound id, 'x = 0', go in the last object on
04127  * the scope chain linked by parent.
04128  *
04129  * ECMA calls that last scoping object the "global object", but note that many
04130  * embeddings have several such objects.  ECMA requires that "global code" be
04131  * executed with the variables object equal to this global object.  But these
04132  * JS API entry points provide freedom to execute code against a "sub-global",
04133  * i.e., a parented or scoped object, in which case the variables object will
04134  * differ from the last object on the scope chain, resulting in confusing and
04135  * non-ECMA explicit vs. implicit variable creation.
04136  *
04137  * Caveat embedders: unless you already depend on this buggy variables object
04138  * binding behavior, you should call JS_SetOptions(cx, JSOPTION_VAROBJFIX) or
04139  * JS_SetOptions(cx, JS_GetOptions(cx) | JSOPTION_VAROBJFIX) -- the latter if
04140  * someone may have set other options on cx already -- for each context in the
04141  * application, if you pass parented objects as the obj parameter, or may ever
04142  * pass such objects in the future.
04143  *
04144  * Why a runtime option?  The alternative is to add six or so new API entry
04145  * points with signatures matching the following six, and that doesn't seem
04146  * worth the code bloat cost.  Such new entry points would probably have less
04147  * obvious names, too, so would not tend to be used.  The JS_SetOption call,
04148  * OTOH, can be more easily hacked into existing code that does not depend on
04149  * the bug; such code can continue to use the familiar JS_EvaluateScript,
04150  * etc., entry points.
04151  */

04153 JS_ExecuteScript(JSContext *cx, JSObject *obj, JSScript *script, jsval *rval);
