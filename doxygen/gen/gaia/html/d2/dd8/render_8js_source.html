00325           function fitText(container, text) {
00326             container.textContent = '';
00327             if (!text)
00328               return null;
00329             var span = document.createElement('span');
00330             span.textContent = text;
00331             container.appendChild(span);
00332 
00333             // Measure the width of the element, and return the scale that
00334             // we can use to make it fit in the container. The return values
00335             // are restricted to a set that matches the standard font sizes
00336             // we use in Gaia.
00337             //
00338             // Note that this only works if the element is display:inline
00339             function getScale(element, container) {
00340               var elementWidth = element.getBoundingClientRect().width;
00341               var s = container.clientWidth / elementWidth;
00342               if (s >= 1)
00343                 return 1;    // 10pt font "Body Large"
00344               if (s >= .8)
00345                 return .8;   // 8pt font "Body"
00346               if (s >= .7)
00347                 return .7;   // 7pt font "Body Medium"
00348               if (s >= .65)
00349                 return .65;  // 6.5pt font "Body Small"
00350               if (s >= .6)
00351                 return .6;   // 6pt font "Body Mini"
00352               return s;      // Something smaller than 6pt.
00353             }
00354 
00355             var limit = .6;  // Dont use a scale smaller than this
00356             var scale = getScale(span, container);

00358             // If the text does not fit within the scaling limit,

00360             // the middle with ...

00362               var charactersReplaced = text.length % 2;
00363               while (scale < limit && charactersReplaced < text.length - 2) {
00364                 charactersReplaced += 2;
00365                 var halflen = (text.length - charactersReplaced) / 2;
00366                 span.textContent = text.substring(0, halflen) +

00368                   text.substring(text.length - halflen);
