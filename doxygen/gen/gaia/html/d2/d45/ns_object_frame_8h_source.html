00121   {
00122     if (mWidget) {

00126         // geometry updates and this should not be called. But apparently we
00127         // have bugs where mWidget sometimes is toplevel here. Bail out.
00128         NS_ERROR("Plugin widgets registered for geometry updates should not be toplevel");
00129         return;
00130       }
00131       nsIWidget::Configuration* configuration = aConfigurations->AppendElement();
00132       configuration->mChild = mWidget;
00133       configuration->mBounds = mNextConfigurationBounds;
00134       configuration->mClipRegion = mNextConfigurationClipRegion;
00135     }
00136   }
