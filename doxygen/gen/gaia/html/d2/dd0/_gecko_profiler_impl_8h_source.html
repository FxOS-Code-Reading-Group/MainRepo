00211 #if defined(PLATFORM_LIKELY_MEMORY_CONSTRAINED)
00212 /* A 1ms sampling interval has been shown to be a large perf hit
00213  * (10fps) on memory-contrained (low-end) platforms, and additionally
00214  * to yield different results from the profiler.  Where this is the
00215  * important case, b2g, there are also many gecko processes which
00216  * magnify these effects. */
00217 # define PROFILE_DEFAULT_INTERVAL 10
00218 #elif defined(ANDROID)
00219 // We use a lower frequency on Android, in order to make things work
00220 // more smoothly on phones.  This value can be adjusted later with
00221 // some libunwind optimizations.
00222 // In one sample measurement on Galaxy Nexus, out of about 700 backtraces,
00223 // 60 of them took more than 25ms, and the average and standard deviation
00224 // were 6.17ms and 9.71ms respectively.
00225 
00226 // For now since we don't support stackwalking let's use 1ms since it's fast
00227 // enough.
00228 #define PROFILE_DEFAULT_INTERVAL 1

00292 inline void* mozilla_sampler_call_enter(const char *aInfo, void *aFrameAddress,
00293                                         bool aCopy, uint32_t line)
00294 {
00295   // check if we've been initialized to avoid calling pthread_getspecific
00296   // with a null tlsStack which will return undefined results.
00297   if (!stack_key_initialized)
00298     return NULL;
00299 
00300   PseudoStack *stack = tlsPseudoStack.get();
00301   // we can't infer whether 'stack' has been initialized
00302   // based on the value of stack_key_intiailized because
00303   // 'stack' is only intialized when a thread is being
00304   // profiled.
00305   if (!stack) {
00306     return stack;
00307   }
00308   stack->push(aInfo, aFrameAddress, aCopy, line);
00309 
00310   // The handle is meant to support future changes
00311   // but for now it is simply use to save a call to
00312   // pthread_getspecific on exit. It also supports the
00313   // case where the sampler is initialized between
00314   // enter and exit.
00315   return stack;
00316 }

00327 inline void mozilla_sampler_add_marker(const char *aMarker)
00328 {
00329   if (!stack_key_initialized)
00330     return;
00331 
00332   // Don't insert a marker if we're not profiling to avoid
00333   // the heap copy (malloc).
00334   if (!profiler_is_active()) {
00335     return;
00336   }
00337 
00338   // Don't add a marker if we don't want to include personal information
00339   if (profiler_in_privacy_mode()) {
00340     return;
00341   }
00342 
00343   PseudoStack *stack = tlsPseudoStack.get();
00344   if (!stack) {
00345     return;
00346   }
00347   stack->addMarker(aMarker);
00348 }
