00007 /* A number of structures to simplify scope-based RAII management. */
00008

00012 /*
00013  * Resource Acquisition Is Initialization is a programming idiom used
00014  * to write robust code that is able to deallocate resources properly,
00015  * even in presence of execution errors or exceptions that need to be
00016  * propagated.  The Scoped* classes defined in this header perform the
00017  * deallocation of the resource they hold once program execution
00018  * reaches the end of the scope for which they have been defined.
00019  *
00020  * This header provides the following RAII classes:
00021  *
00022  * - |ScopedFreePtr| - a container for a pointer, that automatically calls
00023  *   |free()| at the end of the scope;
00024  * - |ScopedDeletePtr| - a container for a pointer, that automatically calls
00025  *   |delete| at the end of the scope;
00026  * - |ScopedDeleteArray| - a container for a pointer to an array, that
00027  *   automatically calls |delete[]| at the end of the scope.
00028  *
00029  * The general scenario for each of the RAII classes is the following:
00030  *
00031  * ScopedClass foo(create_value());
00032  * // ... In this scope, |foo| is defined. Use |foo.get()| or |foo.rwget()|
00033  *        to access the value.
00034  * // ... In case of |return| or |throw|, |foo| is deallocated automatically.
00035  * // ... If |foo| needs to be returned or stored, use |foo.forget()|
00036  *
00037  * Note that the RAII classes defined in this header do _not_ perform any form
00038  * of reference-counting or garbage-collection. These classes have exactly two
00039  * behaviors:
00040  *
00041  * - if |forget()| has not been called, the resource is always deallocated at
00042  *   the end of the scope;
00043  * - if |forget()| has been called, any control on the resource is unbound
00044  *   and the resource is not deallocated by the class.
00045  *
00046  * Extension:
00047  *
00048  * In addition, this header provides class |Scoped| and macros |SCOPED_TEMPLATE|
00049  * and |MOZ_TYPE_SPECIFIC_SCOPED_POINTER_TEMPLATE|  to simplify the definition
00050  * of RAII classes for other scenarios. These macros have been used to
00051  * automatically close file descriptors/file handles when reaching the end of
00052  * the scope, graphics contexts, etc.
00053  */
00054

00061 /*
00062  * Scoped is a helper to create RAII wrappers
00063  * Type argument |Traits| is expected to have the following structure:
00064  *
00065  *   struct Traits {
00066  *     // Define the type of the value stored in the wrapper
00067  *     typedef value_type type;
00068  *     // Returns the value corresponding to the uninitialized or freed state
00069  *     const static type empty();
00070  *     // Release resources corresponding to the wrapped value
00071  *     // This function is responsible for not releasing an |empty| value
00072  *     const static void release(type);
00073  *   }
00074  */
00075 template<typename Traits>
00076 class Scoped
00077 {
00078   public:
00079     typedef typename Traits::type Resource;
00080 
00081     explicit Scoped(MOZ_GUARD_OBJECT_NOTIFIER_ONLY_PARAM)
00082       : value(Traits::empty())
00083     {
00084       MOZ_GUARD_OBJECT_NOTIFIER_INIT;
00085     }
00086     explicit Scoped(const Resource& v
00087                     MOZ_GUARD_OBJECT_NOTIFIER_PARAM)
00088       : value(v)
00089     {
00090       MOZ_GUARD_OBJECT_NOTIFIER_INIT;
00091     }
00092     ~Scoped() {
00093       Traits::release(value);
00094     }
00095 
00096     // Constant getter
00097     operator const Resource&() const { return value; }
00098     const Resource& operator->() const { return value; }
00099     const Resource& get() const { return value; }
00100     // Non-constant getter.
00101     Resource& rwget() { return value; }
00102 
00103     /*
00104      * Forget the resource.
00105      *
00106      * Once |forget| has been called, the |Scoped| is neutralized, i.e. it will
00107      * have no effect at destruction (unless it is reset to another resource by
00108      * |operator=|).
00109      *
00110      * @return The original resource.
00111      */
00112     Resource forget() {
00113       Resource tmp = value;
00114       value = Traits::empty();
00115       return tmp;
00116     }
00117 
00118     /*
00119      * Perform immediate clean-up of this |Scoped|.
00120      *
00121      * If this |Scoped| is currently empty, this method has no effect.
00122      */
00123     void dispose() {
00124       Traits::release(value);
00125       value = Traits::empty();
00126     }
00127 
00128     bool operator==(const Resource& other) const {
00129       return value == other;
00130     }
00131 
00132     /*
00133      * Replace the resource with another resource.
00134      *
00135      * Calling |operator=| has the side-effect of triggering clean-up. If you do
00136      * not want to trigger clean-up, you should first invoke |forget|.
00137      *
00138      * @return this
00139      */
00140     Scoped<Traits>& operator=(const Resource& other) {
00141       return reset(other);
00142     }
00143     Scoped<Traits>& reset(const Resource& other) {
00144       Traits::release(value);
00145       value = other;
00146       return *this;
00147     }
00148 
00149   private:
00150     explicit Scoped(const Scoped<Traits>& value) MOZ_DELETE;
00151     Scoped<Traits>& operator=(const Scoped<Traits>& value) MOZ_DELETE;
00152 
00153   private:
00154     Resource value;
00155     MOZ_DECL_USE_GUARD_OBJECT_NOTIFIER
00156 };

00158 /*
00159  * SCOPED_TEMPLATE defines a templated class derived from Scoped
00160  * This allows to implement templates such as ScopedFreePtr.
00161  *
00162  * @param name The name of the class to define.
00163  * @param Traits A struct implementing clean-up. See the implementations
00164  * for more details.
00165  */
00166 #define SCOPED_TEMPLATE(name, Traits)                          \
00167 template<typename Type>                                        \
00168 struct name : public mozilla::Scoped<Traits<Type> >            \
00169 {                                                              \
00170     typedef mozilla::Scoped<Traits<Type> > Super;              \
00171     typedef typename Super::Resource Resource;                 \
00172     name& operator=(Resource ptr) {                            \
00173       Super::operator=(ptr);                                   \
00174       return *this;                                            \
00175     }                                                          \
00176     explicit name(MOZ_GUARD_OBJECT_NOTIFIER_ONLY_PARAM)        \
00177       : Super(MOZ_GUARD_OBJECT_NOTIFIER_ONLY_PARAM_TO_PARENT)  \
00178     {}                                                         \
00179     explicit name(Resource ptr                                 \
00180                   MOZ_GUARD_OBJECT_NOTIFIER_PARAM)             \
00181       : Super(ptr MOZ_GUARD_OBJECT_NOTIFIER_PARAM_TO_PARENT)   \
00182     {}                                                         \
00183   private:                                                     \
00184     explicit name(name& source) MOZ_DELETE;                    \
00185     name& operator=(name& source) MOZ_DELETE;                  \
00186 };

00188 /*
00189  * ScopedFreePtr is a RAII wrapper for pointers that need to be free()d.
00190  *
00191  *   struct S { ... };
00192  *   ScopedFreePtr<S> foo = malloc(sizeof(S));
00193  *   ScopedFreePtr<char> bar = strdup(str);
00194  */
00195 template<typename T>
00196 struct ScopedFreePtrTraits
00197 {
00198     typedef T* type;
00199     static T* empty() { return nullptr; }
00200     static void release(T* ptr) { free(ptr); }
00201 };
00202 SCOPED_TEMPLATE(ScopedFreePtr, ScopedFreePtrTraits)

00230 /*
00231  * MOZ_TYPE_SPECIFIC_SCOPED_POINTER_TEMPLATE makes it easy to create scoped
00232  * pointers for types with custom deleters; just overload
00233  * TypeSpecificDelete(T*) in the same namespace as T to call the deleter for
00234  * type T.
00235  *
00236  * @param name The name of the class to define.
00237  * @param Type A struct implementing clean-up. See the implementations
00238  * for more details.
00239  * *param Deleter The function that is used to delete/destroy/free a
00240  *        non-null value of Type*.
00241  *
00242  * Example:
00243  *
00244  *   MOZ_TYPE_SPECIFIC_SCOPED_POINTER_TEMPLATE(ScopedPRFileDesc, PRFileDesc, \
00245  *                                             PR_Close)
00246  *   ...
00247  *   {
00248  *       ScopedPRFileDesc file(PR_OpenFile(...));
00249  *       ...
00250  *   } // file is closed with PR_Close here
00251  */
00252 #define MOZ_TYPE_SPECIFIC_SCOPED_POINTER_TEMPLATE(name, Type, Deleter) \
00253 template <> inline void TypeSpecificDelete(Type * value) { Deleter(value); } \
00254 typedef ::mozilla::TypeSpecificScopedPointer<Type> name;
00255 
00256 template <typename T> void TypeSpecificDelete(T * value);
00257 
00258 template <typename T>
00259 struct TypeSpecificScopedPointerTraits
00260 {
00261     typedef T* type;
00262     const static type empty() { return nullptr; }
00263     const static void release(type value)
00264     {
00265       if (value)
00266         TypeSpecificDelete(value);
00267     }
00268 };
00269
