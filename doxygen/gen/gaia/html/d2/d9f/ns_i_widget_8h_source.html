00346 struct InputContextAction {
00351   enum Cause {
00352     // The cause is unknown but originated from content. Focus might have been
00353     // changed by content script.
00354     CAUSE_UNKNOWN,
00355     // The cause is unknown but originated from chrome. Focus might have been
00356     // changed by chrome script.
00357     CAUSE_UNKNOWN_CHROME,
00358     // The cause is user's keyboard operation.
00359     CAUSE_KEY,
00360     // The cause is user's mouse operation.
00361     CAUSE_MOUSE
00362   };
00363   Cause mCause;
00364 
00368   enum FocusChange {
00369     FOCUS_NOT_CHANGED,
00370     // A content got focus.
00371     GOT_FOCUS,
00372     // Focused content lost focus.
00373     LOST_FOCUS,
00374     // Menu got pseudo focus that means focused content isn't changed but
00375     // keyboard events will be handled by menu.
00376     MENU_GOT_PSEUDO_FOCUS,
00377     // Menu lost pseudo focus that means focused content will handle keyboard
00378     // events.
00379     MENU_LOST_PSEUDO_FOCUS
00380   };
00381   FocusChange mFocusChange;
00382 
00383   bool ContentGotFocusByTrustedCause() const {
00384     return (mFocusChange == GOT_FOCUS &&
00385             mCause != CAUSE_UNKNOWN);
00386   }
00387 
00388   bool UserMightRequestOpenVKB() const {
00389     return (mFocusChange == FOCUS_NOT_CHANGED &&
00390             mCause == CAUSE_MOUSE);
00391   }
00392 
00393   InputContextAction() :
00394     mCause(CAUSE_UNKNOWN), mFocusChange(FOCUS_NOT_CHANGED)
00395   {
00396   }
00397

00399                      FocusChange aFocusChange = FOCUS_NOT_CHANGED) :
