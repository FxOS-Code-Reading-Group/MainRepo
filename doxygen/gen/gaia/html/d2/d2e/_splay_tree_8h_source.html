00099     {

00101       MOZ_ASSERT(last, "This tree must contain the element being removed.");
00102       MOZ_ASSERT(Comparator::compare(v, *last) == 0);
00103 
00104       // Splay the tree so that the item to remove is the root.
00105       splay(last);
00106       MOZ_ASSERT(last == root);

00110       // root's right.

00112       T* swapChild;

00127       }

00194     {
00195       MOZ_ASSERT(node);
00196 
00197       while (node != root) {

00218     void rotate(T* node)
00219     {
00220       // Rearrange nodes so that node becomes the parent of its current
00221       // parent, while preserving the sortedness of the tree.
00222       T* parent = node->parent;
00223       if (parent->left == node) {
00224         //     x          y
00225         //   y  c  ==>  a  x
00226         //  a b           b c
00227         parent->left = node->right;
00228         if (node->right)
00229           node->right->parent = parent;
00230         node->right = parent;
00231       } else {
00232         MOZ_ASSERT(parent->right == node);
00233         //   x             y
00234         //  a  y   ==>   x  c
00235         //    b c       a b
00236         parent->right = node->left;
00237         if (node->left)
00238           node->left->parent = parent;
00239         node->left = parent;
00240       }

00242       parent->parent = node;

00244         if (grandparent->left == parent)
