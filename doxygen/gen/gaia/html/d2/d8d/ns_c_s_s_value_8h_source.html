00035 // Deletes a linked list iteratively to avoid blowing up the stack (bug 456196).
00036 #define NS_CSS_DELETE_LIST_MEMBER(type_, ptr_, member_)                        \
00037   {                                                                            \
00038     type_ *cur = (ptr_)->member_;                                              \
00039     (ptr_)->member_ = nullptr;                                                 \
00040     while (cur) {                                                              \
00041       type_ *dlm_next = cur->member_;                                          \
00042       cur->member_ = nullptr;                                                  \
00043       delete cur;                                                              \
00044       cur = dlm_next;                                                          \
00045     }                                                                          \
00046   }

00048 // Clones a linked list iteratively to avoid blowing up the stack.
00049 // If it fails to clone the entire list then 'to_' is deleted and
00050 // we return null.
00051 #define NS_CSS_CLONE_LIST_MEMBER(type_, from_, member_, to_, args_)            \
00052   {                                                                            \
00053     type_ *dest = (to_);                                                       \
00054     (to_)->member_ = nullptr;                                                  \
00055     for (const type_ *src = (from_)->member_; src; src = src->member_) {       \
00056       type_ *clm_clone = src->Clone args_;                                     \
00057       if (!clm_clone) {                                                        \
00058         delete (to_);                                                          \
00059         return nullptr;                                                        \
00060       }                                                                        \
00061       dest->member_ = clm_clone;                                               \
00062       dest = clm_clone;                                                        \
00063     }                                                                          \
00064   }
