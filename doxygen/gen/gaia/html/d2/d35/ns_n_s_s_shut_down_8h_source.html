00166 /*
00167   A class deriving from nsNSSShutDownObject will have its instances
00168   automatically tracked in a list. However, it must follow some rules
00169   to assure correct behaviour.
00170   
00171   The tricky part is that it is not possible to call virtual
00172   functions from a destructor.
00173   
00174   The deriving class must override virtualDestroyNSSReference().
00175   Within this function, it should clean up all resources held to NSS.
00176   The function will be called by the global list, if it is time to
00177   shut down NSS before all references have been freed.
00178 
00179   The same code that goes into virtualDestroyNSSReference must
00180   also be called from the destructor of the deriving class,
00181   which is the standard cleanup (not called from the tracking list).
00182 
00183   Because of that duplication, it is suggested to implement a
00184   function destructorSafeDestroyNSSReference() in the deriving
00185   class, and make the implementation of virtualDestroyNSSReference()
00186   call destructorSafeDestroyNSSReference().
00187 
00188   The destructor of the derived class should call 
00189   destructorSafeDestroyNSSReference() and afterwards call
00190   shutdown(calledFromObject), in order to deregister with the
00191   tracking list, to ensure no additional attempt to free the resources
00192   will be made.
00193   
00194   Function destructorSafeDestroyNSSReference() must
00195   also ensure, that NSS resources have not been freed already.
00196   To achieve this, the deriving class should call 
00197   isAlreadyShutDown() to check.
00198   
00199   It is important that you make your implementation
00200   failsafe, and check whether the resources have already been freed,
00201   in each function that requires the resources.
00202   
00203   class derivedClass : public nsISomeInterface,
00204                        public nsNSSShutDownObject
00205   {
00206     virtual void virtualDestroyNSSReference()
00207     {
00208       destructorSafeDestroyNSSReference();
00209     }
00210     
00211     void destructorSafeDestroyNSSReference()
00212     {
00213       if (isAlreadyShutDown())
00214         return;
00215       
00216       // clean up all NSS resources here
00217     }
00218 
00219     virtual ~derivedClass()
00220     {
00221       destructorSafeDestroyNSSReference();
00222       shutdown(calledFromObject);
00223     }
00224     
00225     NS_IMETHODIMP doSomething()
00226     {
00227       if (isAlreadyShutDown())
00228         return NS_ERROR_NOT_AVAILABLE;
00229       
00230       // use the NSS resources and do something
00231     }
00232   };
00233 */

00239   enum CalledFromType {calledFromList, calledFromObject};
00240 
00241   nsNSSShutDownObject()
00242   {
00243     mAlreadyShutDown = false;
00244     nsNSSShutDownList::remember(this);
00245   }
00246   
00247   virtual ~nsNSSShutDownObject()
00248   {
00249     // the derived class must call 
00250     //   shutdown(calledFromObject);
00251     // in its destructor
00252   }
00253   
00254   void shutdown(CalledFromType calledFrom)
00255   {
00256     if (!mAlreadyShutDown) {
00257       if (calledFromObject == calledFrom) {
00258         nsNSSShutDownList::forget(this);
00259       }
00260       if (calledFromList == calledFrom) {
00261         virtualDestroyNSSReference();
00262       }
00263       mAlreadyShutDown = true;
00264     }
00265   }
00266   
00267   bool isAlreadyShutDown() { return mAlreadyShutDown; }
00268 
00269 protected:
00270   virtual void virtualDestroyNSSReference() = 0;
00271 private:
00272   volatile bool mAlreadyShutDown;
