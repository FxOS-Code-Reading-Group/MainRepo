03870   function estimatePartSizeInBytes(partInfo) {
03871     var encoding = partInfo.encoding.toLowerCase();
03872     // Base64 encodes 3 bytes in 4 characters with padding that always
03873     // causes the encoding to take 4 characters.  The max encoded line length
03874     // (ignoring CRLF) is 76 bytes, with 72 bytes also fairly common.
03875     // As such, a 78=19*4+2 character line encodes 57=19*3 payload bytes and
03876     // we can use that as a rough estimate.
03877     if (encoding === 'base64') {
03878       return Math.floor(partInfo.size * 57 / 78);
03879     }

03881     // to be encoded.  It could be perfectly efficient if the source text
03882     // has a bunch of newlines built-in.
03883     else if (encoding === 'quoted-printable') {
03884       // Let's just provide an upper-bound of perfectly efficient.
03885       return partInfo.size;
