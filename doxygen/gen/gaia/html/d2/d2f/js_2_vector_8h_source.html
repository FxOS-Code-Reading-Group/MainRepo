00020 class TempAllocPolicy;

00022 // If we had C++11 template aliases, we could just use this:
00023 //
00024 //   template <typename T,
00025 //             size_t MinInlineCapacity = 0,
00026 //             class AllocPolicy = TempAllocPolicy>
00027 //   using Vector = mozilla::Vector<T, MinInlineCapacity, AllocPolicy>;
00028 //
00029 // ...and get rid of all the CRTP madness in mozilla::Vector(Base).  But we
00030 // can't because compiler support's not up to snuff.  (Template aliases are in
00031 // gcc 4.7 and clang 3.0 and are expected to be in MSVC 2013.)  Instead, have a
00032 // completely separate class inheriting from mozilla::Vector, and throw CRTP at
00033 // the problem til things work.
00034 //
00035 // This workaround presents a couple issues.  First, because js::Vector is a
00036 // distinct type from mozilla::Vector, overload resolution, method calls, etc.
00037 // are affected.  *Hopefully* this won't be too bad in practice.  (A bunch of
00038 // places had to be fixed when mozilla::Vector was introduced, but it wasn't a
00039 // crazy number.)  Second, mozilla::Vector's interface has to be made subclass-
00040 // ready via CRTP -- or rather, via mozilla::VectorBase, which basically no one
00041 // should use.  :-)  Third, we have to redefine the constructors and the non-
00042 // inherited operators.  Blech.  Happily there aren't too many of these, so it
00043 // isn't the end of the world.
00044 
00045 template <typename T,
00046           size_t MinInlineCapacity = 0,
00047           class AllocPolicy = TempAllocPolicy>
00048 class Vector
00049   : public mozilla::VectorBase<T,
00050                                MinInlineCapacity,
00051                                AllocPolicy,
00052                                Vector<T, MinInlineCapacity, AllocPolicy> >
