00298 
00299     /* The SEPARABLE_CONVOLUTION filter takes the following parameters:
00300      *
00301      *         width:           integer given as 16.16 fixpoint number
00302      *         height:          integer given as 16.16 fixpoint number
00303      *         x_phase_bits:    integer given as 16.16 fixpoint
00304      *         y_phase_bits:    integer given as 16.16 fixpoint
00305      *         xtables:         (1 << x_phase_bits) tables of size width
00306      *         ytables:         (1 << y_phase_bits) tables of size height
00307      *
00308      * When sampling at (x, y), the location is first rounded to one of
00309      * n_x_phases * n_y_phases subpixel positions. These subpixel positions
00310      * determine an xtable and a ytable to use.
00311      *
00312      * Conceptually a width x height matrix is then formed in which each entry
00313      * is the product of the corresponding entries in the x and y tables.
00314      * This matrix is then aligned with the image pixels such that its center
00315      * is as close as possible to the subpixel location chosen earlier. Then
00316      * the image is convolved with the matrix and the resulting pixel returned.
00317      */
00318     PIXMAN_FILTER_SEPARABLE_CONVOLUTION
00319 } pixman_filter_t;

00653                      ((type) << 16) | \
00654                      ((a) << 12) |    \

00912 /* Executive Summary: This function is a no-op that only exists
00913  * for historical reasons.
00914  *
00915  * There used to be a bug in the X server where it would rely on
00916  * out-of-bounds accesses when it was asked to composite with a
00917  * window as the source. It would create a pixman image pointing
00918  * to some bogus position in memory, but then set a clip region
00919  * to the position where the actual bits were.
00920  *
00921  * Due to a bug in old versions of pixman, where it would not clip
00922  * against the image bounds when a clip region was set, this would
00923  * actually work. So when the pixman bug was fixed, a workaround was
00924  * added to allow certain out-of-bound accesses. This function disabled
00925  * those workarounds.
00926  *
00927  * Since 0.21.2, pixman doesn't do these workarounds anymore, so now this
00928  * function is a no-op.
00929  */
00930 void pixman_disable_out_of_bounds_workaround (void);
