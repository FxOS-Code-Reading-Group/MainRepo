00043 // NSPR APIs use PRStatus/PR_GetError and NSS APIs use SECStatus/PR_GetError to
00044 // report success/failure. These funtions make it more convenient and *safer*
00045 // to translate NSPR/NSS results to nsresult. They are safer because they
00046 // refuse to traslate any bad PRStatus/SECStatus into an NS_OK, even when the
00047 // NSPR/NSS function forgot to call PR_SetError.
00048 
00049 // IMPORTANT: This must be called immediately after the function that set the
00050 // error code. Prefer using MapSECStatus to this.
00051 inline nsresult
00052 PRErrorCode_to_nsresult(PRErrorCode error)
00053 {
00054   if (!error) {
00055     MOZ_CRASH("Function failed without calling PR_GetError");
00056   }
00057 
00058   // From NSSErrorsService::GetXPCOMFromNSSError
00059   // XXX Don't make up nsresults, it's supposed to be an enum (bug 778113)
00060   return (nsresult)NS_ERROR_GENERATE_FAILURE(NS_ERROR_MODULE_SECURITY,
00061                                              -1 * error);
00062 }
00063 
00064 // IMPORTANT: This must be called immediately after the function returning the
00065 // SECStatus result. The recommended usage is:
00066 //    nsresult rv = MapSECStatus(f(x, y, z));
00067 inline nsresult
00068 MapSECStatus(SECStatus rv)
00069 {
00070   if (rv == SECSuccess)
00071     return NS_OK;
00072 
00073   PRErrorCode error = PR_GetError();
00074   return PRErrorCode_to_nsresult(error);
00075 }

00174   {
00175     nsresult rv = SetLength(hashAlg);
00176     NS_ENSURE_SUCCESS(rv, rv);
00177     uint32_t len;
00178     rv = MapSECStatus(PK11_DigestFinal(context, item.data, &len, item.len));
00179     NS_ENSURE_SUCCESS(rv, rv);
