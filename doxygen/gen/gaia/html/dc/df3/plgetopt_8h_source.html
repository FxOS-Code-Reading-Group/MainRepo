00079 /*
00080  * PL_GetNextOpt
00081  *
00082  * When this function returns PL_OPT_OK, 
00083  * - opt->option will hold the single-character option name that was parsed, 
00084  *   or zero.  
00085  * When opt->option is zero, the token parsed was either a "long" (keyword) 
00086  *   option or a positional parameter.  
00087  * For a positional parameter, 
00088  * - opt->longOptIndex will contain -1, and
00089  * - opt->value will point to the positional parameter string.
00090  * For a long option name, 
00091  * - opt->longOptIndex will contain the non-negative index of the 
00092  *   PLLongOpt structure in the caller's array of PLLongOpt structures 
00093  *   corresponding to the long option name, and 
00094  * For a single-character or long option, 
00095  * - opt->longOption will contain the value of the single-character option
00096  *   name, or the value of the longOption from the PLLongOpt structure
00097  *   for that long option.  See notes below.
00098  * - opt->value will point to the argument option string, or will
00099  *   be NULL if option does not require argument.  If option requires
00100  *   argument but it is not provided, PL_OPT_BAD is returned.
00101  * When opt->option is non-zero, 
00102  * - opt->longOptIndex will be -1
00103  * When this function returns PL_OPT_EOL, or PL_OPT_BAD, the contents of
00104  *   opt are undefined.
00105  *
00106  * Notes: It is possible to ignore opt->option, and always look at 
00107  *   opt->longOption instead.  opt->longOption will contain the same value
00108  *   as opt->option for single-character option names, and will contain the
00109  *   value of longOption from the PLLongOpt structure for long option names.
00110  * This means that it is possible to equivalence long option names to 
00111  *   single character names by giving the longOption in the PLLongOpt struct
00112  *   the same value as the single-character option name.  
00113  * For long options that are NOT intended to be equivalent to any single-
00114  *   character option, the longOption value should be chosen to not match 
00115  *   any possible single character name.  It might be advisable to choose
00116  *   longOption values greater than 0xff for such long options.
00117  */
00118 PR_EXTERN(PLOptStatus) PL_GetNextOpt(PLOptState *opt);
