00340   inline bool IsAdditive() const {
00341     /*
00342      * Animation is additive if:
00343      *
00344      * (1) additive = "sum" (GetAdditive() == true), or
00345      * (2) it is 'by animation' (by is set, from and values are not)
00346      *
00347      * Although animation is not additive if it is 'to animation'
00348      */
00349     bool isByAnimation = (!HasAttr(nsGkAtoms::values) &&
00350                              HasAttr(nsGkAtoms::by) &&
00351                             !HasAttr(nsGkAtoms::from));
00352     return !IsToAnimation() && (GetAdditive() || isByAnimation);
00353   }
