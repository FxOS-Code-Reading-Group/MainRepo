00127   static Color FromABGR(uint32_t aColor)
00128   {
00129     Color newColor(((aColor >> 0) & 0xff) * (1.0f / 255.0f),
00130                    ((aColor >> 8) & 0xff) * (1.0f / 255.0f),
00131                    ((aColor >> 16) & 0xff) * (1.0f / 255.0f),
00132                    ((aColor >> 24) & 0xff) * (1.0f / 255.0f));
00133 
00134     return newColor;
00135   }
00136 
00137   uint32_t ToABGR() const
00138   {
00139     return uint32_t(r * 255.0f) | uint32_t(g * 255.0f) << 8 |
00140            uint32_t(b * 255.0f) << 16 | uint32_t(a * 255.0f) << 24;
00141   }
