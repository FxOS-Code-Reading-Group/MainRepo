01056   function resizeBlob(blob, target_width, target_height, callback) {
01057     var img = document.createElement('img');
01058     var url = URL.createObjectURL(blob);
01059     img.src = url;
01060     img.onload = function() {
01061       var image_width = img.width;
01062       var image_height = img.height;
01063       var scalex = image_width / target_width;
01064       var scaley = image_height / target_height;
01065       var scale = Math.min(scalex, scaley);
01066 
01067       var w = target_width * scale;
01068       var h = target_height * scale;
01069       var x = (image_width - w) / 2;
01070       var y = (image_height - h) / 2;
01071 
01072       var canvas = document.createElement('canvas');
01073       canvas.width = target_width;
01074       canvas.height = target_height;
01075       var context = canvas.getContext('2d');
01076 
01077       context.drawImage(img, x, y, w, h, 0, 0, target_width, target_height);
01078       URL.revokeObjectURL(url);
01079       canvas.toBlob(callback, 'image/jpeg');
01080     };
01081   }
