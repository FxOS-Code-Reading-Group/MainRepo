00270 inline nsIDocShell*
00271 NS_CP_GetDocShellFromContext(nsISupports *aContext)
00272 {
00273     if (!aContext) {
00274         return nullptr;
00275     }
00276 
00277     nsCOMPtr<nsPIDOMWindow> window = do_QueryInterface(aContext);
00278 
00279     if (!window) {
00280         // our context might be a document (which also QIs to nsIDOMNode), so
00281         // try that first
00282         nsCOMPtr<nsIDocument> doc = do_QueryInterface(aContext);
00283         if (!doc) {
00284             // we were not a document after all, get our ownerDocument,
00285             // hopefully
00286             nsCOMPtr<nsIContent> content = do_QueryInterface(aContext);
00287             if (content) {
00288                 doc = content->OwnerDoc();
00289             }
00290         }
00291 
00292         if (doc) {
00293             if (doc->GetDisplayDocument()) {
00294                 doc = doc->GetDisplayDocument();
00295             }
00296             
00297             window = doc->GetWindow();
00298         }
00299     }
00300 
00301     if (!window) {
00302         return nullptr;
00303     }
00304 
00305     return window->GetDocShell();
