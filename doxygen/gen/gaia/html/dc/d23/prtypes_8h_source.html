00208 /*
00209 ** Starting in NSPR 4.9.5, NSPR's exact-width integer types should match
00210 ** the exact-width integer types defined in <stdint.h>. This allows sloppy
00211 ** code to use PRInt{N} and int{N}_t interchangeably.
00212 **
00213 ** The 8-bit and 16-bit integer types can only be defined using char and
00214 ** short. All platforms define the 32-bit integer types using int. So only
00215 ** the 64-bit integer types could be defined differently.
00216 **
00217 ** NSPR's original strategy was to use the "shortest" 64-bit integer type:
00218 ** if long is 64-bit, then prefer it over long long. This strategy is also
00219 ** used by Linux/glibc, FreeBSD, and NetBSD.
00220 **
00221 ** Other platforms use a different strategy: simply define the 64-bit
00222 ** integer types using long long. We define the PR_ALTERNATE_INT64_TYPEDEF
00223 ** macro on these platforms. Note that PR_ALTERNATE_INT64_TYPEDEF is for
00224 ** internal use by NSPR headers only. Do not define or test this macro in
00225 ** your code.
00226 **
00227 ** NOTE: NSPR can't use <stdint.h> because C99 requires C++ code to define
00228 ** __STDC_LIMIT_MACROS and __STDC_CONSTANT_MACROS to make all the macros
00229 ** defined in <stdint.h> available. This strange requirement is gone in
00230 ** C11. When most platforms ignore this C99 requirement, NSPR will be able
00231 ** to use <stdint.h>. A patch to do that is in NSPR bug 634793.
00232 */
00233 
00234 #if defined(__APPLE__) || defined(__ANDROID__) || defined(__OpenBSD__)
00235 #define PR_ALTERNATE_INT64_TYPEDEF
00236 #endif
