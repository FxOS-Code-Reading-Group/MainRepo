00008 /*
00009  * macros to handle parsing strings of blank sparated arguments.
00010  * Several NSSUTIL_HANDLE_STRING() macros should be places one after another with no intervening
00011  * code. The first ones have precedence over the later ones. The last Macro should be 
00012  * NSSUTIL_HANDLE_FINAL_ARG.
00013  *
00014  *  param is the input parameters. On exit param will point to the next parameter to parse. If the
00015  *      last paramter has been returned, param points to a null byte (*param = '0');
00016  *  target is the location to store any data aquired from the parameter. Caller is responsible to free this data.
00017  *  value is the string value of the parameter.
00018  *  command is any commands you need to run to help process the parameter's data.
00019  */
00020 #define NSSUTIL_HANDLE_STRING_ARG(param,target,value,command) \
00021     if (PORT_Strncasecmp(param,value,sizeof(value)-1) == 0) { \
00022         param += sizeof(value)-1; \
00023         if (target) PORT_Free(target); \
00024         target = NSSUTIL_ArgFetchValue(param,&next); \
00025         param += next; \
00026         command ;\
00027     } else

00030     { param = NSSUTIL_ArgSkipParameter(param); } param = NSSUTIL_ArgStrip(param);
