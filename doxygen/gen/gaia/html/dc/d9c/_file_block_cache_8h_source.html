00019 // Manages file I/O for the media cache. Data comes in over the network
00020 // via callbacks on the main thread, however we don't want to write the
00021 // incoming data to the media cache on the main thread, as this could block
00022 // causing UI jank. 
00023 //
00024 // So FileBlockCache provides an abstraction for a temporary file accessible
00025 // as an array of blocks, which supports a block move operation, and
00026 // allows synchronous reading and writing from any thread, with writes being
00027 // buffered so as not to block.
00028 //
00029 // Writes and cache block moves (which require reading) are deferred to
00030 // their own non-main thread. This object also ensures that data which has
00031 // been scheduled to be written, but hasn't actually *been* written, is read
00032 // as if it had, i.e. pending writes are cached in readable memory until
00033 // they're flushed to file.
00034 //
00035 // To improve efficiency, writes can only be done at block granularity,
00036 // whereas reads can be done with byte granularity.
00037 //
00038 // Note it's also recommended not to read from the media cache from the main
00039 // thread to prevent jank.
00040 //
00041 // When WriteBlock() or MoveBlock() are called, data about how to complete
00042 // the block change is added to mBlockChanges, indexed by block index, and
00043 // the block index is appended to the mChangeIndexList. This enables
00044 // us to quickly tell if a block has been changed, and ensures we can perform
00045 // the changes in the correct order. An event is dispatched to perform the
00046 // changes listed in mBlockChanges to file. Read() checks mBlockChanges and
00047 // determines the current data to return, reading from file or from
00048 // mBlockChanges as necessary.

00083   // Represents a change yet to be made to a block in the file. The change
00084   // is either a write (and the data to be written is stored in this struct)
00085   // or a move (and the index of the source block is stored instead).
00086   struct BlockChange {
00087 
00088     NS_INLINE_DECL_THREADSAFE_REFCOUNTING(BlockChange)
00089 
00090     // This block is waiting in memory to be written.
00091     // Stores a copy of the block, so we can write it asynchronously.
00092     BlockChange(const uint8_t* aData)
00093       : mSourceBlockIndex(-1)
00094     {
00095       mData = new uint8_t[BLOCK_SIZE];
00096       memcpy(mData.get(), aData, BLOCK_SIZE);
00097     }
00098 
00099     // This block's contents are located in another file
00100     // block, i.e. this block has been moved.
00101     BlockChange(int32_t aSourceBlockIndex)
00102       : mSourceBlockIndex(aSourceBlockIndex) {}
00103 
00104     nsAutoArrayPtr<uint8_t> mData;
00105     const int32_t mSourceBlockIndex;
00106 
00107     bool IsMove() const {
00108       return mSourceBlockIndex != -1;
00109     }
00110     bool IsWrite() const {
00111       return mSourceBlockIndex == -1 &&
00112              mData.get() != nullptr;
00113     }
00114   };
