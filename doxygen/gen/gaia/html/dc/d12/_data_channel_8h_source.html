00502   NS_IMETHOD Run()
00503   {
00504     MOZ_ASSERT(NS_IsMainThread());
00505     switch (mType) {
00506       case ON_DATA:
00507       case ON_CHANNEL_OPEN:
00508       case ON_CHANNEL_CLOSED:
00509         {
00510           MutexAutoLock lock(mChannel->mListenerLock);
00511           if (!mChannel->mListener) {
00512             DATACHANNEL_LOG(("DataChannelOnMessageAvailable (%d) with null Listener!",mType));
00513             return NS_OK;
00514           }
00515 
00516           switch (mType) {
00517             case ON_DATA:
00518               if (mLen < 0) {
00519                 mChannel->mListener->OnMessageAvailable(mChannel->mContext, mData);
00520               } else {
00521                 mChannel->mListener->OnBinaryMessageAvailable(mChannel->mContext, mData);
00522               }
00523               break;
00524             case ON_CHANNEL_OPEN:
00525               mChannel->mListener->OnChannelConnected(mChannel->mContext);
00526               break;
00527             case ON_CHANNEL_CLOSED:
00528               mChannel->mListener->OnChannelClosed(mChannel->mContext);
00529               break;
00530           }
00531           break;
00532         }
00533       case ON_DISCONNECTED:
00534         // If we've disconnected, make sure we close all the streams - from mainthread!
00535         mConnection->CloseAll();
00536         // fall through
00537       case ON_CHANNEL_CREATED:
00538       case ON_CONNECTION:
00539         // WeakPtr - only used/modified/nulled from MainThread so we can use a WeakPtr here
00540         if (!mConnection->mListener) {
00541           DATACHANNEL_LOG(("DataChannelOnMessageAvailable (%d) with null Listener",mType));
00542           return NS_OK;
00543         }
00544         switch (mType) {
00545           case ON_CHANNEL_CREATED:
00546             // important to give it an already_AddRefed pointer!
00547             mConnection->mListener->NotifyDataChannel(mChannel.forget());
00548             break;
00549           case ON_CONNECTION:
00550             if (mResult) {
00551               mConnection->mListener->NotifyConnection();
00552             }
00553             // FIX - on mResult false (failure) we should do something.  Needs spec work here
00554             break;
00555           case ON_DISCONNECTED:
00556             mConnection->mListener->NotifyClosedConnection();
00557             break;
00558         }
00559         break;
00560       case START_DEFER:
00561         mConnection->StartDefer();
00562         break;
00563     }
00564     return NS_OK;
00565   }
