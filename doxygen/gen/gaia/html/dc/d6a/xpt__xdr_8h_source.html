00144 /*
00145  * If we're decoding, we want to read the offset before we check
00146  * for already-decoded values.
00147  *
00148  * Then we check for repetition: CheckForRepeat will see if we've already
00149  * encoded/decoded this value, and if so will set offset/addr correctly
00150  * and make already be true.  If not, it will set up the cursor for
00151  * encoding (reserve space) or decoding (seek to correct location) as
00152  * appropriate.  In the encode case, it will also set the addr->offset
00153  * mapping.
00154  */
00155 
00156 #define XPT_PREAMBLE_(cursor, addrp, pool, size, new_curs, already)           \
00157     XPTMode mode = cursor->state->mode;                                       \
00158     if (!(mode == XPT_ENCODE || XPT_Do32(cursor, &new_curs.offset)) ||        \
00159         !CheckForRepeat(cursor, (void **)addrp, pool,                         \
00160                         mode == XPT_ENCODE ? size : 0u, &new_curs,            \
00161                         &already) ||                                          \
00162         !(mode == XPT_DECODE || XPT_Do32(cursor, &new_curs.offset)))          \
00163         return PR_FALSE;                                                      \
00164     if (already)                                                              \
00165         return PR_TRUE;                                                       \
00166 
00167 #define XPT_PREAMBLE_NO_ALLOC(cursor, addrp, pool, size, new_curs, already)   \
00168   {                                                                           \
00169     XPT_PREAMBLE_(cursor, addrp, pool, size, new_curs, already)               \
00170   }
00171 
00172 #define XPT_ERROR_HANDLE(arena, free_it)                                      \
00173  error:                                                                       \
00174     if (cursor->state->mode == XPT_DECODE)                                    \
00175     XPT_FREEIF(arena, free_it);                                               \
00176     return PR_FALSE;
