00029 // Bind an object instance, usually |this|, to a name, usually URL or 
00030 // host name, the instance deals with for its lifetime.  The name string 
00031 // is duplicated.
00032 // IMPORTANT: it is up to the caller to pass the correct static_cast
00033 // of the |instance| pointer to all these macros ; otherwise the linking
00034 // of events and objects will not work!
00035 // The name will show in details of the events on the timeline and also
00036 // will allow events filtering by host names, URLs etc.
00037 #define MOZ_EVENT_TRACER_NAME_OBJECT(instance, name) \
00038   mozilla::eventtracer::Mark(mozilla::eventtracer::eName, instance, name)
00039 
00040 // The same as MOZ_EVENT_TRACER_NAME_OBJECT, just simplifies building
00041 // names when it consists of two parts
00042 #define MOZ_EVENT_TRACER_COMPOUND_NAME(instance, name, name2) \
00043   mozilla::eventtracer::Mark(mozilla::eventtracer::eName, instance, name, name2)
00044 
00045 
00046 // Call the followings with the same |instance| reference as you have 
00047 // previously called MOZ_EVENT_TRACER_NAME_OBJECT.
00048 // Let |name| be the name of the event happening, like "resolving", 
00049 // "connecting", "loading" etc.
00050 
00051 // This will crate a single-point-in-time event marked with an arrow 
00052 // on the timeline, this is a way to indicate an event without a duration.
00053 #define MOZ_EVENT_TRACER_MARK(instance, name) \
00054   mozilla::eventtracer::Mark(mozilla::eventtracer::eShot, instance, name)
00055 
00056 // Following macros are used to log events with duration.
00057 // There always has to be complete WAIT,EXEC,DONE or EXEC,DONE 
00058 // uninterrupted and non-interferring tuple(s) for an event to be correctly 
00059 // shown on the timeline.  Each can be called on any thread, the event tape is
00060 // finally displayed on the thread where it has been EXECuted.
00061 
00062 // Example of 3 phases usage for "HTTP request channel":
00063 // WAIT: we've just created the channel and called AsyncOpen on it
00064 // EXEC: we've just got first call to OnDataAvailable, so we are actually
00065 //       receiving the content after some time like connection establising,
00066 //       opening a cache entry, sending the GET request...
00067 // DONE: we've just got OnStopRequest call that indicates the content
00068 //       has been completely delivered and the request is now finished
00069 
00070 // Indicate an event pending start, on the timeline this will 
00071 // start the event's interval tape with a pale color, the time will
00072 // show in details.  Usually used when an event is being posted or
00073 // we wait for a lock acquisition.
00074 // WAITING is not mandatory, some events don't have a pending phase.
00075 #define MOZ_EVENT_TRACER_WAIT(instance, name) \
00076   mozilla::eventtracer::Mark(mozilla::eventtracer::eWait, instance, name)
00077 
00078 // Indicate start of an event actual execution, on the timeline this will 
00079 // change the event's tape to a deeper color, the time will show in details.
00080 #define MOZ_EVENT_TRACER_EXEC(instance, name) \
00081   mozilla::eventtracer::Mark(mozilla::eventtracer::eExec, instance, name)
00082 
00083 // Indicate the end of an event execution (the event has been done),
00084 // on the timeline this will end the event's tape and show the time in
00085 // event details.
00086 // NOTE: when the event duration is extremely short, like 1ms, it will convert
00087 // to an event w/o a duration - the same as MOZ_EVENT_TRACER_MARK would be used.
00088 #define MOZ_EVENT_TRACER_DONE(instance, name) \
00089   mozilla::eventtracer::Mark(mozilla::eventtracer::eDone, instance, name)
00090 
00091 // The same meaning as the above macros, just for concurent events.
00092 // Concurent event means it can happen for the same instance on more
00093 // then just a single thread, e.g. a service method call, a global lock 
00094 // acquisition, enter and release.
00095 #define MOZ_EVENT_TRACER_WAIT_THREADSAFE(instance, name) \
00096   mozilla::eventtracer::Mark(mozilla::eventtracer::eWait | mozilla::eventtracer::eThreadConcurrent, instance, name)
00097 #define MOZ_EVENT_TRACER_EXEC_THREADSAFE(instance, name) \
00098   mozilla::eventtracer::Mark(mozilla::eventtracer::eExec | mozilla::eventtracer::eThreadConcurrent, instance, name)
00099 #define MOZ_EVENT_TRACER_DONE_THREASAFE(instance, name) \
00100   mozilla::eventtracer::Mark(mozilla::eventtracer::eDone | mozilla::eventtracer::eThreadConcurrent, instance, name)

00129 enum MarkType {
00130   eNone, // No operation, ignored
00131   eName, // This is used to bind an object instance with a name

00133   eShot, // An event with no duration
00134   eWait, // Start of waiting for execution (lock acquire, post...)

00145 // Records an event on the calling thread. 
00146 // @param aType 
00147 //    One of MarkType fields, can be bitwise or'ed with the flags.
00148 // @param aItem
00149 //    Reference to the object we want to bind a name to or the event is
00150 //    happening on.  Can be actually anything, but valid poitners should
00151 //    be used.
00152 // @param aText
00153 //    Text of the name (for eName) or event's name for others.  The string
00154 //    is duplicated.
00155 // @param aText2
00156 //    Optional second part of the instnace name, or event name.
00157 //    Event filtering does apply only to the first part (aText).
00158 void Mark(uint32_t aType, void * aItem, 
00159           const char * aText, const char * aText2 = 0);

00162 // Helper guard object.  Use to mark an event in the constructor and a different
00163 // event in the destructor.

00165 // Example:
00166 // int class::func()

00172 class MOZ_STACK_CLASS AutoEventTracer
00173 {
00174 public:
00175   AutoEventTracer(void * aInstance, 
00176                uint32_t aTypeOn, // MarkType marked in constructor
00177                uint32_t aTypeOff, // MarkType marked in destructor
00178                const char * aName, 
00179                const char * aName2 = 0 
00180                MOZ_GUARD_OBJECT_NOTIFIER_PARAM)
00181     : mInstance(aInstance)
00182     , mName(aName)
00183     , mName2(aName2)
00184     , mTypeOn(aTypeOn)
00185     , mTypeOff(aTypeOff)
00186   {
00187     MOZ_GUARD_OBJECT_NOTIFIER_INIT;
00188 
00189     ::mozilla::eventtracer::Mark(mTypeOn, mInstance, mName, mName2);
00190   }
00191 
00192   ~AutoEventTracer()
00193   {
00194     ::mozilla::eventtracer::Mark(mTypeOff, mInstance, mName, mName2);
00195   }
00196 
00197 private:

00199   const char * mName;

00201   uint32_t mTypeOn;
