00706     if (!tagname || exif[tagname])

00718   function parseValue(data, offset, type, count, byteorder) {
00719     if (type === 2) { // ASCII string
00720       var codes = [];
00721       for (var i = 0; i < count - 1; i++) {
00722         codes[i] = data.getUint8(offset + i);
00723       }
00724       return String.fromCharCode.apply(String, codes);
00725     } else {
00726       if (count == 1) {
00727         return parseOneValue(data, offset, type, byteorder);
00728       } else {
00729         var values = [];
00730         var size = typesize[type];
00731         for (var i = 0; i < count; i++) {
00732           values[i] = parseOneValue(data, offset + size * i, type, byteorder);
00733         }
00734         return values;
00735       }
00736     }

00740     switch (type) {

00742     case 7: // UNDEFINED

00744     case 2: // ASCII
00745       // This case is handed in parseValue
00746       return null;
00747     case 3: // SHORT
00748       return data.getUint16(offset, byteorder);
00749     case 4: // LONG
00750       return data.getUint32(offset, byteorder);
00751     case 5: // RATIONAL
00752       return data.getUint32(offset, byteorder) /
00753         data.getUint32(offset + 4, byteorder);
00754     case 6: // SBYTE
00755       return data.getInt8(offset);
00756     case 8: // SSHORT
00757       return data.getInt16(offset, byteorder);
00758     case 9: // SLONG
00759       return data.getInt32(offset, byteorder);
00760     case 10: // SRATIONAL
00761       return data.getInt32(offset, byteorder) /
00762         data.getInt32(offset + 4, byteorder);
00763     case 11: // FLOAT
00764       return data.getFloat32(offset, byteorder);
00765     case 12: // DOUBLE
00766       return data.getFloat64(offset, byteorder);

00788   function MP4Parser(blob, handlers) {
00789     // Start off with a 1024 chunk from the start of the blob.
00790     BlobView.get(blob, 0, Math.min(1024, blob.size), function(data, error) {
00791       // Make sure that the blob is, in fact, some kind of MP4 file
00792       if (data.byteLength <= 8 || data.getASCIIText(4, 4) !== 'ftyp') {
00793         handlers.errorHandler('not an MP4 file');
00794         return;
00795       }
00796       parseAtom(data);
00797     });

00799     // Call this with a BlobView object that includes the first 16 bytes of
00800     // an atom. It doesn't matter whether the body of the atom is included.
00801     function parseAtom(data) {
00802       var offset = data.sliceOffset + data.viewOffset; // atom position in blob
00803       var size = data.readUnsignedInt();               // atom length
00804       var type = data.readASCIIText(4);                // atom type
00805       var contentOffset = 8;                           // position of content
00806 
00807       if (size === 0) {
00808         // Zero size means the rest of the file
00809         size = blob.size - offset;
00810       }
00811       else if (size === 1) {
00812         // A size of 1 means the size is in bytes 8-15
00813         size = data.readUnsignedInt() * 4294967296 + data.readUnsignedInt();
00814         contentOffset = 16;
00815       }
00816 
00817       var handler = handlers[type] || handlers.defaultHandler;
00818       if (typeof handler === 'function') {
00819         // If the handler is a function, pass that function a
00820         // DataView object that contains the entire atom
00821         // including size and type.  Then use the return value
00822         // of the function as instructions on what to do next.
00823         data.getMore(data.sliceOffset + data.viewOffset, size, function(atom) {
00824           // Pass the entire atom to the handler function
00825           var rv = handler(atom);
00826 
00827           // If the return value is 'done', stop parsing.
00828           // Otherwise, continue with the next atom.
00829           // XXX: For more general parsing we need a way to pop some
00830           // stack levels.  A return value that is an atom name should mean
00831           // pop back up to this atom type and go on to the next atom
00832           // after that.
00833           if (rv !== 'done') {
00834             parseAtomAt(data, offset + size);
00835           }
00836         });
00837       }
00838       else if (handler === 'children') {
00839         // If the handler is this string, then assume that the atom is
00840         // a container atom and do its next child atom next
00841         var skip = (type === 'meta') ? 4 : 0; // special case for meta atoms
00842         parseAtomAt(data, offset + contentOffset + skip);
00843       }
00844       else if (handler === 'skip' || !handler) {
00845         // Skip the atom entirely and go on to the next one.
00846         // If there is no next one, call the eofHandler or just return
00847         parseAtomAt(data, offset + size);
00848       }
00849       else if (handler === 'done') {
00850         // Stop parsing
00851         return;
00852       }
00853     }
00854 
00855     function parseAtomAt(data, offset) {
00856       if (offset >= blob.size) {
00857         if (handlers.eofHandler)
00858           handlers.eofHandler();
00859         return;
00860       }
00861       else {
00862         data.getMore(offset, 16, parseAtom);
00863       }
00864     }
00865   }
00866 
00867   // We want to loop through the top-level atoms until we find the 'moov'
00868   // atom. Then, within this atom, there are one or more 'trak' atoms.
00869   // Each 'trak' should begin with a 'tkhd' atom. The tkhd atom has
00870   // a transformation matrix at byte 48.  The matrix is 9 32 bit integers.
00871   // We'll interpret those numbers as a rotation of 0, 90, 180 or 270.
00872   // If the video has more than one track, we expect all of them to have
00873   // the same rotation, so we'll only look at the first 'trak' atom that
00874   // we find.
00875   MP4Parser(blob, {
00876     errorHandler: function(msg) { rotationCallback(msg); },
00877     eofHandler: function() { rotationCallback(null); },
00878     defaultHandler: 'skip',  // Skip all atoms other than those listed below
00879     moov: 'children',        // Enumerate children of the moov atom
00880     trak: 'children',        // Enumerate children of the trak atom
00881     tkhd: function(data) {   // Pass the tkhd atom to this function
00882       // The matrix begins at byte 48
00883       data.advance(48);
00884 
00885       var a = data.readUnsignedInt();
00886       var b = data.readUnsignedInt();
00887       data.advance(4); // we don't care about this number
00888       var c = data.readUnsignedInt();
00889       var d = data.readUnsignedInt();
00890 
00891       if (a === 0 && d === 0) { // 90 or 270 degrees
00892         if (b === 0x00010000 && c === 0xFFFF0000)
00893           rotationCallback(90);
00894         else if (b === 0xFFFF0000 && c === 0x00010000)
00895           rotationCallback(270);
00896         else
00897           rotationCallback('unexpected rotation matrix');
00898       }
00899       else if (b === 0 && c === 0) { // 0 or 180 degrees
00900         if (a === 0x00010000 && d === 0x00010000)
00901           rotationCallback(0);
00902         else if (a === 0xFFFF0000 && d === 0xFFFF0000)
00903           rotationCallback(180);
00904         else
00905           rotationCallback('unexpected rotation matrix');
00906       }
00907       else {
00908         rotationCallback('unexpected rotation matrix');
00909       }
00910       return 'done';
00911     }
00912   });

00914 /*
00915  * Determine the pixel dimensions of an image without actually
00916  * decoding the image. Passes an object of metadata to the callback
00917  * function on success or an error message to the error function on
00918  * failure. The metadata object will include type, width and height
00919  * properties. Supported image types are GIF, PNG and JPEG. JPEG
00920  * metadata may also include information about an EXIF preview image.
00921  *
00922  * Because of shortcomings in the way Gecko handles images, the
00923  * Gallery app will crash with an OOM error if it attempts to decode
00924  * and display an image that is too big. Images require 4 bytes per
00925  * pixel, so a 10 megapixel photograph requires 40 megabytes of image
00926  * memory. This function gives the gallery app a way to reject images
00927  * that are too large.
00928  *
00929  * Requires the BlobView class from shared/js/blobview.js and the
00930  * parseJPEGMetadata() function from shared/js/media/jpeg_metadata_parser.js
00931  */
00932 function getImageSize(blob, callback, error) {
00933   BlobView.get(blob, 0, Math.min(1024, blob.size), function(data) {
00934     // Make sure we are at least 8 bytes long before reading the first 8 bytes
00935     if (data.byteLength <= 8) {
00936       error('corrupt image file');
00937       return;
00938     }
00939     var magic = data.getASCIIText(0, 8);
00940     if (magic.substring(0, 4) === 'GIF8') {
00941       try {
00942         callback({
00943           type: 'gif',
00944           width: data.getUint16(6, true),
00945           height: data.getUint16(8, true)
00946         });
00947       }
00948       catch (e) {
00949         error(e.toString());
00950       }
00951     }
00952     else if (magic.substring(0, 8) === '\x89PNG\r\n\x1A\n') {
00953       try {
00954         callback({
00955           type: 'png',
00956           width: data.getUint32(16, false),
00957           height: data.getUint32(20, false)
00958         });
00959       }
00960       catch (e) {
00961         error(e.toString());
00962       }
00963     }
00964     else if (magic.substring(0, 2) === '\xFF\xD8') {
00965       parseJPEGMetadata(blob,
00966                         function(metadata) {
00967                           metadata.type = 'jpeg';
00968                           callback(metadata);
00969                         },
00970                         error);
00971     }
00972     else {
00973       error('unknown image type');
00974     }
00975   });
