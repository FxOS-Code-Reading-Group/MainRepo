00054 def parse_ini(filename):
00055     log = logging.getLogger(__name__)
00056     with open(filename) as f:
00057         data = f.readlines()
00058     section = 'default'
00059     imports = { section: [] }
00060     for line in data:
00061         if line.strip() == "" or line.startswith('!') or line.startswith('#'):
00062             continue
00063         elif line.strip().startswith('['): # Section header
00064             section = section_line.search(line).group('section')
00065             imports[section] = []
00066         elif '@import' in line: # Import lines
00067             property_file = import_line.search(line).group('filename')
00068             imports[section].append(property_file)
00069         else:
00070             log.warn('parse_ini - found a line with contents '
00071                      'unaccounted for "%s"', line.strip())
00072     return imports

00031 def parse_manifest_properties(filename):
00032     with open(filename) as f:
00033         data = f.readlines()
00034     strings = {
00035         "default": {},
00036         "entry_points": {},
00037     }
00038     for line in data:
00039         m = property_line.search(line)
00040         if not m or line.strip().startswith('#'):
00041             continue
00042         value = m.group('value').strip()
00043         if '.' in m.group('id'):
00044             entry_point, key = m.group('id').split('.',1)
00045             if entry_point not in strings["entry_points"]:
00046                 strings["entry_points"][entry_point] = {}
