00027 ** Structures returned by network data base library.  All addresses are
00028 ** supplied in host order, and returned in network order (suitable for
00029 ** use in system calls).

00039 #ifdef WIN32
00040     PRInt16 h_addrtype; /* host address type */

00042 #else
00043     PRInt32 h_addrtype; /* host address type */
00044     PRInt32 h_length;   /* length of address */
00045 #endif
00046     char **h_addr_list; /* list of addresses from name server */
00047 } PRHostEnt;
