00042   virtual void ProduceOutput(GraphTime aFrom, GraphTime aTo)
00043   {
00044     nsAutoTArray<bool,8> mappedTracksFinished;
00045     nsAutoTArray<bool,8> mappedTracksWithMatchingInputTracks;
00046     for (uint32_t i = 0; i < mTrackMap.Length(); ++i) {
00047       mappedTracksFinished.AppendElement(true);
00048       mappedTracksWithMatchingInputTracks.AppendElement(false);
00049     }
00050     bool allFinished = true;
00051     bool allHaveCurrentData = true;
00052     for (uint32_t i = 0; i < mInputs.Length(); ++i) {
00053       MediaStream* stream = mInputs[i]->GetSource();
00054       if (!stream->IsFinishedOnGraphThread()) {
00055         allFinished = false;
00056       }
00057       if (!stream->HasCurrentData()) {
00058         allHaveCurrentData = false;
00059       }
00060       for (StreamBuffer::TrackIter tracks(stream->GetStreamBuffer());
00061            !tracks.IsEnded(); tracks.Next()) {
00062         bool found = false;
00063         for (uint32_t j = 0; j < mTrackMap.Length(); ++j) {
00064           TrackMapEntry* map = &mTrackMap[j];
00065           if (map->mInputPort == mInputs[i] && map->mInputTrackID == tracks->GetID()) {
00066             bool trackFinished;
00067             StreamBuffer::Track* outputTrack = mBuffer.FindTrack(map->mOutputTrackID);
00068             if (!outputTrack || outputTrack->IsEnded()) {
00069               trackFinished = true;
00070             } else {
00071               CopyTrackData(tracks.get(), j, aFrom, aTo, &trackFinished);
00072             }
00073             mappedTracksFinished[j] = trackFinished;
00074             mappedTracksWithMatchingInputTracks[j] = true;
00075             found = true;
00076             break;
00077           }
00078         }
00079         if (!found && (!mFilterCallback || mFilterCallback(tracks.get()))) {
00080           bool trackFinished = false;
00081           uint32_t mapIndex = AddTrack(mInputs[i], tracks.get(), aFrom);
00082           CopyTrackData(tracks.get(), mapIndex, aFrom, aTo, &trackFinished);
00083           mappedTracksFinished.AppendElement(trackFinished);
00084           mappedTracksWithMatchingInputTracks.AppendElement(true);
00085         }
00086       }
00087     }
00088     for (int32_t i = mTrackMap.Length() - 1; i >= 0; --i) {
00089       if (mappedTracksFinished[i]) {
00090         EndTrack(i);
00091       } else {
00092         allFinished = false;
00093       }
00094       if (!mappedTracksWithMatchingInputTracks[i]) {
00095         mTrackMap.RemoveElementAt(i);
00096       }
00097     }
00098     if (allFinished && mAutofinish) {
00099       // All streams have finished and won't add any more tracks, and
00100       // all our tracks have actually finished and been removed from our map,
00101       // so we're finished now.
00102       FinishOnGraphThread();
00103     }
00104     mBuffer.AdvanceKnownTracksTime(GraphTimeToStreamTime(aTo));
00105     if (allHaveCurrentData) {
00106       // We can make progress if we're not blocked
00107       mHasCurrentData = true;
00108     }
00109   }

00134   uint32_t AddTrack(MediaInputPort* aPort, StreamBuffer::Track* aTrack,
00135                     GraphTime aFrom)
00136   {
00137     // Use the ID of the source track if we can, otherwise allocate a new
00138     // unique ID
00139     TrackID id = std::max(mMaxTrackID + 1, aTrack->GetID());
00140     mMaxTrackID = id;
00141 
00142     TrackRate rate = aTrack->GetRate();
00143     // Round up the track start time so the track, if anything, starts a
00144     // little later than the true time. This means we'll have enough
00145     // samples in our input stream to go just beyond the destination time.
00146     TrackTicks outputStart = TimeToTicksRoundUp(rate, GraphTimeToStreamTime(aFrom));
00147 
00148     nsAutoPtr<MediaSegment> segment;
00149     segment = aTrack->GetSegment()->CreateEmptyClone();
00150     for (uint32_t j = 0; j < mListeners.Length(); ++j) {
00151       MediaStreamListener* l = mListeners[j];
00152       l->NotifyQueuedTrackChanges(Graph(), id, rate, outputStart,
00153                                   MediaStreamListener::TRACK_EVENT_CREATED,
00154                                   *segment);
00155     }
00156     segment->AppendNullData(outputStart);
00157     StreamBuffer::Track* track =
00158       &mBuffer.AddTrack(id, rate, outputStart, segment.forget());
00159     LOG(PR_LOG_DEBUG, ("TrackUnionStream %p adding track %d for input stream %p track %d, start ticks %lld",
00160                        this, id, aPort->GetSource(), aTrack->GetID(),
00161                        (long long)outputStart));
00162 
00163     TrackMapEntry* map = mTrackMap.AppendElement();
00164     map->mInputPort = aPort;
00165     map->mInputTrackID = aTrack->GetID();
00166     map->mOutputTrackID = track->GetID();
00167     map->mSegment = aTrack->GetSegment()->CreateEmptyClone();
00168     return mTrackMap.Length() - 1;

00172     StreamBuffer::Track* outputTrack = mBuffer.FindTrack(mTrackMap[aIndex].mOutputTrackID);

00187   void CopyTrackData(StreamBuffer::Track* aInputTrack,
00188                      uint32_t aMapIndex, GraphTime aFrom, GraphTime aTo,
00189                      bool* aOutputTrackFinished)
00190   {
00191     TrackMapEntry* map = &mTrackMap[aMapIndex];
00192     StreamBuffer::Track* outputTrack = mBuffer.FindTrack(map->mOutputTrackID);
00193     MOZ_ASSERT(outputTrack && !outputTrack->IsEnded(), "Can't copy to ended track");
00194 
00195     TrackRate rate = outputTrack->GetRate();
00196     MediaSegment* segment = map->mSegment;
00197     MediaStream* source = map->mInputPort->GetSource();
00198 
00199     GraphTime next;
00200     *aOutputTrackFinished = false;
00201     for (GraphTime t = aFrom; t < aTo; t = next) {
00202       MediaInputPort::InputInterval interval = map->mInputPort->GetNextInputInterval(t);
00203       interval.mEnd = std::min(interval.mEnd, aTo);
00204       if (interval.mStart >= interval.mEnd)
00205         break;
00206       next = interval.mEnd;
00207 
00208       // Ticks >= startTicks and < endTicks are in the interval
00209       StreamTime outputEnd = GraphTimeToStreamTime(interval.mEnd);
00210       TrackTicks startTicks = outputTrack->GetEnd();
00211 #ifdef DEBUG
00212       StreamTime outputStart = GraphTimeToStreamTime(interval.mStart);
00213 #endif
00214       NS_ASSERTION(startTicks == TimeToTicksRoundUp(rate, outputStart),
00215                    "Samples missing");
00216       TrackTicks endTicks = TimeToTicksRoundUp(rate, outputEnd);
00217       TrackTicks ticks = endTicks - startTicks;
00218       // StreamTime inputStart = source->GraphTimeToStreamTime(interval.mStart);
00219       StreamTime inputEnd = source->GraphTimeToStreamTime(interval.mEnd);
00220       TrackTicks inputTrackEndPoint = TRACK_TICKS_MAX;
00221 
00222       if (aInputTrack->IsEnded()) {
00223         TrackTicks inputEndTicks = aInputTrack->TimeToTicksRoundDown(inputEnd);
00224         if (aInputTrack->GetEnd() <= inputEndTicks) {
00225           inputTrackEndPoint = aInputTrack->GetEnd();
00226           *aOutputTrackFinished = true;
00227         }
00228       }
00229 
00230       if (interval.mInputIsBlocked) {
00231         // Maybe the input track ended?
00232         segment->AppendNullData(ticks);
00233         LOG(PR_LOG_DEBUG+1, ("TrackUnionStream %p appending %lld ticks of null data to track %d",
00234             this, (long long)ticks, outputTrack->GetID()));
00235       } else {
00236         // Figuring out which samples to use from the input stream is tricky
00237         // because its start time and our start time may differ by a fraction
00238         // of a tick. Assuming the input track hasn't ended, we have to ensure
00239         // that 'ticks' samples are gathered, even though a tick boundary may
00240         // occur between outputStart and outputEnd but not between inputStart
00241         // and inputEnd.
00242         // We'll take the latest samples we can.
00243         TrackTicks inputEndTicks = TimeToTicksRoundUp(rate, inputEnd);
00244         TrackTicks inputStartTicks = inputEndTicks - ticks;
00245         segment->AppendSlice(*aInputTrack->GetSegment(),
00246                              std::min(inputTrackEndPoint, inputStartTicks),
00247                              std::min(inputTrackEndPoint, inputEndTicks));
00248         LOG(PR_LOG_DEBUG+1, ("TrackUnionStream %p appending %lld ticks of input data to track %d",
00249             this, (long long)(std::min(inputTrackEndPoint, inputEndTicks) - std::min(inputTrackEndPoint, inputStartTicks)),
00250             outputTrack->GetID()));
00251       }
00252       ApplyTrackDisabling(outputTrack->GetID(), segment);
00253       for (uint32_t j = 0; j < mListeners.Length(); ++j) {
00254         MediaStreamListener* l = mListeners[j];
00255         l->NotifyQueuedTrackChanges(Graph(), outputTrack->GetID(),
00256                                     outputTrack->GetRate(), startTicks, 0,
00257                                     *segment);
00258       }
00259       outputTrack->GetSegment()->AppendFrom(segment);
00260     }
00261   }
