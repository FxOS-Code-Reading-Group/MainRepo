00076    mime types, file extensions, etc that are required.
00077    Use a vertical bar to separate types, end types with \0.
00078    FileVersion and ProductVersion are 32bit ints, all other
00079    entries are strings that MUST be terminated with a \0.
00080 
00081 AN EXAMPLE:
00082 
00083 RCDATA NP_INFO_ProductVersion { 1,0,0,1,}
00084 
00085 RCDATA NP_INFO_MIMEType    { "video/x-video|",
00086                              "video/x-flick\0" }
00087 RCDATA NP_INFO_FileExtents { "avi|",
00088                              "flc\0" }
00089 RCDATA NP_INFO_FileOpenName{ "MMOS2 video player(*.avi)|",
00090                              "MMOS2 Flc/Fli player(*.flc)\0" }
00091 
00092 RCDATA NP_INFO_FileVersion       { 1,0,0,1 }
00093 RCDATA NP_INFO_CompanyName       { "Netscape Communications\0" }
00094 RCDATA NP_INFO_FileDescription   { "NPAVI32 Extension DLL\0"
00095 RCDATA NP_INFO_InternalName      { "NPAVI32\0" )
00096 RCDATA NP_INFO_LegalCopyright    { "Copyright Netscape Communications \251 1996\0"
00097 RCDATA NP_INFO_OriginalFilename  { "NVAPI32.DLL" }
00098 RCDATA NP_INFO_ProductName       { "NPAVI32 Dynamic Link Library\0" }
00099 */
