00437     bool quickRejectY(SkScalar top, SkScalar bottom) const {
00438         SkASSERT(SkScalarToCompareType(top) <= SkScalarToCompareType(bottom));
00439         const SkRectCompareType& clipR = this->getLocalClipBoundsCompareType();
00440         // In the case where the clip is empty and we are provided with a
00441         // negative top and positive bottom parameter then this test will return
00442         // false even though it will be clipped. We have chosen to exclude that
00443         // check as it is rare and would result double the comparisons.
00444         return SkScalarToCompareType(top) >= clipR.fBottom
00445             || SkScalarToCompareType(bottom) <= clipR.fTop;
00446     }
