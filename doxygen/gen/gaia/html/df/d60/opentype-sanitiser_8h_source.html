00066   // This should be implemented to perform the actual write.
00067   virtual bool WriteRaw(const void *data, size_t length) = 0;

00069   bool Write(const void *data, size_t length) {
00070     if (!length) return false;
00071 
00072     const size_t orig_length = length;
00073     size_t offset = 0;
00074     if (chksum_buffer_offset_) {
00075       const size_t l =
00076         std::min(length, static_cast<size_t>(4) - chksum_buffer_offset_);
00077       std::memcpy(chksum_buffer_ + chksum_buffer_offset_, data, l);
00078       chksum_buffer_offset_ += l;
00079       offset += l;
00080       length -= l;
00081     }
00082 
00083     if (chksum_buffer_offset_ == 4) {
00084       uint32_t chksum;
00085       std::memcpy(&chksum, chksum_buffer_, 4);
00086       chksum_ += ntohl(chksum);
00087       chksum_buffer_offset_ = 0;
00088     }
00089 
00090     while (length >= 4) {
00091       uint32_t tmp;
00092       std::memcpy(&tmp, reinterpret_cast<const uint8_t *>(data) + offset,
00093         sizeof(uint32_t));
00094       chksum_ += ntohl(tmp);
00095       length -= 4;
00096       offset += 4;
00097     }
00098 
00099     if (length) {
00100       if (chksum_buffer_offset_ != 0) return false;  // not reached
00101       if (length > 4) return false;  // not reached
00102       std::memcpy(chksum_buffer_,
00103              reinterpret_cast<const uint8_t*>(data) + offset, length);
00104       chksum_buffer_offset_ = length;
00105     }
00106 
00107     return WriteRaw(data, orig_length);
00108   }

00113   virtual bool Pad(size_t bytes) {
00114     static const uint32_t kZero = 0;
00115     while (bytes >= 4) {
00116       if (!WriteTag(kZero)) return false;
00117       bytes -= 4;
00118     }
00119     while (bytes) {
00120       static const uint8_t kZerob = 0;
00121       if (!Write(&kZerob, 1)) return false;
00122       bytes--;
00123     }
00124     return true;
00125   }
