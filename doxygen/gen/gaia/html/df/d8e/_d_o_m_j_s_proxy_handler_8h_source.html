00106   if (MOZ_LIKELY(JSID_IS_ATOM(id))) {
00107     JSAtom* atom = JSID_TO_ATOM(id);
00108     jschar s = *js::GetAtomChars(atom);
00109     if (MOZ_LIKELY((unsigned)s >= 'a' && (unsigned)s <= 'z'))
00110       return -1;

00112     uint32_t i;
00113     JSLinearString* str = js::AtomToLinearString(JSID_TO_ATOM(id));

00115   }
