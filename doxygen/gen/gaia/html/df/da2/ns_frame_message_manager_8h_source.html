00176       static_cast<nsFrameMessageManager*>(mChildManagers[i - 1])->
00177         Disconnect(false);
00178     }
00179     if (mIsProcessManager) {
00180       if (this == sParentProcessManager) {
00181         sParentProcessManager = nullptr;
00182       }
00183       if (this == sChildProcessManager) {
00184         sChildProcessManager = nullptr;
00185         delete sPendingSameProcessAsyncMessages;
00186         sPendingSameProcessAsyncMessages = nullptr;
