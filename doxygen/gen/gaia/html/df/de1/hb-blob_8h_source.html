00039 /*
00040  * Note re various memory-modes:
00041  *
00042  * - In no case shall the HarfBuzz client modify memory
00043  *   that is passed to HarfBuzz in a blob.  If there is
00044  *   any such possibility, MODE_DUPLICATE should be used
00045  *   such that HarfBuzz makes a copy immediately,
00046  *
00047  * - Use MODE_READONLY otherse, unless you really really
00048  *   really know what you are doing,
00049  *
00050  * - MODE_WRITABLE is appropriate if you relaly made a
00051  *   copy of data solely for the purpose of passing to
00052  *   HarfBuzz and doing that just once (no reuse!),
00053  *
00054  * - If the font is mmap()ed, it's ok to use
00055  *   READONLY_MAY_MAKE_WRITABLE, however, there were
00056  *   design problems with that mode, so HarfBuzz do not
00057  *   really use it anymore.  If not sure, use MODE_READONLY.
00058  */
00059 typedef enum {
00060   HB_MEMORY_MODE_DUPLICATE,
00061   HB_MEMORY_MODE_READONLY,
00062   HB_MEMORY_MODE_WRITABLE,
00063   HB_MEMORY_MODE_READONLY_MAY_MAKE_WRITABLE
00064 } hb_memory_mode_t;
