00029 function SystemMessageManager() {
00030   // If we have a system message handler registered for messages of type
00031   // |type|, this._dispatchers[type] equals {handler, messages, isHandling},
00032   // where
00033   //  - |handler| is the message handler that the page registered,
00034   //  - |messages| is a list of messages which we've received while
00035   //    dispatching messages to the handler, but haven't yet sent, and
00036   //  - |isHandling| indicates whether we're currently dispatching messages
00037   //    to this handler.
00038   this._dispatchers = {};

00063       // Queue up the incomming message if we're currently dispatching a

00068       // and we can then try to send the page another message while it's
00069       // inside this nested event loop.
00070       aDispatcher.messages.push(aMessage);
00071       return;
00072     }

00079     // "@mozilla.org/dom/system-messages/wrapper/TYPE;1" component implementing
00080     // nsISystemMessageWrapper.
