00014  * Memory reporters measure Firefox's memory usage.  They are mainly used to

00016  * https://wiki.mozilla.org/Memory_Reporting before writing a memory

00020 /*
00021  * An nsIMemoryReporter reports a single memory measurement as an object.
00022  * Use this when it makes sense to gather this measurement without gathering
00023  * related measurements at the same time.
00024  *
00025  * Note that the |amount| field may be implemented as a function, and so
00026  * accessing it can trigger significant computation;  the other fields can
00027  * be accessed without triggering this computation.  (Compare and contrast
00028  * this with nsIMemoryMultiReporter.)  
00029  *
00030  * aboutMemory.js is the most important consumer of memory reports.  It
00031  * places the following constraints on reports.
00032  *
00033  * - There must be an "explicit" tree.  It represents non-overlapping
00034  *   regions of memory that have been explicitly allocated with an
00035  *   OS-level allocation (e.g. mmap/VirtualAlloc/vm_allocate) or a
00036  *   heap-level allocation (e.g. malloc/calloc/operator new).  Reporters
00037  *   in this tree must have kind HEAP or NONHEAP, units BYTES, and a
00038  *   description that is a sentence (i.e. starts with a capital letter and
00039  *   ends with a period, or similar).
00040  *
00041  * - The "size", "rss", "pss" and "swap" trees are optional.  They
00042  *   represent regions of virtual memory that the process has mapped.
00043  *   Reporters in this category must have kind NONHEAP, units BYTES, and a
00044  *   non-empty description.
00045  *
00046  * - The "compartments" and "ghost-windows" trees are optional.  They are
00047  *   used by about:compartments.  Reporters in these trees must have kind
00048  *   OTHER, units COUNT, an amount of 1, and a description that's an empty
00049  *   string.
00050  *
00051  * - All other reports are unconstrained except that they must have a
00052  *   description that is a sentence.
00053  */
00054 [scriptable, uuid(b2c39f65-1799-4b92-a806-ab3cf6af3cfa)]
00055 interface nsIMemoryReporter : nsISupports
00056 {
00057   /*
00058    * The name of the process containing this reporter.  Each reporter initially
00059    * has "" in this field, indicating that it applies to the current process.
00060    * (This is true even for reporters in a child process.)  When a reporter
00061    * from a child process is copied into the main process, the copy has its
00062    * 'process' field set appropriately.
00063    */
00064   readonly attribute ACString process;
00065 
00066   /*
00067    * The path that this memory usage should be reported under.  Paths are
00068    * '/'-delimited, eg. "a/b/c".  
00069    *
00070    * Each reporter can be viewed as representing a leaf node in a tree.
00071    * Internal nodes of the tree don't have reporters.  So, for example, the
00072    * reporters "explicit/a/b", "explicit/a/c", "explicit/d/e", and
00073    * "explicit/d/f" define this tree:
00074    *
00075    *   explicit
00076    *   |--a
00077    *   |  |--b [*]
00078    *   |  \--c [*]
00079    *   \--d
00080    *      |--e [*]
00081    *      \--f [*]
00082    *
00083    * Nodes marked with a [*] have a reporter.  Notice that the internal
00084    * nodes are implicitly defined by the paths.
00085    *
00086    * Nodes within a tree should not overlap measurements, otherwise the
00087    * parent node measurements will be double-counted.  So in the example
00088    * above, |b| should not count any allocations counted by |c|, and vice
00089    * versa.
00090    *
00091    * All nodes within each tree must have the same units.
00092    *
00093    * If you want to include a '/' not as a path separator, e.g. because the
00094    * path contains a URL, you need to convert each '/' in the URL to a '\'.
00095    * Consumers of the path will undo this change.  Any other '\' character
00096    * in a path will also be changed.  This is clumsy but hasn't caused any
00097    * problems so far.
00098    *
00099    * The paths of all reporters form a set of trees.  Trees can be
00100    * "degenerate", i.e. contain a single entry with no '/'.
00101    */
00102   readonly attribute AUTF8String path;
00103 
00104   /*
00105    * There are three kinds of memory reporters.
00106    *
00107    *  - HEAP: reporters measuring memory allocated by the heap allocator,
00108    *    e.g. by calling malloc, calloc, realloc, memalign, operator new, or
00109    *    operator new[].  Reporters in this category must have units
00110    *    UNITS_BYTES.
00111    *
00112    *  - NONHEAP: reporters measuring memory which the program explicitly
00113    *    allocated, but does not live on the heap.  Such memory is commonly
00114    *    allocated by calling one of the OS's memory-mapping functions (e.g.
00115    *    mmap, VirtualAlloc, or vm_allocate).  Reporters in this category
00116    *    must have units UNITS_BYTES.
00117    *
00118    *  - OTHER: reporters which don't fit into either of these categories.
00119    *    They can have any units.
00120    *
00121    * The kind only matters for reporters in the "explicit" tree;
00122    * aboutMemory.js uses it to calculate "heap-unclassified".
00123    */
00124   const int32_t KIND_NONHEAP = 0;
00125   const int32_t KIND_HEAP    = 1;
00126   const int32_t KIND_OTHER   = 2;
00127 
00128   /*
00129    * The reporter kind.  See KIND_* above.
00130    */
00131   readonly attribute int32_t kind;
00132 
00133   /*
00134    * The amount reported by a memory reporter must have one of the following
00135    * units, but you may of course add new units as necessary:
00136    *
00137    *  - BYTES: The amount contains a number of bytes.
00138    *
00139    *  - COUNT: The amount is an instantaneous count of things currently in
00140    *    existence.  For instance, the number of tabs currently open would have
00141    *    units COUNT.
00142    *
00143    *  - COUNT_CUMULATIVE: The amount contains the number of times some event
00144    *    has occurred since the application started up.  For instance, the
00145    *    number of times the user has opened a new tab would have units
00146    *    COUNT_CUMULATIVE.
00147    *
00148    *    The amount returned by a reporter with units COUNT_CUMULATIVE must
00149    *    never decrease over the lifetime of the application.
00150    *
00151    *  - PERCENTAGE: The amount contains a fraction that should be expressed as
00152    *    a percentage.  NOTE!  The |amount| field should be given a value 100x
00153    *    the actual percentage;  this number will be divided by 100 when shown.
00154    *    This allows a fractional percentage to be shown even though |amount| is
00155    *    an integer.  E.g. if the actual percentage is 12.34%, |amount| should
00156    *    be 1234.
00157    *
00158    *    Values greater than 100% are allowed.
00159    */
00160   const int32_t UNITS_BYTES = 0;
00161   const int32_t UNITS_COUNT = 1;
00162   const int32_t UNITS_COUNT_CUMULATIVE = 2;
00163   const int32_t UNITS_PERCENTAGE = 3;
00164 
00165   /*
00166    * The units on the reporter's amount.  See UNITS_* above.
00167    */
00168   readonly attribute int32_t units;
00169 
00170   /*
00171    * The numeric value reported by this memory reporter.  Accesses can fail if
00172    * something goes wrong when getting the amount.
00173    */
00174   readonly attribute int64_t amount;
00175 
00176   /*
00177    * A human-readable description of this memory usage report.
00178    */
00179   readonly attribute AUTF8String description;
00180 };
