00110   // Initially the time of the event is always in absolute value.

00241   template<class TimeType>
00242   float GetValueAtTime(TimeType aTime) const
00243   {
00244     const AudioTimelineEvent* previous = nullptr;
00245     const AudioTimelineEvent* next = nullptr;
00246 
00247     bool bailOut = false;
00248     for (unsigned i = 0; !bailOut && i < mEvents.Length(); ++i) {
00249       switch (mEvents[i].mType) {
00250       case AudioTimelineEvent::SetValue:
00251       case AudioTimelineEvent::SetTarget:
00252       case AudioTimelineEvent::LinearRamp:
00253       case AudioTimelineEvent::ExponentialRamp:
00254       case AudioTimelineEvent::SetValueCurve:
00255         if (aTime == mEvents[i].template Time<TimeType>()) {
00256           // Find the last event with the same time
00257           do {
00258             ++i;
00259           } while (i < mEvents.Length() &&
00260                    aTime == mEvents[i].template Time<TimeType>());
00261 
00262           // SetTarget nodes can be handled no matter what their next node is (if they have one)
00263           if (mEvents[i - 1].mType == AudioTimelineEvent::SetTarget) {
00264             // Follow the curve, without regard to the next node
00265             return ExponentialApproach(mEvents[i - 1].template Time<TimeType>(),
00266                                        mValue, mEvents[i - 1].mValue,
00267                                        mEvents[i - 1].mTimeConstant, aTime);
00268           }
00269 
00270           // SetValueCurve events can be handled no mattar what their next node is (if they have one)
00271           if (mEvents[i - 1].mType == AudioTimelineEvent::SetValueCurve) {
00272             return ExtractValueFromCurve(mEvents[i - 1].template Time<TimeType>(),
00273                                          mEvents[i - 1].mCurve,
00274                                          mEvents[i - 1].mCurveLength,
00275                                          mEvents[i - 1].mDuration, aTime);
00276           }
00277 
00278           // For other event types
00279           return mEvents[i - 1].mValue;
00280         }
00281         previous = next;
00282         next = &mEvents[i];
00283         if (aTime < mEvents[i].template Time<TimeType>()) {
00284           bailOut = true;
00285         }
00286         break;
00287       default:
00288         MOZ_ASSERT(false, "unreached");
00289       }
00290     }
00291     // Handle the case where the time is past all of the events
00292     if (!bailOut) {
00293       previous = next;
00294       next = nullptr;
00295     }
00296 
00297     // Just return the default value if we did not find anything
00298     if (!previous && !next) {
00299       return mValue;
00300     }
00301 
00302     // If the requested time is before all of the existing events
00303     if (!previous) {
00304       return mValue;
00305     }
00306 
00307     // SetTarget nodes can be handled no matter what their next node is (if they have one)
00308     if (previous->mType == AudioTimelineEvent::SetTarget) {
00309       // Follow the curve, without regard to the next node
00310       return ExponentialApproach(previous->template Time<TimeType>(), mValue, previous->mValue,
00311                                  previous->mTimeConstant, aTime);
00312     }
00313 
00314     // SetValueCurve events can be handled no mattar what their next node is (if they have one)
00315     if (previous->mType == AudioTimelineEvent::SetValueCurve) {
00316       return ExtractValueFromCurve(previous->template Time<TimeType>(),
00317                                    previous->mCurve, previous->mCurveLength,
00318                                    previous->mDuration, aTime);
00319     }
00320 
00321     // If the requested time is after all of the existing events
00322     if (!next) {
00323       switch (previous->mType) {
00324       case AudioTimelineEvent::SetValue:
00325       case AudioTimelineEvent::LinearRamp:
00326       case AudioTimelineEvent::ExponentialRamp:
00327         // The value will be constant after the last event
00328         return previous->mValue;
00329       case AudioTimelineEvent::SetValueCurve:
00330         return ExtractValueFromCurve(previous->template Time<TimeType>(),
00331                                      previous->mCurve, previous->mCurveLength,
00332                                      previous->mDuration, aTime);
00333       case AudioTimelineEvent::SetTarget:
00334         MOZ_ASSERT(false, "unreached");
00335       }
00336       MOZ_ASSERT(false, "unreached");
00337     }
00338 
00339     // Finally, handle the case where we have both a previous and a next event
00340 
00341     // First, handle the case where our range ends up in a ramp event
00342     switch (next->mType) {
00343     case AudioTimelineEvent::LinearRamp:
00344       return LinearInterpolate(previous->template Time<TimeType>(), previous->mValue, next->template Time<TimeType>(), next->mValue, aTime);
00345     case AudioTimelineEvent::ExponentialRamp:
00346       return ExponentialInterpolate(previous->template Time<TimeType>(), previous->mValue, next->template Time<TimeType>(), next->mValue, aTime);
00347     case AudioTimelineEvent::SetValue:
00348     case AudioTimelineEvent::SetTarget:
00349     case AudioTimelineEvent::SetValueCurve:
00350       break;
00351     }
00352 
00353     // Now handle all other cases
00354     switch (previous->mType) {
00355     case AudioTimelineEvent::SetValue:
00356     case AudioTimelineEvent::LinearRamp:
00357     case AudioTimelineEvent::ExponentialRamp:
00358       // If the next event type is neither linear or exponential ramp, the
00359       // value is constant.
00360       return previous->mValue;
00361     case AudioTimelineEvent::SetValueCurve:
00362       return ExtractValueFromCurve(previous->template Time<TimeType>(),
00363                                    previous->mCurve, previous->mCurveLength,
00364                                    previous->mDuration, aTime);
00365     case AudioTimelineEvent::SetTarget:
00366       MOZ_ASSERT(false, "unreached");
00367     }
00368 
00369     MOZ_ASSERT(false, "unreached");
00370     return 0.0f;
00371   }

00379   static float LinearInterpolate(double t0, float v0, double t1, float v1, double t)
00380   {
00381     return v0 + (v1 - v0) * ((t - t0) / (t1 - t0));
00382   }

00384   static float ExponentialInterpolate(double t0, float v0, double t1, float v1, double t)
00385   {
00386     return v0 * powf(v1 / v0, (t - t0) / (t1 - t0));
00387   }

00389   static float ExponentialApproach(double t0, double v0, float v1, double timeConstant, double t)
00390   {
00391     return v1 + (v0 - v1) * expf(-(t - t0) / timeConstant);
00392   }

00394   static float ExtractValueFromCurve(double startTime, float* aCurve, uint32_t aCurveLength, double duration, double t)
00395   {
00396     if (t >= startTime + duration) {
00397       // After the duration, return the last curve value
00398       return aCurve[aCurveLength - 1];
00399     }
00400     double ratio = (t - startTime) / duration;
00401     MOZ_ASSERT(ratio >= 0.0, "Ratio can never be negative here");
00402     if (ratio >= 1.0) {
00403       return aCurve[aCurveLength - 1];
00404     }
00405     return aCurve[uint32_t(aCurveLength * ratio)];
00406   }

00458   void InsertEvent(const AudioTimelineEvent& aEvent, ErrorResult& aRv)
00459   {
00460     if (!aEvent.IsValid()) {
00461       aRv.Throw(NS_ERROR_DOM_SYNTAX_ERR);
00462       return;
00463     }
00464 
00465     // Make sure that non-curve events don't fall within the duration of a
00466     // curve event.
00467     for (unsigned i = 0; i < mEvents.Length(); ++i) {
00468       if (mEvents[i].mType == AudioTimelineEvent::SetValueCurve &&
00469           mEvents[i].mTime <= aEvent.mTime &&
00470           (mEvents[i].mTime + mEvents[i].mDuration) >= aEvent.mTime) {
00471         aRv.Throw(NS_ERROR_DOM_SYNTAX_ERR);
00472         return;
00473       }
00474     }
00475 
00476     // Make sure that curve events don't fall in a range which includes other
00477     // events.
00478     if (aEvent.mType == AudioTimelineEvent::SetValueCurve) {
00479       for (unsigned i = 0; i < mEvents.Length(); ++i) {
00480         if (mEvents[i].mTime > aEvent.mTime &&
00481             mEvents[i].mTime < (aEvent.mTime + aEvent.mDuration)) {
00482           aRv.Throw(NS_ERROR_DOM_SYNTAX_ERR);
00483           return;
00484         }
00485       }
00486     }
00487 
00488     // Make sure that invalid values are not used for exponential curves
00489     if (aEvent.mType == AudioTimelineEvent::ExponentialRamp) {
00490       if (aEvent.mValue <= 0.f) {
00491         aRv.Throw(NS_ERROR_DOM_SYNTAX_ERR);
00492         return;
00493       }
00494       const AudioTimelineEvent* previousEvent = GetPreviousEvent(aEvent.mTime);
00495       if (previousEvent) {
00496         if (previousEvent->mValue <= 0.f) {
00497           aRv.Throw(NS_ERROR_DOM_SYNTAX_ERR);
00498           return;
00499         }
00500       } else {
00501         if (mValue <= 0.f) {
00502           aRv.Throw(NS_ERROR_DOM_SYNTAX_ERR);
00503           return;
00504         }
00505       }
00506     }
00507 
00508     for (unsigned i = 0; i < mEvents.Length(); ++i) {
00509       if (aEvent.mTime == mEvents[i].mTime) {
00510         if (aEvent.mType == mEvents[i].mType) {
00511           // If times and types are equal, replace the event
00512           mEvents.ReplaceElementAt(i, aEvent);
00513         } else {
00514           // Otherwise, place the element after the last event of another type
00515           do {
00516             ++i;
00517           } while (i < mEvents.Length() &&
00518                    aEvent.mType != mEvents[i].mType &&
00519                    aEvent.mTime == mEvents[i].mTime);
00520           mEvents.InsertElementAt(i, aEvent);
00521         }
00522         return;
00523       }
00524       // Otherwise, place the event right after the latest existing event
00525       if (aEvent.mTime < mEvents[i].mTime) {
00526         mEvents.InsertElementAt(i, aEvent);
00527         return;
00528       }
00529     }
00530 
00531     // If we couldn't find a place for the event, just append it to the list
00532     mEvents.AppendElement(aEvent);
00533   }
