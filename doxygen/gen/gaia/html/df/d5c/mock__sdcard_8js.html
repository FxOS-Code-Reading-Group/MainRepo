MockSdCard updateStorageState
初期値:
 function sd_updateStorageState(state) {
  switch (state) {
    case 'available':
    case 'shared':
      MockSdCard.status = MockSdCard.AVAILABLE;
      break;
    case 'unavailable':
    case 'deleted':
      MockSdCard.status = MockSdCard.NOT_AVAILABLE;
      break;
  }
}
