00013  * nsICommandManager is an interface used to executing user-level commands,
00014  * and getting the state of available commands.
00015  *
00016  * Commands are identified by strings, which are documented elsewhere.
00017  * In addition, the list of required and optional parameters for
00018  * each command, that are passed in via the nsICommandParams, are
00019  * also documented elsewhere. (Where? Need a good location for this).
00020  */
00021  
00022  
00023 [scriptable, uuid(080D2001-F91E-11D4-A73C-F9242928207C)]
00024 interface nsICommandManager : nsISupports
00025 {
00026   /*
00027    * Register an observer on the specified command. The observer's Observe
00028    * method will get called when the state (enabled/disbaled, or toggled etc)
00029    * of the command changes.
00030    *
00031    * You can register the same observer on multiple commmands by calling this
00032    * multiple times.
00033    */
00034   void        addCommandObserver(in nsIObserver aCommandObserver,
00035                                  in string aCommandToObserve);
00036 
00037   /*
00038    * Stop an observer from observering the specified command. If the observer
00039    * was also registered on ther commands, they will continue to be observed.
00040    *
00041    * Passing an empty string in 'aCommandObserved' will remove the observer
00042    * from all commands.
00043    */
00044   void        removeCommandObserver(in nsIObserver aCommandObserver,
00045                                     in string aCommandObserved);
00046 
00047   /*
00048    * Ask the command manager if the specified command is supported.
00049    * If aTargetWindow is null, the focused window is used.
00050    *
00051    */
00052   boolean     isCommandSupported(in string aCommandName,
00053                                  in nsIDOMWindow aTargetWindow);
00054 
00055   /*
00056    * Ask the command manager if the specified command is currently.
00057    * enabled.
00058    * If aTargetWindow is null, the focused window is used.
00059    */
00060   boolean     isCommandEnabled(in string aCommandName,
00061                                in nsIDOMWindow aTargetWindow);
00062 
00063   /*
00064    * Get the state of the specified commands.
00065    *
00066    * On input: aCommandParams filled in with values that the caller cares
00067    * about, most of which are command-specific (see the command documentation
00068    * for details). One boolean value, "enabled", applies to all commands,
00069    * and, in return will be set to indicate whether the command is enabled
00070    * (equivalent to calling isCommandEnabled).
00071    *
00072    * aCommandName is the name of the command that needs the state
00073    * aTargetWindow is the source of command controller 
00074    *      (null means use focus controller)
00075    * On output: aCommandParams: values set by the caller filled in with
00076    * state from the command.
00077    */
00078   void        getCommandState(in string aCommandName,
00079                               in nsIDOMWindow aTargetWindow,
00080                   /* inout */ in nsICommandParams aCommandParams);
00081     
00082   /*
00083    * Execute the specified command.
00084    * The command will be executed in aTargetWindow if it is specified.
00085    * If aTargetWindow is null, it will go to the focused window.
00086    *
00087    * param: aCommandParams, a list of name-value pairs of command parameters,
00088    * may be null for parameter-less commands.
00089    *
00090    */
00091   void        doCommand(in string aCommandName,
00092                         in nsICommandParams aCommandParams,
00093                         in nsIDOMWindow aTargetWindow);
00094 
00095 };
