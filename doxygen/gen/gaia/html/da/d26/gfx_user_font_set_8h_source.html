00087     void ReplaceFontEntry(gfxFontEntry *aOldFontEntry,
00088                           gfxFontEntry *aNewFontEntry) {
00089         uint32_t numFonts = mAvailableFonts.Length();
00090         uint32_t i;
00091         for (i = 0; i < numFonts; i++) {
00092             gfxFontEntry *fe = mAvailableFonts[i];
00093             if (fe == aOldFontEntry) {
00094                 // note that this may delete aOldFontEntry, if there's no
00095                 // other reference to it except from its family
00096                 aNewFontEntry->mFamilyName = Name();
00097                 mAvailableFonts[i] = aNewFontEntry;
00098                 break;
00099             }
00100         }
00101         NS_ASSERTION(i < numFonts, "font entry not found in family!");
00102         ResetCharacterMap();
00103     }
00104 
00105     void RemoveFontEntry(gfxFontEntry *aFontEntry) {
00106         uint32_t numFonts = mAvailableFonts.Length();
00107         for (uint32_t i = 0; i < numFonts; i++) {
00108             gfxFontEntry *fe = mAvailableFonts[i];
00109             if (fe == aFontEntry) {
00110                 mAvailableFonts.RemoveElementAt(i);
00111                 break;
00112             }
00113         }
00114         ResetCharacterMap();
00115     }
