00080 /* Getter/Setter macros.
00081    Usage:
00082    NS_IMPL_[CLASS_]GETTER[_<type>](method, [type,] member);
00083    NS_IMPL_[CLASS_]SETTER[_<type>](method, [type,] member);
00084    NS_IMPL_[CLASS_]GETSET[_<type>]([class, ]postfix, [type,] member);
00085    
00086    where:
00087    CLASS_  - implementation is inside a class definition
00088              (otherwise the class name is needed)
00089              Do NOT use in publicly exported header files, because
00090              the implementation may be included many times over.
00091              Instead, use the non-CLASS_ version.
00092    _<type> - For more complex (STR, IFACE) data types
00093              (otherwise the simple data type is needed)
00094    method  - name of the method, such as GetWidth or SetColor
00095    type    - simple data type if required
00096    member  - class member variable such as m_width or mColor
00097    class   - the class name, such as Window or MyObject
00098    postfix - Method part after Get/Set such as "Width" for "GetWidth"
00099    
00100    Example:
00101    class Window {
00102    public:
00103      NS_IMPL_CLASS_GETSET(Width, int, m_width);
00104      NS_IMPL_CLASS_GETTER_STR(GetColor, m_color);
00105      NS_IMETHOD SetColor(char *color);
00106      
00107    private:
00108      int m_width;     // read/write
00109      char *m_color;   // readonly
00110    };
00111 
00112    // defined outside of class
00113    NS_IMPL_SETTER_STR(Window::GetColor, m_color);
00114 
00115    Questions/Comments to alecf@netscape.com
00116 */
