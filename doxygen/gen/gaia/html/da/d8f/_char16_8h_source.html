00014 /*
00015  * C11 and C++11 introduce a char16_t type and support for UTF-16 string and
00016  * character literals. C++11's char16_t is a distinct builtin type. C11's
00017  * char16_t is a typedef for uint_least16_t. Technically, char16_t is a 16-bit
00018  * code unit of a Unicode code point, not a "character".
00019  *
00020  * For now, Char16.h only supports C++ because we don't want mix different C
00021  * and C++ definitions of char16_t in the same code base.
00022  */
00023 
00024 #ifdef _MSC_VER
00025    /*
00026     * C++11 says char16_t is a distinct builtin type, but Windows's yvals.h
00027     * typedefs char16_t as an unsigned short. We would like to alias char16_t
00028     * to Windows's 16-bit wchar_t so we can declare UTF-16 literals as constant
00029     * expressions (and pass char16_t pointers to Windows APIs). We #define our
00030     * char16_t as a macro to override yval.h's typedef of the same name.
00031     */
00032 #  define MOZ_UTF16_HELPER(s) L##s
00033 #  include <yvals.h>
00034 #  define char16_t wchar_t
00035 #elif defined(__cplusplus) && \
00036       (__cplusplus >= 201103L || defined(__GXX_EXPERIMENTAL_CXX0X__))
00037    /* C++11 has a builtin char16_t type. */
00038 #  define MOZ_UTF16_HELPER(s) u##s
00039 #else
00040 #  error "Char16.h requires C++11 (or something like it) for UTF-16 support."
00041 #endif

00043 /*
00044  * Macro arguments used in concatenation or stringification won't be expanded.
00045  * Therefore, in order for |MOZ_UTF16(FOO)| to work as expected (which is to
00046  * expand |FOO| before doing whatever |MOZ_UTF16| needs to do to it) a helper
00047  * macro, |MOZ_UTF16_HELPER| needs to be inserted in between to allow the macro
00048  * argument to expand. See "3.10.6 Separate Expansion of Macro Arguments" of the
00049  * CPP manual for a more accurate and precise explanation.
00050  */
00051 #define MOZ_UTF16(s) MOZ_UTF16_HELPER(s)
