00013 // Is this pseudo-element a CSS2 pseudo-element that can be specified
00014 // with the single colon syntax (in addition to the double-colon syntax,
00015 // which can be used for all pseudo-elements)?
00016 #define CSS_PSEUDO_ELEMENT_IS_CSS2                (1<<0)
00017 // Is this pseudo-element a pseudo-element that can contain other
00018 // elements?
00019 // (Currently pseudo-elements are either leaves of the tree (relative to
00020 // real elements) or they contain other elements in a non-tree-like
00021 // manner (i.e., like incorrectly-nested start and end tags).  It's
00022 // possible that in the future there might be container pseudo-elements
00023 // that form a properly nested tree structure.  If that happens, we
00024 // should probably split this flag into two.)
00025 #define CSS_PSEUDO_ELEMENT_CONTAINS_ELEMENTS      (1<<1)
