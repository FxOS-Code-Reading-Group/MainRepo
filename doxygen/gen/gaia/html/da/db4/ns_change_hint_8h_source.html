00147 // Merge the "src" hint into the "dst" hint
00148 // Returns true iff the destination changed
00149 inline bool NS_UpdateHint(nsChangeHint& aDest, nsChangeHint aSrc) {
00150   nsChangeHint r = (nsChangeHint)(aDest | aSrc);
00151   bool changed = (int)r != (int)aDest;
00152   aDest = r;
00153   return changed;
00154 }

00156 // Returns true iff the second hint contains all the hints of the first hint
00157 inline bool NS_IsHintSubset(nsChangeHint aSubset, nsChangeHint aSuperSet) {

00182 inline nsChangeHint NS_HintsNotHandledForDescendantsIn(nsChangeHint aChangeHint) {
00183   nsChangeHint result = nsChangeHint(aChangeHint & (
00184     nsChangeHint_UpdateTransformLayer |
00185     nsChangeHint_UpdateEffects |
00186     nsChangeHint_UpdateOpacityLayer |
00187     nsChangeHint_UpdateOverflow |
00188     nsChangeHint_ChildrenOnlyTransform |
00189     nsChangeHint_RecomputePosition |
00190     nsChangeHint_AddOrRemoveTransform |
00191     nsChangeHint_BorderStyleNoneChange));
00192 
00193   if (!NS_IsHintSubset(nsChangeHint_NeedDirtyReflow, aChangeHint) &&
00194       NS_IsHintSubset(nsChangeHint_NeedReflow, aChangeHint)) {
00195     // If NeedDirtyReflow is *not* set, then NeedReflow is a
00196     // non-inherited hint.
00197     NS_UpdateHint(result, nsChangeHint_NeedReflow);
00198   }
00199 
00200   if (!NS_IsHintSubset(nsChangeHint_ClearDescendantIntrinsics, aChangeHint) &&
00201       NS_IsHintSubset(nsChangeHint_ClearAncestorIntrinsics, aChangeHint)) {
00202     // If ClearDescendantIntrinsics is *not* set, then
00203     // ClearAncestorIntrinsics is a non-inherited hint.
00204     NS_UpdateHint(result, nsChangeHint_ClearAncestorIntrinsics);
00205   }
00206 
00207   NS_ABORT_IF_FALSE(NS_IsHintSubset(result,
00208                                     nsChangeHint_Hints_NotHandledForDescendants),
00209                     "something is inconsistent");
00210 
00211   return result;
00212 }
