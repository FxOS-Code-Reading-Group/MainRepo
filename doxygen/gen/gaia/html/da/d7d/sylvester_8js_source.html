00642   inverse: function() {
00643     if (!this.isSquare() || this.isSingular()) { return null; }
00644     var ni = this.elements.length, ki = ni, i, j;
00645     var M = this.augment(Matrix.I(ni)).toRightTriangular();
00646     var np, kp = M.elements[0].length, p, els, divisor;
00647     var inverse_elements = [], new_element;
00648     // Matrix is non-singular so there will be no zeros on the diagonal
00649     // Cycle through rows from last to first
00650     do { i = ni - 1;
00651       // First, normalise diagonal elements to 1
00652       els = []; np = kp;
00653       inverse_elements[i] = [];

00657         els.push(new_element);
00658         // Shuffle of the current row of the right hand side into the results
00659         // array as it will not be modified by later runs through this loop
00660         if (p >= ki) { inverse_elements[i].push(new_element); }
00661       } while (--np);
00662       M.elements[i] = els;
00663       // Then, subtract this row from those above it to
00664       // give the identity matrix on the left hand side
00665       for (j = 0; j < i; j++) {
00666         els = []; np = kp;
00667         do { p = kp - np;
00668           els.push(M.elements[j][p] - M.elements[i][p] * M.elements[j][i]);
00669         } while (--np);
00670         M.elements[j] = els;
00671       }
00672     } while (--ni);

00754 // Rotation matrix about some axis. If no axis is
00755 // supplied, assume we're after a 2D transform
00756 Matrix.Rotation = function(theta, a) {
00757   if (!a) {
00758     return Matrix.create([
00759       [Math.cos(theta),  -Math.sin(theta)],
00760       [Math.sin(theta),   Math.cos(theta)]
00761     ]);
00762   }
00763   var axis = a.dup();
00764   if (axis.elements.length != 3) { return null; }
00765   var mod = axis.modulus();

00768   // Formula derived here: http://www.gamedev.net/reference/articles/article1199.asp
00769   // That proof rotates the co-ordinate system so theta
00770   // becomes -theta and sin becomes -sin here.
00771   return Matrix.create([
00772     [ t*x*x + c, t*x*y - s*z, t*x*z + s*y ],
00773     [ t*x*y + s*z, t*y*y + c, t*y*z - s*x ],
00774     [ t*x*z - s*y, t*y*z + s*x, t*z*z + c ]
00775   ]);
00776 };

01107   // Returns the unique intersection with the argument, if one exists. The result
01108   // will be a vector if a line is supplied, and a line if a plane is supplied.
01109   intersectionWith: function(obj) {
01110     if (!this.intersects(obj)) { return null; }
01111     if (obj.direction) {
01112       // obj is a line
01113       var A = obj.anchor.elements, D = obj.direction.elements,
01114           P = this.anchor.elements, N = this.normal.elements;
01115       var multiplier = (N[0]*(P[0]-A[0]) + N[1]*(P[1]-A[1]) + N[2]*(P[2]-A[2])) / (N[0]*D[0] + N[1]*D[1] + N[2]*D[2]);
01116       return Vector.create([A[0] + D[0]*multiplier, A[1] + D[1]*multiplier, A[2] + D[2]*multiplier]);
01117     } else if (obj.normal) {
01118       // obj is a plane
01119       var direction = this.normal.cross(obj.normal).toUnitVector();
01120       // To find an anchor point, we find one co-ordinate that has a value
01121       // of zero somewhere on the intersection, and remember which one we picked
01122       var N = this.normal.elements, A = this.anchor.elements,
01123           O = obj.normal.elements, B = obj.anchor.elements;
01124       var solver = Matrix.Zero(2,2), i = 0;
01125       while (solver.isSingular()) {
01126         i++;
01127         solver = Matrix.create([
01128           [ N[i%3], N[(i+1)%3] ],
01129           [ O[i%3], O[(i+1)%3]  ]
01130         ]);
01131       }
01132       // Then we solve the simultaneous equations in the remaining dimensions
01133       var inverse = solver.inverse().elements;
01134       var x = N[0]*A[0] + N[1]*A[1] + N[2]*A[2];
01135       var y = O[0]*B[0] + O[1]*B[1] + O[2]*B[2];
01136       var intersection = [
01137         inverse[0][0] * x + inverse[0][1] * y,
01138         inverse[1][0] * x + inverse[1][1] * y
01139       ];
01140       var anchor = [];
01141       for (var j = 1; j <= 3; j++) {
01142         // This formula picks the right element from intersection by
01143         // cycling depending on which element we set to zero above
01144         anchor.push((i == j) ? 0 : intersection[(j + (5 - i)%3)%3]);
01145       }
01146       return Line.create(anchor, direction);
01147     }

01177   // Returns the reflection of the plane in the given point, line or plane.
01178   reflectionIn: function(obj) {
01179     if (obj.normal) {
01180       // obj is a plane
01181       var A = this.anchor.elements, N = this.normal.elements;
01182       var A1 = A[0], A2 = A[1], A3 = A[2], N1 = N[0], N2 = N[1], N3 = N[2];
01183       var newA = this.anchor.reflectionIn(obj).elements;
01184       // Add the plane's normal to its anchor, then mirror that in the other plane
01185       var AN1 = A1 + N1, AN2 = A2 + N2, AN3 = A3 + N3;
01186       var Q = obj.pointClosestTo([AN1, AN2, AN3]).elements;
01187       var newN = [Q[0] + (Q[0] - AN1) - newA[0], Q[1] + (Q[1] - AN2) - newA[1], Q[2] + (Q[2] - AN3) - newA[2]];
01188       return Plane.create(newA, newN);
01189     } else if (obj.direction) {
01190       // obj is a line
01191       return this.rotate(Math.PI, obj);
01192     } else {
01193       // obj is a point
01194       var P = obj.elements || obj;
01195       return Plane.create(this.anchor.reflectionIn([P[0], P[1], (P[2] || 0)]), this.normal);

01199   // Sets the anchor point and normal to the plane. If three arguments are specified,
01200   // the normal is calculated by assuming the three points should lie in the same plane.
01201   // If only two are sepcified, the second is taken to be the normal. Normal vector is
01202   // normalised before storage.
01203   setVectors: function(anchor, v1, v2) {
01204     anchor = Vector.create(anchor);
01205     anchor = anchor.to3D(); if (anchor === null) { return null; }
01206     v1 = Vector.create(v1);
01207     v1 = v1.to3D(); if (v1 === null) { return null; }
01208     if (typeof(v2) == 'undefined') {
01209       v2 = null;
01210     } else {
01211       v2 = Vector.create(v2);
01212       v2 = v2.to3D(); if (v2 === null) { return null; }
01213     }
01214     var A1 = anchor.elements[0], A2 = anchor.elements[1], A3 = anchor.elements[2];
01215     var v11 = v1.elements[0], v12 = v1.elements[1], v13 = v1.elements[2];
01216     var normal, mod;
01217     if (v2 !== null) {
01218       var v21 = v2.elements[0], v22 = v2.elements[1], v23 = v2.elements[2];
01219       normal = Vector.create([
01220         (v12 - A2) * (v23 - A3) - (v13 - A3) * (v22 - A2),
01221         (v13 - A3) * (v21 - A1) - (v11 - A1) * (v23 - A3),
01222         (v11 - A1) * (v22 - A2) - (v12 - A2) * (v21 - A1)
01223       ]);
01224       mod = normal.modulus();
01225       if (mod === 0) { return null; }
01226       normal = Vector.create([normal.elements[0] / mod, normal.elements[1] / mod, normal.elements[2] / mod]);
01227     } else {
01228       mod = Math.sqrt(v11*v11 + v12*v12 + v13*v13);
01229       if (mod === 0) { return null; }
01230       normal = Vector.create([v1.elements[0] / mod, v1.elements[1] / mod, v1.elements[2] / mod]);
01231     }
01232     this.anchor = anchor;
01233     this.normal = normal;
01234     return this;
01235   }
01236 };
