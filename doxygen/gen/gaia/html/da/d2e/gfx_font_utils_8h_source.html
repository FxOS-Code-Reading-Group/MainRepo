00101         uint32_t startBlock, endBlock, blockLen;
00102         
00103         // start point is beyond the end of the block array? return false immediately
00104         startBlock = aStart >> BLOCK_INDEX_SHIFT;
00105         blockLen = mBlocks.Length();
00106         if (startBlock >= blockLen) return false;
00107         
00108         // check for blocks in range, if none, return false
00109         uint32_t blockIndex;
00110         bool hasBlocksInRange = false;
00111 
00112         endBlock = aEnd >> BLOCK_INDEX_SHIFT;
00113         blockIndex = startBlock;
00114         for (blockIndex = startBlock; blockIndex <= endBlock; blockIndex++) {
00115             if (blockIndex < blockLen && mBlocks[blockIndex])
00116                 hasBlocksInRange = true;
00117         }
00118         if (!hasBlocksInRange) return false;
00119 
00120         Block *block;
00121         uint32_t i, start, end;
00122         
00123         // first block, check bits
00124         if ((block = mBlocks[startBlock])) {
00125             start = aStart;
00126             end = std::min(aEnd, ((startBlock+1) << BLOCK_INDEX_SHIFT) - 1);
00127             for (i = start; i <= end; i++) {
00128                 if ((block->mBits[(i>>3) & (BLOCK_SIZE - 1)]) & (1 << (i & 0x7)))
00129                     return true;
00130             }
00131         }
00132         if (endBlock == startBlock) return false;
00133 
00134         // [2..n-1] blocks check bytes
00135         for (blockIndex = startBlock + 1; blockIndex < endBlock; blockIndex++) {
00136             uint32_t index;
00137             
00138             if (blockIndex >= blockLen || !(block = mBlocks[blockIndex])) continue;
00139             for (index = 0; index < BLOCK_SIZE; index++) {
00140                 if (block->mBits[index]) 
00141                     return true;
00142             }
00143         }
00144         
00145         // last block, check bits
00146         if (endBlock < blockLen && (block = mBlocks[endBlock])) {
00147             start = endBlock << BLOCK_INDEX_SHIFT;
00148             end = aEnd;
00149             for (i = start; i <= end; i++) {
00150                 if ((block->mBits[(i>>3) & (BLOCK_SIZE - 1)]) & (1 << (i & 0x7)))
00151                     return true;
00152             }
00153         }

00282     // set this bitset to the union of its current contents and another
00283     void Union(const gfxSparseBitSet& aBitset) {
00284         // ensure mBlocks is large enough
00285         uint32_t blockCount = aBitset.mBlocks.Length();
00286         if (blockCount > mBlocks.Length()) {
00287             uint32_t needed = blockCount - mBlocks.Length();
00288             nsAutoPtr<Block> *blocks = mBlocks.AppendElements(needed);
00289             if (MOZ_UNLIKELY(!blocks)) { // OOM
00290                 return;
00291             }
00292         }
00293         // for each block that may be present in aBitset...
00294         for (uint32_t i = 0; i < blockCount; ++i) {
00295             // if it is missing (implicitly empty), just skip
00296             if (!aBitset.mBlocks[i]) {
00297                 continue;
00298             }
00299             // if the block is missing in this set, just copy the other
00300             if (!mBlocks[i]) {
00301                 mBlocks[i] = new Block(*aBitset.mBlocks[i]);
00302                 continue;
00303             }
00304             // else set existing block to the union of both
00305             uint32_t *dst = reinterpret_cast<uint32_t*>(mBlocks[i]->mBits);
00306             const uint32_t *src =
00307                 reinterpret_cast<const uint32_t*>(aBitset.mBlocks[i]->mBits);
00308             for (uint32_t j = 0; j < BLOCK_SIZE / 4; ++j) {
00309                 dst[j] |= src[j];
00310             }
00311         }
00312     }

00621 // Return just the highest bit of the given value, i.e., the highest
00622 // power of 2 that is <= value, or zero if the input value is zero.
00623 inline uint32_t
00624 FindHighestBit(uint32_t value)
00625 {
00626     // propagate highest bit into all lower bits of the value
00627     value |= (value >> 1);
00628     value |= (value >> 2);
00629     value |= (value >> 4);
00630     value |= (value >> 8);
00631     value |= (value >> 16);
00632     // isolate the leftmost bit
00633     return (value & ~(value >> 1));
00634 }

00966         stateTimerOnInterval,

00971         mInterval(0), mState(stateInitial)
00972     {

00977     // start timer with an initial delay, then call Run method at regular intervals
00978     void StartLoader(uint32_t aDelay, uint32_t aInterval) {
00979         mInterval = aInterval;
00980 
00981         // sanity check
00982         if (mState != stateInitial && mState != stateTimerOff)
00983             CancelLoader();

00985         // set up timer
00986         if (!mTimer) {
00987             mTimer = do_CreateInstance("@mozilla.org/timer;1");
00988             if (!mTimer) {
00989                 NS_WARNING("Failure to create font info loader timer");
00990                 return;
00991             }
00992         }
00993 
00994         // need an initial delay?
00995         uint32_t timerInterval;
00996 
00997         if (aDelay) {
00998             mState = stateTimerOnDelay;
00999             timerInterval = aDelay;
01000         } else {
01001             mState = stateTimerOnInterval;
01002             timerInterval = mInterval;
01003         }
