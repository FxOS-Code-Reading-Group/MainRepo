00091     enum RecordingFlags {
00092         /*  This flag specifies that when clipPath() is called, the path will
00093             be faithfully recorded, but the recording canvas' current clip will
00094             only see the path's bounds. This speeds up the recording process
00095             without compromising the fidelity of the playback. The only side-
00096             effect for recording is that calling getTotalClip() or related
00097             clip-query calls will reflect the path's bounds, not the actual
00098             path.
00099          */
00100         kUsePathBoundsForClip_RecordingFlag = 0x01,
00101         /*  This flag causes the picture to compute bounding boxes and build
00102             up a spatial hierarchy (currently an R-Tree), plus a tree of Canvas'
00103             usually stack-based clip/etc state. This requires an increase in
00104             recording time (often ~2x; likely more for very complex pictures),
00105             but allows us to perform much faster culling at playback time, and
00106             completely avoid some unnecessary clips and other operations. This
00107             is ideal for tiled rendering, or any other situation where you're
00108             drawing a fraction of a large scene into a smaller viewport.
00109 
00110             In most cases the record cost is offset by the playback improvement
00111             after a frame or two of tiled rendering (and complex pictures that
00112             induce the worst record times will generally get the largest
00113             speedups at playback time).
00114 
00115             Note: Currently this is not serializable, the bounding data will be
00116             discarded if you serialize into a stream and then deserialize.
00117         */
00118         kOptimizeForClippedPlayback_RecordingFlag = 0x02
00119     };
