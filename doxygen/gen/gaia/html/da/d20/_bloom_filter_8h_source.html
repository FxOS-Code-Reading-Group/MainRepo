00026 /*
00027  * This class implements a counting Bloom filter as described at
00028  * <http://en.wikipedia.org/wiki/Bloom_filter#Counting_filters>, with
00029  * 8-bit counters.  This allows quick probabilistic answers to the
00030  * question "is object X in set Y?" where the contents of Y might not
00031  * be time-invariant.  The probabilistic nature of the test means that
00032  * sometimes the answer will be "yes" when it should be "no".  If the
00033  * answer is "no", then X is guaranteed not to be in Y.
00034  *
00035  * The filter is parametrized on KeySize, which is the size of the key
00036  * generated by each of hash functions used by the filter, in bits,
00037  * and the type of object T being added and removed.  T must implement
00038  * a |uint32_t hash() const| method which returns a uint32_t hash key
00039  * that will be used to generate the two separate hash functions for
00040  * the Bloom filter.  This hash key MUST be well-distributed for good
00041  * results!  KeySize is not allowed to be larger than 16.
00042  *
00043  * The filter uses exactly 2**KeySize bytes of memory.  From now on we
00044  * will refer to the memory used by the filter as M.
00045  *
00046  * The expected rate of incorrect "yes" answers depends on M and on
00047  * the number N of objects in set Y.  As long as N is small compared
00048  * to M, the rate of such answers is expected to be approximately
00049  * 4*(N/M)**2 for this filter.  In practice, if Y has a few hundred
00050  * elements then using a KeySize of 12 gives a reasonably low
00051  * incorrect answer rate.  A KeySize of 12 has the additional benefit
00052  * of using exactly one page for the filter in typical hardware
00053  * configurations.
00054  */
00055 
00056 template<unsigned KeySize, class T>
00057 class BloomFilter
00058 {
00059     /*
00060      * A counting Bloom filter with 8-bit counters.  For now we assume
00061      * that having two hash functions is enough, but we may revisit that
00062      * decision later.
00063      *
00064      * The filter uses an array with 2**KeySize entries.
00065      *
00066      * Assuming a well-distributed hash function, a Bloom filter with
00067      * array size M containing N elements and
00068      * using k hash function has expected false positive rate exactly
00069      *
00070      * $  (1 - (1 - 1/M)^{kN})^k  $
00071      *
00072      * because each array slot has a
00073      *
00074      * $  (1 - 1/M)^{kN}  $
00075      *
00076      * chance of being 0, and the expected false positive rate is the
00077      * probability that all of the k hash functions will hit a nonzero
00078      * slot.
00079      *
00080      * For reasonable assumptions (M large, kN large, which should both
00081      * hold if we're worried about false positives) about M and kN this
00082      * becomes approximately
00083      *
00084      * $$  (1 - \exp(-kN/M))^k   $$
00085      *
00086      * For our special case of k == 2, that's $(1 - \exp(-2N/M))^2$,
00087      * or in other words
00088      *
00089      * $$    N/M = -0.5 * \ln(1 - \sqrt(r))   $$
00090      *
00091      * where r is the false positive rate.  This can be used to compute
00092      * the desired KeySize for a given load N and false positive rate r.
00093      *
00094      * If N/M is assumed small, then the false positive rate can
00095      * further be approximated as 4*N^2/M^2.  So increasing KeySize by
00096      * 1, which doubles M, reduces the false positive rate by about a
00097      * factor of 4, and a false positive rate of 1% corresponds to
00098      * about M/N == 20.
00099      *
00100      * What this means in practice is that for a few hundred keys using a
00101      * KeySize of 12 gives false positive rates on the order of 0.25-4%.
00102      *
00103      * Similarly, using a KeySize of 10 would lead to a 4% false
00104      * positive rate for N == 100 and to quite bad false positive
00105      * rates for larger N.
00106      */
00107   public:
00108     BloomFilter() {
00109         static_assert(KeySize <= keyShift, "KeySize too big");
00110 
00111         // Should we have a custom operator new using calloc instead and
00112         // require that we're allocated via the operator?
00113         clear();
00114     }
00115 
00116     /*
00117      * Clear the filter.  This should be done before reusing it, because
00118      * just removing all items doesn't clear counters that hit the upper
00119      * bound.
00120      */
00121     void clear();
00122 
00123     /*
00124      * Add an item to the filter.
00125      */
00126     void add(const T* t);
00127 
00128     /*
00129      * Remove an item from the filter.
00130      */
00131     void remove(const T* t);
00132 
00133     /*
00134      * Check whether the filter might contain an item.  This can
00135      * sometimes return true even if the item is not in the filter,
00136      * but will never return false for items that are actually in the
00137      * filter.
00138      */
00139     bool mightContain(const T* t) const;
00140 
00141     /*
00142      * Methods for add/remove/contain when we already have a hash computed
00143      */
00144     void add(uint32_t hash);
00145     void remove(uint32_t hash);
00146     bool mightContain(uint32_t hash) const;
00147 
00148   private:
00149     static const size_t arraySize = (1 << KeySize);
00150     static const uint32_t keyMask = (1 << KeySize) - 1;
00151     static const uint32_t keyShift = 16;
00152 
00153     static uint32_t hash1(uint32_t hash) { return hash & keyMask; }
00154     static uint32_t hash2(uint32_t hash) { return (hash >> keyShift) & keyMask; }
00155 
00156     uint8_t& firstSlot(uint32_t hash) { return counters[hash1(hash)]; }
00157     uint8_t& secondSlot(uint32_t hash) { return counters[hash2(hash)]; }
00158     const uint8_t& firstSlot(uint32_t hash) const { return counters[hash1(hash)]; }
00159     const uint8_t& secondSlot(uint32_t hash) const { return counters[hash2(hash)]; }
00160 
00161     static bool full(const uint8_t& slot) { return slot == UINT8_MAX; }
00162 
00163     uint8_t counters[arraySize];
00164 };
