00021 // Greatest Common Divisor
00022 template<typename IntegerType>
00023 MOZ_ALWAYS_INLINE IntegerType
00024 EuclidGCD(IntegerType a, IntegerType b)
00025 {
00026   // Euclid's algorithm; O(N) in the worst case.  (There are better
00027   // ways, but we don't need them for the current use of this algo.)
00028   MOZ_ASSERT(a > 0);
00029   MOZ_ASSERT(b > 0);
00030 
00031   while (a != b) {
00032     if (a > b) {
00033       a = a - b;
00034     } else {
00035       b = b - a;
00036     }
00037   }
00038 
00039   return a;
00040 }

00042 // Least Common Multiple
00043 template<typename IntegerType>
00044 MOZ_ALWAYS_INLINE IntegerType
00045 EuclidLCM(IntegerType a, IntegerType b)
00046 {
00047   // Divide first to reduce overflow risk.
00048   return (a / EuclidGCD(a, b)) * b;

00067 // DO NOT USE DeprecatedAbs.  It exists only until its callers can be converted
00068 // to Abs below, and it will be removed when all callers have been changed.
00069 template<typename T>
00070 inline typename mozilla::EnableIf<detail::AllowDeprecatedAbs<T>::value, T>::Type
00071 DeprecatedAbs(const T t)
00072 {
00073   // The absolute value of the smallest possible value of a signed-integer type
00074   // won't fit in that type (on twos-complement systems -- and we're blithely
00075   // assuming we're on such systems, for the non-<stdint.h> types listed above),
00076   // so assert that the input isn't that value.
00077   //
00078   // This is the case if: the value is non-negative; or if adding one (giving a
00079   // value in the range [-maxvalue, 0]), then negating (giving a value in the
00080   // range [0, maxvalue]), doesn't produce maxvalue (because in twos-complement,
00081   // (minvalue + 1) == -maxvalue).
00082   MOZ_ASSERT(t >= 0 ||
00083              -(t + 1) != T((1ULL << (CHAR_BIT * sizeof(T) - 1)) - 1),
00084              "You can't negate the smallest possible negative integer!");
00085   return t >= 0 ? t : -t;
00086 }
