00136   isVisible: function fv_isVisible(element) {
00137     // scrollable frames can be ignored we just care about iframes...
00138     let rect = element.getBoundingClientRect();
00139     let parent = element.ownerDocument.defaultView;
00140 
00141     // used to calculate the inner position of frames / scrollables.
00142     // The intent was to use this information to scroll either up or down.
00143     // scrollIntoView(true) will _break_ some web content so we can't do
00144     // this today. If we want that functionality we need to manually scroll
00145     // the individual elements.
00146     let pos = {
00147       top: rect.top,
00148       height: rect.height,
00149       width: rect.width
00150     };

00153 
00154     do {
00155       let frame = parent.frameElement;
00156       visible = visible &&
00157                 this.yAxisVisible(pos.top, pos.height, parent.innerHeight) &&
00158                 this.scrollablesVisible(element, pos);
00159

00161       // In the future we can use this information to scroll
00162       // only the elements we need to at this point as we should
00163       // have all the details we need to figure out how to scroll.
00164       if (!visible)
00165         return false;

00954 function setSelectionRange(element, start, end) {
00955   let isTextField = isPlainTextField(element);
00956 
00957   // Check the parameters
00958 
00959   if (!isTextField && !isContentEditable(element)) {
00960     // Skip HTMLOptionElement and HTMLSelectElement elements, as they don't
00961     // support the operation of setSelectionRange
00962     return;
00963   }
00964 
00965   let text = isTextField ? element.value : getContentEditableText(element);
00966   let length = text.length;
00967   if (start < 0) {

00969   }

00971     end = length;
00972   }
00973   if (start > end) {
00974     start = end;

00979     element.setSelectionRange(start, end, "forward");

00988       sel.modify("move", "forward", "character");

00992       sel.modify("move", "forward", "character");
00993     }
00994 
00995     // Extend the selection to the end position
00996     for (let i = start; i < end; i++) {
