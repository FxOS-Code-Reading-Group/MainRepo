00058 template<typename T, RefCountAtomicity Atomicity>
00059 class RefCounted
00060 {
00061     friend class RefPtr<T>;
00062 
00063   protected:
00064     RefCounted() : refCnt(0) { }
00065     ~RefCounted() {
00066       MOZ_ASSERT(refCnt == detail::DEAD);
00067     }
00068 
00069   public:
00070     // Compatibility with nsRefPtr.
00071     void AddRef() {
00072       MOZ_ASSERT(refCnt >= 0);
00073       ++refCnt;
00074     }
00075 
00076     void Release() {
00077       MOZ_ASSERT(refCnt > 0);
00078       if (0 == --refCnt) {
00079 #ifdef DEBUG
00080         refCnt = detail::DEAD;
00081 #endif
00082         delete static_cast<T*>(this);
00083       }
00084     }
00085 
00086     // Compatibility with wtf::RefPtr.
00087     void ref() { AddRef(); }
00088     void deref() { Release(); }
00089     int refCount() const { return refCnt; }
00090     bool hasOneRef() const {
00091       MOZ_ASSERT(refCnt > 0);
00092       return refCnt == 1;
00093     }
00094 
00095   private:
00096     typename Conditional<Atomicity == AtomicRefCount, Atomic<int>, int>::Type refCnt;
00097 };
