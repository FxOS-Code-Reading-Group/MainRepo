00273 /* We assume that right shift corresponds to signed division by 2 with
00274  * rounding towards minus infinity.  This is correct for typical "arithmetic
00275  * shift" instructions that shift in copies of the sign bit.  But some
00276  * C compilers implement >> with an unsigned shift.  For these machines you
00277  * must define RIGHT_SHIFT_IS_UNSIGNED.
00278  * RIGHT_SHIFT provides a proper signed right shift of an INT32 quantity.
00279  * It is only applied with constant shift counts.  SHIFT_TEMPS must be
00280  * included in the variables of any routine using RIGHT_SHIFT.
00281  */
00282 
00283 #ifdef RIGHT_SHIFT_IS_UNSIGNED
00284 #define SHIFT_TEMPS INT32 shift_temp;
00285 #define RIGHT_SHIFT(x,shft)  \
00286     ((shift_temp = (x)) < 0 ? \
00287      (shift_temp >> (shft)) | ((~((INT32) 0)) << (32-(shft))) : \
00288      (shift_temp >> (shft)))
00289 #else
00290 #define SHIFT_TEMPS
00291 #define RIGHT_SHIFT(x,shft) ((x) >> (shft))
00292 #endif
