00017 /*
00018 ** An array of these structures defines an encoding for an object using DER.
00019 ** The array usually starts with a dummy entry whose kind is DER_SEQUENCE;
00020 ** such an array is terminated with an entry where kind == 0.  (An array
00021 ** which consists of a single component does not require a second dummy
00022 ** entry -- the array is only searched as long as previous component(s)
00023 ** instruct it.)
00024 */
00025 struct DERTemplateStr {
00026     /*
00027     ** Kind of item being decoded/encoded, including tags and modifiers.
00028     */
00029     unsigned long kind;
00030 
00031     /*
00032     ** Offset from base of structure to field that holds the value
00033     ** being decoded/encoded.
00034     */
00035     unsigned int offset;
00036 
00037     /*
00038     ** When kind suggests it (DER_POINTER, DER_INDEFINITE, DER_INLINE),
00039     ** this points to a sub-template for nested encoding/decoding.
00040     */
00041     DERTemplate *sub;
00042 
00043     /*
00044     ** Argument value, dependent on "kind" and/or template placement
00045     ** within an array of templates:
00046     **  - In the first element of a template array, the value is the
00047     **    size of the structure to allocate when this template is being
00048     **    referenced by another template via DER_POINTER or DER_INDEFINITE.
00049     **  - In a component of a DER_SET or DER_SEQUENCE which is *not* a
00050     **    DER_UNIVERSAL type (that is, it has a class tag for either
00051     **    DER_APPLICATION, DER_CONTEXT_SPECIFIC, or DER_PRIVATE), the
00052     **    value is the underlying type of item being decoded/encoded.
00053     */
00054     unsigned long arg;
00055 };
