00140       // parse the *.properties file into an associative array
00141       function parseRawLines(rawText, extendedSyntax) {
00142         var entries = rawText.replace(reBlank, '').split(/[\r\n]+/);
00143         var currentLang = '*';
00144         var genericLang = lang.replace(/-[a-z]+$/i, '');
00145         var skipLang = false;
00146         var match = '';
00147 
00148         for (var i = 0; i < entries.length; i++) {
00149           var line = entries[i];
00150 
00151           // comment or blank line?
00152           if (reComment.test(line))
00153             continue;
00154 
00155           // the extended syntax supports [lang] sections and @import rules
00156           if (extendedSyntax) {
00157             if (reSection.test(line)) { // section start?
00158               match = reSection.exec(line);
00159               currentLang = match[1];
00160               skipLang = (currentLang !== '*') &&
00161                   (currentLang !== lang) && (currentLang !== genericLang);
00162               continue;
00163             } else if (skipLang) {
00164               continue;
00165             }
00166             if (reImport.test(line)) { // @import rule?
00167               match = reImport.exec(line);
00168               loadImport(baseURL + match[1]); // load the resource synchronously
00169             }
00170           }
00171 
00172           // key-value pair
00173           var tmp = line.match(reSplit);
00174           if (tmp && tmp.length == 3)
00175             dictionary[tmp[1]] = evalString(tmp[2]);
00176         }
00177       }

00801   // replace {{arguments}} with their values
00802   function substArguments(str, args) {
00803     var reArgs = /\{\{\s*([a-zA-Z\.]+)\s*\}\}/;
00804     var match = reArgs.exec(str);
00805     while (match) {
00806       if (!match || match.length < 2)
00807         return str; // argument key not found
00808 
00809       var arg = match[1];
00810       var sub = '';
00811       if (arg in args) {
00812         sub = args[arg];
00813       } else if (arg in gL10nData) {
00814         sub = gL10nData[arg][gTextProp];
00815       } else {
00816         consoleWarn('could not find argument {{' + arg + '}}');
00817         return str;
00818       }
00819 
00820       str = str.substring(0, match.index) + sub +
00821             str.substr(match.index + match[0].length);
00822       match = reArgs.exec(str);
00823     }
00824     return str;
00825   }
