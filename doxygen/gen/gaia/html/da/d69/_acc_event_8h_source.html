00045      //    node, only the umbrella event on the ancestor will be emitted.
00046     eCoalesceReorder,
00047 
00048      // eCoalesceMutationTextChange : coalesce text change events caused by
00049      // tree mutations of the same tree level.
00050     eCoalesceMutationTextChange,
00051 
00052     // eCoalesceOfSameType : For events of the same type, only the newest event
00053     // will be processed.
00054     eCoalesceOfSameType,
00055 
00056     // eCoalesceSelectionChange: coalescence of selection change events.
