00096 /*
00097  * This structure is used to send stroke options that are used in stroking
00098  * operations. It consists of the following:
00099  *
00100  * mLineWidth    - Width of the stroke in userspace.
00101  * mLineJoin     - Join style used for joining lines.
00102  * mLineCap      - Cap style used for capping lines.
00103  * mMiterLimit   - Miter limit in units of linewidth
00104  * mDashPattern  - Series of on/off userspace lengths defining dash.
00105  *                 Owned by the caller; must live at least as long as
00106  *                 this StrokeOptions.
00107  *                 mDashPattern != null <=> mDashLength > 0.
00108  * mDashLength   - Number of on/off lengths in mDashPattern.
00109  * mDashOffset   - Userspace offset within mDashPattern at which stroking
00110  *                 begins.
00111  */
00112 struct StrokeOptions {
00113   StrokeOptions(Float aLineWidth = 1.0f,
00114                 JoinStyle aLineJoin = JOIN_MITER_OR_BEVEL,
00115                 CapStyle aLineCap = CAP_BUTT,
00116                 Float aMiterLimit = 10.0f,
00117                 size_t aDashLength = 0,
00118                 const Float* aDashPattern = 0,
00119                 Float aDashOffset = 0.f)
00120     : mLineWidth(aLineWidth)
00121     , mMiterLimit(aMiterLimit)
00122     , mDashPattern(aDashLength > 0 ? aDashPattern : 0)
00123     , mDashLength(aDashLength)
00124     , mDashOffset(aDashOffset)
00125     , mLineJoin(aLineJoin)
00126     , mLineCap(aLineCap)
00127   {
00128     MOZ_ASSERT(aDashLength == 0 || aDashPattern);
00129   }
00130 
00131   Float mLineWidth;
00132   Float mMiterLimit;
00133   const Float* mDashPattern;
00134   size_t mDashLength;
00135   Float mDashOffset;
00136   JoinStyle mLineJoin : 4;
00137   CapStyle mLineCap : 3;

01039   // We can optionally Init after construction in
01040   // case we don't know what the DT will be at construction
01041   // time.
01042   CGContextRef Init(DrawTarget *aDT)
01043   {
01044     MOZ_ASSERT(!mDT, "Can't initialize twice!");
01045     mDT = aDT;
01046     cg = BorrowCGContextFromDrawTarget(aDT);
01047     return cg;

01051   // they are done with the context. This is currently explicit

01056   {
01057     if (cg) {
