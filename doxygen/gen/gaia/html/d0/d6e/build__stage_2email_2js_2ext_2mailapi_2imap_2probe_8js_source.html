00843   function processData(data) {
00844     if (data.length === 0) return;
00845     var idxCRLF = null, literalInfo, curReq;
00846 
00847     // - Accumulate data until newlines when not in a literal
00848     if (self._state.curExpected === null) {
00849       var indexFrom = 0;
00850       // If we have an inline literal requested then integrate it into our
00851       // buffer, then let the newline logic come into play.
00852       if (self._state.curDataMinLength) {
00853         // not enough data, buffer and bail
00854         if (self._state.curData.length + data.length <
00855               self._state.curDataMinLength) {
00856           bufferStateData(data);
00857           return;
00858         }
00859         // enough data, do the CRLF check starting after the last byte of our
00860         // data.
00861         indexFrom = self._state.curDataMinLength - self._state.curData.length;
00862         self._state.curDataMinLength = 0;
00863       }
00864 
00865       // Just buffer and bail if:
00866       // * there's no newline (we want a newline)
00867       // * there is a newline, but we're waiting for some minimal amount of data
00868       //   and this doesn't put us over that threshold.
00869       if ((idxCRLF = bufferIndexOfCRLF(data, indexFrom)) === -1) {
00870         bufferStateData(data);
00871         return;
00872       }
00873       // Save everything after the newline off for the next call to this.  We
00874       // used to keep that extra data around, but it creates permutations that
00875       // require extra unit tests for no good reason.
00876       if (data.length >= idxCRLF + 2) {
00877         self._unprocessed.unshift(data.slice(idxCRLF + 2));
00878       }
00879       data = data.slice(0, idxCRLF);
00880       // And buffer/unify the buffer
00881       data = bufferStateData(data);
00882 
00883       // note: We are leaving all our data in self._state.curData, so consumers
00884       // of the data need to call resetStateData() if they process the entirety
00885       // of the line!
00886     }
00887 
00888     // -- Literal
00889     // Don't mess with incoming data if it's part of a literal
00890     curReq = curReq || self._state.requests[0];
00891     if (curReq && self._state.curExpected !== null) {
00892       var chunk = data;
00893       self._state.curXferred += data.length;
00894       if (self._state.curXferred > self._state.curExpected) {
00895         var pos = data.length
00896               - (self._state.curXferred - self._state.curExpected),
00897             extra = data.slice(pos);
00898 
00899         if (pos > 0)
00900           chunk = data.slice(0, pos);
00901         else
00902           chunk = undefined;
00903         // It's not guaranteed that extra will actually have everything
00904         // required to close out our processing, so we instead unprocess it
00905         // if present, then just set _done on the req
00906         if (extra.length) {
00907           self._unprocessed.unshift(extra);
00908         }
00909         curReq._done = 1;
00910         self._state.curExpected = null;
00911       }
00912 
00913       if (chunk && chunk.length) {
00914         if (self._LOG) self._LOG.data(chunk.length, chunk);
00915         if (curReq._msgtype === 'headers') {
00916           chunk.copy(self._state.curData, curReq.curPos, 0);
00917           curReq.curPos += chunk.length;
00918         }
00919         else
00920           curReq._msg.emit('data', chunk);
00921       }
00922 
00923       if (curReq._done) {
00924         // Save off the headers before we reset the state.
00925         if (curReq._msgtype === 'headers')
00926           curReq._headers = self._state.curData.toString('ascii');
00927         resetStateData();
00928       }
00929       return;
00930     }
00931     // -- Done (post-literal)
00932     if (curReq && curReq._done === 1) {
00933       var restDesc;
00934 
00935       if (data[data.length - 1] === CHARCODE_RPAREN) {
00936         // eat up to, but not including, the right paren
00937         restDesc = data.toString('ascii', 0, data.length - 1).trim();
00938         if (restDesc.length) {
00939           curReq._desc += ' ' + restDesc;
00940         }
00941 
00942         parseFetch(curReq._desc, curReq._headers, curReq._msg);
00943         curReq._done = false;
00944         self._state.curXferred = 0;
00945         self._state.curExpected = null;
00946         resetStateData();
00947         curReq._msg.emit('end', curReq._msg);
00948       }
00949       return;
00950     }
00951 
00952     // -- Check for fetch w/literal
00953     // The general idea here is that if we see a line that ends with a CRLF
00954     // (known if we are at this point) and starts with an asterisk, then we can
00955     // make one of four decisions about what we're looking at:
00956     // 1) This line is a FETCH result line with a literal that is the
00957     //    'payload' of the FETCH.  Either a list of headers or the BODY.
00958     //    We currently do not support a fetch that returns more than one
00959     //    payload.  This code will be discarded before that ever happens.
00960     // 2) This line is a FETCH result line with an inline literal that
00961     //    is part of the response.  We want to make sure that we fetch the
00962     //    literal and then resume processing so that we can handle more of
00963     //    this case #2 or our termination case of #1.
00964     // 3) This line is a FETCH result line with no literals and we want to
00965     //    leave this up to processResponse to handle.
00966     // 4) This line is some other result that we want processResponse to
00967     //    handle.
00968     //
00969     // In this case we only need to handle #1 and #2
00970     else if (data[0] === CHARCODE_ASTERISK &&
00971              // regexps need strings!
00972              (strdata = data.toString('ascii')) &&
00973              reFetchStart.test(strdata) &&
00974              (literalInfo = reEndsWithLiteral.exec(strdata))) {
00975       var strdata; // this gets hoisted so our use above works, but is evil.
00976       var literalLength = parseInt(literalInfo[1], 10);
00977 
00978       var bodyInfo = reBodyFetch.exec(data);
00979       // - Case #2: inline literal
00980       if (!bodyInfo) {
00981         // We want to accumulate the literal into our buffer. So:
00982         // Put a CRLF into our buffer; it has semantic meaning for parseExpr but
00983         // we stripped it off of data before.  (We could elide it, but then
00984         // debugging would end up more confusing as our log output would be
00985         // misleading.)
00986         data = bufferStateData(CRLF_BUFFER);
00987         // Make sure that the buffer accumulation logic above does not call us
00988         // again until we have received our requested data (and another CRLF
00989         // pair)
00990         self._state.curDataMinLength = data.length + literalLength;
00991         return;
00992       }
00993 
00994       // - Case #1: body literal.
00995 
00996       // At this point, the literal points to the FETCH body.
00997       self._state.curExpected = literalLength;
00998 
00999       var desc = strdata.substring(strdata.indexOf('(')+1).trim();
01000       var type = bodyInfo[1];
01001       var uid = reUid.exec(desc)[1];
01002 
01003       // figure out which request this belongs to. If its not assigned to a
01004       // specific uid then send it to the first pending request...
01005       curReq = self._findFetchRequest(uid, type) || self._state.requests[0];
01006       var msg = new ImapMessage();
01007 
01008       // because we push data onto the unprocessed queue for any literals and
01009       // processData lacks any context, we need to reorder the request to be
01010       // first if it is not already first.  (Storing the request along-side
01011       // the data in insufficient because if the literal data is fragmented
01012       // at all, the context will be lost.)
01013       if (self._state.requests[0] !== curReq) {
01014         self._state.requests.splice(self._state.requests.indexOf(curReq), 1);
01015         self._state.requests.unshift(curReq);
01016       }
01017 
01018       msg.seqno = parseInt(literalInfo[1], 10);
01019       curReq._desc = desc;
01020       curReq._msg = msg;
01021       msg.size = self._state.curExpected;
01022 
01023       curReq._fetcher.emit('message', msg);
01024 
01025       curReq._msgtype = (type.indexOf('HEADER') === 0 ? 'headers' : 'body');
01026       // Headers get accumulated into self._state.curData, so we want to
01027       // allocate a buffer for that.  The body contents get directly emitted
01028       // to consumers as 'data' events, so we don't need to resetStateData for
01029       // that.
01030       if (curReq._msgtype === 'headers') {
01031         resetStateData(self._state.curExpected);
01032         curReq.curPos = 0;
01033       }
01034       if (self._LOG) self._LOG.data(strdata.length, strdata);
01035       return;
01036     }
01037 
01038     // consume the data
01039     resetStateData();
01040 
01041     // no need to process zero-length buffers
01042     if (!data.length)
01043       return;
01044 
01045     var response = data.toString('ascii');
01046 
01047     if (self._LOG) self._LOG.data(response.length, response);
01048     processResponse(stringExplode(response, ' ', 3));
01049   };

01756       data = request.cmddata,

02373   var inQuote = false, lastPos = start - 1, isTop = false;
02374   if (!result)
02375     result = new Array();
02376

02384     if (!inQuote) {

02386         inQuote = true;
02387       } else if (o.str[i] === ' ' || o.str[i] === ')' || o.str[i] === ']') {
02388         if (i - (lastPos+1) > 0)
02389           result.push(convStr(o.str.substring(lastPos+1, i)));
02390         if (o.str[i] === ')' || o.str[i] === ']')
02391           return i;

02394         var innerResult = [];
02395         i = parseExpr(o, innerResult, i+1);
02396         lastPos = i;
02397         result.push(innerResult);
02398       } else if (o.str[i] === '{') {

02400         var idxCRLF = o.str.indexOf('\r\n', i);

02403           if (match) {

02405             if (!isNaN(count)) {
02406               result.push(convStr(o.str.substr(idxCRLF + 2, count)));
02407               // Consume through the last byte of the literal.  Subtract off 1
02408               // since when we i++ in the loop, we will then be pointing at the
02409               // first byte after the literal, as desired.
02410               // NOTE: literals do *not* need a CRLF after them even though
02411               // protocol traces may make it look like they do; it just happens
02412               // that most literal payloads end with a CRLF/newline!
02413               i = (idxCRLF + 2 + count) - 1;
02414               lastPos = i;
02415             }

02425       result.push(convStr(o.str.substring(lastPos+1)));
02426     }
02427   }

02677 var normalizeError = exports.normalizeError = function normalizeError(err) {
02678   var errName, reachable = false, retry = true, reportProblem = false;
02679   // We want to produce error-codes as defined in `MailApi.js` for
02680   // tryToCreateAccount.  We have also tried to make imap.js produce
02681   // error codes of the right type already, but for various generic paths
02682   // (like saying 'NO'), there isn't currently a good spot for that.
02683   switch (err.type) {
02684     // dovecot says after a delay and does not terminate the connection:
02685     //   NO [AUTHENTICATIONFAILED] Authentication failed.
02686     // zimbra 7.2.x says after a delay and DOES terminate the connection:
02687     //   NO LOGIN failed
02688     //   * BYE Zimbra IMAP server terminating connection
02689     // yahoo says after a delay and does not terminate the connection:
02690     //   NO [AUTHENTICATIONFAILED] Incorrect username or password.
02691   case 'NO':
02692   case 'no':
02693     reachable = true;
02694     if (!err.serverResponse) {
02695       errName = 'unknown';
02696       reportProblem = false;
02697     }
02698     else {
02699       // All of these require user action to resolve.
02700       reportProblem = true;
02701       retry = false;
02702       if (err.serverResponse.indexOf(
02703         '[ALERT] Application-specific password required') !== -1)
02704         errName = 'needs-app-pass';
02705       else if (err.serverResponse.indexOf(
02706             '[ALERT] Your account is not enabled for IMAP use.') !== -1 ||
02707           err.serverResponse.indexOf(
02708             '[ALERT] IMAP access is disabled for your domain.') !== -1)
02709         errName = 'imap-disabled';
02710       else
02711         errName = 'bad-user-or-pass';
02712     }
02713     break;
02714   // It's an ill-behaved server that reports BAD on bad credentials; the fake
02715   // IMAP server was doing this and has now been fixed.  However, it raises a
02716   // good point that on login it's probably best if we interpret BAD as a
02717   // credentials problem, at least until we start using other forms of
02718   // authentication.
02719   case 'BAD':
02720   case 'bad':
02721     errName = 'bad-user-or-pass';
02722     reportProblem = true;
02723     retry = false;
02724     break;
02725   case 'server-maintenance':
02726     errName = err.type;
02727     reachable = true;
02728     // do retry
02729     break;
02730   // An SSL error is either something we just want to report (probe), or
02731   // something that is currently probably best treated as a network failure.  We
02732   // could tell the user they may be experiencing a MITM attack, but that's not
02733   // really something they can do anything about and we have protected them from
02734   // it currently.
02735   case 'bad-security':
02736     errName = err.type;
02737     reachable = true;
02738     retry = false;
02739     break;
02740   case 'unresponsive-server':
02741   case 'timeout':
02742     errName = 'unresponsive-server';
02743     break;
02744   default:
02745     errName = 'unknown';
02746     break;
02747   }
02748 
02749   return {
02750     name: errName,
02751     reachable: reachable,
02752     retry: retry,
02753     reportProblem: reportProblem,
02754   };

02886   // If there is no SPECIAL-USE capability, there is nothing for us to check.
02887   if (!conn.hasCapability('SPECIAL-USE')) {

02891   function hasInbox(boxesRoot) {
02892     for (var boxName in boxesRoot) {
02893       if (boxName.toLowerCase() === 'inbox') {
02894         return true;
02895       }

02901   // Inbox, then it's not broken.  This is the simplest and most straightforward
02902   // test because RFC 3501 demands that Inbox exists at that path, avoiding
02903   // namespace complications.  (The personal namespace can be rooted under
02904   // INBOX, but INBOX is still exists and is chock full of messages.)
02905   conn.getBoxes(function(err, boxesRoot) {

02907     if (!hasInbox(boxesRoot)) {

02914         // we ever encounter a server that we actually can't support at all.
02915         else
02916           callback('server-problem', null);

02918       return;
