00002 //
00003 // This system module takes a screenshot of the currently running app
00004 // or homescreen and stores it with DeviceStorage when the user
00005 // presses the home and sleep buttons at the same time. It communicates
00006 // with gecko code running in b2g/chrome/content/shell.js using a private
00007 // event-based API. It is the gecko code that creates the screenshot.
00008 //
00009 // This script must be used with the defer attribute.
00010 //
00011 //
00012 (function() {
00013   window.addEventListener('home+sleep', takeScreenshot);
00014 
00015   // Assume that the maximum screenshot size is 4 bytes per pixel
00016   // plus a bit extra. In practice, with compression, our PNG files will be
00017   // much smaller than this.
00018   var MAX_SCREENSHOT_SIZE = window.innerWidth * window.innerHeight * 4 + 4096;
00019 
00020   function takeScreenshot() {
00021     // Give feedback that the screenshot request was received
00022     navigator.vibrate(100);
00023 
00024     // We don't need device storage here, but check to see that
00025     // it is available before sending the screenshot request to chrome.
00026     // If device storage is available, the callback will be called.
00027     // Otherwise, an error message notification will be displayed.
00028     getDeviceStorage(function() {
00029       // Let chrome know we'd like a screenshot.
00030       // This is a completely non-standard undocumented API
00031       // for communicating with our chrome code.
00032       var screenshotProps = {
00033         detail: {
00034           type: 'take-screenshot'
00035         }
00036       };
00037       window.dispatchEvent(new CustomEvent('mozContentEvent', screenshotProps));
00038     });
00039   }
00040 
00041   // Display a screenshot success or failure notification.
00042   // Localize the first argument, and localize the third if the second is null
00043   function notify(titleid, body, bodyid) {
00044     var title = navigator.mozL10n.get(titleid) || titleid;
00045     body = body || navigator.mozL10n.get(bodyid);
00046     navigator.mozNotification.createNotification(
00047       title, body, 'style/icons/Gallery.png').show();
00048   }
00049 
00050   // Get a DeviceStorage object and pass it to the callback.
00051   // Or, if device storage is not available, display a notification.
00052   function getDeviceStorage(callback) {
00053     var storage = navigator.getDeviceStorage('pictures');
00054     var availreq = storage.available();
00055     availreq.onsuccess = function() {
00056       var state = availreq.result;
00057       if (state === 'unavailable') {
00058         notify('screenshotFailed', null, 'screenshotNoSDCard');
00059       }
00060       else if (state === 'shared') {
00061         notify('screenshotFailed', null, 'screenshotSDCardInUse');
00062       }
00063       else if (state === 'available') {
00064         var freereq = storage.freeSpace();
00065         freereq.onsuccess = function() {
00066           if (freereq.result < MAX_SCREENSHOT_SIZE) {
00067             notify('screenshotFailed', null, 'screenshotSDCardLow');
00068           }
00069           else {
00070             callback(storage);
00071           }
00072         };
00073         freereq.onerror = function() {
00074           notify('screenshotFailed', freereq.error && freereq.error.name);
00075         };
00076       }
00077     };
00078     availreq.onerror = function() {
00079       notify('screenshotFailed', availreq.error && availreq.error.name);
00080     };
00081   }
00082 
00083   // Handle the event we get from chrome with the screenshot
00084   window.addEventListener('mozChromeEvent', function ss_onMozChromeEvent(e) {
00085     try {
00086       if (e.detail.type === 'take-screenshot-success') {

00088           var d = new Date();
00089           d = new Date(d.getTime() - d.getTimezoneOffset() * 60000);
00090           var filename = 'screenshots/' +
00091             d.toISOString().slice(0, -5).replace(/[:T]/g, '-') +
00092             '.png';
00093 
00094           var saveRequest = storage.addNamed(e.detail.file, filename);
00095 
00096           saveRequest.onsuccess = function ss_onsuccess() {
00097             // Vibrate again when the screenshot is saved
00098             navigator.vibrate(100);
00099 
00100             // Display filename in a notification
00101             notify('screenshotSaved', filename);
00102           };
00103 
00104           saveRequest.onerror = function ss_onerror() {
00105             notify('screenshotFailed', saveRequest.error.name);
00106           };
00107         });
00108       }
00109       else if (e.detail.type === 'take-screenshot-error') {
00110         notify('screenshotFailed', e.detail.error);
00111       }
00112     }
00113     catch (e) {
00114       console.log('exception in screenshot handler', e);
00115       notify('screenshotFailed', e.toString());
00116     }
