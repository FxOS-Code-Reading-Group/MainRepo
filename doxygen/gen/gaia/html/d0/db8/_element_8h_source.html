00072 // Element-specific flags
00073 enum {
00074   // Set if the element has a pending style change.
00075   ELEMENT_HAS_PENDING_RESTYLE =                 ELEMENT_FLAG_BIT(0),
00076 
00077   // Set if the element is a potential restyle root (that is, has a style
00078   // change pending _and_ that style change will attempt to restyle
00079   // descendants).
00080   ELEMENT_IS_POTENTIAL_RESTYLE_ROOT =           ELEMENT_FLAG_BIT(1),
00081 
00082   // Set if the element has a pending animation style change.
00083   ELEMENT_HAS_PENDING_ANIMATION_RESTYLE =       ELEMENT_FLAG_BIT(2),
00084 
00085   // Set if the element is a potential animation restyle root (that is,
00086   // has an animation style change pending _and_ that style change
00087   // will attempt to restyle descendants).
00088   ELEMENT_IS_POTENTIAL_ANIMATION_RESTYLE_ROOT = ELEMENT_FLAG_BIT(3),
00089 
00090   // All of those bits together, for convenience.
00091   ELEMENT_ALL_RESTYLE_FLAGS = ELEMENT_HAS_PENDING_RESTYLE |
00092                               ELEMENT_IS_POTENTIAL_RESTYLE_ROOT |
00093                               ELEMENT_HAS_PENDING_ANIMATION_RESTYLE |
00094                               ELEMENT_IS_POTENTIAL_ANIMATION_RESTYLE_ROOT,
00095 
00096   // Just the HAS_PENDING bits, for convenience
00097   ELEMENT_PENDING_RESTYLE_FLAGS = ELEMENT_HAS_PENDING_RESTYLE |
00098                                   ELEMENT_HAS_PENDING_ANIMATION_RESTYLE,
00099 
00100   // Remaining bits are for subclasses
00101   ELEMENT_TYPE_SPECIFIC_BITS_OFFSET = NODE_TYPE_SPECIFIC_BITS_OFFSET + 4
00102 };
00103

00293   inline void SetDirectionality(Directionality aDir, bool aNotify) {
00294     UnsetFlags(NODE_ALL_DIRECTION_FLAGS);
00295     if (!aNotify) {
00296       RemoveStatesSilently(DIRECTION_STATES);
00297     }
00298 
00299     switch (aDir) {
00300       case (eDir_RTL):
00301         SetFlags(NODE_HAS_DIRECTION_RTL);
00302         if (!aNotify) {
00303           AddStatesSilently(NS_EVENT_STATE_RTL);
00304         }
00305         break;
00306 
00307       case(eDir_LTR):
00308         SetFlags(NODE_HAS_DIRECTION_LTR);
00309         if (!aNotify) {
00310           AddStatesSilently(NS_EVENT_STATE_LTR);
00311         }
00312         break;
00313 
00314       default:
00315         break;
00316     }
00317 
00318     /* 
00319      * Only call UpdateState if we need to notify, because we call
00320      * SetDirectionality for every element, and UpdateState is very very slow
00321      * for some elements.
00322      */
00323     if (aNotify) {
00324       UpdateState(true);
00325     }
00326   }
