00126 {
00127     uint32_t a = SkAbs32(value);
00128     uint32_t ah = a >> 16;
00129     uint32_t al = a & 0xFFFF;
00130     SkFixed result = ah * a + al * ah + (al * al >> 16);
00131     if (result >= 0)
00132         return result;
00133     else // Overflow.
00134         return SK_FixedMax;
