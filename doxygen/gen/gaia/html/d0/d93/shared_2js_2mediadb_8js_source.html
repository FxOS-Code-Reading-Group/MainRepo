00696       // Given a storage name -> availability map figure out what state
00697       // the mediadb object should be in
00698       function getState(availability) {
00699         var n = 0;   // total number of storages
00700         var a = 0;   // # that are available
00701         var u = 0;   // # that are unavailable
00702         var s = 0;   // # that are shared
00703 
00704         for (var name in availability) {
00705           n++;
00706           switch (availability[name]) {
00707           case 'available':
00708             a++;
00709             break;
00710           case 'unavailable':
00711             u++;
00712             break;
00713           case 'shared':
00714             s++;
00715             break;
00716           }
00717         }
00718 
00719         // If any volume is shared, then behave as if they are all shared
00720         // and make the entire MediaDB shared. This is because shared volumes
00721         // are generally shared transiently and most files on the volume will
00722         // probably still be there when the volume comes back. If we want to
00723         // keep the MediaDB available while one volume is shared we have to
00724         // rescan and discard all the files on the shared volume, which means
00725         // it will take much longer to recover when the volume comes back. So
00726         // it is better to just act as if all volumes are shared together
00727         if (s > 0)
00728           return MediaDB.UNMOUNTED;
00729 
00730         // If all volumes are unavailable, then MediaDB is unavailable
00731         if (u === n)
00732           return MediaDB.NOCARD;
00733 
00734         // Otherwise, there is at least one available volume, so MediaDB
00735         // is available.
00736         return MediaDB.READY;
00737       }

00757       function volumeChangeHandler(e) {
00758         var storageName = e.target.storageName;
00759 
00760         // If nothing changed, ignore this event
00761         if (details.availability[storageName] === e.reason)
00762           return;
00763 
00764         var oldState = media.state;
00765 
00766         // Record the new availability of the volume that changed.
00767         details.availability[storageName] = e.reason;
00768 
00769         // And figure out what our new state is
00770         var newState = getState(details.availability);
00771 
00772         // If the state changed, send out an event about it
00773         if (newState !== oldState) {
00774           changeState(media, newState);
00775 
00776           // Start scanning if we're available, and cancel scanning otherwise
00777           if (newState === MediaDB.READY) {
00778             if (media.autoscan)
00779               scan(media);
00780           }
00781           else {
00782             endscan(media);
00783           }
00784         }
00785         else if (newState === MediaDB.READY) {
00786           // In this case, the state did not change. But we may still need to
00787           // send out an event. If both states are READY, then the user
00788           // just inserted or removed an sdcard. If the user just added a
00789           // card, then we want to send another available event and start a
00790           // scan.  If the user just removed a card then we need to immediately
00791           // tell the client that happened so the music app (for example) can
00792           // stop playing. If it is playing a file that just disappeared it is
00793           // in danger of crashing. Also, in this case we must delete the
00794           // records (and send events) for all of the files on that card.
00795           if (e.reason === 'available') {
00796             // An SD card was just inserted, so send another ready event.
00797             dispatchEvent(media, 'ready');
00798 
00799             // And if we're automatically scanning, start the scan now.
00800             // It would be more efficient if the scan() function could scan
00801             // just one storage area at a time. But SD card insertion should
00802             // be rare enough that efficiency is not so important.
00803             if (media.autoscan)
00804               scan(media);
00805           }
00806           else if (e.reason === 'unavailable') {
00807             // An SD card was just removed. First send an event.
00808             dispatchEvent(media, 'cardremoved');

01400       function compareLists(dbfiles, dsfiles) {
01401         // The dbfiles are sorted when we get them from the db.
01402         // But the ds files are not sorted
01403         dsfiles.sort(function(a, b) {
01404           if (a.name < b.name)
01405             return -1;
01406           else
01407             return 1;
01408         });
01409 
01410         // Loop through both the dsfiles and dbfiles lists
01411         var dsindex = 0, dbindex = 0;
01412         while (true) {
01413           // Get the next DeviceStorage file or null
01414           var dsfile;
01415           if (dsindex < dsfiles.length)
01416             dsfile = dsfiles[dsindex];
01417           else
01418             dsfile = null;
01419 
01420           // Get the next DB file or null
01421           var dbfile;
01422           if (dbindex < dbfiles.length)
01423             dbfile = dbfiles[dbindex];
01424           else
01425             dbfile = null;
01426 
01427           // Case 1: both files are null.  If so, we're done.
01428           if (dsfile === null && dbfile === null)
01429             break;
01430 
01431           // Case 2: no more files in the db.  This means that
01432           // the file from ds is a new one
01433           if (dbfile === null) {
01434             insertRecord(media, dsfile);
01435             dsindex++;
01436             continue;
01437           }
01438 
01439           // Case 3: no more files in ds. This means that the db file
01440           // has been deleted
01441           if (dsfile === null) {
01442             deleteRecord(media, dbfile.name);
01443             dbindex++;
01444             continue;
01445           }
01446 
01447           // Case 4: two files with the same name.
01448           // 4a: date and size are the same for both: do nothing
01449           // 4b: file has changed: it is both a deletion and a creation
01450           if (dsfile.name === dbfile.name) {
01451             var lastModified = dsfile.lastModifiedDate;
01452             if ((lastModified && lastModified.getTime() !== dbfile.date) ||
01453                 dsfile.size !== dbfile.size) {
01454               deleteRecord(media, dbfile.name);
01455               insertRecord(media, dsfile);
01456             }
01457             dsindex++;
01458             dbindex++;
01459             continue;
01460           }
01461 
01462           // Case 5: the dsfile name is less than the dbfile name.
01463           // This means that the dsfile is new.  Like case 2
01464           if (dsfile.name < dbfile.name) {
01465             insertRecord(media, dsfile);
01466             dsindex++;
01467             continue;
01468           }
01469 
01470           // Case 6: the dsfile name is greater than the dbfile name.
01471           // this means that the dbfile no longer exists on disk
01472           if (dsfile.name > dbfile.name) {
01473             deleteRecord(media, dbfile.name);
01474             dbindex++;
01475             continue;
01476           }
01477 
01478           // That should be an exhaustive set of possiblities
01479           // and we should never reach this point.
01480           console.error('Assertion failed');
01481         }
01482 
01483         // Push a special value onto the queue so that when it is
01484         // processed we can trigger a 'scanend' event
01485         insertRecord(media, null);
01486       }
01487     }

01691     function storeRecord(fileinfo) {
01692       if (media.details.firstscan) {
01693         // If this is the first scan then we know this is a new file and
01694         // we can assume that adding it to the db will succeed.

01696         // waiting for a db operation.

01700         }

01702         next();

01704       else {

01706         // record for this new file. In that case, the call to add() above
01707         // is going to fail. We need to handle that case, so we can't send
01708         // out the new file notification until we get a response to the add().

01718           next();
01719         };

01722           // file already exists in the database. So try again, using put()
01723           // instead of add(). If that succeeds, then queue a delete
01724           // notification along with the insert notification.  If the
01725           // second try fails, or if the error was something different
01726           // then issue a warning and continue with the next.
01727           if (request.error.name === 'ConstraintError') {

01729             event.stopPropagation();
01730             // And don't spew a default error message to the console either
01731             event.preventDefault();

01733             putrequest.onsuccess = function() {
01734               queueDeleteNotification(media, fileinfo.name);
01735               if (!fileinfo.fail)

01737               next();

01741               console.error('MediaDB: unexpected ConstraintError',
