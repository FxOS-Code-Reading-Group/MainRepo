00024 struct FrameMetrics {
00025 public:
00026   // We use IDs to identify frames across processes.
00027   typedef uint64_t ViewID;
00028   static const ViewID NULL_SCROLL_ID;   // This container layer does not scroll.
00029   static const ViewID ROOT_SCROLL_ID;   // This is the root scroll frame.
00030   static const ViewID START_SCROLL_ID;  // This is the ID that scrolling subframes
00031                                         // will begin at.
00032 
00033   FrameMetrics()
00034     : mCompositionBounds(0, 0, 0, 0)
00035     , mDisplayPort(0, 0, 0, 0)
00036     , mCriticalDisplayPort(0, 0, 0, 0)
00037     , mViewport(0, 0, 0, 0)
00038     , mScrollOffset(0, 0)
00039     , mScrollId(NULL_SCROLL_ID)
00040     , mScrollableRect(0, 0, 0, 0)
00041     , mResolution(1)
00042     , mZoom(1)
00043     , mDevPixelsPerCSSPixel(1)
00044     , mMayHaveTouchListeners(false)
00045     , mPresShellId(-1)
00046   {}
00047 
00048   // Default copy ctor and operator= are fine
00049 
00050   bool operator==(const FrameMetrics& aOther) const
00051   {
00052     return mCompositionBounds.IsEqualEdges(aOther.mCompositionBounds) &&
00053            mDisplayPort.IsEqualEdges(aOther.mDisplayPort) &&
00054            mCriticalDisplayPort.IsEqualEdges(aOther.mCriticalDisplayPort) &&
00055            mViewport.IsEqualEdges(aOther.mViewport) &&
00056            mScrollOffset == aOther.mScrollOffset &&
00057            mScrollId == aOther.mScrollId &&
00058            mScrollableRect.IsEqualEdges(aOther.mScrollableRect) &&
00059            mResolution == aOther.mResolution &&
00060            mDevPixelsPerCSSPixel == aOther.mDevPixelsPerCSSPixel &&
00061            mMayHaveTouchListeners == aOther.mMayHaveTouchListeners &&
00062            mPresShellId == aOther.mPresShellId;
00063   }
00064   bool operator!=(const FrameMetrics& aOther) const
00065   {
00066     return !operator==(aOther);
00067   }

00122   // ---------------------------------------------------------------------------
00123   // The following metrics are all in widget space/device pixels.
00124   //
00125 
00126   // This is the area within the widget that we're compositing to, which means
00127   // that it is the visible region of this frame. It is not relative to
00128   // anything.
00129   // So { 0, 0, [compositeArea.width], [compositeArea.height] }.
00130   //
00131   // This is useful because, on mobile, the viewport and composition dimensions
00132   // are not always the same. In this case, we calculate the displayport using
00133   // an area bigger than the region we're compositing to. If we used the
00134   // viewport dimensions to calculate the displayport, we'd run into situations
00135   // where we're prerendering the wrong regions and the content may be clipped,
00136   // or too much of it prerendered. If the displayport is the same as the
00137   // viewport, there is no need for this and we can just use the viewport
00138   // instead.
00139   //
00140   // This is only valid on the root layer. Nested iframes do not need this
00141   // metric as they do not have a displayport set. See bug 775452.
00142   ScreenIntRect mCompositionBounds;
00143 
00144   // ---------------------------------------------------------------------------
00145   // The following metrics are all in CSS pixels. They are not in any uniform
00146   // space, so each is explained separately.
00147   //
00148 
00149   // The area of a frame's contents that has been painted, relative to the
00150   // viewport. It is in the same coordinate space as |mViewport|. For example,
00151   // if it is at 0,0, then it's at the same place at the viewport, which is at
00152   // the top-left in the layer, and at the same place as the scroll offset of
00153   // the document.
00154   //
00155   // Note that this is structured in such a way that it doesn't depend on the
00156   // method layout uses to scroll content.
00157   //
00158   // May be larger or smaller than |mScrollableRect|.
00159   //
00160   // To pre-render a margin of 100 CSS pixels around the window,
00161   // { x = -100, y = - 100,
00162   //   width = window.innerWidth + 200, height = window.innerHeight + 200 }
00163   //
00164   // This is only valid on the root layer. Nested iframes do not have a
00165   // displayport set on them. See bug 775452.
00166   CSSRect mDisplayPort;
00167 
00168   // If non-empty, the area of a frame's contents that is considered critical
00169   // to paint. Area outside of this area (i.e. area inside mDisplayPort, but
00170   // outside of mCriticalDisplayPort) is considered low-priority, and may be
00171   // painted with lower precision, or not painted at all.
00172   //
00173   // The same restrictions for mDisplayPort apply here.
00174   CSSRect mCriticalDisplayPort;
00175 
00176   // The CSS viewport, which is the dimensions we're using to constrain the
00177   // <html> element of this frame, relative to the top-left of the layer. Note
00178   // that its offset is structured in such a way that it doesn't depend on the
00179   // method layout uses to scroll content.
00180   //
00181   // This is mainly useful on the root layer, however nested iframes can have
00182   // their own viewport, which will just be the size of the window of the
00183   // iframe. For layers that don't correspond to a document, this metric is
00184   // meaningless and invalid.
00185   CSSRect mViewport;
00186 
00187   // The position of the top-left of the CSS viewport, relative to the document
00188   // (or the document relative to the viewport, if that helps understand it).
00189   //
00190   // Thus it is relative to the document. It is in the same coordinate space as
00191   // |mScrollableRect|, but a different coordinate space than |mViewport| and
00192   // |mDisplayPort|.
00193   //
00194   // It is required that the rect:
00195   // { x = mScrollOffset.x, y = mScrollOffset.y,
00196   //   width = mCompositionBounds.x / mResolution.scale,
00197   //   height = mCompositionBounds.y / mResolution.scale }
00198   // Be within |mScrollableRect|.
00199   //
00200   // This is valid for any layer, but is always relative to this frame and
00201   // not any parents, regardless of parent transforms.
00202   CSSPoint mScrollOffset;
00203 
00204   // A unique ID assigned to each scrollable frame (unless this is
00205   // ROOT_SCROLL_ID, in which case it is not unique).
00206   ViewID mScrollId;
00207 
00208   // The scrollable bounds of a frame. This is determined by reflow.
00209   // For the top-level |window|,
00210   // { x = window.scrollX, y = window.scrollY, // could be 0, 0
00211   //   width = window.innerWidth, height = window.innerHeight }
00212   //
00213   // This is relative to the document. It is in the same coordinate space as
00214   // |mScrollOffset|, but a different coordinate space than |mViewport| and
00215   // |mDisplayPort|. Note also that this coordinate system is understood by
00216   // window.scrollTo().
00217   //
00218   // This is valid on any layer unless it has no content.
00219   CSSRect mScrollableRect;
00220 
00221   // ---------------------------------------------------------------------------
00222   // The following metrics are dimensionless.
00223   //
00224 
00225   // The resolution, along both axes, that the current frame has been painted
00226   // at.
00227   //
00228   // Every time this frame is composited and the compositor samples its
00229   // transform, this metric is used to create a transform which is
00230   // post-multiplied into the parent's transform. Since this only happens when
00231   // we walk the layer tree, the resulting transform isn't stored here. Thus the
00232   // resolution of parent layers is opaque to this metric.
00233   LayoutDeviceToLayerScale mResolution;
00234 
00235   // The resolution-independent "user zoom".  For example, if a page
00236   // configures the viewport to a zoom value of 2x, then this member
00237   // will always be 2.0 no matter what the viewport or composition
00238   // bounds.
00239   //
00240   // In the steady state (no animations), the following is usually true
00241   //
00242   //  intrinsicScale = (mCompositionBounds / mViewport)
00243   //  mResolution = mZoom * intrinsicScale / mDevPixelsPerCSSPixel
00244   //
00245   // When this is not true, we're probably asynchronously sampling a
00246   // zoom animation for content.
00247   ScreenToScreenScale mZoom;
00248 
00249   // The conversion factor between CSS pixels and device pixels for this frame.
00250   // This can vary based on a variety of things, such as reflowing-zoom. The
00251   // conversion factor for device pixels to layers pixels is just the
00252   // resolution.
00253   CSSToLayoutDeviceScale mDevPixelsPerCSSPixel;
00254 
00255   // Whether or not this frame may have touch listeners.
00256   bool mMayHaveTouchListeners;
00257 
00258   uint32_t mPresShellId;
00259 };
