00139                                    bool *shouldDelete = nullptr)
00140   {
00141     MOZ_ASSERT(get() > 0);
00142     if (!IsInPurpleBuffer()) {
00143       mRefCntAndFlags -= NS_REFCOUNT_CHANGE;
00144       mRefCntAndFlags |= (NS_IN_PURPLE_BUFFER | NS_IS_PURPLE);
00145       uintptr_t retval = NS_REFCOUNT_VALUE(mRefCntAndFlags);
00146       // Suspect may delete 'owner' and 'this'!
00147       NS_CycleCollectorSuspect3(owner, p, this, shouldDelete);

00149     }

00151     mRefCntAndFlags |= (NS_IN_PURPLE_BUFFER | NS_IS_PURPLE);
