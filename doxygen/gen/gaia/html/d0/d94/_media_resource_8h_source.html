00051 class MediaChannelStatistics {
00052 public:
00053   MediaChannelStatistics() { Reset(); }
00054 
00055   MediaChannelStatistics(MediaChannelStatistics * aCopyFrom)
00056   {
00057     MOZ_ASSERT(aCopyFrom);
00058     mAccumulatedBytes = aCopyFrom->mAccumulatedBytes;
00059     mAccumulatedTime = aCopyFrom->mAccumulatedTime;
00060     mLastStartTime = aCopyFrom->mLastStartTime;
00061     mIsStarted = aCopyFrom->mIsStarted;
00062   }
00063 
00064   NS_INLINE_DECL_THREADSAFE_REFCOUNTING(MediaChannelStatistics)
00065 
00066   void Reset() {
00067     mLastStartTime = TimeStamp();
00068     mAccumulatedTime = TimeDuration(0);
00069     mAccumulatedBytes = 0;
00070     mIsStarted = false;
00071   }
00072   void Start() {
00073     if (mIsStarted)
00074       return;
00075     mLastStartTime = TimeStamp::Now();
00076     mIsStarted = true;
00077   }
00078   void Stop() {
00079     if (!mIsStarted)
00080       return;
00081     mAccumulatedTime += TimeStamp::Now() - mLastStartTime;
00082     mIsStarted = false;
00083   }
00084   void AddBytes(int64_t aBytes) {
00085     if (!mIsStarted) {
00086       // ignore this data, it may be related to seeking or some other
00087       // operation we don't care about
00088       return;
00089     }
00090     mAccumulatedBytes += aBytes;
00091   }
00092   double GetRateAtLastStop(bool* aReliable) {
00093     double seconds = mAccumulatedTime.ToSeconds();
00094     *aReliable = (seconds >= 1.0) ||
00095                  (mAccumulatedBytes >= RELIABLE_DATA_THRESHOLD);
00096     if (seconds <= 0.0)
00097       return 0.0;
00098     return static_cast<double>(mAccumulatedBytes)/seconds;
00099   }
00100   double GetRate(bool* aReliable) {
00101     TimeDuration time = mAccumulatedTime;
00102     if (mIsStarted) {
00103       time += TimeStamp::Now() - mLastStartTime;
00104     }
00105     double seconds = time.ToSeconds();
00106     *aReliable = (seconds >= 3.0) ||
00107                  (mAccumulatedBytes >= RELIABLE_DATA_THRESHOLD);
00108     if (seconds <= 0.0)
00109       return 0.0;
00110     return static_cast<double>(mAccumulatedBytes)/seconds;
00111   }
00112 private:
00113   int64_t      mAccumulatedBytes;
00114   TimeDuration mAccumulatedTime;
00115   TimeStamp    mLastStartTime;
00116   bool         mIsStarted;
00117 };
