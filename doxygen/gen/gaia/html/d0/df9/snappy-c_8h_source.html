00052 /*
00053  * Takes the data stored in "input[0..input_length-1]" and stores
00054  * it in the array pointed to by "compressed".
00055  *
00056  * <compressed_length> signals the space available in "compressed".
00057  * If it is not at least equal to "snappy_max_compressed_length(input_length)",
00058  * SNAPPY_BUFFER_TOO_SMALL is returned. After successful compression,
00059  * <compressed_length> contains the true length of the compressed output,
00060  * and SNAPPY_OK is returned.
00061  *
00062  * Example:
00063  *   size_t output_length = snappy_max_compressed_length(input_length);
00064  *   char* output = (char*)malloc(output_length);
00065  *   if (snappy_compress(input, input_length, output, &output_length)
00066  *       == SNAPPY_OK) {
00067  *     ... Process(output, output_length) ...
00068  *   }
00069  *   free(output);
00070  */
00071 snappy_status snappy_compress(const char* input,
00072                               size_t input_length,
00073                               char* compressed,
00074                               size_t* compressed_length);
00075 
00076 /*
00077  * Given data in "compressed[0..compressed_length-1]" generated by
00078  * calling the snappy_compress routine, this routine stores
00079  * the uncompressed data to
00080  *   uncompressed[0..uncompressed_length-1].
00081  * Returns failure (a value not equal to SNAPPY_OK) if the message
00082  * is corrupted and could not be decrypted.
00083  *
00084  * <uncompressed_length> signals the space available in "uncompressed".
00085  * If it is not at least equal to the value returned by
00086  * snappy_uncompressed_length for this stream, SNAPPY_BUFFER_TOO_SMALL
00087  * is returned. After successful decompression, <uncompressed_length>
00088  * contains the true length of the decompressed output.
00089  *
00090  * Example:
00091  *   size_t output_length;
00092  *   if (snappy_uncompressed_length(input, input_length, &output_length)
00093  *       != SNAPPY_OK) {
00094  *     ... fail ...
00095  *   }
00096  *   char* output = (char*)malloc(output_length);
00097  *   if (snappy_uncompress(input, input_length, output, &output_length)
00098  *       == SNAPPY_OK) {
00099  *     ... Process(output, output_length) ...
00100  *   }
00101  *   free(output);
00102  */
00103 snappy_status snappy_uncompress(const char* compressed,
00104                                 size_t compressed_length,
00105                                 char* uncompressed,
00106                                 size_t* uncompressed_length);
