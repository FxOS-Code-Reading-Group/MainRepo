00061 class MediaEngineSource : public nsISupports
00062 {
00063 public:
00064   virtual ~MediaEngineSource() {}
00065 
00066   /* Populate the human readable name of this device in the nsAString */
00067   virtual void GetName(nsAString&) = 0;
00068 
00069   /* Populate the UUID of this device in the nsAString */
00070   virtual void GetUUID(nsAString&) = 0;
00071 
00072   /* This call reserves but does not start the device. */
00073   virtual nsresult Allocate(const MediaEnginePrefs &aPrefs) = 0;
00074 
00075   /* Release the device back to the system. */
00076   virtual nsresult Deallocate() = 0;
00077 
00078   /* Start the device and add the track to the provided SourceMediaStream, with
00079    * the provided TrackID. You may start appending data to the track
00080    * immediately after. */
00081   virtual nsresult Start(SourceMediaStream*, TrackID) = 0;
00082 
00083   /* Take a snapshot from this source. In the case of video this is a single
00084    * image, and for audio, it is a snippet lasting aDuration milliseconds. The
00085    * duration argument is ignored for a MediaEngineVideoSource.
00086    */
00087   virtual nsresult Snapshot(uint32_t aDuration, nsIDOMFile** aFile) = 0;
00088 
00089   /* Called when the stream wants more data */
00090   virtual void NotifyPull(MediaStreamGraph* aGraph,
00091                           SourceMediaStream *aSource,
00092                           TrackID aId,
00093                           StreamTime aDesiredTime,
00094                           TrackTicks &aLastEndTime) = 0;
00095 
00096   /* Stop the device and release the corresponding MediaStream */
00097   virtual nsresult Stop(SourceMediaStream *aSource, TrackID aID) = 0;
00098 
00099   /* Change device configuration.  */
00100   virtual nsresult Config(bool aEchoOn, uint32_t aEcho,
00101                           bool aAgcOn, uint32_t aAGC,
00102                           bool aNoiseOn, uint32_t aNoise) = 0;
00103 
00104   /* Returns true if a source represents a fake capture device and
00105    * false otherwise
00106    */
00107   virtual bool IsFake() = 0;
00108 
00109   /* Return false if device is currently allocated or started */
00110   bool IsAvailable() {
00111     if (mState == kAllocated || mState == kStarted) {
00112       return false;
00113     } else {
00114       return true;
00115     }
00116   }
00117 
00118   /* It is an error to call Start() before an Allocate(), and Stop() before
00119    * a Start(). Only Allocate() may be called after a Deallocate(). */
00120 
00121 protected:

00123 };
