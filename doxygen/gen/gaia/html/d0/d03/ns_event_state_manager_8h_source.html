00808 class nsAutoHandlingUserInputStatePusher
00809 {
00810 public:
00811   nsAutoHandlingUserInputStatePusher(bool aIsHandlingUserInput,
00812                                      nsEvent* aEvent,
00813                                      nsIDocument* aDocument)
00814     : mIsHandlingUserInput(aIsHandlingUserInput),
00815       mIsMouseDown(aEvent && aEvent->message == NS_MOUSE_BUTTON_DOWN),
00816       mResetFMMouseDownState(false)
00817   {
00818     if (aIsHandlingUserInput) {
00819       nsEventStateManager::StartHandlingUserInput();
00820       if (mIsMouseDown) {
00821         nsIPresShell::SetCapturingContent(nullptr, 0);
00822         nsIPresShell::AllowMouseCapture(true);
00823         if (aDocument && aEvent->mFlags.mIsTrusted) {
00824           nsFocusManager* fm = nsFocusManager::GetFocusManager();
00825           if (fm) {
00826             fm->SetMouseButtonDownHandlingDocument(aDocument);
00827             mResetFMMouseDownState = true;
00828           }
00829         }
00830       }
00831     }
00832   }
00833 
00834   ~nsAutoHandlingUserInputStatePusher()
00835   {
00836     if (mIsHandlingUserInput) {
00837       nsEventStateManager::StopHandlingUserInput();
