00147 /*
00148  * The FIPS 186-1 algorithm for generating primes P and Q allows only 9
00149  * distinct values for the length of P, and only one value for the
00150  * length of Q.
00151  * The algorithm uses a variable j to indicate which of the 9 lengths
00152  * of P is to be used.
00153  * The following table relates j to the lengths of P and Q in bits.
00154  *
00155  *  j   bits in P   bits in Q
00156  *  _   _________   _________
00157  *  0    512        160
00158  *  1    576        160
00159  *  2    640        160
00160  *  3    704        160
00161  *  4    768        160
00162  *  5    832        160
00163  *  6    896        160
00164  *  7    960        160
00165  *  8   1024        160
00166  *
00167  * The FIPS-186-1 compliant PQG generator takes j as an input parameter.
00168  *
00169  * FIPS 186-3 algorithm specifies 4 distinct P and Q sizes:
00170  *
00171  *     bits in P       bits in Q
00172  *     _________       _________
00173  *      1024           160
00174  *      2048           224
00175  *      2048           256
00176  *      3072           256
00177  *
00178  * The FIPS-186-3 complaiant PQG generator (PQG V2) takes arbitrary p and q
00179  * lengths as input and returns an error if they aren't in this list.
00180  */
00181 
00182 #define DSA1_Q_BITS      160
00183 #define DSA_MAX_P_BITS  3072
00184 #define DSA_MIN_P_BITS   512
00185 #define DSA_MAX_Q_BITS   256
00186 #define DSA_MIN_Q_BITS   160
