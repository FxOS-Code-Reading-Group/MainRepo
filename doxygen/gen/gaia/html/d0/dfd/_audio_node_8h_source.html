00065 public:
00066   SelfCountedReference() : mRefCnt(0) {}
00067   ~SelfCountedReference()
00068   {
00069     NS_ASSERTION(mRefCnt == 0, "Forgot to drop the self reference?");
00070   }
00071 
00072   void Take(T* t)
00073   {
00074     if (mRefCnt++ == 0) {
00075       t->AddRef();
00076     }
00077   }
00078   void Drop(T* t)
00079   {
00080     if (mRefCnt > 0) {
00081       --mRefCnt;
00082       if (mRefCnt == 0) {
00083         t->Release();
00084       }
00085     }
00086   }
