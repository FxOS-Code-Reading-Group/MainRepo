00035 // RepeatedField and RepeatedPtrField are used by generated protocol message
00036 // classes to manipulate repeated fields.  These classes are very similar to
00037 // STL's vector, but include a number of optimizations found to be useful
00038 // specifically in the case of Protocol Buffers.  RepeatedPtrField is
00039 // particularly different from STL vector as it manages ownership of the
00040 // pointers that it contains.
00041 //
00042 // Typically, clients should not need to access RepeatedField objects directly,
00043 // but should instead use the accessor functions generated automatically by the
00044 // protocol compiler.

00067 // RepeatedField is used to represent repeated fields of a primitive type (in
00068 // other words, everything except strings and nested Messages).  Most users will
00069 // not ever use a RepeatedField directly; they will use the get-by-index,
00070 // set-by-index, and add accessors that are generated for all repeated fields.
00071 template <typename Element>
00072 class RepeatedField {
00073  public:
00074   RepeatedField();
00075   RepeatedField(const RepeatedField& other);
00076   ~RepeatedField();
00077 
00078   RepeatedField& operator=(const RepeatedField& other);
00079 
00080   int size() const;
00081 
00082   const Element& Get(int index) const;
00083   Element* Mutable(int index);
00084   void Set(int index, const Element& value);
00085   void Add(const Element& value);
00086   Element* Add();
00087   // Remove the last element in the array.
00088   // We don't provide a way to remove any element other than the last
00089   // because it invites inefficient use, such as O(n^2) filtering loops
00090   // that should have been O(n).  If you want to remove an element other
00091   // than the last, the best way to do it is to re-arrange the elements
00092   // so that the one you want removed is at the end, then call RemoveLast().
00093   void RemoveLast();
00094   void Clear();
00095   void MergeFrom(const RepeatedField& other);
00096   void CopyFrom(const RepeatedField& other);
00097 
00098   // Reserve space to expand the field to at least the given size.  If the
00099   // array is grown, it will always be at least doubled in size.
00100   void Reserve(int new_size);
00101 
00102   // Resize the RepeatedField to a new, smaller size.  This is O(1).
00103   void Truncate(int new_size);
00104 
00105   void AddAlreadyReserved(const Element& value);
00106   Element* AddAlreadyReserved();
00107   int Capacity() const;
00108 
00109   // Gets the underlying array.  This pointer is possibly invalidated by
00110   // any add or remove operation.
00111   Element* mutable_data();
00112   const Element* data() const;
00113 
00114   // Swap entire contents with "other".
00115   void Swap(RepeatedField* other);
00116 
00117   // Swap two elements.
00118   void SwapElements(int index1, int index2);
00119 
00120   // STL-like iterator support
00121   typedef Element* iterator;
00122   typedef const Element* const_iterator;
00123   typedef Element value_type;
00124 
00125   iterator begin();
00126   const_iterator begin() const;
00127   iterator end();
00128   const_iterator end() const;
00129 
00130   // Returns the number of bytes used by the repeated field, excluding
00131   // sizeof(*this)
00132   int SpaceUsedExcludingSelf() const;
00133 
00134  private:
00135   static const int kInitialSize = 4;
00136 
00137   Element* elements_;
00138   int      current_size_;
00139   int      total_size_;
00140 
00141   Element  initial_space_[kInitialSize];
00142 
00143   // Move the contents of |from| into |to|, possibly clobbering |from| in the
00144   // process.  For primitive types this is just a memcpy(), but it could be
00145   // specialized for non-primitive types to, say, swap each element instead.
00146   void MoveArray(Element to[], Element from[], int size);
00147 
00148   // Copy the elements of |from| into |to|.
00149   void CopyArray(Element to[], const Element from[], int size);
00150 };

00596 // Avoid inlining of Reserve(): new, memcpy, and delete[] lead to a significant
00597 // amount of code bloat.
00598 template <typename Element>
00599 void RepeatedField<Element>::Reserve(int new_size) {
00600   if (total_size_ >= new_size) return;
00601 
00602   Element* old_elements = elements_;
00603   total_size_ = max(total_size_ * 2, new_size);
00604   elements_ = new Element[total_size_];
00605   MoveArray(elements_, old_elements, current_size_);
00606   if (old_elements != initial_space_) {
00607     delete [] old_elements;
00608   }
00609 }

00760 template <typename TypeHandler>
00761 void RepeatedPtrFieldBase::AddAllocated(
00762     typename TypeHandler::Type* value) {
00763   // Make room for the new pointer.
00764   if (current_size_ == total_size_) {
00765     // The array is completely full with no cleared objects, so grow it.
00766     Reserve(total_size_ + 1);
00767     ++allocated_size_;
00768   } else if (allocated_size_ == total_size_) {
00769     // There is no more space in the pointer array because it contains some
00770     // cleared objects awaiting reuse.  We don't want to grow the array in this
00771     // case because otherwise a loop calling AddAllocated() followed by Clear()
00772     // would leak memory.
00773     TypeHandler::Delete(cast<TypeHandler>(elements_[current_size_]));
00774   } else if (current_size_ < allocated_size_) {
00775     // We have some cleared objects.  We don't care about their order, so we
00776     // can just move the first one to the end to make space.
00777     elements_[allocated_size_] = elements_[current_size_];
00778     ++allocated_size_;
00779   } else {
00780     // There are no cleared objects.
00781     ++allocated_size_;
00782   }
00783 
00784   elements_[current_size_++] = value;
00785 }

00961 // refer to this class directly; use RepeatedPtrField<T>::iterator instead.
00962 //
00963 // The iterator for RepeatedPtrField<T>, RepeatedPtrIterator<T>, is
00964 // very similar to iterator_ptr<T**> in util/gtl/iterator_adaptors-inl.h,
00965 // but adds random-access operators and is modified to wrap a void** base

00981   // write "typename" in front of them everywhere.
