00015 /*
00016 ** An array of these structures defines a BER/DER encoding for an object.
00017 **
00018 ** The array usually starts with a dummy entry whose kind is SEC_ASN1_SEQUENCE;
00019 ** such an array is terminated with an entry where kind == 0.  (An array
00020 ** which consists of a single component does not require a second dummy
00021 ** entry -- the array is only searched as long as previous component(s)
00022 ** instruct it.)
00023 */
00024 typedef struct sec_ASN1Template_struct {
00025     /*
00026     ** Kind of item being decoded/encoded, including tags and modifiers.
00027     */
00028     unsigned long kind;
00029 
00030     /*
00031     ** The value is the offset from the base of the structure to the
00032     ** field that holds the value being decoded/encoded.
00033     */
00034     unsigned long offset;
00035 
00036     /*
00037     ** When kind suggests it (SEC_ASN1_POINTER, SEC_ASN1_GROUP, SEC_ASN1_INLINE,
00038     ** or a component that is *not* a SEC_ASN1_UNIVERSAL), this points to
00039     ** a sub-template for nested encoding/decoding,
00040     ** OR, iff SEC_ASN1_DYNAMIC is set, then this is a pointer to a pointer
00041     ** to a function which will return the appropriate template when called
00042     ** at runtime.  NOTE! that explicit level of indirection, which is
00043     ** necessary because ANSI does not allow you to store a function
00044     ** pointer directly as a "void *" so we must store it separately and
00045     ** dereference it to get at the function pointer itself.
00046     */
00047     const void *sub;
00048 
00049     /*
00050     ** In the first element of a template array, the value is the size
00051     ** of the structure to allocate when this template is being referenced
00052     ** by another template via SEC_ASN1_POINTER or SEC_ASN1_GROUP.
00053     ** In all other cases, the value is ignored.
00054     */
00055     unsigned int size;
00056 } SEC_ASN1Template;

00234 /*
00235  * Type of the function pointer used for grabbing encoded bytes.
00236  * This can be used during either encoding or decoding, as follows...
00237  *
00238  * When decoding, this can be used to filter the encoded bytes as they
00239  * are parsed.  This is what you would do if you wanted to process the data
00240  * along the way (like to decrypt it, or to perform a hash on it in order
00241  * to do a signature check later).  See SEC_ASN1DecoderSetFilterProc().
00242  * When processing only part of the encoded bytes is desired, you "watch"
00243  * for the field(s) you are interested in with a "notify proc" (see
00244  * SEC_ASN1DecoderSetNotifyProc()) and for even finer granularity (e.g. to
00245  * ignore all by the contents bytes) you pay attention to the "data_kind"
00246  * parameter.
00247  *
00248  * When encoding, this is the specification for the output function which
00249  * will receive the bytes as they are encoded.  The output function can
00250  * perform any postprocessing necessary (like hashing (some of) the data
00251  * to create a digest that gets included at the end) as well as shoving
00252  * the data off wherever it needs to go.  (In order to "tune" any processing,
00253  * you can set a "notify proc" as described above in the decoding case.)
00254  *
00255  * The parameters:
00256  * - "arg" is an opaque pointer that you provided at the same time you
00257  *   specified a function of this type
00258  * - "data" is a buffer of length "len", containing the encoded bytes
00259  * - "depth" is how deep in a nested encoding we are (it is not usually
00260  *   valuable, but can be useful sometimes so I included it)
00261  * - "data_kind" tells you if these bytes are part of the ASN.1 encoded
00262  *   octets for identifier, length, contents, or end-of-contents
00263  */ 
00264 typedef void (* SEC_ASN1WriteProc)(void *arg,
00265                    const char *data, unsigned long len,
00266                    int depth, SEC_ASN1EncodingPart data_kind);
