00491 {
00492 protected:
00493     // The default implementation uses nsAutoRefTrait<T>.
00494     // Specializations need not define this typedef.
00495     typedef nsAutoRefTraits<T> Traits;
00496     // The type of the handle to the resource.
00497     // A specialization must provide a typedef for RawRef.
00498     typedef typename Traits::RawRef RawRef;
00499 
00500     // Construct with no resource.
00501     //
00502     // If this constructor is not accessible then some limited nsAutoRef
00503     // functionality will still be available, but the default constructor,
00504     // |reset|, and most transfer of ownership methods will not be available.
00505     nsSimpleRef()
00506         : mRawRef(Traits::Void())
00507     {
00508     }
00509     // Construct with a handle to a resource.
00510     // A specialization must provide this. 
00511     nsSimpleRef(RawRef aRawRef)
00512         : mRawRef(aRawRef)
00513     {
00514     }
00515 
00516     // Test whether there is an associated resource.  A specialization must
00517     // provide this.  The function is permitted to always return true if the
00518     // default constructor is not accessible, or if Release (and AddRef) can
00519     // deal with void handles.
00520     bool HaveResource() const
00521     {
00522         return mRawRef != Traits::Void();
00523     }
00524 
00525 public:
00526     // A specialization must provide get() or loose some functionality.  This
00527     // is inherited by derived classes and the specialization may choose
00528     // whether it is public or protected.
00529     RawRef get() const
00530     {
00531         return mRawRef;
00532     }

00629     // the reference must be stored in a POD type object, such as may be
