00148 /*
00149  * PK11_ATTR_XXX
00150  *
00151  * The following PK11_ATTR_XXX bitflags are used to specify
00152  * PKCS #11 object attributes that have Boolean values.  Some NSS
00153  * functions have a "PK11AttrFlags attrFlags" parameter whose value
00154  * is the logical OR of these bitflags.  NSS use these bitflags on
00155  * private keys or secret keys.  Some of these bitflags also apply
00156  * to the public keys associated with the private keys.
00157  *
00158  * For each PKCS #11 object attribute, we need two bitflags to
00159  * specify not only "true" and "false" but also "default".  For
00160  * example, PK11_ATTR_PRIVATE and PK11_ATTR_PUBLIC control the
00161  * CKA_PRIVATE attribute.  If PK11_ATTR_PRIVATE is set, we add
00162  *     { CKA_PRIVATE, &cktrue, sizeof(CK_BBOOL) }
00163  * to the template.  If PK11_ATTR_PUBLIC is set, we add
00164  *     { CKA_PRIVATE, &ckfalse, sizeof(CK_BBOOL) }
00165  * to the template.  If neither flag is set, we don't add any
00166  * CKA_PRIVATE entry to the template.
00167  */
00168

00174 /*
00175  * PK11_ATTR_TOKEN
00176  * PK11_ATTR_SESSION
00177  *
00178  * These two flags determine whether the object is a token or
00179  * session object.
00180  *
00181  * These two flags are related and cannot both be set.
00182  * If the PK11_ATTR_TOKEN flag is set, the object is a token
00183  * object.  If the PK11_ATTR_SESSION flag is set, the object is
00184  * a session object.  If neither flag is set, the object is *by
00185  * default* a session object.
00186  *
00187  * These two flags specify the value of the PKCS #11 CKA_TOKEN
00188  * attribute.
00189  */
00190 #define PK11_ATTR_TOKEN         0x00000001L
00191 #define PK11_ATTR_SESSION       0x00000002L
00192 
00193 /*
00194  * PK11_ATTR_PRIVATE
00195  * PK11_ATTR_PUBLIC
00196  *
00197  * These two flags determine whether the object is a private or
00198  * public object.  A user may not access a private object until the
00199  * user has authenticated to the token.
00200  *
00201  * These two flags are related and cannot both be set.
00202  * If the PK11_ATTR_PRIVATE flag is set, the object is a private
00203  * object.  If the PK11_ATTR_PUBLIC flag is set, the object is a
00204  * public object.  If neither flag is set, it is token-specific
00205  * whether the object is private or public.
00206  *
00207  * These two flags specify the value of the PKCS #11 CKA_PRIVATE
00208  * attribute.  NSS only uses this attribute on private and secret
00209  * keys, so public keys created by NSS get the token-specific
00210  * default value of the CKA_PRIVATE attribute.
00211  */
00212 #define PK11_ATTR_PRIVATE       0x00000004L
00213 #define PK11_ATTR_PUBLIC        0x00000008L

00215 /*
00216  * PK11_ATTR_MODIFIABLE
00217  * PK11_ATTR_UNMODIFIABLE
00218  *
00219  * These two flags determine whether the object is modifiable or
00220  * read-only.
00221  *
00222  * These two flags are related and cannot both be set.
00223  * If the PK11_ATTR_MODIFIABLE flag is set, the object can be
00224  * modified.  If the PK11_ATTR_UNMODIFIABLE flag is set, the object
00225  * is read-only.  If neither flag is set, the object is *by default*
00226  * modifiable.
00227  *
00228  * These two flags specify the value of the PKCS #11 CKA_MODIFIABLE

00236 /*
00237  * PK11_ATTR_SENSITIVE
00238  * PK11_ATTR_INSENSITIVE
00239  *
00240  * These two flags are related and cannot both be set.
00241  * If the PK11_ATTR_SENSITIVE flag is set, the key is sensitive.
00242  * If the PK11_ATTR_INSENSITIVE flag is set, the key is not
00243  * sensitive.  If neither flag is set, it is token-specific whether
00244  * the key is sensitive or not.
00245  *
00246  * If a key is sensitive, certain attributes of the key cannot be
00247  * revealed in plaintext outside the token.
00248  *
00249  * This flag specifies the value of the PKCS #11 CKA_SENSITIVE
00250  * attribute.  Although the default value of the CKA_SENSITIVE
00251  * attribute for secret keys is CK_FALSE per PKCS #11, some FIPS
00252  * tokens set the default value to CK_TRUE because only CK_TRUE
00253  * is allowed.  So in practice the default value of this attribute
00254  * is token-specific, hence the need for two bitflags.
00255  */
00256 #define PK11_ATTR_SENSITIVE     0x00000040L
00257 #define PK11_ATTR_INSENSITIVE   0x00000080L
00258 
00259 /*
00260  * PK11_ATTR_EXTRACTABLE
00261  * PK11_ATTR_UNEXTRACTABLE
00262  *
00263  * These two flags are related and cannot both be set.
00264  * If the PK11_ATTR_EXTRACTABLE flag is set, the key is extractable
00265  * and can be wrapped.  If the PK11_ATTR_UNEXTRACTABLE flag is set,
00266  * the key is not extractable, and certain attributes of the key
00267  * cannot be revealed in plaintext outside the token (just like a
00268  * sensitive key).  If neither flag is set, it is token-specific
00269  * whether the key is extractable or not.
00270  *
00271  * These two flags specify the value of the PKCS #11 CKA_EXTRACTABLE
00272  * attribute.
00273  */
00274 #define PK11_ATTR_EXTRACTABLE   0x00000100L
00275 #define PK11_ATTR_UNEXTRACTABLE 0x00000200L
