00269         strings.init();

00272     ZoneStats(mozilla::MoveRef<ZoneStats> other)
00273         : ZoneStatsPod(other),
00274           strings(mozilla::Move(other->strings)),
00275           notableStrings(mozilla::Move(other->notableStrings))
00276     {}
00277 
00278     // Add other's numbers to this object's numbers.  Both objects'
00279     // notableStrings vectors must be empty at this point, because we can't
00280     // merge them.  (A NotableStringInfo contains only a prefix of the string,
00281     // so we can't tell whether two NotableStringInfo objects correspond to the
00282     // same string.)
00283     void add(const ZoneStats &other) {
00284         ZoneStatsPod::add(other);
00285 
00286         MOZ_ASSERT(notableStrings.empty());
00287         MOZ_ASSERT(other.notableStrings.empty());
00288 
00289         for (StringsHashMap::Range r = other.strings.all(); !r.empty(); r.popFront()) {
00290             StringsHashMap::AddPtr p = strings.lookupForAdd(r.front().key);
00291             if (p) {
00292                 // We've seen this string before; add its size to our tally.
00293                 p->value.add(r.front().value);
00294             } else {
00295                 // We haven't seen this string before; add it to the hashtable.
00296                 strings.add(p, r.front().key, r.front().value);
00297             }
00298         }
00299     }
