00391 // This class is designed to be used when you have an event class E that has a
00392 // pointer back to resource class R.  If R goes away while E is still pending,
00393 // then it is important to "revoke" E so that it does not try use R after R has
00394 // been destroyed.  nsRevocableEventPtr makes it easy for R to manage such
00395 // situations:
00396 //
00397 //   class R;
00398 //
00399 //   class E : public nsRunnable {
00400 //   public:
00401 //     void Revoke() {
00402 //       mResource = nullptr;
00403 //     }
00404 //   private:
00405 //     R *mResource;
00406 //   };
00407 //
00408 //   class R {
00409 //   public:
00410 //     void EventHandled() {
00411 //       mEvent.Forget();
00412 //     }
00413 //   private:
00414 //     nsRevocableEventPtr<E> mEvent;
00415 //   };
00416 //
00417 //   void R::PostEvent() {
00418 //     // Make sure any pending event is revoked.
00419 //     mEvent->Revoke();
00420 //
00421 //     nsCOMPtr<nsIRunnable> event = new E();
00422 //     if (NS_SUCCEEDED(NS_DispatchToCurrentThread(event))) {
00423 //       // Keep pointer to event so we can revoke it.
00424 //       mEvent = event;
00425 //     }
00426 //   }
00427 //
00428 //   NS_IMETHODIMP E::Run() {
00429 //     if (!mResource)
00430 //       return NS_OK;
00431 //     ...
00432 //     mResource->EventHandled();
00433 //     return NS_OK;
00434 //   }
00435 //
00436 template <class T>
00437 class nsRevocableEventPtr {
00438 public:
00439   nsRevocableEventPtr()
00440     : mEvent(nullptr) {
00441   }
00442 
00443   ~nsRevocableEventPtr() {
00444     Revoke();
00445   }
00446 
00447   const nsRevocableEventPtr& operator=(T *event) {
00448     if (mEvent != event) {
00449       Revoke();
00450       mEvent = event;
00451     }
00452     return *this;
00453   }

00456     if (mEvent) {
00457       mEvent->Revoke();
