00023 /*
00024  * The following classes are designed to cause assertions to detect
00025  * inadvertent use of guard objects as temporaries. In other words,
00026  * when we have a guard object whose only purpose is its constructor and
00027  * destructor (and is never otherwise referenced), the intended use
00028  * might be:
00029  *
00030  *   AutoRestore savePainting(mIsPainting);
00031  *
00032  * but is is easy to accidentally write:
00033  *
00034  *   AutoRestore(mIsPainting);
00035  *
00036  * which compiles just fine, but runs the destructor well before the
00037  * intended time.
00038  *
00039  * They work by adding (#ifdef DEBUG) an additional parameter to the
00040  * guard object's constructor, with a default value, so that users of
00041  * the guard object's API do not need to do anything. The default value
00042  * of this parameter is a temporary object. C++ (ISO/IEC 14882:1998),
00043  * section 12.2 [class.temporary], clauses 4 and 5 seem to assume a
00044  * guarantee that temporaries are destroyed in the reverse of their
00045  * construction order, but I actually can't find a statement that that
00046  * is true in the general case (beyond the two specific cases mentioned
00047  * there). However, it seems to be true.
00048  *
00049  * These classes are intended to be used only via the macros immediately
00050  * below them:
00051  *
00052  *   MOZ_DECL_USE_GUARD_OBJECT_NOTIFIER declares (ifdef DEBUG) a member
00053  *     variable, and should be put where a declaration of a private
00054  *     member variable would be placed.
00055  *   MOZ_GUARD_OBJECT_NOTIFIER_PARAM should be placed at the end of the
00056  *     parameters to each constructor of the guard object; it declares
00057  *     (ifdef DEBUG) an additional parameter. (But use the *_ONLY_PARAM
00058  *     variant for constructors that take no other parameters.)
00059  *   MOZ_GUARD_OBJECT_NOTIFIER_PARAM_IN_IMPL should likewise be used in
00060  *     the implementation of such constructors when they are not inline.
00061  *   MOZ_GUARD_OBJECT_NOTIFIER_PARAM_TO_PARENT should be used in
00062  *     the implementation of such constructors to pass the parameter to
00063  *     a base class that also uses these macros
00064  *   MOZ_GUARD_OBJECT_NOTIFIER_INIT is a statement that belongs in each
00065  *     constructor. It uses the parameter declared by
00066  *     MOZ_GUARD_OBJECT_NOTIFIER_PARAM.
00067  *
00068  * For more details, and examples of using these macros, see
00069  * https://developer.mozilla.org/en/Using_RAII_classes_in_Mozilla
00070  */
00071 class MOZ_EXPORT GuardObjectNotifier
00072 {
00073   private:
00074     bool* statementDone;
00075 
00076   public:
00077     GuardObjectNotifier() : statementDone(nullptr) { }
00078 
00079     ~GuardObjectNotifier() {
00080       *statementDone = true;
00081     }
00082 
00083     void setStatementDone(bool* statementIsDone) {
00084       statementDone = statementIsDone;
00085     }
00086 };
00087 
00088 class MOZ_EXPORT GuardObjectNotificationReceiver
00089 {
00090   private:
00091     bool statementDone;
00092 
00093   public:
00094     GuardObjectNotificationReceiver() : statementDone(false) { }
00095 
00096     ~GuardObjectNotificationReceiver() {
00097       /*
00098        * Assert that the guard object was not used as a temporary.  (Note that
00099        * this assert might also fire if init is not called because the guard
00100        * object's implementation is not using the above macros correctly.)
00101        */
00102       MOZ_ASSERT(statementDone);
00103     }
00104 
00105     void init(const GuardObjectNotifier& constNotifier) {
00106       /*
00107        * constNotifier is passed as a const reference so that we can pass a
00108        * temporary, but we really intend it as non-const.
00109        */
00110       GuardObjectNotifier& notifier = const_cast<GuardObjectNotifier&>(constNotifier);
00111       notifier.setStatementDone(&statementDone);
00112     }
00113 };
