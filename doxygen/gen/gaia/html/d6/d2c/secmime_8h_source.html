00019 /*
00020  * Initialize the local recording of the user S/MIME cipher preferences.
00021  * This function is called once for each cipher, the order being
00022  * important (first call records greatest preference, and so on).
00023  * When finished, it is called with a "which" of CIPHER_FAMILID_MASK.
00024  * If the function is called again after that, it is assumed that
00025  * the preferences are being reset, and the old preferences are
00026  * discarded.
00027  *
00028  * XXX This is for a particular user, and right now the storage is
00029  * XXX local, static.  The preference should be stored elsewhere to allow
00030  * XXX for multiple uses of one library?  How does SSL handle this;
00031  * XXX it has something similar?
00032  *
00033  *  - The "which" values are defined in ciferfam.h (the SMIME_* values,
00034  *    for example SMIME_DES_CBC_56).
00035  *  - If "on" is non-zero then the named cipher is enabled, otherwise
00036  *    it is disabled.  (It is not necessary to call the function for
00037  *    ciphers that are disabled, however, as that is the default.)
00038  *
00039  * If the cipher preference is successfully recorded, SECSuccess
00040  * is returned.  Otherwise SECFailure is returned.  The only errors
00041  * are due to failure allocating memory or bad parameters/calls:
00042  *  SEC_ERROR_XXX ("which" is not in the S/MIME cipher family)
00043  *  SEC_ERROR_XXX (function is being called more times than there
00044  *      are known/expected ciphers)
00045  */
00046 extern SECStatus SECMIME_EnableCipher(long which, int on);
00047 
00048 /*
00049  * Initialize the local recording of the S/MIME policy.
00050  * This function is called to enable/disable a particular cipher.
00051  * (S/MIME encryption or decryption using a particular cipher is only
00052  * allowed if that cipher is currently enabled.)  At startup, all S/MIME
00053  * ciphers are disabled.  From that point, this function can be called
00054  * to enable a cipher -- it is not necessary to call this to disable
00055  * a cipher unless that cipher was previously, explicitly enabled via
00056  * this function.
00057  *
00058  * XXX This is for a the current module, I think, so local, static storage
00059  * XXX is okay.  Is that correct, or could multiple uses of the same
00060  * XXX library expect to operate under different policies?
00061  *
00062  *  - The "which" values are defined in ciferfam.h (the SMIME_* values,
00063  *    for example SMIME_DES_CBC_56).
00064  *  - If "on" is non-zero then the named cipher is enabled, otherwise
00065  *    it is disabled.
00066  *
00067  * If the cipher is successfully enabled/disabled, SECSuccess is
00068  * returned.  Otherwise SECFailure is returned.  The only errors
00069  * are due to bad parameters:
00070  *  SEC_ERROR_XXX ("which" is not in the S/MIME cipher family)
00071  *  SEC_ERROR_XXX ("which" exceeds expected maximum cipher; this is
00072  *      really an internal error)
00073  */
00074 extern SECStatus SECMIME_SetPolicy(long which, int on);

00082 /*
00083  * Does the current policy allow *any* S/MIME encryption (or decryption)?
00084  *
00085  * This tells whether or not *any* S/MIME encryption can be done,
00086  * according to policy.  Callers may use this to do nicer user interface
00087  * (say, greying out a checkbox so a user does not even try to encrypt
00088  * a message when they are not allowed to) or for any reason they want
00089  * to check whether S/MIME encryption (or decryption, for that matter)
00090  * may be done.
00091  *
00092  * It takes no arguments.  The return value is a simple boolean:
00093  *   PR_TRUE means encryption (or decryption) is *possible*
00094  *  (but may still fail due to other reasons, like because we cannot
00095  *  find all the necessary certs, etc.; PR_TRUE is *not* a guarantee)
00096  *   PR_FALSE means encryption (or decryption) is not permitted
00097  *
00098  * There are no errors from this routine.
00099  */
00100 extern PRBool SECMIME_EncryptionPossible(void);

00102 /*
00103  * Start an S/MIME encrypting context.
00104  *
00105  * "scert" is the cert for the sender.  It will be checked for validity.
00106  * "rcerts" are the certs for the recipients.  They will also be checked.
00107  *
00108  * "certdb" is the cert database to use for verifying the certs.
00109  * It can be NULL if a default database is available (like in the client).
00110  *
00111  * This function already does all of the stuff specific to S/MIME protocol
00112  * and local policy; the return value just needs to be passed to
00113  * SEC_PKCS7Encode() or to SEC_PKCS7EncoderStart() to create the encoded data,
00114  * and finally to SEC_PKCS7DestroyContentInfo().
00115  *
00116  * An error results in a return value of NULL and an error set.
00117  * (Retrieve specific errors via PORT_GetError()/XP_GetError().)
00118  */
00119 extern SEC_PKCS7ContentInfo *SECMIME_CreateEncrypted(CERTCertificate *scert,
00120                              CERTCertificate **rcerts,
00121                              CERTCertDBHandle *certdb,
00122                              SECKEYGetPasswordKey pwfn,
00123                              void *pwfn_arg);

00125 /*
00126  * Start an S/MIME signing context.
00127  *
00128  * "scert" is the cert that will be used to sign the data.  It will be
00129  * checked for validity.
00130  *
00131  * "certdb" is the cert database to use for verifying the cert.
00132  * It can be NULL if a default database is available (like in the client).
00133  * 
00134  * "digestalg" names the digest algorithm.  (It should be SEC_OID_SHA1;
00135  * XXX There should be SECMIME functions for hashing, or the hashing should
00136  * be built into this interface, which we would like because we would
00137  * support more smartcards that way, and then this argument should go away.)
00138  *
00139  * "digest" is the actual digest of the data.  It must be provided in
00140  * the case of detached data or NULL if the content will be included.
00141  *
00142  * This function already does all of the stuff specific to S/MIME protocol
00143  * and local policy; the return value just needs to be passed to
00144  * SEC_PKCS7Encode() or to SEC_PKCS7EncoderStart() to create the encoded data,
00145  * and finally to SEC_PKCS7DestroyContentInfo().
00146  *
00147  * An error results in a return value of NULL and an error set.
00148  * (Retrieve specific errors via PORT_GetError()/XP_GetError().)
00149  */
00150 extern SEC_PKCS7ContentInfo *SECMIME_CreateSigned(CERTCertificate *scert,

00152                           CERTCertDBHandle *certdb,
00153                           SECOidTag digestalg,
00154                           SECItem *digest,
