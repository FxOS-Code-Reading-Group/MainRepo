00014 enum nsStyleUnit {
00015   eStyleUnit_Null         = 0,      // (no value) value is not specified
00016   eStyleUnit_Normal       = 1,      // (no value)
00017   eStyleUnit_Auto         = 2,      // (no value)

00019   eStyleUnit_Percent      = 10,     // (float) 1.0 == 100%
00020   eStyleUnit_Factor       = 11,     // (float) a multiplier
00021   eStyleUnit_Degree       = 12,     // (float) angle in degrees
00022   eStyleUnit_Grad         = 13,     // (float) angle in grads

00026   eStyleUnit_Integer      = 30,     // (int) value is simple integer

00029   // The following are allocated types.  They are weak pointers to
00030   // values allocated by nsStyleContext::Alloc.

00032                                     // to distinguish 50% from calc(50%), etc.

00035 typedef union {
00036   int32_t     mInt;   // nscoord is a int32_t for now
00037   float       mFloat;
00038   // An mPointer is a weak pointer to a value that is guaranteed to
00039   // outlive the nsStyleCoord.  In the case of nsStyleCoord::Calc*, it
00040   // is a pointer owned by the style context, allocated through
00041   // nsStyleContext::Alloc (and, therefore, is never stored in the rule
00042   // tree).
00043   void*       mPointer;
