00092         let app = apps[i];

00095         if (entryPoints) {

00097             let currentEntryPoint = entryPoints[ep];
00098             let appName = currentEntryPoint.name;
00099 
00100             if (normalizedSearchName === GaiaApps.normalizeName(appName)) {
00101               return GaiaApps.sendLocateResponse(aCallback, app, appName, ep);
00102             }
00103           }
00104         } else {
00105           let appName = app.manifest.name;
00106 
00107           if (normalizedSearchName === GaiaApps.normalizeName(appName)) {
00108             return GaiaApps.sendLocateResponse(aCallback, app, appName);
00109           }
00110         }
