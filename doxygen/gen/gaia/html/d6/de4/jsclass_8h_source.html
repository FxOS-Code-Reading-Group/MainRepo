00032 /*
00033  * We partition the ways to refer to a property into three: by an index
00034  * (uint32_t); by a string whose characters do not represent an index
00035  * (PropertyName, see vm/String.h); and by various special values.
00036  *
00037  * Special values are encoded using SpecialId, which is layout-compatible but
00038  * non-interconvertible with jsid.  A SpecialId is used for JSID_VOID, which
00039  * does not occur in JS scripts but may be used to indicate the absence of a
00040  * valid identifier.  In the future, a SpecialId may also be an object used by
00041  * Harmony-proposed private names.
00042  */
00043 class SpecialId
00044 {
00045     uintptr_t bits_;
00046 
00047     /* Needs access to raw bits. */
00048     friend JS_ALWAYS_INLINE jsid SPECIALID_TO_JSID(const SpecialId &sid);
00049     friend class PropertyId;
00050 
00051     static const uintptr_t TYPE_VOID = JSID_TYPE_VOID;
00052     static const uintptr_t TYPE_OBJECT = JSID_TYPE_OBJECT;
00053     static const uintptr_t TYPE_MASK = JSID_TYPE_MASK;
00054 
00055     SpecialId(uintptr_t bits) : bits_(bits) { }
00056 
00057   public:
00058     SpecialId() : bits_(TYPE_VOID) { }
00059 
00060     /* Object-valued */
00061 
00062     SpecialId(JSObject &obj)
00063       : bits_(uintptr_t(&obj) | TYPE_OBJECT)
00064     {
00065         JS_ASSERT(&obj != NULL);
00066         JS_ASSERT((uintptr_t(&obj) & TYPE_MASK) == 0);
00067     }
00068 
00069     bool isObject() const {
00070         return (bits_ & TYPE_MASK) == TYPE_OBJECT && bits_ != TYPE_OBJECT;
00071     }
00072 
00073     JSObject *toObject() const {
00074         JS_ASSERT(isObject());
00075         return reinterpret_cast<JSObject *>(bits_ & ~TYPE_MASK);
00076     }
00077 
00078     /* Empty */
00079 
00080     static SpecialId empty() {
00081         SpecialId sid(TYPE_OBJECT);
00082         JS_ASSERT(sid.isEmpty());
00083         return sid;
00084     }
00085 
00086     bool isEmpty() const {
00087         return bits_ == TYPE_OBJECT;
00088     }
00089 
00090     /* Void */
00091 
00092     static SpecialId voidId() {
00093         SpecialId sid(TYPE_VOID);
00094         JS_ASSERT(sid.isVoid());
00095         return sid;
00096     }
00097 
00098     bool isVoid() const {
00099         return bits_ == TYPE_VOID;
00100     }
00101 };

00103 static JS_ALWAYS_INLINE jsid
00104 SPECIALID_TO_JSID(const SpecialId &sid)
00105 {
00106     jsid id;
00107     JSID_BITS(id) = sid.bits_;
00108     JS_ASSERT_IF(sid.isObject(), JSID_IS_OBJECT(id) && JSID_TO_OBJECT(id) == sid.toObject());
00109     JS_ASSERT_IF(sid.isVoid(), JSID_IS_VOID(id));
00110     JS_ASSERT_IF(sid.isEmpty(), JSID_IS_EMPTY(id));
00111     return id;
00112 }

00120 static JS_ALWAYS_INLINE SpecialId
00121 JSID_TO_SPECIALID(jsid id)
00122 {
00123     JS_ASSERT(JSID_IS_SPECIAL(id));
00124     if (JSID_IS_OBJECT(id))
00125         return SpecialId(*JSID_TO_OBJECT(id));
00126     if (JSID_IS_EMPTY(id))
00127         return SpecialId::empty();

00129     return SpecialId::voidId();
