00021 inline void
00022 NS_SetAuthInfo(nsIAuthInformation* aAuthInfo, const nsString& user,
00023                const nsString& password)
00024 {
00025   uint32_t flags;
00026   aAuthInfo->GetFlags(&flags);
00027   if (flags & nsIAuthInformation::NEED_DOMAIN) {
00028     // Domain is separated from username by a backslash
00029     int32_t idx = user.FindChar(PRUnichar('\\'));
00030     if (idx == kNotFound) {
00031       aAuthInfo->SetUsername(user);
00032     } else {
00033       aAuthInfo->SetDomain(Substring(user, 0, idx));
00034       aAuthInfo->SetUsername(Substring(user, idx + 1));
00035     }
00036   } else {
00037     aAuthInfo->SetUsername(user);
00038   }
00039   aAuthInfo->SetPassword(password);
00040 }

00055 inline void
00056 NS_GetAuthHostPort(nsIChannel* aChannel, nsIAuthInformation* aAuthInfo,
00057                    bool machineProcessing, nsCString& host, int32_t* port)
00058 {
00059   nsCOMPtr<nsIURI> uri;
00060   nsresult rv = aChannel->GetURI(getter_AddRefs(uri));
00061   if (NS_FAILED(rv))
00062     return;
00063 
00064   // Have to distinguish proxy auth and host auth here...
00065   uint32_t flags;
00066   aAuthInfo->GetFlags(&flags);
00067   if (flags & nsIAuthInformation::AUTH_PROXY) {
00068     nsCOMPtr<nsIProxiedChannel> proxied(do_QueryInterface(aChannel));
00069     NS_ASSERTION(proxied, "proxy auth needs nsIProxiedChannel");
00070 
00071     nsCOMPtr<nsIProxyInfo> info;
00072     proxied->GetProxyInfo(getter_AddRefs(info));
00073     NS_ASSERTION(info, "proxy auth needs nsIProxyInfo");
00074 
00075     nsAutoCString idnhost;
00076     info->GetHost(idnhost);
00077     info->GetPort(port);
00078 
00079     if (machineProcessing) {
00080       nsCOMPtr<nsIIDNService> idnService =
00081         do_GetService(NS_IDNSERVICE_CONTRACTID);
00082       if (idnService) {
00083         idnService->ConvertUTF8toACE(idnhost, host);
00084       } else {
00085         // Not much we can do here...
00086         host = idnhost;
00087       }
00088     } else {
00089       host = idnhost;
00090     }
00091   } else {
00092     if (machineProcessing) {
00093       uri->GetAsciiHost(host);
00094       *port = NS_GetRealPort(uri);
00095     } else {
00096       uri->GetHost(host);
00097       uri->GetPort(port);
00098     }
00099   }
00100 }

00107 inline void
00108 NS_GetAuthKey(nsIChannel* aChannel, nsIAuthInformation* aAuthInfo,
00109               nsCString& key)
00110 {
00111   // HTTP does this differently from other protocols
00112   nsCOMPtr<nsIHttpChannel> http(do_QueryInterface(aChannel));
00113   if (!http) {
00114     nsCOMPtr<nsIURI> uri;
00115     aChannel->GetURI(getter_AddRefs(uri));
00116     uri->GetPrePath(key);
00117     return;
00118   }
00119 
00120   // NOTE: For backwards-compatibility reasons, this must be the ASCII host.
00121   nsCString host;
00122   int32_t port = -1;
00123 
00124   NS_GetAuthHostPort(aChannel, aAuthInfo, true, host, &port);
00125 
00126   nsAutoString realm;
00127   aAuthInfo->GetRealm(realm);
00128   
00129   // Now assemble the key: host:port (realm)
00130   key.Append(host);
00131   key.Append(':');
00132   key.AppendInt(port);
00133   key.AppendLiteral(" (");
00134   AppendUTF16toUTF8(realm, key);
