00031 inline float NS_IEEEPositiveInfinity() {
00032   union { uint32_t mPRUint32; float mFloat; } pun;
00033   pun.mPRUint32 = 0x7F800000;
00034   return pun.mFloat;

00036 inline bool NS_IEEEIsNan(float aF) {
00037   union { uint32_t mBits; float mFloat; } pun;
00038   pun.mFloat = aF;
00039   return (pun.mBits & 0x7F800000) == 0x7F800000 &&
00040     (pun.mBits & 0x007FFFFF) != 0;
00041 }

00102 inline nscoord _nscoordSaturatingMultiply(nscoord aCoord, float aScale,
00103                                           bool requireNotNegative) {
00104   VERIFY_COORD(aCoord);
00105   if (requireNotNegative) {
00106     NS_ABORT_IF_FALSE(aScale >= 0.0f,
00107                       "negative scaling factors must be handled manually");
00108   }
00109 #ifdef NS_COORD_IS_FLOAT
00110   return floorf(aCoord * aScale);
00111 #else
00112   // This one's only a warning because it may be possible to trigger it with
00113   // valid inputs.
00114   NS_WARN_IF_FALSE((requireNotNegative
00115                     ? aCoord > 0
00116                     : (aCoord > 0) == (aScale > 0))
00117                    ? floorf(aCoord * aScale) < nscoord_MAX
00118                    : ceilf(aCoord * aScale) > nscoord_MIN,
00119                    "nscoord multiplication capped");
00120 
00121   float product = aCoord * aScale;
00122   if (requireNotNegative ? aCoord > 0 : (aCoord > 0) == (aScale > 0))
00123     return NSToCoordRoundWithClamp(std::min<float>(nscoord_MAX, product));
00124   return NSToCoordRoundWithClamp(std::max<float>(nscoord_MIN, product));
00125 #endif
00126 }

00155 inline nscoord
00156 NSCoordSaturatingAdd(nscoord a, nscoord b)
00157 {
00158   VERIFY_COORD(a);
00159   VERIFY_COORD(b);
00160   NS_ASSERTION(a != nscoord_MIN && b != nscoord_MIN,
00161                "NSCoordSaturatingAdd got nscoord_MIN as argument");
00162 
00163 #ifdef NS_COORD_IS_FLOAT
00164   // Float math correctly handles a+b, given that neither is -infinity.
00165   return a + b;
00166 #else
00167   if (a == nscoord_MAX || b == nscoord_MAX) {
00168     // infinity + anything = anything + infinity = infinity
00169     return nscoord_MAX;
00170   } else {
00171     // a + b = a + b
00172     NS_ASSERTION(a < nscoord_MAX && b < nscoord_MAX,
00173                  "Doing nscoord addition with values > nscoord_MAX");
00174     NS_ASSERTION((int64_t)a + (int64_t)b > (int64_t)nscoord_MIN,
00175                  "nscoord addition will reach or pass nscoord_MIN");
00176     // This one's only a warning because the std::min below means that
00177     // we'll handle this case correctly.
00178     NS_WARN_IF_FALSE((int64_t)a + (int64_t)b < (int64_t)nscoord_MAX,
00179                      "nscoord addition capped to nscoord_MAX");
00180 
00181     // Cap the result, just in case we're dealing with numbers near nscoord_MAX
00182     return std::min(nscoord_MAX, a + b);
00183   }
00184 #endif
00185 }

00204 NSCoordSaturatingSubtract(nscoord a, nscoord b, 
00205                           nscoord infMinusInfResult)
00206 {
00207   VERIFY_COORD(a);
00208   VERIFY_COORD(b);
00209   NS_ASSERTION(a != nscoord_MIN && b != nscoord_MIN,
00210                "NSCoordSaturatingSubtract got nscoord_MIN as argument");
00211 
00212   if (b == nscoord_MAX) {
00213     if (a == nscoord_MAX) {
00214       // case (a)
00215       return infMinusInfResult;
00216     } else {
00217       // case (b)
00218       NS_NOTREACHED("Attempted to subtract [n - nscoord_MAX]");
00219       return 0;
00220     }
00221   } else {
00222 #ifdef NS_COORD_IS_FLOAT
00223     // case (c) and (d) for floats.  (float math handles both)
00224     return a - b;
00225 #else
00226     if (a == nscoord_MAX) {
00227       // case (c) for integers
00228       return nscoord_MAX;
00229     } else {
00230       // case (d) for integers
00231       NS_ASSERTION(a < nscoord_MAX && b < nscoord_MAX,
00232                    "Doing nscoord subtraction with values > nscoord_MAX");
00233       NS_ASSERTION((int64_t)a - (int64_t)b > (int64_t)nscoord_MIN,
00234                    "nscoord subtraction will reach or pass nscoord_MIN");
00235       // This one's only a warning because the std::min below means that
00236       // we'll handle this case correctly.
00237       NS_WARN_IF_FALSE((int64_t)a - (int64_t)b < (int64_t)nscoord_MAX,
00238                        "nscoord subtraction capped to nscoord_MAX");
00239 
00240       // Cap the result, in case we're dealing with numbers near nscoord_MAX
00241       return std::min(nscoord_MAX, a - b);
