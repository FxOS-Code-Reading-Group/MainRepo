00127   bool
00128   ApplyGCSetting(JSGCParamKey aKey, uint32_t aValue)
00129   {
00130     JSSettings::JSGCSetting* firstEmptySetting = nullptr;
00131     JSSettings::JSGCSetting* foundSetting = nullptr;
00132 
00133     for (uint32_t index = 0; index < ArrayLength(gcSettings); index++) {
00134       JSSettings::JSGCSetting& setting = gcSettings[index];
00135       if (setting.key == aKey) {
00136         foundSetting = &setting;
00137         break;
00138       }
00139       if (!firstEmptySetting && !setting.IsSet()) {
00140         firstEmptySetting = &setting;
00141       }
00142     }
00143 
00144     if (aValue) {
00145       if (!foundSetting) {
00146         foundSetting = firstEmptySetting;
00147         if (!foundSetting) {
00148           NS_ERROR("Not enough space for this value!");
00149           return false;
00150         }
00151       }
00152       foundSetting->key = aKey;
00153       foundSetting->value = aValue;
00154       return true;
00155     }
00156 
00157     if (foundSetting) {
00158       foundSetting->Unset();
00159       return true;
00160     }
00161 
00162     return false;
00163   }
