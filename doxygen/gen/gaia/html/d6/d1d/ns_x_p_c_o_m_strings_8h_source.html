00713 enum nsCStringEncoding {
00714   /* Conversion between ASCII and UTF-16 assumes that all bytes in the source
00715    * string are 7-bit ASCII and can be inflated to UTF-16 by inserting null
00716    * bytes.  Reverse conversion is done by truncating every other byte.  The
00717    * conversion may result in loss and/or corruption of information if the
00718    * strings do not strictly contain ASCII data. */
00719   NS_CSTRING_ENCODING_ASCII = 0,
00720 
00721   /* Conversion between UTF-8 and UTF-16 is non-lossy. */
00722   NS_CSTRING_ENCODING_UTF8 = 1,
00723 
00724   /* Conversion from UTF-16 to the native filesystem charset may result in a
00725    * loss of information.  No attempt is made to protect against data loss in
00726    * this case.  The native filesystem charset applies to strings passed to
00727    * the "Native" method variants on nsIFile. */
00728   NS_CSTRING_ENCODING_NATIVE_FILESYSTEM = 2
00729 };
