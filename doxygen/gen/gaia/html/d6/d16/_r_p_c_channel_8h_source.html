00217                       const Message* msg) : mThat(that) {
00218             mThat.AssertWorkerThread();

00220             if (mThat.mCxxStackFrames.empty())
00221                 mThat.EnteredCxxStack();
00222 
00223             mThat.mCxxStackFrames.push_back(RPCFrame(direction, msg));
00224             const RPCFrame& frame = mThat.mCxxStackFrames.back();
00225 
00226             if (frame.IsRPCIncall())
00227                 mThat.EnteredCall();
00228 
00229             mThat.mSawRPCOutMsg |= frame.IsRPCOutcall();
00230         }
00231 
00232         ~CxxStackFrame() {
00233             bool exitingCall = mThat.mCxxStackFrames.back().IsRPCIncall();
00234             mThat.mCxxStackFrames.pop_back();
00235             bool exitingStack = mThat.mCxxStackFrames.empty();

00238             // RPCChannel code was still on the stack
00239             if (!mThat.mListener)
00240                 return;
00241 
00242             mThat.AssertWorkerThread();
00243             if (exitingCall)
00244                 mThat.ExitedCall();

00272     // 
00273     // Queue of all incoming messages, except for replies to sync
00274     // messages, which are delivered directly to the SyncChannel
00275     // through its mRecvd member.
00276     //
00277     // If both this side and the other side are functioning correctly,
00278     // the queue can only be in certain configurations.  Let
00279     // 
00280     //   |A<| be an async in-message,
00281     //   |S<| be a sync in-message,
00282     //   |C<| be an RPC in-call,
00283     //   |R<| be an RPC reply.
00284     // 
00285     // The queue can only match this configuration
00286     // 
00287     //  A<* (S< | C< | R< (?{mStack.size() == 1} A<* (S< | C<)))
00288     //
00289     // The other side can send as many async messages |A<*| as it
00290     // wants before sending us a blocking message.
00291     //
00292     // The first case is |S<|, a sync in-msg.  The other side must be
00293     // blocked, and thus can't send us any more messages until we
00294     // process the sync in-msg.
00295     //
00296     // The second case is |C<|, an RPC in-call; the other side must be
00297     // blocked.  (There's a subtlety here: this in-call might have
00298     // raced with an out-call, but we detect that with the mechanism
00299     // below, |mRemoteStackDepth|, and races don't matter to the
00300     // queue.)
00301     //
00302     // Final case, the other side replied to our most recent out-call
00303     // |R<|.  If that was the *only* out-call on our stack,
00304     // |?{mStack.size() == 1}|, then other side "finished with us,"
00305     // and went back to its own business.  That business might have
00306     // included sending any number of async message |A<*| until
00307     // sending a blocking message |(S< | C<)|.  If we had more than
00308     // one RPC call on our stack, the other side *better* not have
00309     // sent us another blocking message, because it's blocked on a
00310     // reply from us.
00311     //
00312     typedef std::deque<Message> MessageQueue;
00313     MessageQueue mPending;
00314 
00315     // List of async and sync messages that have been received while waiting
00316     // for an urgent reply, that need to be deferred until that reply has been
00317     // received.
00318     MessageQueue mNonUrgentDeferred;

00340     //
00341     // This is what we think the RPC stack depth is on the "other
00342     // side" of this RPC channel.  We maintain this variable so that
00343     // we can detect racy RPC calls.  With each RPC out-call sent, we
00344     // send along what *we* think the stack depth of the remote side
00345     // is *before* it will receive the RPC call.
00346     //
00347     // After sending the out-call, our stack depth is "incremented"
00348     // by pushing that pending message onto mPending.
00349     //
00350     // Then when processing an in-call |c|, it must be true that
00351     //
00352     //   mStack.size() == c.remoteDepth
00353     //
00354     // i.e., my depth is actually the same as what the other side
00355     // thought it was when it sent in-call |c|.  If this fails to
00356     // hold, we have detected racy RPC calls.
00357     //
00358     // We then increment mRemoteStackDepth *just before* processing
00359     // the in-call, since we know the other side is waiting on it, and
00360     // decrement it *just after* finishing processing that in-call,
00361     // since our response will pop the top of the other side's
00362     // |mPending|.
00363     //
00364     // One nice aspect of this race detection is that it is symmetric;
00365     // if one side detects a race, then the other side must also 
00366     // detect the same race.
00367     //
00368     size_t mRemoteStackDepthGuess;
