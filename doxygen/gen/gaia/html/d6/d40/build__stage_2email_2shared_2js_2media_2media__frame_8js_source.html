00001 /*
00002  * media_frame.js:
00003  *
00004  * A MediaFrame displays a photo or a video. The gallery app uses
00005  * three side by side to support smooth panning from one item to the
00006  * next.  The Camera app uses one for image and video preview. The
00007  * Gallery app's open activity uses one of these to display the opened
00008  * item.
00009  *
00010  * MediaFrames have different behavior depending on whether they display
00011  * images or videos. Photo frames allow the user to zoom and pan on the photo.
00012  * Video frames allow the user to play and pause but don't allow zooming.
00013  *
00014  * When a frame is displaying a video, it handles mouse events.
00015  * When display a picture, however, it expects the client to handle events
00016  * and call the pan() and zoom() methods.
00017  *
00018  * The pan() method is a little unusual. It "uses" as much of the pan
00019  * event as it can, and returns a number indicating how much of the
00020  * horizontal motion it did not use. The gallery uses this returned
00021  * value for transitioning between frames.  If a frame displays a
00022  * photo that is not zoomed in at all, then it can't use any of the
00023  * pan, and returns the full amount which the gallery app turns into a
00024  * panning motion between frames.  But if the photo is zoomed in, then
00025  * the MediaFrame will just move the photo within itself, if it can, and
00026  * return 0.
00027  *
00028  * Much of the code in this file used to be part of the PhotoState class.
00029  */

00415   if (!this.displayingImage)
00416     return;
00417 
00418   // If we were displaying the preview, switch to the full-size image
00419   if (this.displayingPreview)

00422   // Never zoom in farther than the native resolution of the image
00423   if (this.fit.scale * scale > 1) {
00424     scale = 1 / (this.fit.scale);
00425   }
00426   // And never zoom out to make the image smaller than it would normally be
00427   else if (this.fit.scale * scale < this.fit.baseScale) {
00428     scale = this.fit.baseScale / this.fit.scale;
00429   }
00430

00432

00434   this.fit.width = Math.floor(this.itemWidth * this.fit.scale);
00435   this.fit.height = Math.floor(this.itemHeight * this.fit.scale);

00438   // These are the photo coordinates displayed at that point in the viewport
00439   var photoX = fixedX - this.fit.left;
00440   var photoY = fixedY - this.fit.top;
00441 
00442   // After zooming, these are the new photo coordinates.
00443   // Note we just use the relative scale amount here, not this.fit.scale
00444   photoX = Math.floor(photoX * scale);
00445   photoY = Math.floor(photoY * scale);
00446 
00447   // To keep that point still, here are the new left and top values we need
00448   this.fit.left = fixedX - photoX;
00449   this.fit.top = fixedY - photoY;

00503 MediaFrame.prototype.pan = function(dx, dy) {
00504   // We can only pan images, so return the entire dx amount
00505   if (!this.displayingImage) {
00506     return dx;
00507   }
00508 
00509   // Handle panning in the y direction first, since it is easier.
00510   // Don't pan in the y direction if we already fit on the screen
00511   if (this.fit.height > this.viewportHeight) {
00512     this.fit.top += dy;
00513 
00514     // Don't let the top of the photo be below the top of the screen
00515     if (this.fit.top > 0)
00516       this.fit.top = 0;
00517 
00518     // bottom of photo shouldn't be above the bottom of screen
00519     if (this.fit.top + this.fit.height < this.viewportHeight)
00520       this.fit.top = this.viewportHeight - this.fit.height;
00521   }
00522 
00523   // Now handle the X dimension. If we've already panned as far as we can
00524   // within the image (or if it isn't zoomed in) then return the "extra"
00525   // unused dx amount to the caller so that the caller can use them to
00526   // shift the frame left or right.
00527   var extra = 0;
00528 
00529   if (this.fit.width <= this.viewportWidth) {
00530     // In this case, the photo isn't zoomed in, so it is all extra
00531     extra = dx;
00532   }
00533   else {
00534     this.fit.left += dx;
00535 
00536     // If this would take the left edge of the photo past the
00537     // left edge of the screen, then some of the motion is extra
00538     if (this.fit.left > 0) {
00539       extra = this.fit.left;
00540       this.fit.left = 0;
00541     }
00542 
00543     // Or, if this would take the right edge of the photo past the
00544     // right edge of the screen, then we've got extra.
00545     if (this.fit.left + this.fit.width < this.viewportWidth) {
00546       extra = this.fit.left + this.fit.width - this.viewportWidth;
00547       this.fit.left = this.viewportWidth - this.fit.width;
00548     }
00549   }
00550
