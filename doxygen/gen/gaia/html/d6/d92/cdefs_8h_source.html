00049 /*
00050  * The __CONCAT macro is used to concatenate parts of symbol names, e.g.
00051  * with "#define OLD(foo) __CONCAT(old,foo)", OLD(foo) produces oldfoo.
00052  * The __CONCAT macro is a bit tricky -- make sure you don't put spaces
00053  * in between its arguments.  __CONCAT can also concatenate double-quoted
00054  * strings produced by the __STRING macro, but this only works with ANSI C.
00055  */
00056 #if defined(__STDC__) || defined(__cplusplus) || defined(_WINDOWS) || defined(XP_OS2)
00057 #define __P(protos) protos      /* full-blown ANSI C */
00058 #define __CONCAT(x,y)   x ## y
00059 #define __STRING(x) #x

00061 /* On HP-UX 11.00, <sys/stdsyms.h> defines __const. */
00062 #ifndef __const
00063 #define __const     const       /* define reserved names to standard */
00064 #endif  /* __const */
00065 #define __signed    signed
00066 #define __volatile  volatile
00067 #ifndef _WINDOWS
00068 #if defined(__cplusplus)
00069 #define __inline    inline      /* convert to C++ keyword */
00070 #else
00071 #if !defined(__GNUC__) && !defined(__MWERKS__)
00072 #define __inline            /* delete GCC keyword */
00073 #endif /* !__GNUC__ */
00074 #endif /* !__cplusplus */
00075 #endif /* !_WINDOWS */
00076 
00077 #else   /* !(__STDC__ || __cplusplus) */
00078 #define __P(protos) ()      /* traditional C preprocessor */
00079 #define __CONCAT(x,y)   xy
00080 #define __STRING(x) "x"

00087 /*
00088  * In non-ANSI C environments, new programs will want ANSI-only C keywords
00089  * deleted from the program and old programs will want them left alone.
00090  * When using a compiler other than gcc, programs using the ANSI C keywords
00091  * const, inline etc. as normal identifiers should define -DNO_ANSI_KEYWORDS.
00092  * When using "gcc -traditional", we assume that this is the intent; if
00093  * __GNUC__ is defined but __STDC__ is not, we leave the new keywords alone.
00094  */

00096 #define const               /* delete ANSI C keywords */

00100 #endif

00104 /*
00105  * GCC1 and some versions of GCC2 declare dead (non-returning) and
00106  * pure (no side effects) functions using "volatile" and "const";
00107  * unfortunately, these then cause warnings under "-ansi -pedantic".
00108  * GCC2 uses a new, peculiar __attribute__((attrs)) style.  All of
00109  * these work for GNU C++ (modulo a slight glitch in the C++ grammar
00110  * in the distribution version of 2.5.5).
00111  */
00112 #if !defined(__GNUC__) || __GNUC__ < 2 || __GNUC_MINOR__ < 5
00113 #define __attribute__(x)    /* delete __attribute__ if non-gcc or gcc1 */
00114 #if defined(__GNUC__) && !defined(__STRICT_ANSI__)
00115 #define __dead      __volatile
00116 #define __pure      __const
00117 #endif
00118 #endif
