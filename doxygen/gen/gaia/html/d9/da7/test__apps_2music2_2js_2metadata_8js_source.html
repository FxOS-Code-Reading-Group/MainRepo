00826     function getImage(file) {
00827       // Get the embedded image from the music file
00828       var embedded = file.slice(metadata.picture.start,
00829                                 metadata.picture.end,
00830                                 metadata.picture.type);
00831       // Convert to a blob url
00832       var embeddedURL = URL.createObjectURL(embedded);
00833       // Load it into an image element
00834       offscreenImage.src = embeddedURL;
00835       offscreenImage.onerror = function() {
00836         URL.revokeObjectURL(embeddedURL);
00837         offscreenImage.removeAttribute('src');
00838         // Something went wrong reading the embedded image.
00839         // Return a default one instead
00840         console.warn('Album cover art failed to load', file.name);
00841         callback(null);
00842       };
00843       offscreenImage.onload = function() {
00844         // We've loaded the image, now copy it to a canvas
00845         var canvas = document.createElement('canvas');
00846         canvas.width = THUMBNAIL_WIDTH;
00847         canvas.height = THUMBNAIL_HEIGHT;
00848         var context = canvas.getContext('2d');
00849         var scalex = canvas.width / offscreenImage.width;
00850         var scaley = canvas.height / offscreenImage.height;
00851 
00852         // Take the larger of the two scales: we crop the image to the thumbnail
00853         var scale = Math.max(scalex, scaley);
00854 
00855         // If the image was already thumbnail size, it is its own thumbnail
00856         if (scale >= 1) {
00857           offscreenImage.removeAttribute('src');
00858           cacheThumbnail(key, embedded, embeddedURL);
00859           callback(embeddedURL);
00860           return;
00861         }
00862 
00863         // Calculate the region of the image that will be copied to the
00864         // canvas to create the thumbnail
00865         var w = Math.round(THUMBNAIL_WIDTH / scale);
00866         var h = Math.round(THUMBNAIL_HEIGHT / scale);
00867         var x = Math.round((offscreenImage.width - w) / 2);
00868         var y = Math.round((offscreenImage.height - h) / 2);
00869 
00870         // Draw that region of the image into the canvas, scaling it down
00871         context.drawImage(offscreenImage, x, y, w, h,
00872                           0, 0, THUMBNAIL_WIDTH, THUMBNAIL_HEIGHT);
00873 
00874         // We're done with the image now
00875         offscreenImage.removeAttribute('src');
00876         URL.revokeObjectURL(embeddedURL);
00877 
00878         canvas.toBlob(function(blob) {
00879           var url = URL.createObjectURL(blob);
00880           cacheThumbnail(key, blob, url);
00881           callback(url);
00882         }, 'image/jpeg');
