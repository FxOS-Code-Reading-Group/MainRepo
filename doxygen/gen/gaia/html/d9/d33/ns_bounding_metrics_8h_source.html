00012 /* Struct used for accurate measurements of a string, in order to
00013  * allow precise positioning when processing MathML.  This is in its
00014  * own header file because some very-widely-included headers need it
00015  * but not the rest of nsFontMetrics, or vice versa.
00016  */
00017 
00018 struct nsBoundingMetrics {
00019 
00021     // Metrics that _exactly_ enclose the text:
00022 
00023     // The character coordinate system is the one used on X Windows:
00024     // 1. The origin is located at the intersection of the baseline
00025     //    with the left of the character's cell.
00026     // 2. All horizontal bearings are oriented from left to right.
00027     // 2. All horizontal bearings are oriented from left to right.
00028     // 3. The ascent is oriented from bottom to top (being 0 at the orgin).
00029     // 4. The descent is oriented from top to bottom (being 0 at the origin).
00030 
00031     // Note that Win32/Mac/PostScript use a different convention for
00032     // the descent (all vertical measurements are oriented from bottom
00033     // to top on these palatforms). Make sure to flip the sign of the
00034     // descent on these platforms for cross-platform compatibility.
00035 
00036     // Any of the following member variables listed here can have
00037     // positive or negative value.
00038 
00039     nscoord leftBearing;
00040     /* The horizontal distance from the origin of the drawing
00041        operation to the left-most part of the drawn string. */
00042 
00043     nscoord rightBearing;
00044     /* The horizontal distance from the origin of the drawing
00045        operation to the right-most part of the drawn string.
00046        The _exact_ width of the string is therefore:
00047        rightBearing - leftBearing */
00048 
00049     nscoord ascent;
00050     /* The vertical distance from the origin of the drawing
00051        operation to the top-most part of the drawn string. */
00052 
00053     nscoord descent;
00054     /* The vertical distance from the origin of the drawing
00055        operation to the bottom-most part of the drawn string.
00056        The _exact_ height of the string is therefore:
00057        ascent + descent */
00058 
00059     nscoord width;
00060     /* The horizontal distance from the origin of the drawing
00061        operation to the correct origin for drawing another string
00062        to follow the current one. Depending on the font, this
00063        could be greater than or less than the right bearing. */
00064 
00065     nsBoundingMetrics() : leftBearing(0), rightBearing(0),
00066                           ascent(0), descent(0), width(0)
00067     {}
00068 
00069     void
00070     operator += (const nsBoundingMetrics& bm) {
00071         if (ascent + descent == 0 && rightBearing - leftBearing == 0) {
00072             ascent = bm.ascent;
00073             descent = bm.descent;
00074             leftBearing = width + bm.leftBearing;
00075             rightBearing = width + bm.rightBearing;
00076         }
00077         else {
00078             if (ascent < bm.ascent) ascent = bm.ascent;
00079             if (descent < bm.descent) descent = bm.descent;
00080             leftBearing = std::min(leftBearing, width + bm.leftBearing);
00081             rightBearing = std::max(rightBearing, width + bm.rightBearing);
00082         }
00083         width += bm.width;
00084     }
00085 };
