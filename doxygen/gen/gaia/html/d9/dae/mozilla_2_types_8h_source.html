00023 /*
00024  * MOZ_EXPORT is used to declare and define a symbol or type which is externally
00025  * visible to users of the current library.  It encapsulates various decorations
00026  * needed to properly export the method's symbol.
00027  *
00028  *   api.h:
00029  *     extern MOZ_EXPORT int MeaningOfLife(void);
00030  *     extern MOZ_EXPORT int LuggageCombination;
00031  *
00032  *   api.c:
00033  *     int MeaningOfLife(void) { return 42; }
00034  *     int LuggageCombination = 12345;
00035  *
00036  * If you are merely sharing a method across files, just use plain |extern|.
00037  * These macros are designed for use by library interfaces -- not for normal
00038  * methods or data used cross-file.
00039  */
00040 #if defined(WIN32) || defined(XP_OS2)
00041 #  define MOZ_EXPORT   __declspec(dllexport)
00042 #else /* Unix */
00043 #  ifdef HAVE_VISIBILITY_ATTRIBUTE
00044 #    define MOZ_EXPORT       __attribute__((visibility("default")))
00045 #  elif defined(__SUNPRO_C) || defined(__SUNPRO_CC)
00046 #    define MOZ_EXPORT      __global
00047 #  else
00048 #    define MOZ_EXPORT /* nothing */
00049 #  endif
00050 #endif
00051 
00052 
00053 /*
00054  * Whereas implementers use MOZ_EXPORT to declare and define library symbols,
00055  * users use MOZ_IMPORT_API and MOZ_IMPORT_DATA to access them.  Most often the
00056  * implementer of the library will expose an API macro which expands to either
00057  * the export or import version of the macro, depending upon the compilation
00058  * mode.
00059  */
00060 #ifdef _WIN32
00061 #  if defined(__MWERKS__)
00062 #    define MOZ_IMPORT_API /* nothing */
00063 #  else
00064 #    define MOZ_IMPORT_API __declspec(dllimport)
00065 #  endif
00066 #elif defined(XP_OS2)
00067 #  define MOZ_IMPORT_API  __declspec(dllimport)
00068 #else
00069 #  define MOZ_IMPORT_API MOZ_EXPORT
00070 #endif
00071 
00072 #if defined(_WIN32) && !defined(__MWERKS__)
00073 #  define MOZ_IMPORT_DATA  __declspec(dllimport)
00074 #elif defined(XP_OS2)
00075 #  define MOZ_IMPORT_DATA  __declspec(dllimport)
00076 #else
00077 #  define MOZ_IMPORT_DATA  MOZ_EXPORT
00078 #endif

00080 /*
00081  * Consistent with the above comment, the MFBT_API and MFBT_DATA macros expose
00082  * export mfbt declarations when building mfbt, and they expose import mfbt
00083  * declarations when using mfbt.
00084  */

00086 #  define MFBT_API     MOZ_EXPORT
00087 #  define MFBT_DATA    MOZ_EXPORT

00089   /*
00090    * On linux mozglue is linked in the program and we link libxul.so with
00091    * -z,defs. Normally that causes the linker to reject undefined references in
00092    * libxul.so, but as a loophole it allows undefined references to weak
00093    * symbols. We add the weak attribute to the import version of the MFBT API
00094    * macros to exploit this.

00098 #    define MFBT_DATA  __attribute__((weak)) MOZ_IMPORT_DATA
00099 #  else
00100 #    define MFBT_API   MOZ_IMPORT_API
00101 #    define MFBT_DATA  MOZ_IMPORT_DATA
00102 #  endif
00103 #endif

00105 /*
00106  * C symbols in C++ code must be declared immediately within |extern "C"|
00107  * blocks.  However, in C code, they need not be declared specially.  This
00108  * difference is abstracted behind the MOZ_BEGIN_EXTERN_C and MOZ_END_EXTERN_C
00109  * macros, so that the user need not know whether he is being used in C or C++
00110  * code.
00111  *
00112  *   MOZ_BEGIN_EXTERN_C
00113  *
00114  *   extern MOZ_EXPORT int MostRandomNumber(void);
00115  *   ...other declarations...
00116  *
00117  *   MOZ_END_EXTERN_C
00118  *
00119  * This said, it is preferable to just use |extern "C"| in C++ header files for
00120  * its greater clarity.
00121  */
00122 #ifdef __cplusplus
00123 #  define MOZ_BEGIN_EXTERN_C    extern "C" {
00124 #  define MOZ_END_EXTERN_C      }
00125 #else
00126 #  define MOZ_BEGIN_EXTERN_C
00127 #  define MOZ_END_EXTERN_C
00128 #endif
