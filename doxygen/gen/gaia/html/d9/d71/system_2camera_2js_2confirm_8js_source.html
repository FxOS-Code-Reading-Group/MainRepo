00001 /*
00002  * When the Camera app is invoked by a "pick" activity, it uses this module
00003  * to allow the user to preview the photo they just took or the video they
00004  * just recorded. The user is given the choice to select (i.e. pick) the
00005  * photo or video or to reject and retake it.
00006  *
00007  * This module uses shared/js/media/media_frame.js, so the user should
00008  * have the ability to zoom and pan the photo and play the video before
00009  * making their decision.
00010  */
00011 var ConfirmDialog = (function() {
00012 
00013   // These are the document elements we need
00014   var confirm = document.getElementById('confirm');
00015   var mediaFrame = document.getElementById('confirm-media-frame');
00016   var retakeButton = document.getElementById('retake-button');
00017   var selectButton = document.getElementById('select-button');
00018 
00019   // Create the MediaFrame for confirmations
00020   var frame = new MediaFrame(mediaFrame);
00021 
00022   // Enable panning and zooming for images
00023   addPanAndZoomHandlers(frame);
00024 
00025   // confirmImage() and confirmVideo() store the callback functions
00026   // in these variables so that the button callbacks can invoke them
00027   var retakecb, selectcb;
00028 
00029   // Event handlers for the two buttons
00030   retakeButton.onclick = function() {
00031     confirm.hidden = true;
00032     retakecb();
00033   };
00034   selectButton.onclick = function() {
00035     confirm.hidden = true;
00036     selectcb();
00037   };
00038 
00039   function confirmImage(blob, selectCallback, retakeCallback) {
00040     selectcb = selectCallback;
00041     retakecb = retakeCallback;
00042 
00043     parseJPEGMetadata(blob, function(metadata) {
00044       // Show the confirm pane
00045       confirm.hidden = false;
00046       frame.displayImage(blob, metadata.width, metadata.height,
00047                          metadata.preview, metadata.rotation,
00048                          metadata.mirrored);
00049     });
00050   }
00051 
00052   function confirmVideo(blob, posterBlob, width, height, rotation,
00053                         selectCallback, retakeCallback)
00054   {
00055     selectcb = selectCallback;
00056     retakecb = retakeCallback;
00057     // Show the dialog and then display the video in it.
00058     // We have to do it in this order, or the video won't display correctly.
00059     confirm.hidden = false;
00060     frame.displayVideo(blob, posterBlob, width, height, rotation);
00061   }
00062 
00063   function isShowing() {
00064     return !confirm.hidden;
00065   }
00066 
00067   // Export the public functions of this module
00068   return {
00069     confirmImage: confirmImage,
00070     confirmVideo: confirmVideo,
00071     isShowing: isShowing
00072   };
00073 })();
