00045 class GenericRefCounted : public GenericRefCountedBase

00055     virtual void AddRef() {
00056       MOZ_ASSERT(refCnt >= 0);
00057       ++refCnt;

00060     virtual void Release() {
00061       MOZ_ASSERT(refCnt > 0);
00062       if (0 == --refCnt) {
00063 #ifdef DEBUG
00064         refCnt = detail::DEAD;
00065 #endif
00066         delete this;
00067       }
00068     }

00070     int refCount() const { return refCnt; }
00071     bool hasOneRef() const {
00072       MOZ_ASSERT(refCnt > 0);
00073       return refCnt == 1;
00074     }
