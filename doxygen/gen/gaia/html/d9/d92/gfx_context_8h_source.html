00930     gfxContextAutoDisableSubpixelAntialiasing(gfxContext *aContext, bool aDisable)
00931     {
00932         if (aDisable) {
00933             if (aContext->IsCairo()) {
00934                 mSurface = aContext->CurrentSurface();
00935                 if (!mSurface) {
00936                   return;
00937                 }
00938                 mSubpixelAntialiasingEnabled = mSurface->GetSubpixelAntialiasingEnabled();
00939                 mSurface->SetSubpixelAntialiasingEnabled(false);
00940             } else {
00941                 mDT = aContext->GetDrawTarget();
00942 
00943                 mSubpixelAntialiasingEnabled = mDT->GetPermitSubpixelAA();
00944                 mDT->SetPermitSubpixelAA(false);
00945             }
00946         }
00947     }
00948     ~gfxContextAutoDisableSubpixelAntialiasing()
00949     {
00950         if (mSurface) {
00951             mSurface->SetSubpixelAntialiasingEnabled(mSubpixelAntialiasingEnabled);
00952         } else if (mDT) {
00953             mDT->SetPermitSubpixelAA(mSubpixelAntialiasingEnabled);
00954         }
00955     }
