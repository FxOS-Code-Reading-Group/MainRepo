00072   URIPrincipalAndCORSModeHashKey* GetKey() const {
00073     return const_cast<URIPrincipalAndCORSModeHashKey*>(this);
00074   }
00075   const URIPrincipalAndCORSModeHashKey* GetKeyPointer() const { return this; }
00076 
00077   bool KeyEquals(const URIPrincipalAndCORSModeHashKey* aKey) const {
00078     if (!nsURIHashKey::KeyEquals(aKey->mKey)) {
00079       return false;
00080     }
00081 
00082     if (!mPrincipal != !aKey->mPrincipal) {
00083       // One or the other has a principal, but not both... not equal
00084       return false;
00085     }
00086 
00087     if (mCORSMode != aKey->mCORSMode) {
00088       // Different CORS modes; we don't match
00089       return false;
00090     }
00091 
00092     bool eq;
00093     return !mPrincipal ||
00094       (NS_SUCCEEDED(mPrincipal->Equals(aKey->mPrincipal, &eq)) && eq);
00095   }
00096 
00097   static const URIPrincipalAndCORSModeHashKey*
00098   KeyToPointer(URIPrincipalAndCORSModeHashKey* aKey) { return aKey; }
00099   static PLDHashNumber HashKey(const URIPrincipalAndCORSModeHashKey* aKey) {
00100     return nsURIHashKey::HashKey(aKey->mKey);
00101   }
00102
