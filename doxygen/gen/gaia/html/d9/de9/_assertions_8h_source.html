00042 /*
00043  * MOZ_STATIC_ASSERT may be used to assert a condition *at compile time* in C.
00044  * In C++11, static_assert is provided by the compiler to the same effect.
00045  * This can be useful when you make certain assumptions about what must hold for
00046  * optimal, or even correct, behavior.  For example, you might assert that the
00047  * size of a struct is a multiple of the target architecture's word size:
00048  *
00049  *   struct S { ... };
00050  *   // C
00051  *   MOZ_STATIC_ASSERT(sizeof(S) % sizeof(size_t) == 0,
00052  *                     "S should be a multiple of word size for efficiency");
00053  *   // C++11
00054  *   static_assert(sizeof(S) % sizeof(size_t) == 0,
00055  *                 "S should be a multiple of word size for efficiency");
00056  *
00057  * This macro can be used in any location where both an extern declaration and a
00058  * typedef could be used.
00059  */
00060 #ifndef __cplusplus
00061    /*
00062     * Some of the definitions below create an otherwise-unused typedef.  This
00063     * triggers compiler warnings with some versions of gcc, so mark the typedefs
00064     * as permissibly-unused to disable the warnings.
00065     */
00066 #  if defined(__GNUC__)
00067 #    define MOZ_STATIC_ASSERT_UNUSED_ATTRIBUTE __attribute__((unused))
00068 #  else
00069 #    define MOZ_STATIC_ASSERT_UNUSED_ATTRIBUTE /* nothing */
00070 #  endif
00071 #  define MOZ_STATIC_ASSERT_GLUE1(x, y)          x##y
00072 #  define MOZ_STATIC_ASSERT_GLUE(x, y)           MOZ_STATIC_ASSERT_GLUE1(x, y)
00073 #  if defined(__SUNPRO_CC)
00074      /*
00075       * The Sun Studio C++ compiler is buggy when declaring, inside a function,
00076       * another extern'd function with an array argument whose length contains a
00077       * sizeof, triggering the error message "sizeof expression not accepted as
00078       * size of array parameter".  This bug (6688515, not public yet) would hit
00079       * defining moz_static_assert as a function, so we always define an extern
00080       * array for Sun Studio.
00081       *
00082       * We include the line number in the symbol name in a best-effort attempt
00083       * to avoid conflicts (see below).
00084       */
00085 #    define MOZ_STATIC_ASSERT(cond, reason) \
00086        extern char MOZ_STATIC_ASSERT_GLUE(moz_static_assert, __LINE__)[(cond) ? 1 : -1]
00087 #  elif defined(__COUNTER__)
00088      /*
00089       * If there was no preferred alternative, use a compiler-agnostic version.
00090       *
00091       * Note that the non-__COUNTER__ version has a bug in C++: it can't be used
00092       * in both |extern "C"| and normal C++ in the same translation unit.  (Alas
00093       * |extern "C"| isn't allowed in a function.)  The only affected compiler
00094       * we really care about is gcc 4.2.  For that compiler and others like it,
00095       * we include the line number in the function name to do the best we can to
00096       * avoid conflicts.  These should be rare: a conflict would require use of
00097       * MOZ_STATIC_ASSERT on the same line in separate files in the same
00098       * translation unit, *and* the uses would have to be in code with
00099       * different linkage, *and* the first observed use must be in C++-linkage
00100       * code.
00101       */
00102 #    define MOZ_STATIC_ASSERT(cond, reason) \
00103        typedef int MOZ_STATIC_ASSERT_GLUE(moz_static_assert, __COUNTER__)[(cond) ? 1 : -1] MOZ_STATIC_ASSERT_UNUSED_ATTRIBUTE
00104 #  else
00105 #    define MOZ_STATIC_ASSERT(cond, reason) \
00106        extern void MOZ_STATIC_ASSERT_GLUE(moz_static_assert, __LINE__)(int arg[(cond) ? 1 : -1]) MOZ_STATIC_ASSERT_UNUSED_ATTRIBUTE
00107 #  endif

00156     * On MSVC use the __debugbreak compiler intrinsic, which produces an inline
00157     * (not nested in a system function) breakpoint.  This distinctively invokes
00158     * Breakpad without requiring system library symbols on all stack-processing
00159     * machines, as a nested breakpoint would require.
00160     *
00161     * We use TerminateProcess with the exit code aborting would generate
00162     * because we don't want to invoke atexit handlers, destructors, library
00163     * unload handlers, and so on when our process might be in a compromised
00164     * state.
00165     *
00166     * We don't use abort() because it'd cause Windows to annoyingly pop up the
00167     * process error dialog multiple times.  See bug 345118 and bug 426163.
00168     *

00170     * compiler doesn't hassle us to provide a return statement after a
00171     * MOZ_REALLY_CRASH() call.
00172     *
00173     * (Technically these are Windows requirements, not MSVC requirements.  But
00174     * practically you need MSVC for debugging, and we only ship builds created
00175     * by MSVC, so doing it this way reduces complexity.)

00213 /*
00214  * MOZ_CRASH([explanation-string]) crashes the program, plain and simple, in a
00215  * Breakpad-compatible way, in both debug and release builds.
00216  *
00217  * MOZ_CRASH is a good solution for "handling" failure cases when you're
00218  * unwilling or unable to handle them more cleanly -- for OOM, for likely memory
00219  * corruption, and so on.  It's also a good solution if you need safe behavior
00220  * in release builds as well as debug builds.  But if the failure is one that
00221  * should be debugged and fixed, MOZ_ASSERT is generally preferable.
00222  *
00223  * The optional explanation-string, if provided, must be a string literal
00224  * explaining why we're crashing.  This argument is intended for use with
00225  * MOZ_CRASH() calls whose rationale is non-obvious; don't use it if it's
00226  * obvious why we're crashing.
00227  *
00228  * If we're a DEBUG build and we crash at a MOZ_CRASH which provides an
00229  * explanation-string, we print the string to stderr.  Otherwise, we don't
00230  * print anything; this is because we want MOZ_CRASH to be 100% safe in release
00231  * builds, and it's hard to print to stderr safely when memory might have been
00232  * corrupted.
00233  */
00234 #ifndef DEBUG
00235 #  define MOZ_CRASH(...) MOZ_REALLY_CRASH()
00236 #else
00237 #  define MOZ_CRASH(...) \
00238      do { \
00239        MOZ_ReportCrash("" __VA_ARGS__, __FILE__, __LINE__); \
00240        MOZ_REALLY_CRASH(); \
00241      } while(0)
00242 #endif

00248 /*
00249  * MOZ_ASSERT(expr [, explanation-string]) asserts that |expr| must be truthy in
00250  * debug builds.  If it is, execution continues.  Otherwise, an error message
00251  * including the expression and the explanation-string (if provided) is printed,
00252  * an attempt is made to invoke any existing debugger, and execution halts.
00253  * MOZ_ASSERT is fatal: no recovery is possible.  Do not assert a condition
00254  * which can correctly be falsy.
00255  *
00256  * The optional explanation-string, if provided, must be a string literal
00257  * explaining the assertion.  It is intended for use with assertions whose
00258  * correctness or rationale is non-obvious, and for assertions where the "real"
00259  * condition being tested is best described prosaically.  Don't provide an
00260  * explanation if it's not actually helpful.
00261  *
00262  *   // No explanation needed: pointer arguments often must not be NULL.
00263  *   MOZ_ASSERT(arg);
00264  *
00265  *   // An explanation can be helpful to explain exactly how we know an
00266  *   // assertion is valid.
00267  *   MOZ_ASSERT(state == WAITING_FOR_RESPONSE,
00268  *              "given that <thingA> and <thingB>, we must have...");
00269  *
00270  *   // Or it might disambiguate multiple identical (save for their location)
00271  *   // assertions of the same expression.
00272  *   MOZ_ASSERT(getSlot(PRIMITIVE_THIS_SLOT).isUndefined(),
00273  *              "we already set [[PrimitiveThis]] for this Boolean object");
00274  *   MOZ_ASSERT(getSlot(PRIMITIVE_THIS_SLOT).isUndefined(),
00275  *              "we already set [[PrimitiveThis]] for this String object");
00276  *
00277  * MOZ_ASSERT has no effect in non-debug builds.  It is designed to catch bugs
00278  * *only* during debugging, not "in the field".
00279  */
00280 #ifdef DEBUG
00281    /* First the single-argument form. */
00282 #  define MOZ_ASSERT_HELPER1(expr) \
00283      do { \
00284        if (MOZ_UNLIKELY(!(expr))) { \
00285          MOZ_ReportAssertionFailure(#expr, __FILE__, __LINE__); \
00286          MOZ_REALLY_CRASH(); \
00287        } \
00288      } while (0)
00289    /* Now the two-argument form. */
00290 #  define MOZ_ASSERT_HELPER2(expr, explain) \
00291      do { \
00292        if (MOZ_UNLIKELY(!(expr))) { \
00293          MOZ_ReportAssertionFailure(#expr " (" explain ")", __FILE__, __LINE__); \
00294          MOZ_REALLY_CRASH(); \
00295        } \
00296      } while (0)

00299     * Count the number of arguments passed to MOZ_ASSERT, very carefully
00300     * tiptoeing around an MSVC bug where it improperly expands __VA_ARGS__ as a

00302     *
00303     *   http://connect.microsoft.com/VisualStudio/feedback/details/380090/variadic-macro-replacement
00304     *   http://cplusplus.co.il/2010/07/17/variadic-macro-to-count-number-of-arguments/#comment-644
00305     */

00307      count
00308 #  define MOZ_COUNT_ASSERT_ARGS_IMPL(args) \
00309      MOZ_COUNT_ASSERT_ARGS_IMPL2 args
00310 #  define MOZ_COUNT_ASSERT_ARGS(...) \
00311      MOZ_COUNT_ASSERT_ARGS_IMPL((__VA_ARGS__, 2, 1, 0))

00313 #  define MOZ_ASSERT_CHOOSE_HELPER2(count) MOZ_ASSERT_HELPER##count
00314 #  define MOZ_ASSERT_CHOOSE_HELPER1(count) MOZ_ASSERT_CHOOSE_HELPER2(count)
00315 #  define MOZ_ASSERT_CHOOSE_HELPER(count) MOZ_ASSERT_CHOOSE_HELPER1(count)

00345  * MOZ_ASSUME_UNREACHABLE_MARKER() expands to an expression which states that it is
00346  * undefined behavior for execution to reach this point.  No guarantees are made
00347  * about what will happen if this is reached at runtime.  Most code should
00348  * probably use the higher level MOZ_ASSUME_UNREACHABLE, which uses this when
00349  * appropriate.

00353 #elif defined(__GNUC__)

00366 #    endif
00367 #  endif

00369 #  define MOZ_ASSUME_UNREACHABLE_MARKER() __assume(0)
00370 #else

00378 /*
00379  * MOZ_ASSUME_UNREACHABLE([reason]) tells the compiler that it can assume that
00380  * the macro call cannot be reached during execution.  This lets the compiler
00381  * generate better-optimized code under some circumstances, at the expense of
00382  * the program's behavior being undefined if control reaches the
00383  * MOZ_ASSUME_UNREACHABLE.
00384  *
00385  * In Gecko, you probably should not use this macro outside of performance- or
00386  * size-critical code, because it's unsafe.  If you don't care about code size
00387  * or performance, you should probably use MOZ_ASSERT or MOZ_CRASH.
00388  *
00389  * SpiderMonkey is a different beast, and there it's acceptable to use
00390  * MOZ_ASSUME_UNREACHABLE more widely.
00391  *
00392  * Note that MOZ_ASSUME_UNREACHABLE is noreturn, so it's valid not to return a
00393  * value following a MOZ_ASSUME_UNREACHABLE call.
00394  *
00395  * Example usage:
00396  *
00397  *   enum ValueType {
00398  *     VALUE_STRING,
00399  *     VALUE_INT,
00400  *     VALUE_FLOAT
00401  *   };
00402  *
00403  *   int ptrToInt(ValueType type, void* value) {
00404  *   {
00405  *     // We know for sure that type is either INT or FLOAT, and we want this
00406  *     // code to run as quickly as possible.
00407  *     switch (type) {
00408  *     case VALUE_INT:
00409  *       return *(int*) value;
00410  *     case VALUE_FLOAT:
00411  *       return (int) *(float*) value;
00412  *     default:
00413  *       MOZ_ASSUME_UNREACHABLE("can only handle VALUE_INT and VALUE_FLOAT");
00414  *     }
00415  *   }
00416  */
00417 #if defined(DEBUG)
00418 #  define MOZ_ASSUME_UNREACHABLE(...) \
00419      do { \
00420        MOZ_ASSERT(false, "MOZ_ASSUME_UNREACHABLE(" __VA_ARGS__ ")"); \
00421        MOZ_ASSUME_UNREACHABLE_MARKER(); \
00422      } while (0)
00423 #else
00424 #  define MOZ_ASSUME_UNREACHABLE(reason)  MOZ_ASSUME_UNREACHABLE_MARKER()
00425 #endif
