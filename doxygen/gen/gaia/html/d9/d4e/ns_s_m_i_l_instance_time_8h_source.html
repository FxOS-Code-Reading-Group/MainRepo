00016 //----------------------------------------------------------------------
00017 // nsSMILInstanceTime
00018 //
00019 // An instant in document simple time that may be used in creating a new
00020 // interval.
00021 //
00022 // For an overview of how this class is related to other SMIL time classes see
00023 // the documentation in nsSMILTimeValue.h
00024 //
00025 // These objects are owned by an nsSMILTimedElement but MAY also be referenced
00026 // by:
00027 //
00028 // a) nsSMILIntervals that belong to the same nsSMILTimedElement and which refer
00029 //    to the nsSMILInstanceTimes which form the interval endpoints; and/or
00030 // b) nsSMILIntervals that belong to other nsSMILTimedElements but which need to
00031 //    update dependent instance times when they change or are deleted.
00032 //    E.g. for begin='a.begin', 'a' needs to inform dependent
00033 //    nsSMILInstanceTimes if its begin time changes. This notification is
00034 //    performed by the nsSMILInterval.
00035 
00036 class nsSMILInstanceTime
00037 {
00038 public:
00039   // Instance time source. Times generated by events, syncbase relationships,
00040   // and DOM calls behave differently in some circumstances such as when a timed
00041   // element is reset.
00042   enum nsSMILInstanceTimeSource {
00043     // No particularly significant source, e.g. offset time, 'indefinite'
00044     SOURCE_NONE,
00045     // Generated by a DOM call such as beginElement
00046     SOURCE_DOM,
00047     // Generated by a syncbase relationship
00048     SOURCE_SYNCBASE,
00049     // Generated by an event
00050     SOURCE_EVENT
00051   };

00053   nsSMILInstanceTime(const nsSMILTimeValue& aTime,
00054                      nsSMILInstanceTimeSource aSource = SOURCE_NONE,

00060                              bool aBeginObjectChanged,
00061                              bool aEndObjectChanged);

00107   // Internal flags used to represent the behaviour of different instance times
00108   enum {
00109     // Indicates that this instance time was generated by an event or a DOM
00110     // call. Such instance times require special handling when (i) the owning
00111     // element is reset, (ii) when they are to be added as a new end instance
00112     // times (as per SMIL's event sensitivity contraints), and (iii) when
00113     // a backwards seek is performed and the timing model is reconstructed.
00114     kDynamic = 1,
00115 
00116     // Indicates that this instance time is referred to by an
00117     // nsSMILTimeValueSpec and as such may be updated. Such instance time should
00118     // not be filtered out by the nsSMILTimedElement even if they appear to be
00119     // in the past as they may be updated to a future time.
00120     kMayUpdate = 2,
00121 
00122     // Indicates that this instance time was generated from the DOM as opposed
00123     // to an nsSMILTimeValueSpec. When a 'begin' or 'end' attribute is set or
00124     // reset we should clear all the instance times that have been generated by
00125     // that attribute (and hence an nsSMILTimeValueSpec), but not those from the
00126     // DOM.
00127     kFromDOM = 4,
00128 
00129     // Indicates that this instance time was used as the endpoint of an interval
00130     // that has been filtered or removed. However, since it is a dynamic time it
00131     // should be preserved and not filtered.
00132     kWasDynamicEndpoint = 8
00133   };
00134   uint8_t       mFlags;   // Combination of kDynamic, kMayUpdate, etc.
00135   mutable bool  mVisited; // Cycle tracking
00136 
00137   // Additional reference count to determine if this instance time is currently
00138   // used as a fixed endpoint in any intervals. Instance times that are used in
00139   // this way should not be removed when the owning nsSMILTimedElement removes
00140   // instance times in response to a restart or in an attempt to free up memory
00141   // by filtering out old instance times.
00142   //
00143   // Instance times are only shared in a few cases, namely:
00144   // a) early ends,
00145   // b) zero-duration intervals,
00146   // c) momentarily whilst establishing new intervals and updating the current
00147   //    interval, and
00148   // d) trimmed intervals
00149   // Hence the limited range of a uint16_t should be more than adequate.
00150   uint16_t      mFixedEndpointRefCnt;
00151 
00152   uint32_t      mSerial; // A serial number used by the containing class to

00155

00158                                  // instance times.)
00159   nsSMILInterval* mBaseInterval; // Interval from which this time is derived
00160                                  // (only used for syncbase instance times)
00161 };
