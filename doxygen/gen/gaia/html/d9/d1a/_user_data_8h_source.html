00027   /* Attaches untyped userData associated with key. destroy is called on destruction */
00028   void Add(UserDataKey *key, void *userData, destroyFunc destroy)
00029   {
00030     for (int i=0; i<count; i++) {
00031       if (key == entries[i].key) {
00032         if (entries[i].destroy) {
00033           entries[i].destroy(entries[i].userData);
00034         }
00035         entries[i].userData = userData;
00036         entries[i].destroy = destroy;
00037         return;
00038       }
00039     }
00040 
00041     // We could keep entries in a std::vector instead of managing it by hand
00042     // but that would propagate an stl dependency out which we'd rather not
00043     // do (see bug 666609). Plus, the entries array is expect to stay small
00044     // so doing a realloc everytime we add a new entry shouldn't be too costly
00045     entries = static_cast<Entry*>(realloc(entries, sizeof(Entry)*(count+1)));
00046 
00047     if (!entries) {
00048       MOZ_CRASH();
00049     }

00052     entries[count].userData = userData;
00053     entries[count].destroy  = destroy;
00054 
00055     count++;
00056   }

00058   /* Remove and return user data associated with key, without destroying it */
00059   void* Remove(UserDataKey *key)
00060   {
00061     for (int i=0; i<count; i++) {
00062       if (key == entries[i].key) {
00063         void *userData = entries[i].userData;
00064         // decrement before looping so entries[i+1] doesn't read past the end:
00065         --count;
00066         for (;i<count; i++) {
00067           entries[i] = entries[i+1];
00068         }
00069         return userData;
00070       }
00071     }
00072     return nullptr;
00073   }
