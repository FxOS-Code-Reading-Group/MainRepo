00139 // A SkBaseMutex is a POD structure that can be directly initialized
00140 // at declaration time with SK_DECLARE_STATIC/GLOBAL_MUTEX. This avoids the
00141 // generation of a static initializer in the final machine code (and
00142 // a corresponding static finalizer).
00143 //
00144 struct SkBaseMutex {
00145     void    acquire() { pthread_mutex_lock(&fMutex); }
00146     void    release() { pthread_mutex_unlock(&fMutex); }
00147     pthread_mutex_t  fMutex;
00148 };
