00213 /*
00214  * Load a shared library called "newShLibName" in the same directory as
00215  * a shared library that is already loaded, called existingShLibName.
00216  * A pointer to a static function in that shared library,
00217  * staticShLibFunc, is required.
00218  *
00219  * existingShLibName:
00220  *   The file name of the shared library that shall be used as the 
00221  *   "reference library". The loader will attempt to load the requested
00222  *   library from the same directory as the reference library.
00223  *
00224  * staticShLibFunc:
00225  *   Pointer to a static function in the "reference library".
00226  *
00227  * newShLibName:
00228  *   The simple file name of the new shared library to be loaded.
00229  *
00230  * We use PR_GetLibraryFilePathname to get the pathname of the loaded 
00231  * shared lib that contains this function, and then do a
00232  * PR_LoadLibraryWithFlags with an absolute pathname for the shared
00233  * library to be loaded.
00234  *
00235  * On Windows, the "alternate search path" strategy is employed, if available.
00236  * On Unix, if existingShLibName is a symbolic link, and no link exists for the
00237  * new library, the original link will be resolved, and the new library loaded
00238  * from the resolved location.
00239  *
00240  * If the new shared library is not found in the same location as the reference
00241  * library, it will then be loaded from the normal system library path.
00242  */
00243 PRLibrary *
00244 PORT_LoadLibraryFromOrigin(const char* existingShLibName,
00245                  PRFuncPtr staticShLibFunc,
00246                  const char *newShLibName);
