00006 /*
00007 ** File:        pralarm.h
00008 ** Description: API to periodic alarms.
00009 **
00010 **
00011 ** Alarms are defined to invoke some client specified function at 
00012 ** a time in the future. The notification may be a one time event
00013 ** or repeated at a fixed interval. The interval at which the next
00014 ** notification takes place may be modified by the client code only
00015 ** during the respective notification.
00016 **
00017 ** The notification is delivered on a thread that is part of the
00018 ** alarm context (PRAlarm). The thread will inherit the priority
00019 ** of the Alarm creator.
00020 **
00021 ** Any number of periodic alarms (PRAlarmID) may be created within
00022 ** the context of a single alarm (PRAlarm). The notifications will be
00023 ** scheduled as close to the desired time as possible.
00024 **
00025 ** Repeating periodic notifies are expected to run at a fixed rate.
00026 ** That rate is expressed as some number of notifies per period where
00027 ** the period is much larger than a PRIntervalTime (see prinrval.h).
00028 */

00093 /***********************************************************************
00094 ** FUNCTION:    PR_SetAlarm
00095 ** DESCRIPTION:
00096 **  Creates a periodic notifier that is to be delivered to a specified
00097 **  function at some fixed interval.
00098 ** INPUTS:      PRAlarm *alarm              Parent alarm context
00099 **              PRIntervalTime period       Interval over which the notifies
00100 **                                          are delivered.
00101 **              PRUint32 rate               The rate within the interval that
00102 **                                          the notifies will be delivered.
00103 **              PRPeriodicAlarmFn function  Entry point where the notifies
00104 **                                          will be delivered.
00105 ** OUTPUTS:     None
00106 ** RETURN:      PRAlarmID*                  Handle to the notifier just created
00107 **                                          or NULL if the request failed.
00108 **  
00109 ** SIDE EFFECTS:
00110 **  A periodic notifier is created. The notifications will be delivered
00111 **  by the alarm's internal thread at a fixed interval whose rate is the
00112 **  number of interrupts per interval specified. The first notification
00113 **  will be delivered as soon as possible, and they will continue until
00114 **  the notifier routine indicates that they should cease of the alarm
00115 **  context is destroyed (PR_DestroyAlarm).
00116 ** RESTRICTIONS:
00117 **  None. 
00118 ** MEMORY:      Memory for the notifier object.
00119 ** ALGORITHM:   The rate at which notifications are delivered are stated
00120 **              to be "'rate' notifies per 'interval'". The exact time of
00121 **              the notification is computed based on a epoch established
00122 **              when the notifier was set. Each notification is delivered
00123 **              not ealier than the epoch plus the fixed rate times the
00124 **              notification sequence number. Such notifications have the
00125 **              potential to be late by not more than 'interval'/'rate'.
00126 **              The amount of lateness of one notification is taken into
00127 **              account on the next in an attempt to avoid long term slew.  
00128 ***********************************************************************/
00129 NSPR_API(PRAlarmID*) PR_SetAlarm(
00130     PRAlarm *alarm, PRIntervalTime period, PRUint32 rate,
00131     PRPeriodicAlarmFn function, void *clientData);

00150 ** RESTRICTIONS:

00154 ***********************************************************************/
00155 NSPR_API(PRStatus) PR_ResetAlarm(
