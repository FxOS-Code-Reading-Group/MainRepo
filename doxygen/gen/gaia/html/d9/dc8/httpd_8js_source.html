04288     // occur or when copying is drawing to a finish.

04302     //     act as though the stream closed except that we notify of completion

04745   normalizeFieldName: function(fieldName)
04746   {
04747     if (fieldName == "")
04748       throw Cr.NS_ERROR_INVALID_ARG;
04749

04751     {
04752       if (!IS_TOKEN_ARRAY[fieldName.charCodeAt(i)])
04753       {

04756       }
04757     }

04775   normalizeFieldValue: function(fieldValue)
04776   {
04777     // field-value    = *( field-content | LWS )
04778     // field-content  = <the OCTETs making up the field-value
04779     //                  and consisting of either *TEXT or combinations
04780     //                  of token, separators, and quoted-string>
04781     // TEXT           = <any OCTET except CTLs,
04782     //                  but including LWS>
04783     // LWS            = [CRLF] 1*( SP | HT )
04784     //
04785     // quoted-string  = ( <"> *(qdtext | quoted-pair ) <"> )
04786     // qdtext         = <any TEXT except <">>
04787     // quoted-pair    = "\" CHAR
04788     // CHAR           = <any US-ASCII character (octets 0 - 127)>
04789 
04790     // Any LWS that occurs between field-content MAY be replaced with a single
04791     // SP before interpreting the field value or forwarding the message
04792     // downstream (section 4.2); we replace 1*LWS with a single SP
04793     var val = fieldValue.replace(/(?:(?:\r\n)?[ \t]+)+/g, " ");
04794 
04795     // remove leading/trailing LWS (which has been converted to SP)
04796     val = val.replace(/^ +/, "").replace(/ +$/, "");
04797 
04798     // that should have taken care of all CTLs, so val should contain no CTLs
04799     for (var i = 0, len = val.length; i < len; i++)
04800       if (isCTL(val.charCodeAt(i)))
04801         throw Cr.NS_ERROR_INVALID_ARG;
04802 
04803     // XXX disallows quoted-pair where CHAR is a CTL -- will not invalidly
04804     //     normalize, however, so this can be construed as a tightening of the
04805     //     spec and not entirely as a bug
04806     return val;
04807   }
