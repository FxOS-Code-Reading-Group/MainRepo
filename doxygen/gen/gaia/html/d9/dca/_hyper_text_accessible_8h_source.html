00299   int32_t AdjustCaretOffset(int32_t aOffset)
00300   {
00301     // It is the same character offset when the caret is visually at the very
00302     // end of a line or the start of a new line (soft line break). Getting text
00303     // at the line should provide the line with the visual caret, otherwise
00304     // screen readers will announce the wrong line as the user presses up or
00305     // down arrow and land at the end of a line.
00306     if (aOffset > 0) {
00307       nsRefPtr<nsFrameSelection> frameSelection = FrameSelection();
00308       if (frameSelection &&
00309           frameSelection->GetHint() == nsFrameSelection::HINTLEFT) {
00310         return aOffset - 1;
00311       }
00312     }
00313     return aOffset;
00314   }
