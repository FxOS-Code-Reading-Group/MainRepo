00397     ResourceAcquisitionArray* CheckAcquisition(const T* aLast,
00398                                                const T* aProposed,
00399                                                const CallStack& aCallContext)
00400     {
00401         NS_ASSERTION(aProposed, "null resource");
00402         PRAutoLock _(mLock);
00403 
00404         PLHashEntry* second = *GetEntry(aProposed);
00405         OrderingEntry* e = static_cast<OrderingEntry*>(second->value);
00406         if (CallStack::kNone == e->mFirstSeen)
00407             e->mFirstSeen = aCallContext;
00408 
00409         if (!aLast)
00410             // don't check if |0 < proposed|; just vamoose
00411             return 0;
00412 
00413         PLHashEntry* first = *GetEntry(aLast);
00414 
00415         // this is the crux of the deadlock detector algorithm
00416 
00417         if (first == second) {
00418             // reflexive deadlock.  fastpath b/c InTransitiveClosure is
00419             // not applicable here.
00420             ResourceAcquisitionArray* cycle = new ResourceAcquisitionArray();
00421             if (!cycle)
00422                 NS_RUNTIMEABORT("can't allocate dep. cycle array");
00423             cycle->AppendElement(MakeResourceAcquisition(first));
00424             cycle->AppendElement(ResourceAcquisition(aProposed,
00425                                                      aCallContext));
00426             return cycle;
00427         }
00428         if (InTransitiveClosure(first, second)) {
00429             // we've already established |last < proposed|.  all is well.
00430             return 0;
00431         }
00432         if (InTransitiveClosure(second, first)) {
00433             // the order |proposed < last| has been deduced, perhaps
00434             // transitively.  we're attempting to violate that
00435             // constraint by acquiring resources in the order
00436             // |last < proposed|, and thus we may deadlock under the
00437             // right conditions.
00438             ResourceAcquisitionArray* cycle = GetDeductionChain(second, first);
00439             // show how acquiring |proposed| would complete the cycle
00440             cycle->AppendElement(ResourceAcquisition(aProposed,
00441                                                      aCallContext));
00442             return cycle;
00443         }
00444         // |last|, |proposed| are unordered according to our
00445         // poset.  this is fine, but we now need to add this
00446         // ordering constraint.
00447         AddOrder(first, second);
00448         return 0;
00449     }

00457     bool InTransitiveClosure(const PLHashEntry* aStart,
00458                              const PLHashEntry* aTarget) const
00459     {
00460         if (IsOrdered(aStart, aTarget))

00502     // precondition: |aStart != aTarget|
00503     // invariant: |aStart| is the last element in |aChain|
00504     bool GetDeductionChain_Helper(const PLHashEntry* aStart,
00505                                   const PLHashEntry* aTarget,
00506                                   ResourceAcquisitionArray* aChain)
00507     {
00508         if (IsOrdered(aStart, aTarget)) {
00509             aChain->AppendElement(MakeResourceAcquisition(aTarget));
00510             return true;
00511         }
00512 
00513         index_type i = 0;
00514         size_type len = NumOrders(aStart);
00515         for (const PLHashEntry* const* it = GetOrders(aStart);
00516              i < len; ++i, ++it) {
00517             aChain->AppendElement(MakeResourceAcquisition(*it));
00518             if (GetDeductionChain_Helper(*it, aTarget, aChain))
00519                 return true;
00520             aChain->RemoveElementAt(aChain->Length() - 1);
00521         }
00522         return false;
00523     }
