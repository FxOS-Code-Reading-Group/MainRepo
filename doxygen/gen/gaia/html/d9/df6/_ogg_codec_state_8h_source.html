00051 // A queue of ogg_packets. When we read a page, we extract the page's packets
00052 // and buffer them in the owning stream's OggCodecState. This is because
00053 // if we're skipping up to the next keyframe in very large frame sized videos,
00054 // there may be several megabytes of data between keyframes, and the
00055 // ogg_stream_state would end up resizing its buffer every time we added a
00056 // new 4KB page to the bitstream, which kills performance on Windows. This
00057 // also gives us the option to timestamp packets rather than decoded
00058 // frames/samples, reducing the amount of frames/samples we must decode to
00059 // determine start-time at a particular offset, and gives us finer control
00060 // over memory usage.
00061 class OggPacketQueue : private nsDeque {
00062 public:
00063   OggPacketQueue() : nsDeque(new OggPacketDeallocator()) {}
00064   ~OggPacketQueue() { Erase(); }
00065   bool IsEmpty() { return nsDeque::GetSize() == 0; }
00066   void Append(ogg_packet* aPacket);
00067   ogg_packet* PopFront() { return static_cast<ogg_packet*>(nsDeque::PopFront()); }
00068   ogg_packet* PeekFront() { return static_cast<ogg_packet*>(nsDeque::PeekFront()); }
00069   void PushFront(ogg_packet* aPacket) { nsDeque::PushFront(aPacket); }
00070   void PushBack(ogg_packet* aPacket) { nsDeque::PushFront(aPacket); }
00071   void Erase() { nsDeque::Erase(); }
00072 };
