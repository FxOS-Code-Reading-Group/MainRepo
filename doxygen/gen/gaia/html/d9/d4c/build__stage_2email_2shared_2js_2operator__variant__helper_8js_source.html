00036 function OperatorVariantHelper(listener, persistKey, checkNow) {
00037   var errMsg = null;
00038 
00039   // The IccHelper should be available. If it's not, we will report a warning
00040   // only. Under b2g desktop and fxos simulator it can be totally normal for the
00041   // IccHelper to be absent.
00042   if (IccHelper === undefined) {
00043     // Warn the developer.
00044     errMsg = 'Expected IccHelper to be present. This can be normal when using' +
00045              'B2G Desktop or FxOS Simulator.';
00046     console.warn(errMsg);
00047     // Bail out!
00048     return;
00049   }
00050 
00051   // ... and it should be enabled.
00052   if (!IccHelper.enabled) {
00053     // Indicate this may be an expected exception.
00054     errMsg = 'Expected IccHelper to be enabled. This error is expected and ' +
00055              'normal if you are running B2G Desktop or FxOS Simulator.';
00056     console.error(errMsg);
00057     // Exceptional errors require exceptions. :P
00058     throw new Error(errMsg);
00059   }
00060 
00061   // And mozSettings must be present.
00062   var settings = window.navigator.mozSettings;
00063   if (!settings) {
00064     errMsg = 'Expected mozSettings to be present.';
00065     console.error(errMsg);
00066     throw new Error(errMsg);
00067   }
00068 
00069   if (listener === undefined || typeof listener !== 'function') {
00070     throw new Error('listener argument must be a function!');
00071   }
00072 
00073   this._listener = listener;
00074 
00075   if (persistKey === undefined || typeof persistKey !== 'string') {
00076     throw new Error('persistKey argument must be a string!');
00077   }
00078 
00079   this._persistKey = persistKey;
00080 
00081   if (!!checkNow) {
00082     this.customize();
00083   }
00084 }
