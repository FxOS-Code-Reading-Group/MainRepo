00123 /*
00124  * Often you have to cast an implementation pointer, e.g., |this|, to an
00125  * |nsISupports*|, but because you have multiple inheritance, a simple cast
00126  * is ambiguous.  One could simply say, e.g., (given a base |nsIBase|),
00127  * |static_cast<nsIBase*>(this)|; but that disguises the fact that what
00128  * you are really doing is disambiguating the |nsISupports|.  You could make
00129  * that more obvious with a double cast, e.g., |static_cast<nsISupports*>
00130                                                            (* static_cast<nsIBase*>(this))|, but that is bulky and harder to read...
00131  *
00132  * The following macro is clean, short, and obvious.  In the example above,
00133  * you would use it like this: |NS_ISUPPORTS_CAST(nsIBase*, this)|.
00134  */

00136 #define NS_ISUPPORTS_CAST(__unambiguousBase, __expr) \
00137   static_cast<nsISupports*>(static_cast<__unambiguousBase>(__expr))
