00257     enum PluginQuirks {
00258         QUIRKS_NOT_INITIALIZED                          = 0,
00259         // Silverlight assumes it is transparent in windowless mode. This quirk
00260         // matches the logic in nsNPAPIPluginInstance::SetWindowless.
00261         QUIRK_SILVERLIGHT_DEFAULT_TRANSPARENT           = 1 << 0,
00262         // Win32: Hook TrackPopupMenu api so that we can swap out parent
00263         // hwnds. The api will fail with parents not associated with our
00264         // child ui thread. See WinlessHandleEvent for details.
00265         QUIRK_WINLESS_TRACKPOPUP_HOOK                   = 1 << 1,
00266         // Win32: Throttle flash WM_USER+1 heart beat messages to prevent
00267         // flooding chromium's dispatch loop, which can cause ipc traffic
00268         // processing lag.
00269         QUIRK_FLASH_THROTTLE_WMUSER_EVENTS              = 1 << 2,
00270         // Win32: Catch resets on our subclass by hooking SetWindowLong.
00271         QUIRK_FLASH_HOOK_SETLONGPTR                     = 1 << 3,
00272         // X11: Work around a bug in Flash up to 10.1 d51 at least, where
00273         // expose event top left coordinates within the plugin-rect and
00274         // not at the drawable origin are misinterpreted.
00275         QUIRK_FLASH_EXPOSE_COORD_TRANSLATION            = 1 << 4,
00276         // Win32: Catch get window info calls on the browser and tweak the
00277         // results so mouse input works when flash is displaying it's settings
00278         // window.
00279         QUIRK_FLASH_HOOK_GETWINDOWINFO                  = 1 << 5,
00280         // Win: Addresses a flash bug with mouse capture and full screen
00281         // windows.
00282         QUIRK_FLASH_FIXUP_MOUSE_CAPTURE                 = 1 << 6,
00283         // Win: QuickTime steals focus on SetWindow calls even if it's hidden.
00284         // Avoid calling SetWindow in that case.
00285         QUIRK_QUICKTIME_AVOID_SETWINDOW                 = 1 << 7,
00286         // Win: Check to make sure the parent window has focus before calling
00287         // set focus on the child. Addresses a full screen dialog prompt
00288         // problem in Silverlight.
00289         QUIRK_SILVERLIGHT_FOCUS_CHECK_PARENT            = 1 << 8,
00290         // Mac: Allow the plugin to use offline renderer mode.
00291         // Use this only if the plugin is certified the support the offline renderer.
00292         QUIRK_ALLOW_OFFLINE_RENDERER                    = 1 << 9,
00293         // Mac: Work around a Flash bug that can cause plugin process crashes
00294         // in CoreGraphics mode:  The Flash plugin sometimes accesses the
00295         // CGContextRef we pass to it in NPP_HandleEvent(NPCocoaEventDrawRect)
00296         // outside of that call.  See bug 804606.
00297         QUIRK_FLASH_AVOID_CGMODE_CRASHES                = 1 << 10,

00342     // If a plugin spins a nested glib event loop in response to a
00343     // synchronous IPC message from the browser, the loop might break
00344     // only after the browser responds to a request sent by the
00345     // plugin.  This can happen if a plugin uses gtk's synchronous
00346     // copy/paste, for example.  But because the browser is blocked on
00347     // a condvar, it can't respond to the request.  This situation
00348     // isn't technically a deadlock, but the symptoms are basically
00349     // the same from the user's perspective.
00350     //
00351     // We take two steps to prevent this
00352     //
00353     //  (1) Detect nested event loops spun by the plugin.  This is
00354     //      done by scheduling a glib timer event in the plugin
00355     //      process whenever the browser might block on the plugin.
00356     //      If the plugin indeed spins a nested loop, this timer event
00357     //      will fire "soon" thereafter.
00358     //
00359     //  (2) When a nested loop is detected, deschedule the
00360     //      nested-loop-detection timer and in its place, schedule
00361     //      another timer that periodically calls back into the
00362     //      browser and spins a mini event loop.  This mini event loop
00363     //      processes a handful of pending native events.
00364     //
00365     // Because only timer (1) or (2) (or neither) may be active at any
00366     // point in time, we use the same member variable
00367     // |mNestedLoopTimerId| to refer to both.
00368     //
00369     // When the browser no longer might be blocked on a plugin's IPC
00370     // response, we deschedule whichever of (1) or (2) is active.
00371     guint mNestedLoopTimerId;
00372 #  ifdef DEBUG
00373     // Depth of the stack of calls to g_main_context_dispatch before any
00374     // nested loops are run.  This is 1 when IPC calls are dispatched from
00375     // g_main_context_iteration, or 0 when dispatched directly from
00376     // MessagePumpForUI.
00377     int mTopLoopDepth;
00378 #  endif
00379 #elif defined (MOZ_WIDGET_QT)
00380     NestedLoopTimer *mNestedLoopTimerObject;
00381 #endif
