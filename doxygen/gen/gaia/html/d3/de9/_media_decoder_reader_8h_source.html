00360   // Extracts elements from the queue into aResult, in order.
00361   // Elements whose start time is before aTime are ignored.
00362   void GetElementsAfter(int64_t aTime, nsTArray<T*>* aResult) {
00363     ReentrantMonitorAutoEnter mon(mReentrantMonitor);
00364     if (!GetSize())
00365       return;
00366     int32_t i;
00367     for (i = GetSize() - 1; i > 0; --i) {
00368       T* v = static_cast<T*>(ObjectAt(i));
00369       if (v->GetEnd() < aTime)
00370         break;
00371     }
00372     // Elements less than i have a end time before aTime. It's also possible
00373     // that the element at i has a end time before aTime, but that's OK.
00374     for (; i < GetSize(); ++i) {
00375       aResult->AppendElement(static_cast<T*>(ObjectAt(i)));
00376     }
00377   }

00385     }
