00286   /*
00287    * The keyboard calls this method to tell us about user input.
00288    *
00289    * What we do with the input depends on various things:
00290    *
00291    * - whether we are suggesting, correcting, punctuating and/or capitalizing:
00292    *   these are controlled by settings, input mode and input type
00293    *
00294    * - whether there is a selected region in the input field
00295    *
00296    * - whether there is an auto-correction ready (when input is space
00297    *   or punctuation).
00298    *
00299    * - whether the last action was an autocorrection (when input is backspace)
00300    *
00301    * - the cursor position (affects suggestions, capitalization, etc.)
00302    *
00303    * If there is a selection just handle simple insertions and deletions
00304    * with no extra behavior. (I think)
00305    *
00306    * If input is a space or punctuation:
00307    *
00308    *  If there is a autocorrection ready, and we are correcting and
00309    *  the cursor is at the end of a word, make the correction
00310    *
00311    *  If the previous character is a space, fix the punctuation. Note that
00312    *  this only works for a subset of the punctuation characters.
00313    *
00314    *  Otherwise just insert it.
00315    *
00316    *  If we're correcting and corrections are temporarily diabled, turn them
00317    *  back on.
00318    *
00319    * If input is a backspace:
00320    *
00321    *  If we just did an auto-correction, revert it and turn off corrections
00322    *  until the next space or punctuation character.
00323    *
00324    *  If we just inserted a suggested word that the user selected, revert
00325    *  the insertion, but don't disable autocorrect.
00326    *
00327    *  Should we undo punctuation corrections this way, too?
00328    *
00329    *  Otherwise, just delete the character before the cursor
00330    *
00331    * For any other input character, just insert it.
00332    *
00333    * Reset the backspace reversion state
00334    *
00335    * Update the capitalization state, if we're capitalizing
00336    */

00356       case COMMA:

00392   // Handle any key (including backspace) and do the right thing even if
00393   // there is a selection in the text field. This method does not perform
00394   // auto-correction or auto-punctuation.
00395   function handleKey(keycode) {
00396     // First, update our internal state
00397     if (keycode === BACKSPACE) {
00398       if (selection) {
00399         // backspace while a region is selected erases the selection
00400         // and leaves the cursor at the selection start
00401         inputText = inputText.substring(0, cursor) +
00402           inputText.substring(selection);
00403         selection = 0;
00404       } else if (cursor > 0) {
00405         cursor--;
00406         inputText = inputText.substring(0, cursor) +
00407           inputText.substring(cursor + 1);
00408 
00409         // If we have temporarily disabled auto correction for the current word
00410         // and we've just backspaced over the entire word, then we can
00411         // re-enabled it again
00412         if (correctionDisabled && !wordBeforeCursor())
00413           correctionDisabled = false;
00414       }
00415     } else {
00416       if (selection) {
00417         inputText =
00418           inputText.substring(0, cursor) +
00419           String.fromCharCode(keycode) +
00420           inputText.substring(selection);
00421         selection = 0;
00422       } else {
00423         inputText =
00424           inputText.substring(0, cursor) +
00425           String.fromCharCode(keycode) +
00426           inputText.substring(cursor);
00427       }
00428       cursor++;
00429     }
00430 
00431     // Generate the key event
00432     keyboard.sendKey(keycode);
00433   }

00435   // Assuming that the word before the cursor is oldWord, send a
00436   // minimal number of key events to change it to newWord in the text
00437   // field. Also update our internal state to match the new textfield
00438   // content and cursor position.
00439   function replaceBeforeCursor(oldWord, newWord) {
00440     var oldWordLen = oldWord.length;
00441     if (keyboard.replaceSurroundingText) {
00442       keyboard.replaceSurroundingText(newWord, oldWordLen, 0);
00443     }
00444     else {
00445       // Find the first character in currentWord and newWord that differs
00446       // so we know how many backspaces we need to send.
00447       for (var firstdiff = 0; firstdiff < oldWordLen; firstdiff++) {
00448         if (oldWord[firstdiff] !== newWord[firstdiff])
00449           break;
00450       }
00451 
00452       // Backspace as far as that first difference
00453       for (var i = oldWordLen; i > firstdiff; i--)
00454         keyboard.sendKey(BACKSPACE);
00455 
00456       // And send the first different character and all that follow
00457       keyboard.sendString(newWord.substring(firstdiff));
00458     }
00459 
00460     // Now update internal state
00461     inputText =
00462       inputText.substring(0, cursor - oldWordLen) +
00463       newWord +
00464       inputText.substring(cursor);
00465     cursor += newWord.length - oldWordLen;
00466   }

00513   // Perform an autocorrection. Assumes that all pre-conditions for
00514   // auto-correction have been met.
00515   function autoCorrect(keycode) {
00516     // Get the word before the cursor
00517     var currentWord = wordBeforeCursor();
00518 
00519     // The space or punctuation that triggered the autocorrect
00520     var delimiter = String.fromCharCode(keycode);
00521 
00522     // Figure out the auto correction text
00523     var newWord = autoCorrection + delimiter;
00524 
00525     // Make the correction
00526     replaceBeforeCursor(currentWord, newWord);
00527 
00528     // Remember the change we just made so we can revert it if the
00529     // user types backspace
00530     revertTo = currentWord + delimiter;
00531     revertFrom = newWord;
00532     justAutoCorrected = true;

00535   // Auto punctuate, converting space punctuation to punctuation space
00536   // or converting space space to period space if the two spaces were
00537   // close enough together. Assumes that pre-conditions for auto punctuation
00538   // have been met.
00539   function autoPunctuate(keycode) {
00540     switch (keycode) {
00541     case SPACE:
00542       if (Date.now() - lastSpaceTimestamp < DOUBLE_SPACE_TIME)
00543         fixPunctuation(PERIOD, SPACE);
00544       else
00545         handleKey(keycode);
00546       break;
00547 
00548     case PERIOD:
00549     case QUESTION:
00550     case EXCLAMATION:
00551     case COMMA:
00552       fixPunctuation(keycode);
00553       break;
00554 
00555     default:
00556       // colon and semicolon don't auto-punctuate because they're
00557       // used after spaces for smileys.
00558       handleKey(keycode);
00559       break;
00560     }

00584   // When the worker thread sends us a batch of suggestions, deal
00585   // with them here.
00586   function handleSuggestions(input, suggestions) {
00587     // If we didn't get any suggestions just send the empty array to
00588     // clear any suggestions that are currently displayed. Do the same
00589     // if the word before the cursor has changed since we requested
00590     // these suggestions. That is, if the user has typed faster than we could
00591     // offer suggestions, ignore them.
00592     if (suggestions.length === 0 || wordBeforeCursor() !== input) {
00593       keyboard.sendCandidates([]); // Clear any displayed suggestions
00594       return;
00595     }
00596 
00597     // Loop through the suggestions discarding the weights
00598     var lcinput = input.toLowerCase();
00599     for (var i = 0; i < suggestions.length; i++) {
00600       suggestions[i] = suggestions[i][0];
00601     }
00602 
00603     // Now figure out if the input is one of the suggestions
00604     var inputindex = suggestions.indexOf(input);
00605 
00606     // If the input is just one character long then we never want to
00607     // auto-correct to more than one character (because it makes it hard
00608     // to type abbreviations and other single-characters uses).  So if the
00609     // first suggestion is not also one-character, then make the input
00610     // into the first suggestion.
00611     if (input.length === 1 && suggestions[0].length !== 1) {
00612       if (inputindex === -1)                // If the input is not a suggestion
00613         suggestions.pop();                  // Remove the last suggestion
00614       else                                  // Otherwise
00615         suggestions.splice(inputindex, 1);  // Remove the input
00616       suggestions.unshift(input);           // Then add it at the start
00617       inputindex = 0;                       // Update inputindex
00618     }
00619 
00620     switch (inputindex) {
00621     case -1:
00622       // Input is not a word: show it second in the list
00623       if (suggestions.length > 1)
00624         suggestions = [suggestions[0], input, suggestions[1]];
00625       else
00626         suggestions = [suggestions[0], input];
00627       break;
00628     case 0:
00629     case 1:
00630       // Input is the same as the first suggestion or is already in
00631       // the second position: use the suggestions unmodified
00632       break;
00633     case 2:
00634       // Input is the 3rd suggestion: swap to make it second
00635       suggestions = [suggestions[0], input, suggestions[1]];
00636       break;
00637     }
00638 
00639     // If we're going to use the first suggestion as an auto-correction

00704   function nearbyKeys(layout) {
00705     var nearbyKeys = {};
00706     var keys = layout.keyArray;
00707     var keysize = Math.min(layout.keyWidth, layout.keyHeight) * 1.2;
00708     var threshold = keysize * keysize;
00709 
00710     // Make sure that all the keycodes are lowercase, not uppercase
00711     for (var n = 0; n < keys.length; ++n) {
00712       keys[n].code =
00713         String.fromCharCode(keys[n].code).toLowerCase().charCodeAt(0);
00714     }
00715 
00716     // For each key, calculate the keys nearby.
00717     for (var n = 0; n < keys.length; ++n) {
00718       var key1 = keys[n];
00719       if (SpecialKey(key1))
00720         continue;
00721       var nearby = {};
00722       for (var m = 0; m < keys.length; ++m) {
00723         if (m === n)
00724           continue; // don't compare a key to itself
00725         var key2 = keys[m];
00726         if (SpecialKey(key2))
00727           continue;
00728         var d = distance(key1, key2);
00729         if (d !== 0)
00730           nearby[key2.code] = d;
00731       }
00732       nearbyKeys[key1.code] = nearby;
00733     }
00734

00737     // Compute the inverse square distance between the center point of
00738     // two keys, using the radius of the key (where radius is defined
00739     // as the distance from the center of key1 to a corner of key1)
00740     // as the unit of measure. If the distance is greater than 2.5
00741     // times the radius return 0 instead.
00742     function distance(key1, key2) {
00743       var cx1 = key1.x + key1.width / 2;
00744       var cy1 = key1.y + key1.height / 2;
00745       var cx2 = key2.x + key2.width / 2;
00746       var cy2 = key2.y + key2.height / 2;
00747       var radius = Math.sqrt(key1.width * key1.width / 4 +
00748                              key1.height * key1.height / 4);
00749 
00750       var dx = (cx1 - cx2) / radius;
00751       var dy = (cy1 - cy2) / radius;
00752       var distanceSquared = dx * dx + dy * dy;
00753 
00754       if (distanceSquared < 1) {
00755         console.warn('Keys too close',
00756                      JSON.stringify(key1), JSON.stringify(key2));
00757         return 0;
00758       }
00759 
00760       if (distanceSquared > 2.5 * 2.5)
00761         return 0;
00762       else
00763         return 1 / distanceSquared;
00764     }
