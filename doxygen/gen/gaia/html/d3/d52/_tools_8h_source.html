00051 static inline void
00052 NudgeToInteger(float *aVal)
00053 {
00054   float r = floorf(*aVal + 0.5f);
00055   // The error threshold should be proportional to the rounded value. This
00056   // bounds the relative error introduced by the nudge operation. However,
00057   // when the rounded value is 0, the error threshold can't be proportional
00058   // to the rounded value (we'd never round), so we just choose the same
00059   // threshold as for a rounded value of 1.
00060   if (FuzzyEqual(r, *aVal, r == 0.0f ? 1e-6f : fabs(r*1e-6f))) {
00061     *aVal = r;
00062   }
00063 }
