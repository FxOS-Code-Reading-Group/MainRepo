02718     function parseStatement() {
02719         var token = lookahead(),
02720             expr,
02721             labeledBody;
02722 
02723         if (token.type === Token.EOF) {
02724             throwUnexpected(token);
02725         }
02726 
02727         if (token.type === Token.Punctuator) {
02728             switch (token.value) {
02729             case ';':
02730                 return parseEmptyStatement();
02731             case '{':
02732                 return parseBlock();
02733             case '(':
02734                 return parseExpressionStatement();
02735             default:
02736                 break;
02737             }
02738         }
02739 
02740         if (token.type === Token.Keyword) {
02741             switch (token.value) {
02742             case 'break':
02743                 return parseBreakStatement();
02744             case 'continue':
02745                 return parseContinueStatement();
02746             case 'debugger':
02747                 return parseDebuggerStatement();
02748             case 'do':
02749                 return parseDoWhileStatement();
02750             case 'for':
02751                 return parseForStatement();
02752             case 'function':
02753                 return parseFunctionDeclaration();
02754             case 'if':
02755                 return parseIfStatement();
02756             case 'return':
02757                 return parseReturnStatement();
02758             case 'switch':
02759                 return parseSwitchStatement();
02760             case 'throw':
02761                 return parseThrowStatement();
02762             case 'try':
02763                 return parseTryStatement();
02764             case 'var':
02765                 return parseVariableStatement();
02766             case 'while':
02767                 return parseWhileStatement();
02768             case 'with':
02769                 return parseWithStatement();
02770             default:
02771                 break;
02772             }
02773         }
02774 
02775         expr = parseExpression();
02776 
02777         // 12.12 Labelled Statements
02778         if ((expr.type === Syntax.Identifier) && match(':')) {
02779             lex();
02780 
02781             if (Object.prototype.hasOwnProperty.call(state.labelSet, expr.name)) {
02782                 throwError({}, Messages.Redeclaration, 'Label', expr.name);
02783             }
02784 
02785             state.labelSet[expr.name] = true;
02786             labeledBody = parseStatement();
02787             delete state.labelSet[expr.name];
02788 
02789             return {
02790                 type: Syntax.LabeledStatement,
02791                 label: expr,
02792                 body: labeledBody
02793             };
02794         }
02795 
02796         consumeSemicolon();
02797 
02798         return {
02799             type: Syntax.ExpressionStatement,
02800             expression: expr
02801         };
02802     }

03110         assert(typeof start === 'number', 'Comment must have valid position');

03119             }

03130     function scanComment() {
03131         var comment, ch, loc, start, blockComment, lineComment;
03132 
03133         comment = '';
03134         blockComment = false;
03135         lineComment = false;
03136 
03137         while (index < length) {
03138             ch = source[index];
03139 
03140             if (lineComment) {
03141                 ch = source[index++];
03142                 if (isLineTerminator(ch)) {
03143                     loc.end = {
03144                         line: lineNumber,
03145                         column: index - lineStart - 1
03146                     };
03147                     lineComment = false;
03148                     addComment('Line', comment, start, index - 1, loc);
03149                     if (ch === '\r' && source[index] === '\n') {
03150                         ++index;
03151                     }
03152                     ++lineNumber;
03153                     lineStart = index;
03154                     comment = '';
03155                 } else if (index >= length) {
03156                     lineComment = false;
03157                     comment += ch;
03158                     loc.end = {
03159                         line: lineNumber,
03160                         column: length - lineStart
03161                     };
