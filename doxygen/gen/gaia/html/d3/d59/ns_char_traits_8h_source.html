00035 /*
00036  * Some macros for converting PRUnichar (UTF-16) to and from Unicode scalar
00037  * values.
00038  *
00039  * Note that UTF-16 represents all Unicode scalar values up to U+10FFFF by
00040  * using "surrogate pairs". These consist of a high surrogate, i.e. a code
00041  * point in the range U+D800 - U+DBFF, and a low surrogate, i.e. a code point
00042  * in the range U+DC00 - U+DFFF, like this:
00043  *
00044  *  U+D800 U+DC00 =  U+10000
00045  *  U+D800 U+DC01 =  U+10001
00046  *  ...
00047  *  U+DBFF U+DFFE = U+10FFFE
00048  *  U+DBFF U+DFFF = U+10FFFF
00049  *
00050  * These surrogate code points U+D800 - U+DFFF are not themselves valid Unicode
00051  * scalar values and are not well-formed UTF-16 except as high-surrogate /
00052  * low-surrogate pairs.
00053  */
00054 
00055 #define PLANE1_BASE          uint32_t(0x00010000)
00056 // High surrogates are in the range 0xD800 -- OxDBFF
00057 #define NS_IS_HIGH_SURROGATE(u) ((uint32_t(u) & 0xFFFFFC00) == 0xD800)
00058 // Low surrogates are in the range 0xDC00 -- 0xDFFF
00059 #define NS_IS_LOW_SURROGATE(u)  ((uint32_t(u) & 0xFFFFFC00) == 0xDC00)
00060 // Faster than testing NS_IS_HIGH_SURROGATE || NS_IS_LOW_SURROGATE
00061 #define IS_SURROGATE(u)      ((uint32_t(u) & 0xFFFFF800) == 0xD800)
00062 
00063 // Everything else is not a surrogate: 0x000 -- 0xD7FF, 0xE000 -- 0xFFFF
00064 
00065 // N = (H - 0xD800) * 0x400 + 0x10000 + (L - 0xDC00)
00066 // I wonder whether we could somehow assert that H is a high surrogate
00067 // and L is a low surrogate
00068 #define SURROGATE_TO_UCS4(h, l) (((uint32_t(h) & 0x03FF) << 10) + \
00069                                  (uint32_t(l) & 0x03FF) + PLANE1_BASE)

00071 // Extract surrogates from a UCS4 char
00072 // Reference: the Unicode standard 4.0, section 3.9
00073 // Since (c - 0x10000) >> 10 == (c >> 10) - 0x0080 and 
00074 // 0xD7C0 == 0xD800 - 0x0080,
00075 // ((c - 0x10000) >> 10) + 0xD800 can be simplified to
00076 #define H_SURROGATE(c) PRUnichar(PRUnichar(uint32_t(c) >> 10) + \
00077                                  PRUnichar(0xD7C0)) 
00078 // where it's to be noted that 0xD7C0 is not bitwise-OR'd
00079 // but added.
00080 
00081 // Since 0x10000 & 0x03FF == 0, 
00082 // (c - 0x10000) & 0x03FF == c & 0x03FF so that
00083 // ((c - 0x10000) & 0x03FF) | 0xDC00 is equivalent to
00084 #define L_SURROGATE(c) PRUnichar(PRUnichar(uint32_t(c) & uint32_t(0x03FF)) | \
00085                                  PRUnichar(0xDC00))
