00027 typedef uint32_t TextureFlags;
00028 // Use nearest-neighbour texture filtering (as opposed to linear filtering).
00029 const TextureFlags UseNearestFilter           = 1 << 0;
00030 // The texture should be flipped around the y-axis when composited.
00031 const TextureFlags NeedsYFlip                 = 1 << 1;
00032 // Force the texture to be represented using a single tile (note that this means
00033 // tiled textures, not tiled layers).
00034 const TextureFlags TEXTURE_DISALLOW_BIGIMAGE  = 1 << 2;
00035 // Allow using 'repeat' mode for wrapping.
00036 const TextureFlags AllowRepeat                = 1 << 3;
00037 // The texture represents a tile which is newly created.
00038 const TextureFlags NewTile                    = 1 << 4;
00039 // The texture is part of a component-alpha pair
00040 const TextureFlags ComponentAlpha             = 1 << 5;
00041 // The buffer will be treated as if the RB bytes are swapped.
00042 // This is useful for rendering using Cairo/Thebes, because there is no
00043 // BGRX Android pixel format, and so we have to do byte swapping.
00044 //
00045 // For example, if the GraphicBuffer has an Android pixel format of
00046 // PIXEL_FORMAT_RGBA_8888 and isRBSwapped is true, when it is sampled
00047 // (for example, with GL), a BGRA shader should be used.
00048 const TextureFlags TEXTURE_RB_SWAPPED         = 1 << 6;
00049 
00050 // A texture host that supports tiling
00051 const TextureFlags TEXTURE_FRONT              = 1 << 12;
00052 // A texture host on white for component alpha
00053 const TextureFlags TEXTURE_ON_WHITE           = 1 << 13;
00054  // A texture host on black for component alpha
00055 const TextureFlags TEXTURE_ON_BLACK           = 1 << 14;
00056 // A texture host that supports tiling
00057 const TextureFlags TEXTURE_TILE               = 1 << 15;
00058 // Texture contents should be initialized
00059 // from the previous texture.
00060 const TextureFlags TEXTURE_COPY_PREVIOUS      = 1 << 24;
00061 // Who is responsible for deallocating the shared data.
00062 // if none of the following two flags is set, the shared data will not be
00063 // deallocated by the layers system. It is not necessarily a leak, it could
00064 // be a choice from another part of gecko that wants to keep the data alive
00065 // for some reason. The default behaviour is to deallocate on the host side.
00066 const TextureFlags TEXTURE_DEALLOCATE_CLIENT  = 1 << 25;
00067 const TextureFlags TEXTURE_DEALLOCATE_HOST    = 1 << 26;
00068 // After being shared ith the compositor side, an immutable texture is never
00069 // modified, it can only be read. It is safe to not Lock/Unlock immutable
00070 // textures.
00071 const TextureFlags TEXTURE_IMMUTABLE          = 1 << 27;
00072 // The contents of the texture must be uploaded or copied immediately
00073 // during the transaction, because the producer may want to write
00074 // to it again.
00075 const TextureFlags TEXTURE_IMMEDIATE_UPLOAD   = 1 << 28;
00076 // The texture is going to be used as part of a double
00077 // buffered pair, and so we can guarantee that the producer/consumer
00078 // won't be racing to access its contents.
00079 const TextureFlags TEXTURE_DOUBLE_BUFFERED    = 1 << 29;
00080 
00081 // the default flags
00082 const TextureFlags TEXTURE_FLAGS_DEFAULT = TEXTURE_DEALLOCATE_HOST
00083                                          | TEXTURE_FRONT;

00085 static inline bool
00086 TextureRequiresLocking(TextureFlags aFlags)
00087 {
00088   // If we're not double buffered, or uploading
00089   // within a transaction, then we need to support
00090   // locking correctly.
00091   return !(aFlags & (TEXTURE_IMMEDIATE_UPLOAD |
00092                      TEXTURE_DOUBLE_BUFFERED |
00093                      TEXTURE_IMMUTABLE));
00094 }
