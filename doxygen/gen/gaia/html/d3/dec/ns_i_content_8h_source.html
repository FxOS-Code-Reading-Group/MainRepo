00852   void GetLang(nsAString& aResult) const {
00853     for (const nsIContent* content = this; content; content = content->GetParent()) {
00854       if (content->GetAttrCount() > 0) {
00855         // xml:lang has precedence over lang on HTML elements (see
00856         // XHTML1 section C.7).
00857         bool hasAttr = content->GetAttr(kNameSpaceID_XML, nsGkAtoms::lang,
00858                                           aResult);
00859         if (!hasAttr && (content->IsHTML() || content->IsSVG())) {
00860           hasAttr = content->GetAttr(kNameSpaceID_None, nsGkAtoms::lang,
00861                                      aResult);
00862         }
00863         NS_ASSERTION(hasAttr || aResult.IsEmpty(),
00864                      "GetAttr that returns false should not make string non-empty");
00865         if (hasAttr) {
00866           return;
00867         }
00868       }
00869     }
00870   }
