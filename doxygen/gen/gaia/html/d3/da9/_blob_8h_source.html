00053   // BaseType on the parent side is a bit more complicated than for the child
00054   // side. In the case of nsIInputStreams backed by files we need to ensure that
00055   // the files are actually opened and closed on a background thread before we
00056   // can send their file handles across to the child. The child process could
00057   // crash during this process so we need to make sure we cancel the intended
00058   // response in such a case. We do that by holding an array of
00059   // nsRevocableEventPtr. If the child crashes then this actor will be destroyed
00060   // and the nsRevocableEventPtr destructor will cancel any stream events that
00061   // are currently in flight.
00062   class BaseType : public ProtocolType
00063   {
00064   public:
00065     static const ChildBlobConstructorParams&
00066     GetBlobConstructorParams(const ConstructorParamsType& aParams)
00067     {
00068       return aParams.blobParams();
00069     }
00070 
00071     static void
00072     SetBlobConstructorParams(ConstructorParamsType& aParams,
00073                              const ChildBlobConstructorParams& aBlobParams)
00074     {
00075       aParams.blobParams() = aBlobParams;
00076       aParams.optionalInputStreamParams() = mozilla::void_t();
00077     }
00078 
00079     static void
00080     SetBlobConstructorParams(OtherSideConstructorParamsType& aParams,
00081                              const ChildBlobConstructorParams& aBlobParams)
00082     {
00083       aParams = aBlobParams;
00084     }
00085 
00086   protected:
00087     virtual StreamType*
00088     AllocPBlobStreamParent() MOZ_OVERRIDE;
00089 
00090     virtual bool
00091     DeallocPBlobStreamParent(StreamType* aActor) MOZ_OVERRIDE;
00092 
00093     BaseType();

00103   };
