00447 // Loads a shader program by scouring the current document,
00448 // looking for a script with the specified ID.

00450 function getShader(gl, id) {
00451   var shaderScript = document.getElementById(id);
00452   
00453   // Didn't find an element with the specified ID; abort.
00454   
00455   if (!shaderScript) {
00456     return null;
00457   }
00458   
00459   // Walk through the source element's children, building the
00460   // shader source string.
00461   
00462   var theSource = "";
00463   var currentChild = shaderScript.firstChild;
00464   
00465   while(currentChild) {
00466     if (currentChild.nodeType == 3) {
00467       theSource += currentChild.textContent;
00468     }
00469     
00470     currentChild = currentChild.nextSibling;
00471   }
00472   
00473   // Now figure out what type of shader script we have,
00474   // based on its MIME type.
00475   
00476   var shader;
00477   
00478   if (shaderScript.type == "x-shader/x-fragment") {
00479     shader = gl.createShader(gl.FRAGMENT_SHADER);
00480   } else if (shaderScript.type == "x-shader/x-vertex") {
00481     shader = gl.createShader(gl.VERTEX_SHADER);
00482   } else {
00483     return null;  // Unknown shader type
00484   }
00485   
00486   // Send the source to the shader object
00487   
00488   gl.shaderSource(shader, theSource);
00489   
00490   // Compile the shader program
00491   
00492   gl.compileShader(shader);
00493   
00494   // See if it compiled successfully
00495   
00496   if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
