00136   // This function takes as input the start time, duration, and direction of an
00137   // animation and returns the position in the current iteration.  Note that
00138   // this only works when we know that the animation is currently running.
00139   // This way of calling the function can be used from the compositor.  Note
00140   // that if the animation has not started yet, has already ended, or is paused,
00141   // it should not be run from the compositor.  When this function is called 
00142   // from the main thread, we need the actual ElementAnimation* in order to 
00143   // get correct animation-fill behavior and to fire animation events.
00144   // This function returns -1 for the position if the animation should not be
00145   // run (because it is not currently active and has no fill behavior), but
00146   // only does so if aAnimation is non-null; with a null aAnimation it is an
00147   // error to give aCurrentTime < aStartTime, and fill-forwards is assumed.
00148   // After calling GetPositionInIteration with non-null aAnimation and aEa, be
00149   // sure to call CheckNeedsRefresh on the animation manager afterwards.

00207     if (!aContent->MayHaveAnimations())

00210       aContent->GetProperty(nsGkAtoms::animationsProperty));

00212       return nullptr;
00213     bool propertyMatches = animations->HasAnimationOfProperty(aProperty);
00214     return (propertyMatches &&
00215             animations->CanPerformOnCompositorThread(
00216               mozilla::css::CommonElementAnimationData::CanAnimate_AllowPartial))
00217            ? animations
00218            : nullptr;
