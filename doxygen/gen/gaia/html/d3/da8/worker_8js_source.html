00001 //
00002 // The Latin input method provides word suggestions/auto completions
00003 // using the algorithm in predictions.js. For efficiency, however, this code
00004 // runs in a different thread. This is the web worker wrapper around
00005 // predictions.js. This file defines the code that interfaces predictions.js
00006 // with latin.js.
00007 //
00008 // This worker uses the convention that all messages send or received have
00009 // "cmd" and "args" fields. cmd is the message name or command. args is
00010 // an array of arguments required by the command, in a form suitable for
00011 // passing to Function.apply().
00012 //
00013 // This worker responds to the following messages:
00014 //
00015 //  setLanguage: the input method uses this message to tell the worker
00016 //    to load a dictionary for the prediction engine. If no such dictionary
00017 //    is found or if the dictionary is invalid, the worker responds with
00018 //    an "error" message.
00019 //
00020 //  setNearbyKeys: the input method uses this message to pass information about
00021 //    what keys are near each other in the current keyboard layout
00022 //
00023 //  predict: the input method uses this message to ask the prediction
00024 //    engine to suggestion completions for (or corrections to) the specified
00025 //    string. The worker responds with a "predictions" message whose argument
00026 //    is an array of up to 3 predicted words. If the the specified string
00027 //    is a word in the dictionary, the worker may send a "word" message
00028 //    first to quickly tell the keyboard that the word is valid.
00029 //
00030 
00031 'use strict';
00032 
00033 // Load the predictions.js module.  This defines the Predictions object.
00034 importScripts('predictions.js');
00035 
00036 // When we receive messages, translate them into function calls to one
00037 // of the functions in the Commands object below.
00038 self.onmessage = function(e) {
00039   Commands[e.data.cmd].apply(null, e.data.args);
00040 };
00041 
00042 // Send console messages back to the main thread with this method
00043 function log(msg) {
00044   postMessage({cmd: 'log', message: msg});
00045 }
00046 
00047 // Track our current language so we don't load dictionaries more often
00048 // than we have to.
00049 var currentLanguage;
00050 
00051 // The prediction that is currently running, if any. So that it can be cancelled
00052 var pendingPrediction;
00053 
00054 var Commands = {
00055   setLanguage: function setLanguage(language) {
00056     if (language !== currentLanguage) {
00057       currentLanguage = language;
00058 
00059       try {
00060         var dicturl = 'dictionaries/' + language + '.dict';
00061         var xhr = new XMLHttpRequest();
00062         xhr.open('GET', dicturl, false);
00063         xhr.responseType = 'arraybuffer';
00064         xhr.send();
00065         //
00066         // XXX
00067         // https://bugzilla.mozilla.org/show_bug.cgi?id=804395
00068         // The app protocol doesn't seem to return a status code and
00069         // we just get a zero-length array if the url is undefined
00070         //
00071         if (xhr.response && xhr.response.byteLength) {
00072           Predictions.setDictionary(xhr.response);
00073         }
00074         else {
00075           postMessage({
00076             cmd: 'error',
00077             message: 'setLanguage: Unknown language: ' + language
00078           });
00079         }
00080       }
00081       catch (e) {
00082         postMessage({
00083           cmd: 'error',
00084           message: 'setLanguage: Unknown language: ' + language + ': ' + e
00085         });
00086       }
00087     }
00088   },
00089 
00090   setNearbyKeys: function setNearbyKeys(nearbyKeys) {
00091     try {
00092       Predictions.setNearbyKeys(nearbyKeys);
00093     }
00094     catch (e) {
00095       postMessage({cmd: 'error', message: 'setNearbyKeys: ' + e.message});
00096     }
00097   },
00098 
00099   predict: function predict(prefix) {
00100     if (pendingPrediction)  // Make sure we're not still running a previous one
00101       pendingPrediction.abort();
00102 
00103     // Ask for 3 predictions, considering 24 candidates that and considering
00104     // only words with an edit distance of 1 (i.e. make only one correction
00105     // per word)
00106     pendingPrediction = Predictions.predict(prefix, 3, 24, 1,
00107                                             success, error);
00108 
00109     function success(words) {
00110       if (words.length) {
00111         postMessage({ cmd: 'predictions', input: prefix, suggestions: words });
00112         return;
00113       }
00114       else {
00115         // If we didn't find anything, try more candidates and a larger
00116         // edit distance to enlarge the search space.
00117         pendingPrediction =
00118           Predictions.predict(prefix, 3, 60, 2,
00119                               function(words) {
00120                                 postMessage({ cmd: 'predictions',
00121                                               input: prefix,
00122                                               suggestions: words });
00123                               }, error);
00124       }
00125     }
00126 
00127     function error(msg) {
00128       log('Error in Predictions.predict(): ' + msg);
00129     }
00130   }
00131 };
