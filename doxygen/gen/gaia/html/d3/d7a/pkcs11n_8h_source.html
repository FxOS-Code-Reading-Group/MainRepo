00194 /* Constant-time MAC mechanisms:
00195  *
00196  * These operations verify a padded, MAC-then-encrypt block of data in
00197  * constant-time. Because of the order of operations, the padding bytes are not
00198  * protected by the MAC. However, disclosing the value of the padding bytes
00199  * gives an attacker the ability to decrypt ciphertexts. Such disclosure can be
00200  * as subtle as taking slightly less time to perform the MAC when the padding
00201  * is one byte longer. See https://www.isg.rhul.ac.uk/tls/
00202  *
00203  * CKM_NSS_HMAC_CONSTANT_TIME: performs an HMAC authentication.
00204  * CKM_NSS_SSL3_MAC_CONSTANT_TIME: performs an authentication with SSLv3 MAC.
00205  *
00206  * Parameter type: CK_NSS_MAC_CONSTANT_TIME_PARAMS
00207  */
00208 #define CKM_NSS_HMAC_CONSTANT_TIME      (CKM_NSS + 19)

00262 /* macAlg: the MAC algorithm to use. This determines the hash function used in
00263  *     the HMAC/SSLv3 MAC calculations.
00264  * ulBodyTotalLen: the total length of the data, including padding bytes and
00265  *     padding length.
00266  * pHeader: points to a block of data that contains additional data to
00267  *     authenticate. For TLS this includes the sequence number etc. For SSLv3,
00268  *     this also includes the initial padding bytes.
00269  *
00270  * NOTE: the softoken's implementation of CKM_NSS_HMAC_CONSTANT_TIME and
00271  * CKM_NSS_SSL3_MAC_CONSTANT_TIME requires that the sum of ulBodyTotalLen
00272  * and ulHeaderLen be much smaller than 2^32 / 8 bytes because it uses an
00273  * unsigned int variable to represent the length in bits. This should not
00274  * be a problem because the SSL/TLS protocol limits the size of an SSL
00275  * record to something considerably less than 2^32 bytes.
00276  */
00277 typedef struct CK_NSS_MAC_CONSTANT_TIME_PARAMS {
00278     CK_MECHANISM_TYPE macAlg;   /* in */
00279     CK_ULONG ulBodyTotalLen;    /* in */
00280     CK_BYTE * pHeader;          /* in */
00281     CK_ULONG ulHeaderLen;       /* in */
00282 } CK_NSS_MAC_CONSTANT_TIME_PARAMS;
