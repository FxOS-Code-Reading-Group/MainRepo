00115     def wait_for_email(self, subject, timeout=60):
00116         timeout = float(timeout) + time.time()
00117         while time.time() < timeout:
00118             time.sleep(5)
00119             self.toolbar.tap_refresh()
00120             self.wait_for_emails_to_sync()
00121             emails = self.mails
00122             if subject in [mail.subject for mail in emails]:
00123                 break
00124         else:
00125             raise TimeoutException('Email %s not received before timeout' % subject)
