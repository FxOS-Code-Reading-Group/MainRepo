00055   {
00056     // If the option is empty, store the index.  If not, store the value.
00057     if (aValue.IsEmpty()) {
00058       mIndices.Put(aIndex);
00059     } else {
00060       mValues.Put(aValue);
00061     }
00062   }

00065   {
00066     return mValues.Contains(aValue) || mIndices.Contains(aIndex);
00067   }
