00094   /* Produces results in the range [1, 2**31 - 2].
00095      Algorithm is: r = (a * r) mod m
00096      where a = 16807 and m = 2**31 - 1 = 2147483647
00097      See [Park & Miller], CACM vol. 31 no. 10 p. 1195, Oct. 1988
00098      To test: the algorithm should produce the result 1043618065
00099      as the 10,000th generated number if the original seed is 1.
00100   */
00101 #define RAND_M 2147483647   /* 2**31 - 1 */
00102 #define RAND_A 16807        /* 7**5; primitive root of m */
00103 #define RAND_Q 127773       /* m / a */
00104 #define RAND_R 2836     /* m % a */
00105 
00106   int32_t SetupSeed(int32_t aSeed) {
00107     if (aSeed <= 0)
00108       aSeed = -(aSeed % (RAND_M - 1)) + 1;
00109     if (aSeed > RAND_M - 1)
00110       aSeed = RAND_M - 1;
00111     return aSeed;
00112   }
00113 
00114   uint32_t Random(uint32_t aSeed) {
00115     int32_t result = RAND_A * (aSeed % RAND_Q) - RAND_R * (aSeed / RAND_Q);
00116     if (result <= 0)
00117       result += RAND_M;
00118     return result;
00119   }
00120 #undef RAND_M
00121 #undef RAND_A
00122 #undef RAND_Q
00123 #undef RAND_R
