00028 // Represents (affine) transforms that are calculated from a content view.
00029 struct ViewTransform {
00030   ViewTransform(LayerPoint aTranslation = LayerPoint(),
00031                 LayoutDeviceToScreenScale aScale = LayoutDeviceToScreenScale())
00032     : mTranslation(aTranslation)
00033     , mScale(aScale)
00034   {}
00035 
00036   operator gfx3DMatrix() const
00037   {
00038     return
00039       gfx3DMatrix::Translation(mTranslation.x, mTranslation.y, 0) *
00040       gfx3DMatrix::ScalingMatrix(mScale.scale, mScale.scale, 1);
00041   }
00042 
00043   bool operator==(const ViewTransform& rhs) const {
00044     return mTranslation == rhs.mTranslation && mScale == rhs.mScale;
00045   }
00046 
00047   bool operator!=(const ViewTransform& rhs) const {
00048     return !(*this == rhs);
00049   }
00050 
00051   LayerPoint mTranslation;
00052   LayoutDeviceToScreenScale mScale;
00053 };
