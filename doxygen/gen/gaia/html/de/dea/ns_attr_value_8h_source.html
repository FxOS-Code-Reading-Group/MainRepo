00474 {
00475   switch (Type()) {
00476     case eString:
00477     {
00478       nsStringBuffer* str = static_cast<nsStringBuffer*>(GetPtr());
00479       if (str) {
00480         aResult.SetStringBuffer(str, str->StorageSize()/sizeof(PRUnichar) - 1);
00481       }
00482       // else aResult is already empty
00483       return;
00484     }
00485     case eAtom:
00486     {
00487       nsIAtom *atom = static_cast<nsIAtom*>(GetPtr());
00488       aResult.SetStringBuffer(atom->GetStringBuffer(), atom->GetLength());
00489       break;
00490     }
00491     default:
00492     {
00493       ToString(aResult.AsAString());
