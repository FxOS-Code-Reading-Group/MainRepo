00001 /*
00002  * Determine the pixel dimensions of an image without actually
00003  * decoding the image. Passes an object of metadata to the callback
00004  * function on success or an error message to the error function on
00005  * failure. The metadata object will include type, width and height
00006  * properties. Supported image types are GIF, PNG and JPEG. JPEG
00007  * metadata may also include information about an EXIF preview image.
00008  *
00009  * Because of shortcomings in the way Gecko handles images, the
00010  * Gallery app will crash with an OOM error if it attempts to decode
00011  * and display an image that is too big. Images require 4 bytes per
00012  * pixel, so a 10 megapixel photograph requires 40 megabytes of image
00013  * memory. This function gives the gallery app a way to reject images
00014  * that are too large.
00015  *
00016  * Requires the BlobView class from shared/js/blobview.js and the
00017  * parseJPEGMetadata() function from shared/js/media/jpeg_metadata_parser.js
00018  */
00019 function getImageSize(blob, callback, error) {
00020   BlobView.get(blob, 0, Math.min(1024, blob.size), function(data) {
00021     // Make sure we are at least 8 bytes long before reading the first 8 bytes
00022     if (data.byteLength <= 8) {
00023       error('corrupt image file');
00024       return;
00025     }
00026     var magic = data.getASCIIText(0, 8);
00027     if (magic.substring(0, 4) === 'GIF8') {
00028       try {
00029         callback({
00030           type: 'gif',
00031           width: data.getUint16(6, true),
00032           height: data.getUint16(8, true)
00033         });
00034       }
00035       catch (e) {
00036         error(e.toString());
00037       }
00038     }
00039     else if (magic.substring(0, 8) === '\x89PNG\r\n\x1A\n') {
00040       try {
00041         callback({
00042           type: 'png',
00043           width: data.getUint32(16, false),
00044           height: data.getUint32(20, false)
00045         });
00046       }
00047       catch (e) {
00048         error(e.toString());
00049       }
00050     }
00051     else if (magic.substring(0, 2) === '\xFF\xD8') {
00052       parseJPEGMetadata(blob,
00053                         function(metadata) {
00054                           metadata.type = 'jpeg';
00055                           callback(metadata);
00056                         },
00057                         error);
00058     }
00059     else {
00060       error('unknown image type');
00061     }
00062   });
00063 }
