00007 /* A type-safe doubly-linked list class. */
00008 
00009 /*
00010  * The classes LinkedList<T> and LinkedListElement<T> together form a
00011  * convenient, type-safe doubly-linked list implementation.
00012  *
00013  * The class T which will be inserted into the linked list must inherit from
00014  * LinkedListElement<T>.  A given object may be in only one linked list at a
00015  * time.
00016  *
00017  * A LinkedListElement automatically removes itself from the list upon
00018  * destruction, and a LinkedList will fatally assert in debug builds if it's
00019  * non-empty when it's destructed.
00020  *
00021  * For example, you might use LinkedList in a simple observer list class as
00022  * follows.
00023  *
00024  *   class Observer : public LinkedListElement<Observer>
00025  *   {
00026  *     public:
00027  *       void observe(char* topic) { ... }
00028  *   };
00029  *
00030  *   class ObserverContainer
00031  *   {
00032  *     private:
00033  *       LinkedList<Observer> list;
00034  *
00035  *     public:
00036  *       void addObserver(Observer* observer) {
00037  *         // Will assert if |observer| is part of another list.
00038  *         list.insertBack(observer);
00039  *       }
00040  *
00041  *       void removeObserver(Observer* observer) {
00042  *         // Will assert if |observer| is not part of some list.
00043  *         observer.remove();
00044  *         // Or, will assert if |observer| is not part of |list| specifically.
00045  *         // observer.removeFrom(list);
00046  *       }
00047  *
00048  *       void notifyObservers(char* topic) {
00049  *         for (Observer* o = list.getFirst(); o != nullptr; o = o->getNext())
00050  *           o->observe(topic);
00051  *       }
00052  *   };
00053  *
00054  */

00073     /*
00074      * It's convenient that we return nullptr when getNext() or getPrevious()
00075      * hits the end of the list, but doing so costs an extra word of storage in
00076      * each linked list node (to keep track of whether |this| is the sentinel
00077      * node) and a branch on this value in getNext/getPrevious.
00078      *
00079      * We could get rid of the extra word of storage by shoving the "is
00080      * sentinel" bit into one of the pointers, although this would, of course,
00081      * have performance implications of its own.
00082      *
00083      * But the goal here isn't to win an award for the fastest or slimmest
00084      * linked list; rather, we want a *convenient* linked list.  So we won't
00085      * waste time guessing which micro-optimization strategy is best.
00086      *
00087      *
00088      * Speaking of unnecessary work, it's worth addressing here why we wrote
00089      * mozilla::LinkedList in the first place, instead of using stl::list.
00090      *
00091      * The key difference between mozilla::LinkedList and stl::list is that
00092      * mozilla::LinkedList stores the prev/next pointers in the object itself,
00093      * while stl::list stores the prev/next pointers in a list element which
00094      * itself points to the object being stored.
00095      *
00096      * mozilla::LinkedList's approach makes it harder to store an object in more
00097      * than one list.  But the upside is that you can call next() / prev() /
00098      * remove() directly on the object.  With stl::list, you'd need to store a
00099      * pointer to its iterator in the object in order to accomplish this.  Not
00100      * only would this waste space, but you'd have to remember to update that
00101      * pointer every time you added or removed the object from a list.
00102      *
00103      * In-place, constant-time removal is a killer feature of doubly-linked
00104      * lists, and supporting this painlessly was a key design criterion.
00105      */
