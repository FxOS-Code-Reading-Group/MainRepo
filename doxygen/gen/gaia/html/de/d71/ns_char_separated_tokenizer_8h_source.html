00088     const nsDependentSubstring nextToken()
00089     {
00090         nsSubstring::const_char_iterator end = mIter, begin = mIter;
00091 
00092         NS_ASSERTION(mIter == mEnd || !IsWhitespace(*mIter),
00093                      "Should be at beginning of token if there is one");
00094 
00095         // Search until we hit separator or end (or whitespace, if separator
00096         // isn't required -- see clause with 'break' below).
00097         while (mIter != mEnd && *mIter != mSeparatorChar) {
00098           // Skip to end of current word.
00099           while (mIter != mEnd &&
00100                  !IsWhitespace(*mIter) && *mIter != mSeparatorChar) {
00101               ++mIter;
00102           }
00103           end = mIter;
00104 
00105           // Skip whitespace after current word.
00106           mLastTokenEndedWithWhitespace = false;
00107           while (mIter != mEnd && IsWhitespace(*mIter)) {
00108               mLastTokenEndedWithWhitespace = true;
00109               ++mIter;
00110           }
00111           if (mFlags & SEPARATOR_OPTIONAL) {
00112             // We've hit (and skipped) whitespace, and that's sufficient to end
00113             // our token, regardless of whether we've reached a SeparatorChar.
00114             break;
00115           } // (else, we'll keep looping until we hit mEnd or SeparatorChar)
00116         }
00117 
00118         mLastTokenEndedWithSeparator = (mIter != mEnd &&
00119                                         *mIter == mSeparatorChar);
00120         NS_ASSERTION((mFlags & SEPARATOR_OPTIONAL) ||
00121                      (mLastTokenEndedWithSeparator == (mIter != mEnd)),
00122                      "If we require a separator and haven't hit the end of "
00123                      "our string, then we shouldn't have left the loop "
00124                      "unless we hit a separator");
00125 
00126         // Skip separator (and any whitespace after it), if we're at one.
00127         if (mLastTokenEndedWithSeparator) {
00128             ++mIter;
00129 
00130             while (mIter != mEnd && IsWhitespace(*mIter)) {
00131                 ++mIter;
00132             }
00133         }
00134 
00135         return Substring(begin, end);
00136     }
