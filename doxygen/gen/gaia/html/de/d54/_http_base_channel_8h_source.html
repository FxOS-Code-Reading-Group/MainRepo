00312 // Share some code while working around C++'s absurd inability to handle casting
00313 // of member functions between base/derived types.
00314 // - We want to store member function pointer to call at resume time, but one
00315 //   such function--HandleAsyncAbort--we want to share between the
00316 //   nsHttpChannel/HttpChannelChild.  Can't define it in base class, because
00317 //   then we'd have to cast member function ptr between base/derived class
00318 //   types.  Sigh...
00319 template <class T>
00320 class HttpAsyncAborter
00321 {
00322 public:
00323   HttpAsyncAborter(T *derived) : mThis(derived), mCallOnResume(0) {}
00324 
00325   // Aborts channel: calls OnStart/Stop with provided status, removes channel
00326   // from loadGroup.
00327   nsresult AsyncAbort(nsresult status);
00328 
00329   // Does most the actual work.
00330   void HandleAsyncAbort();
00331 
00332   // AsyncCall calls a member function asynchronously (via an event).
00333   // retval isn't refcounted and is set only when event was successfully
00334   // posted, the event is returned for the purpose of cancelling when needed
00335   nsresult AsyncCall(void (T::*funcPtr)(),
00336                      nsRunnableMethod<T> **retval = nullptr);
00337 private:
00338   T *mThis;
00339 
00340 protected:
00341   // Function to be called at resume time
00342   void (T::* mCallOnResume)(void);
00343 };
00344 
00345 template <class T>
00346 nsresult HttpAsyncAborter<T>::AsyncAbort(nsresult status)
00347 {
00348   PR_LOG(gHttpLog, 4,
00349          ("HttpAsyncAborter::AsyncAbort [this=%p status=%x]\n", mThis, status));
00350 
00351   mThis->mStatus = status;
00352   mThis->mIsPending = false;
00353 
00354   // if this fails?  Callers ignore our return value anyway....
00355   return AsyncCall(&T::HandleAsyncAbort);
00356 }

00358 // Each subclass needs to define its own version of this (which just calls this
00359 // base version), else we wind up casting base/derived member function ptrs
00360 template <class T>
00361 inline void HttpAsyncAborter<T>::HandleAsyncAbort()
00362 {
00363   NS_PRECONDITION(!mCallOnResume, "How did that happen?");
00364 
00365   if (mThis->mSuspendCount) {
00366     PR_LOG(gHttpLog, 4,
00367            ("Waiting until resume to do async notification [this=%p]\n", mThis));
00368     mCallOnResume = &T::HandleAsyncAbort;
00369     return;
00370   }
00371 
00372   mThis->DoNotifyListener();
00373 
00374   // finally remove ourselves from the load group.
00375   if (mThis->mLoadGroup)
00376     mThis->mLoadGroup->RemoveRequest(mThis, nullptr, mThis->mStatus);
00377 }

00379 template <class T>
00380 nsresult HttpAsyncAborter<T>::AsyncCall(void (T::*funcPtr)(),
00381                                    nsRunnableMethod<T> **retval)
00382 {
00383   nsresult rv;
00384 
00385   nsRefPtr<nsRunnableMethod<T> > event = NS_NewRunnableMethod(mThis, funcPtr);
00386   rv = NS_DispatchToCurrentThread(event);
00387   if (NS_SUCCEEDED(rv) && retval) {
00388     *retval = event;
00389   }
00390 
00391   return rv;
00392 }
