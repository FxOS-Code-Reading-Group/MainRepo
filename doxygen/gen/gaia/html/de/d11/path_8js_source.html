00030 // resolves . and .. elements in a path array with directory names there
00031 // must be no slashes, empty elements, or device names (c:\) in the array
00032 // (so also no leading and trailing slashes - it does not distinguish
00033 // relative and absolute paths)
00034 function normalizeArray(parts, allowAboveRoot) {
00035   // if the path tries to go above the root, `up` ends up > 0
00036   var up = 0;
00037   for (var i = parts.length; i >= 0; i--) {
00038     var last = parts[i];
00039     if (last == '.') {
00040       parts.splice(i, 1);
00041     } else if (last === '..') {
00042       parts.splice(i, 1);
00043       up++;
00044     } else if (up) {
00045       parts.splice(i, 1);
00046       up--;
00047     }
00048   }
00049 
00050   // if the path is allowed to go above the root, restore leading ..s
00051   if (allowAboveRoot) {
00052     for (; up--; up) {
00053       parts.unshift('..');
00054     }
00055   }
00056 
00057   return parts;
00058 }

00087 // Normalize the path

00089     return !!p;

00126 exports.dirname = function(path) {
00127   var dir = splitPathRe.exec(path)[1] || '';
00128   var isWindows = false;
00129   if (!dir) {
00130     // No dirname
00131     return '.';
00132   } else if (dir.length === 1 ||
00133       (isWindows && dir.length <= 3 && dir.charAt(1) === ':')) {
00134     // It is just a slash or a drive letter with a slash
00135     return dir;
00136   } else {
00137     // It is a full dirname, strip trailing slash
00138     return dir.substring(0, dir.length - 1);
00139   }
00140 };
