00037 class UTF8CharEnumerator
00038 {
00039 public:
00040   static uint32_t NextChar(const char **buffer, const char *end,
00041                            bool *err)
00042   {
00043     NS_ASSERTION(buffer && *buffer, "null buffer!");
00044 
00045     const char *p = *buffer;
00046     *err = false;
00047 
00048     if (p >= end)
00049       {
00050         *err = true;
00051 
00052         return 0;
00053       }
00054 
00055     char c = *p++;
00056 
00057     if ( UTF8traits::isASCII(c) )
00058       {
00059         *buffer = p;
00060         return c;
00061       }
00062 
00063     uint32_t ucs4;
00064     uint32_t minUcs4;
00065     int32_t state = 0;
00066 
00067     if (!CalcState(c, ucs4, minUcs4, state)) {
00068         NS_ERROR("Not a UTF-8 string. This code should only be used for converting from known UTF-8 strings.");
00069         *err = true;
00070 
00071         return 0;
00072     }
00073 
00074     while ( state-- )
00075       {
00076         if (p == end)
00077           {
00078             *err = true;
00079 
00080             return 0;
00081           }
00082 
00083         c = *p++;
00084 
00085         if (!AddByte(c, state, ucs4))
00086           {
00087             *err = true;
00088 
00089             return 0;
00090           }
00091       }
00092 
00093       if ( ucs4 < minUcs4 )
00094         {
00095           // Overlong sequence
00096           ucs4 = UCS2_REPLACEMENT_CHAR;
00097         }
00098       else if ( ucs4 >= 0xD800 &&
00099                 (ucs4 <= 0xDFFF || ucs4 >= UCS_END))
00100         {
00101           // Surrogates and code points outside the Unicode range.
00102           ucs4 = UCS2_REPLACEMENT_CHAR;
00103         }
00104 
00105     *buffer = p;
00106     return ucs4;
00107   }
00108 
00109 private:
00110   static bool CalcState(char c, uint32_t& ucs4, uint32_t& minUcs4,
00111                           int32_t& state)
00112   {
00113     if ( UTF8traits::is2byte(c) )
00114       {
00115         ucs4 = (uint32_t(c) << 6) & 0x000007C0L;
00116         state = 1;
00117         minUcs4 = 0x00000080;
00118       }
00119     else if ( UTF8traits::is3byte(c) )
00120       {
00121         ucs4 = (uint32_t(c) << 12) & 0x0000F000L;
00122         state = 2;
00123         minUcs4 = 0x00000800;
00124       }
00125     else if ( UTF8traits::is4byte(c) )
00126       {
00127         ucs4 = (uint32_t(c) << 18) & 0x001F0000L;
00128         state = 3;
00129         minUcs4 = 0x00010000;
00130       }
00131     else if ( UTF8traits::is5byte(c) )
00132       {
00133         ucs4 = (uint32_t(c) << 24) & 0x03000000L;
00134         state = 4;
00135         minUcs4 = 0x00200000;
00136       }
00137     else if ( UTF8traits::is6byte(c) )
00138       {
00139         ucs4 = (uint32_t(c) << 30) & 0x40000000L;
00140         state = 5;
00141         minUcs4 = 0x04000000;
00142       }
00143     else
00144       {
00145         return false;
00146       }
00147 
00148     return true;
00149   }
00150 
00151   static bool AddByte(char c, int32_t state, uint32_t& ucs4)
00152   {
00153     if ( UTF8traits::isInSeq(c) )
00154       {
00155         int32_t shift = state * 6;
00156         ucs4 |= (uint32_t(c) & 0x3F) << shift;
00157         return true;
00158       }
00159 
00160     return false;
00161   }

00173 {

00175   static uint32_t NextChar(const PRUnichar **buffer, const PRUnichar *end,
00176                            bool *err = nullptr)
00177   {
00178     NS_ASSERTION(buffer && *buffer, "null buffer!");
00179 
00180     const PRUnichar *p = *buffer;
00181 
00182     if (p >= end)
00183       {
00184         NS_ERROR("No input to work with");
00185         if (err)
00186           *err = true;
00187 
00188         return 0;
00189       }
00190 
00191     PRUnichar c = *p++;
00192 
00193     if (!IS_SURROGATE(c)) // U+0000 - U+D7FF,U+E000 - U+FFFF
00194       {
00195         if (err)
00196           *err = false;
00197         *buffer = p;
00198         return c;
00199       }
00200     else if (NS_IS_HIGH_SURROGATE(c)) // U+D800 - U+DBFF
00201       {
00202         if (p == end)
00203           {
00204             // Found a high surrogate the end of the buffer. Flag this
00205             // as an error and return the Unicode replacement
00206             // character 0xFFFD.
00207 
00208             NS_WARNING("Unexpected end of buffer after high surrogate");
00209 
00210             if (err)
00211               *err = true;
00212             *buffer = p;
00213             return 0xFFFD;
00214           }
00215 
00216         // D800- DBFF - High Surrogate
00217         PRUnichar h = c;
00218 
00219         c = *p++;
00220 
00221         if (NS_IS_LOW_SURROGATE(c))
00222           {
00223             // DC00- DFFF - Low Surrogate
00224             // N = (H - D800) *400 + 10000 + (L - DC00)
00225             uint32_t ucs4 = SURROGATE_TO_UCS4(h, c);
00226             if (err)
00227               *err = false;
00228             *buffer = p;
00229             return ucs4;
00230           }
00231         else
00232           {
00233             // Found a high surrogate followed by something other than
00234             // a low surrogate. Flag this as an error and return the
00235             // Unicode replacement character 0xFFFD.  Note that the
00236             // pointer to the next character points to the second 16-bit
00237             // value, not beyond it, as per Unicode 5.0.0 Chapter 3 C10,
00238             // only the first code unit of an illegal sequence must be
00239             // treated as an illegally terminated code unit sequence
00240             // (also Chapter 3 D91, "isolated [not paired and ill-formed]
00241             // UTF-16 code units in the range D800..DFFF are ill-formed").
00242             NS_WARNING("got a High Surrogate but no low surrogate");
00243 
00244             if (err)
00245               *err = true;
00246             *buffer = p - 1;
00247             return 0xFFFD;
00248           }
00249       }
00250     else // U+DC00 - U+DFFF
00251       {
00252         // DC00- DFFF - Low Surrogate
00253 
00254         // Found a low surrogate w/o a preceding high surrogate. Flag
00255         // this as an error and return the Unicode replacement
00256         // character 0xFFFD.
00257 
00258         NS_WARNING("got a low Surrogate but no high surrogate");
00259         if (err)
00260           *err = true;
00261         *buffer = p;
00262         return 0xFFFD;
00263       }
00264 
00265     if (err)
00266       *err = true;
00267     return 0;
00268   }

00339 class CalculateUTF8Length
00340   {
00341     public:
00342       typedef char value_type;
00343 
00344     CalculateUTF8Length() : mLength(0), mErrorEncountered(false) { }
00345 
00346     size_t Length() const { return mLength; }
00347 
00348     void write( const value_type* start, uint32_t N )
00349       {
00350           // ignore any further requests
00351         if ( mErrorEncountered )
00352             return;
00353 
00354         // algorithm assumes utf8 units won't
00355         // be spread across fragments
00356         const value_type* p = start;
00357         const value_type* end = start + N;
00358         for ( ; p < end /* && *p */; ++mLength )
00359           {
00360             if ( UTF8traits::isASCII(*p) )
00361                 p += 1;
00362             else if ( UTF8traits::is2byte(*p) )
00363                 p += 2;
00364             else if ( UTF8traits::is3byte(*p) )
00365                 p += 3;
00366             else if ( UTF8traits::is4byte(*p) ) {
00367                 // Because a UTF-8 sequence of 4 bytes represents a codepoint
00368                 // greater than 0xFFFF, it will become a surrogate pair in the
00369                 // UTF-16 string, so add 1 more to mLength.
00370                 // This doesn't happen with is5byte and is6byte because they
00371                 // are illegal UTF-8 sequences (greater than 0x10FFFF) so get
00372                 // converted to a single replacement character.
00373 
00374                 // However, there is one case when a 4 byte UTF-8 sequence will
00375                 // only generate 2 UTF-16 bytes. If we have a properly encoded
00376                 // sequence, but with an invalid value (too small or too big),
00377                 // that will result in a replacement character being written
00378                 // This replacement character is encoded as just 1 single
00379                 // UTF-16 character, which is 2 bytes.
00380 
00381                 // The below code therefore only adds 1 to mLength if the UTF8
00382                 // data will produce a decoded character which is greater than
00383                 // or equal to 0x010000 and less than 0x0110000.
00384 
00385                 // A 4byte UTF8 character is encoded as
00386                 // 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
00387                 // Bit 1-3 on the first byte, and bit 5-6 on the second byte,
00388                 // map to bit 17-21 in the final result. If these bits are
00389                 // between 0x01 and 0x11, that means that the final result is
00390                 // between 0x010000 and 0x110000. The below code reads these
00391                 // bits out and assigns them to c, but shifted up 4 bits to
00392                 // avoid having to shift twice.
00393 
00394                 // It doesn't matter what to do in the case where p + 4 > end
00395                 // since no UTF16 characters will be written in that case by
00396                 // ConvertUTF8toUTF16. Likewise it doesn't matter what we do if
00397                 // any of the surrogate bits are wrong since no UTF16
00398                 // characters will be written in that case either.
00399 
00400                 if (p + 4 <= end) {
00401                   uint32_t c = ((uint32_t)(p[0] & 0x07)) << 6 |
00402                                ((uint32_t)(p[1] & 0x30));
00403                   if (c >= 0x010 && c < 0x110)
00404                     ++mLength;
00405                 }
00406 
00407                 p += 4;
00408             }
00409             else if ( UTF8traits::is5byte(*p) )
00410                 p += 5;
00411             else if ( UTF8traits::is6byte(*p) )
00412                 p += 6;
00413             else // error
00414               {
00415                 ++mLength; // to account for the decrement below
00416                 break;
00417               }
00418           }
00419         if ( p != end )
00420           {
00421             NS_ERROR("Not a UTF-8 string. This code should only be used for converting from known UTF-8 strings.");
00422             --mLength; // The last multi-byte char wasn't complete, discard it.
00423             mErrorEncountered = true;
00424           }
00425       }
00426 
00427     private:
00428       size_t mLength;
00429       bool mErrorEncountered;
00430   };
