00611     function showDevicePaired(paired, errorMessage) {
00612       // If we don't know the pairing device address,
00613       // it means the pair request is handled by interface level.
00614       // So we just need to update paired list.

00618       }
00619       // clear pairingAddress first to prevent execute
00620       // the same status update twice.
00621       var workingAddress = pairingAddress;
00622       pairingAddress = null;
00623       if (paired) {
00624         // if the device is on the list, remove it.
00625         // it will show on paired list later.
00626         if (openList.index[workingAddress]) {
00627           var device = openList.index[workingAddress].device;
00628           var item = openList.index[workingAddress].item;
00629           openList.list.removeChild(item);
00630           delete openList.index[workingAddress];
00631           connectingAddress = workingAddress;
00632         }
00633       } else {
00634         // if the attention screen still open, close it
00635         if (childWindow) {
00636           childWindow.PairView.closeInput();
00637           childWindow.close();
00638         }
00639         // display failure only when active request
00640         if (pairingMode === 'active' && !userCanceledPairing) {
00641           // show pair process fail.
00642           var msg = _('error-pair-title');
00643           if (errorMessage === 'Repeated Attempts') {
00644             msg = msg + '\n' + _('error-pair-toofast');
00645           } else if (errorMessage === 'Authentication Failed') {
00646             msg = msg + '\n' + _('error-pair-pincode');
