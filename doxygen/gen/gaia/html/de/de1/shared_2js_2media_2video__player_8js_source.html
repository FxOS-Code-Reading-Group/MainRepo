00004 // Create a <video> element and  <div> containing a video player UI and
00005 // add them to the specified container. The UI requires a GestureDetector
00006 // to be running for the container or one of its ancestors.
00007 //
00008 // Some devices have only a single hardware video decoder and can only
00009 // have one video tag playing anywhere at once. So this class is careful
00010 // to only load content into a <video> element when the user really wants
00011 // to play it. At other times it displays a poster image for the video.
00012 // Initially, it displays the poster image. Pressing play starts the video.
00013 // Pausing pauses the video but does not revert to the poster. Finishing the
00014 // video reverts to the initial state with the poster image displayed.
00015 // If we get a visiblitychange event saying that we've been hidden, we
00016 // remember the playback position, pause the video take a temporary
00017 // screenshot and display it, and unload the video. If shown again

00327     case 0:

00330       height = videoheight;

00334       width = videoheight;

00435   function formatTime(time) {
00436     function padLeft(num, length) {
00437       var r = String(num);
00438       while (r.length < length) {
00439         r = '0' + r;
00440       }
00441       return r;
00442     }
00443 
00444     time = Math.round(time);
00445     var minutes = Math.floor(time / 60);
00446     var seconds = time % 60;
00447     if (minutes < 60) {
00448       return padLeft(minutes, 2) + ':' + padLeft(seconds, 2);
00449     } else {
00450       var hours = Math.floor(minutes / 60);
00451       minutes = Math.round(minutes % 60);
00452       return hours + ':' + padLeft(minutes, 2) + ':' + padLeft(seconds, 2);
00453     }
00454     return '';
00455   }
