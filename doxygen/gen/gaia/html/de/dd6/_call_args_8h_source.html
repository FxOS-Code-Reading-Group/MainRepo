00007 /*
00008  * Helper classes encapsulating access to the callee, |this| value, arguments,
00009  * and argument count for a function call.
00010  *
00011  * The intent of JS::CallArgs and JS::CallReceiver is that they be used to
00012  * encapsulate access to the un-abstracted |unsigned argc, Value *vp| arguments
00013  * to a function.  It's possible (albeit deprecated) to manually index into
00014  * |vp| to access the callee, |this|, and arguments of a function, and to set
00015  * its return value.  It's also possible to use the supported API of JS_CALLEE,
00016  * JS_THIS, JS_ARGV, JS_RVAL and JS_SET_RVAL to the same ends.  But neither API
00017  * has the error-handling or moving-GC correctness of CallArgs or CallReceiver.
00018  * New code should use CallArgs and CallReceiver instead whenever possible.
00019  *
00020  * The eventual plan is to change JSNative to take |const CallArgs&| directly,
00021  * for automatic assertion of correct use and to make calling functions more
00022  * efficient.  Embedders should start internally switching away from using
00023  * |argc| and |vp| directly, except to create a |CallArgs|.  Then, when an
00024  * eventual release making that change occurs, porting efforts will require
00025  * changing methods' signatures but won't require invasive changes to the
00026  * methods' implementations, potentially under time pressure.
00027  */
00028

00087 /*
00088  * JS::CallReceiver encapsulates access to the callee, |this|, and eventual
00089  * return value for a function call.  The principal way to create a
00090  * CallReceiver is using JS::CallReceiverFromVp:
00091  *
00092  *   static JSBool
00093  *   FunctionReturningThis(JSContext *cx, unsigned argc, JS::Value *vp)
00094  *   {
00095  *       JS::CallReceiver rec = JS::CallReceiverFromVp(vp);
00096  *
00097  *       // Access to the callee must occur before accessing/setting
00098  *       // the return value.
00099  *       JSObject &callee = rec.callee();
00100  *       rec.rval().set(JS::ObjectValue(callee));
00101  *
00102  *       // callee() and calleev() will now assert.
00103  *
00104  *       // It's always fine to access thisv().
00105  *       HandleValue thisv = rec.thisv();
00106  *       rec.rval().set(thisv);
00107  *
00108  *       // As the return value was last set to |this|, returns |this|.
00109  *       return true;
00110  *   }
00111  *
00112  * A note on JS_ComputeThis and JS_THIS_OBJECT: these methods currently aren't
00113  * part of the CallReceiver interface.  We will likely add them at some point.
00114  * Until then, you should probably continue using |vp| directly for these two
00115  * cases.
00116  *
00117  * CallReceiver is exposed publicly and used internally.  Not all parts of its
00118  * public interface are meant to be used by embedders!  See inline comments to
00119  * for details.
00120  */

00280 /*
00281  * JS::CallArgs encapsulates everything JS::CallReceiver does, plus access to
00282  * the function call's arguments.  The principal way to create a CallArgs is
00283  * like so, using JS::CallArgsFromVp:
00284  *
00285  *   static JSBool
00286  *   FunctionReturningArgcTimesArg0(JSContext *cx, unsigned argc, JS::Value *vp)
00287  *   {
00288  *       JS::CallArgs args = JS::CallArgsFromVp(argc, vp);
00289  *
00290  *       // Guard against no arguments or a non-numeric arg0.
00291  *       if (args.length() == 0 || !args[0].isNumber()) {
00292  *           args.rval().setInt32(0);
00293  *           return true;
00294  *       }
00295  *
00296  *       args.rval().set(JS::NumberValue(args.length() * args[0].toNumber()));
00297  *       return true;
00298  *   }
00299  *
00300  * CallArgs is exposed publicly and used internally.  Not all parts of its
00301  * public interface are meant to be used by embedders!  See inline comments to
00302  * for details.
00303  */

00311 {

00315   public:

00319     /* Returns the i-th zero-indexed argument. */
00320     MutableHandleValue operator[](unsigned i) const {
00321         MOZ_ASSERT(i < argc_);
00322         return MutableHandleValue::fromMarkedLocation(&this->argv_[i]);
00323     }

00325     /*
00326      * Returns the i-th zero-indexed argument, or |undefined| if there's no

00330         return i < length()
00331                ? HandleValue::fromMarkedLocation(&this->argv_[i])
00332                : UndefinedHandleValue;

00340         return i < argc_ && !this->argv_[i].isUndefined();
