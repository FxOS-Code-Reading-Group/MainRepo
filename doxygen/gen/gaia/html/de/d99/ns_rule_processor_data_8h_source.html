00044   /* Check whether we might have an ancestor matching one of the given
00045      atom hashes.  |hashes| must have length hashListLength */
00046   template<size_t hashListLength>
00047     bool MightHaveMatchingAncestor(const uint32_t* aHashes) const
00048   {
00049     MOZ_ASSERT(mFilter);
00050     for (size_t i = 0; i < hashListLength && aHashes[i]; ++i) {
00051       if (!mFilter->mightContain(aHashes[i])) {
00052         return false;
00053       }
00054     }
00055 
00056     return true;
00057   }
00058 
00059   bool HasFilter() const { return mFilter; }
00060 
00061 #ifdef DEBUG
00062   void AssertHasAllAncestors(mozilla::dom::Element *aElement) const;
00063 #endif
00064   
00065  private:
00066   // Using 2^12 slots makes the Bloom filter a nice round page in
00067   // size, so let's do that.  We get a false positive rate of 1% or
00068   // less even with several hundred things in the filter.  Note that
00069   // we allocate the filter lazily, because not all tree match
00070   // contexts can use one effectively.
00071   typedef mozilla::BloomFilter<12, nsIAtom> Filter;
00072   nsAutoPtr<Filter> mFilter;
00073 
00074   // Stack of indices to pop to.  These are indices into mHashes.
00075   nsTArray<uint32_t> mPopTargets;
00076 
00077   // List of hashes; this is what we pop using mPopTargets.  We store
00078   // hashes of our ancestor element tag names, ids, and classes in
00079   // here.
00080   nsTArray<uint32_t> mHashes;
00081 
00082   // A debug-only stack of Elements for use in assertions
00083 #ifdef DEBUG
00084   nsTArray<mozilla::dom::Element*> mElements;
00085 #endif

00161   bool PopStyleScopeForSelectorMatching(mozilla::dom::Element* aElement)
00162   {
00163     NS_ASSERTION(mForScopedStyle, "only call PopStyleScopeForSelectorMatching "
00164                                   "when mForScopedStyle is true");
00165 
00166     if (!mCurrentStyleScope) {
00167       return false;
00168     }

00170       mCurrentStyleScope = nullptr;
00171     }
00172     return true;

00175   bool SetStyleScopeForSelectorMatching(mozilla::dom::Element* aSubject,
00176                                         mozilla::dom::Element* aScope)
00177   {
00178     mForScopedStyle = !!aScope;
00179     if (!aScope) {
00180       // This is not for a scoped style sheet; return true, as we want
00181       // selector matching to proceed.
00182       mCurrentStyleScope = nullptr;
00183       return true;
00184     }
00185     if (aScope == aSubject) {
00186       // Although the subject is the same element as the scope, as soon
00187       // as we continue with selector matching up the tree we don't want
00188       // to match any more elements.  So we return true to indicate that
00189       // we want to do the initial selector matching, but set
00190       // mCurrentStyleScope to null so that no ancestor elements will match.
00191       mCurrentStyleScope = nullptr;
00192       return true;
00193     }
00194     if (mStyleScopes.Contains(aScope)) {
00195       // mStyleScopes contains all of the scope elements that are ancestors of
00196       // aSubject, so if aScope is in mStyleScopes, then we do want selector
00197       // matching to proceed.
00198       mCurrentStyleScope = aScope;
00199       return true;
00200     }
00201     // Otherwise, we're not in the scope, and we don't want to proceed
00202     // with selector matching.
00203     mCurrentStyleScope = nullptr;
00204     return false;
00205   }
