00003   monitorTagVisibility
00004     generalized function to watch children of a container
00005 
00006     ------------------------------------
00007 
00008     terminology
00009       - child: any descendent of a node
00010 
00011     ------------------------------------
00012 
00013     arguments
00014       container
00015         - the scrollable element we want to monitor the children of
00016       tag
00017         - the tag to monitor
00018       scrollMargin
00019         - how close an element needs to be to the container edge to be
00020           considered onscreen
00021       scrollDelta,
00022         - how much the container needs to be scrolled before onscreen
00023           and offscreen are recalculated
00024         - higher value means callbacks fired less frequently, but there
00025           are more of them when they are fired
00026       onscreenCallback,
00027         - called with the element that is now onscreen
00028       offscreenCallback
00029         - called with the element that is now offscreen
00030 
00031     ------------------------------------
00032 
00033     returns an object with the following properties:
00034         pauseMonitoringMutations
00035           - a function, that when called, disables mutation monitoring
00036         resumeMonitoringMutations
00037           - a function, that when called, enables mutation monitoring
00038         stop
00039           - a funtion, that when called, stops and cleans up the module instance
00040           - performs full visibility update if first argument is true
00041 
00042     ------------------------------------
00043 
00044     assumptions
00045       children flow top to bottom
00046       children are not absolutely positioned
00047       children do not move on their own
00048       children don't change size or become hidden
00049       the container is not statically positioned
00050       the container element only changes size on a resize event
00051 
00052     ------------------------------------
00053 
00054     performance
00055       - if you can, when adding a node to the dom, add children to the node
00056         before adding the node to the dom.
00057       - if you're certain a change won't change visibility, surround it with
00058         calls to pause and resume monitoring mutations.
00059 
00060     ------------------------------------
00061 
00062     high level overview of implementation
00063       more details can be found in comments closer to the implementation itself
00064 
00065       the central algorithm contains 2 stages
00066         1) visibility computation (recomputeFirstAndLastOnscreen)
00067           - computing the first and last visible element
00068         2) notification of offscreen / onscreen (callCallbacks)
00069           - notifying elements whose visibility has changed
00070 
00071         - the central algorithm is triggered whenever an event is received that
00072           suggests the visibility may have changed
00073             - onscroll
00074             - onresize
00075             - mutation of any child (mutation observer)
00076 
00077 ====================================*/

00210   function fixRange(mutations) {
00211     var numNodesAdded = 0;
00212     for (var i = 0; i < mutations.length; i++) {
00213       var mutation = mutations[i];
00214       if (mutation.addedNodes) {
00215         for (var j = 0; j < mutation.addedNodes.length; j++) {
00216           var child = mutation.addedNodes[j];
00217           if (child.nodeType === Node.ELEMENT_NODE &&
00218             child.tagName === tag)
00219             numNodesAdded += 1;
00220         }
00221       }
00222     }
00223 
00224     // if a node was deleted, we'll have to use its sibling, but what if its
00225     //  sibling is deleted? Then we need the mutation event for that sibling
00226     //  so we can get its sibling. Using WeakMaps allow us to use the deleted
00227     //  node as a key to get the mutation for that deleted node

00229     for (var i = 0; i < mutations.length; i++) {
00230       var mutation = mutations[i];

00232         for (var j = 0; j < mutation.removedNodes.length; j++) {
00233           var child = mutation.removedNodes[j];

00241     //    takes a possibly invalid index, and returns the next valid index
00242     //    - if the node at our index (target) was deleted, we have to find

00249       if (dir == FORWARDS)

00252         sibling = 'previousSibling';

00258       var limit = getLimit(dir);
00259       if (dir == FORWARDS && index > limit)
00260         index = limit;
00261       while (state.children[index] !== target && index !== limit)
00262         index += dir;

00264     }

00323   function computeBorderVisibilityIndex(guess, visibilityPosition, dir) {
00324     // move in direction past container
00325     var limit = getLimit(dir);
00326     while (relativeVisibilityPosition(container, state.children[guess]) !==
00327            visibilityPosition) {
00328       if (guess === limit)
00329         break;
00330       guess += dir;
00331     }
00332     // move back towards container
00333     limit = getLimit(-dir);
00334     while (relativeVisibilityPosition(container, state.children[guess]) ===
00335            visibilityPosition) {
00336       if (guess === limit)
00337         break;
00338       guess += -dir;
00339     }
00340     return guess;
00341   }

00352   //---------------------------------------
00353   //  callCallbacks
00354   //    - notifies elements between the new [first/last] onscreen and the old
00355   //      [first/last] onscreen as either onscreen or offscreen
00356   //
00357   //    3 possibilities
00358   //      1) there is no old [first/last] onscreen
00359   //          - notify between new firstOnscreen and new lastOnscreen as
00360   //            onscreen
00361   //      2) the new first->last onscreen and old first->last onscreen are
00362   //         disjoint
00363   //          - notify the old first->last onscreen as offscreen
00364   //          - notify the new first->last onscreen as onscreen
00365   //      3) the new first->last onscreen and the old first->last onscreen
00366   //         overlap
00367   //          - consider the two ranges (new first->old first), (new last->old
00368   //            last) separately
00369   //          - for each of the two ranges, compare the relative positions of
00370   //            the old and new
00371   function callCallbacks() {
00372 
00373     if (state.pendingCallCallbacksTimeoutId !== null) {
00374       window.clearTimeout(state.pendingCallCallbacksTimeoutId);
00375       state.pendingCallCallbacksTimeoutId = null;
00376     }
00377 
00378     if (state.firstChildIndex === null || state.lastChildIndex === null) {
00379       // nothing on screen, nothing to do
00380     }
00381     else if (
00382               state.prev.firstChildIndex === null ||
00383               state.prev.lastChildIndex === null
00384     ) {
00385       notifyOnscreenInRange(state.firstChildIndex, state.lastChildIndex);
00386     }
00387     else if (
00388       state.lastChildIndex < state.prev.firstChildIndex ||
00389       state.firstChildIndex > state.prev.lastChildIndex
00390     ) { // disjoint
00391       notifyOnscreenInRange(state.firstChildIndex,
00392                             state.lastChildIndex);
00393       notifyOffscreenInRange(state.prev.firstChildIndex,
00394                             state.prev.lastChildIndex);
00395     }
00396     else { // overlapping
00397       // onscreen at top
00398       if (state.firstChildIndex < state.prev.firstChildIndex)
00399         notifyOnscreenInRange(state.firstChildIndex,
00400                               state.prev.firstChildIndex - 1);
00401       // onscreen at bottom
00402       if (state.lastChildIndex > state.prev.lastChildIndex)
00403         notifyOnscreenInRange(state.prev.lastChildIndex + 1,
00404                               state.lastChildIndex);
00405       // offscreen at top
00406       if (state.firstChildIndex > state.prev.firstChildIndex)
00407         notifyOffscreenInRange(state.prev.firstChildIndex,
00408                                state.firstChildIndex - 1);
00409       // offscreen at bottom
00410       if (state.lastChildIndex < state.prev.lastChildIndex)
00411         notifyOffscreenInRange(state.lastChildIndex + 1,
00412                                state.prev.lastChildIndex);
00413     }
00414 
00415     state.prev.firstChildIndex = state.firstChildIndex;
00416     state.prev.lastChildIndex = state.lastChildIndex;
00417     state.prev.firstChild = state.firstChild;
00418     state.prev.lastChild = state.lastChild;
00419   }

00425   function relativeVisibilityPosition(container, child) {
00426 
00427     var scrollTop = container.scrollTop;
00428     var screenTop = scrollTop - scrollMargin;
00429     var screenBottom = scrollTop + container.clientHeight + scrollMargin;
00430 
00431     var childTop = child.offsetTop;
00432     var childBottom = childTop + child.offsetHeight;
00433     if (childBottom <= screenTop)
00434       return BEFORE;
00435     if (childTop >= screenBottom)
00436       return AFTER;
00437     return ON;
00438   }
