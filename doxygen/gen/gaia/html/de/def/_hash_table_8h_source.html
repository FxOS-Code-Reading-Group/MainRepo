00473 // Hash Policy
00474 //
00475 // A hash policy P for a hash table with key-type Key must provide:
00476 //  - a type |P::Lookup| to use to lookup table entries;
00477 //  - a static member function |P::hash| with signature
00478 //
00479 //      static js::HashNumber hash(Lookup)
00480 //
00481 //    to use to hash the lookup type; and
00482 //  - a static member function |P::match| with signature
00483 //
00484 //      static bool match(Key, Lookup)
00485 //
00486 //    to use to test equality of key and lookup values.
00487 //
00488 // Normally, Lookup = Key. In general, though, different values and types of
00489 // values can be used to lookup and store. If a Lookup value |l| is != to the
00490 // added Key value |k|, the user must ensure that |P::match(k,l)|. E.g.:
00491 //
00492 //   js::HashSet<Key, P>::AddPtr p = h.lookup(l);
00493 //   if (!p) {
00494 //     assert(P::match(k, l));  // must hold
00495 //     h.add(p, k);
00496 //   }
00497 
00498 // Pointer hashing policy that strips the lowest zeroBits when calculating the
00499 // hash to improve key distribution.
00500 template <typename Key, size_t zeroBits>
00501 struct PointerHasher
00502 {
00503     typedef Key Lookup;
00504     static HashNumber hash(const Lookup &l) {
00505         JS_ASSERT(!JS::IsPoisonedPtr(l));
00506         size_t word = reinterpret_cast<size_t>(l) >> zeroBits;
00507         JS_STATIC_ASSERT(sizeof(HashNumber) == 4);
00508 #if JS_BYTES_PER_WORD == 4
00509         return HashNumber(word);
00510 #else
00511         JS_STATIC_ASSERT(sizeof word == 8);
00512         return HashNumber((word >> 32) ^ word);
00513 #endif
00514     }
00515     static bool match(const Key &k, const Lookup &l) {
00516         JS_ASSERT(!JS::IsPoisonedPtr(k));
00517         JS_ASSERT(!JS::IsPoisonedPtr(l));
00518         return k == l;
00519     }
00520 };
00521 
00522 // Default hash policy: just use the 'lookup' value. This of course only
00523 // works if the lookup value is integral. HashTable applies ScrambleHashCode to
00524 // the result of the 'hash' which means that it is 'ok' if the lookup value is
00525 // not well distributed over the HashNumber domain.
00526 template <class Key>
00527 struct DefaultHasher
00528 {
00529     typedef Key Lookup;
00530     static HashNumber hash(const Lookup &l) {
00531         // Hash if can implicitly cast to hash number type.
00532         return l;
00533     }
00534     static bool match(const Key &k, const Lookup &l) {
00535         // Use builtin or overloaded operator==.
00536         return k == l;
00537     }
00538 };
00539 
00540 // Specialize hashing policy for pointer types. It assumes that the type is
00541 // at least word-aligned. For types with smaller size use PointerHasher.
00542 template <class T>
00543 struct DefaultHasher<T *> : PointerHasher<T *, mozilla::tl::FloorLog2<sizeof(void *)>::value>
00544 {};
00545 
00546 // For doubles, we can xor the two uint32s.
00547 template <>
00548 struct DefaultHasher<double>
00549 {
00550     typedef double Lookup;
00551     static HashNumber hash(double d) {
00552         JS_STATIC_ASSERT(sizeof(HashNumber) == 4);
00553         uint64_t u = mozilla::BitwiseCast<uint64_t>(d);
00554         return HashNumber(u ^ (u >> 32));
00555     }
00556     static bool match(double lhs, double rhs) {
00557         return mozilla::BitwiseCast<uint64_t>(lhs) == mozilla::BitwiseCast<uint64_t>(rhs);
00558     }
00559 };

01042     Entry &lookup(const Lookup &l, HashNumber keyHash, unsigned collisionBit) const
01043     {
01044         JS_ASSERT(isLiveHash(keyHash));
01045         JS_ASSERT(!(keyHash & sCollisionBit));
01046         JS_ASSERT(collisionBit == 0 || collisionBit == sCollisionBit);
01047         JS_ASSERT(table);
01048         METER(stats.searches++);
01049 
01050         // Compute the primary hash address.
01051         HashNumber h1 = hash1(keyHash);
01052         Entry *entry = &table[h1];
01053 
01054         // Miss: return space for a new entry.
01055         if (entry->isFree()) {
01056             METER(stats.misses++);
01057             return *entry;
01058         }
01059 
01060         // Hit: return entry.
01061         if (entry->matchHash(keyHash) && match(*entry, l)) {
01062             METER(stats.hits++);
01063             return *entry;
01064         }
01065 
01066         // Collision: double hash.
01067         DoubleHash dh = hash2(keyHash);
01068 
01069         // Save the first removed entry pointer so we can recycle later.
01070         Entry *firstRemoved = NULL;
01071 
01072         while(true) {
01073             if (JS_UNLIKELY(entry->isRemoved())) {
01074                 if (!firstRemoved)
01075                     firstRemoved = entry;
01076             } else {
01077                 entry->setCollision(collisionBit);
01078             }
01079 
01080             METER(stats.steps++);
01081             h1 = applyDoubleHash(h1, dh);
01082 
01083             entry = &table[h1];
01084             if (entry->isFree()) {
01085                 METER(stats.misses++);
01086                 return firstRemoved ? *firstRemoved : *entry;
01087             }
01088 
01089             if (entry->matchHash(keyHash) && match(*entry, l)) {
01090                 METER(stats.hits++);
01091                 return *entry;
01092             }
01093         }
01094     }

01097     //   1. the lookup is a lookupForAdd

01101     // from entries, which allows more flexible Lookup/Key types.

01103     {
01104         JS_ASSERT(!(keyHash & sCollisionBit));

01106         METER(stats.searches++);

01108         // We assume 'keyHash' has already been distributed.
01109 
01110         // Compute the primary hash address.
01111         HashNumber h1 = hash1(keyHash);
01112         Entry *entry = &table[h1];
01113 
01114         // Miss: return space for a new entry.
01115         if (!entry->isLive()) {
01116             METER(stats.misses++);
01117             return *entry;
01118         }
01119 
01120         // Collision: double hash.
01121         DoubleHash dh = hash2(keyHash);
01122 
01123         while(true) {
01124             JS_ASSERT(!entry->isRemoved());
01125             entry->setCollision();
01126 
01127             METER(stats.steps++);
01128             h1 = applyDoubleHash(h1, dh);
01129 
01130             entry = &table[h1];

01140     RebuildStatus changeTableSize(int deltaLog2)
01141     {
01142         // Look, but don't touch, until we succeed in getting new entry store.
01143         Entry *oldTable = table;
01144         uint32_t oldCap = capacity();
01145         uint32_t newLog2 = sHashBits - hashShift + deltaLog2;
01146         uint32_t newCapacity = JS_BIT(newLog2);
01147         if (newCapacity > sMaxCapacity) {
01148             this->reportAllocOverflow();
01149             return RehashFailed;
01150         }
01151 
01152         Entry *newTable = createTable(*this, newCapacity);
01153         if (!newTable)
01154             return RehashFailed;
01155 
01156         // We can't fail from here on, so update table parameters.
01157         setTableSizeLog2(newLog2);
01158         removedCount = 0;
01159         gen++;
01160         table = newTable;
01161 
01162         // Copy only live entries, leaving removed ones behind.
01163         for (Entry *src = oldTable, *end = src + oldCap; src < end; ++src) {
01164             if (src->isLive()) {
01165                 HashNumber hn = src->getKeyHash();
01166                 findFreeEntry(hn).setLive(hn, mozilla::Move(src->get()));
01167                 src->destroy();
01168             }
01169         }
01170 
01171         // All entries have been destroyed, no need to destroyTable.
01172         this->free_(oldTable);
01173         return Rehashed;
01174     }

01227     // Resize the table down to the largest capacity which doesn't underload the
01228     // table.  Since we call checkUnderloaded() on every remove, you only need
01229     // to call this after a bulk removal of items done without calling remove().
01230     void compactIfUnderloaded()
01231     {
01232         int32_t resizeLog2 = 0;
01233         uint32_t newCapacity = capacity();
01234         while (wouldBeUnderloaded(newCapacity, entryCount)) {
01235             newCapacity = newCapacity >> 1;
01236             resizeLog2--;

01244     // This is identical to changeTableSize(currentSize), but without requiring
01245     // a second table.  We do this by recycling the collision bits to tell us if
01246     // the element is already inserted or still waiting to be inserted.  Since
01247     // already-inserted elements win any conflicts, we get the same table as we
01248     // would have gotten through random insertion order.
01249     void rehashTableInPlace()
01250     {
01251         METER(stats.rehashes++);
01252         removedCount = 0;
01253         for (size_t i = 0; i < capacity(); ++i)
01254             table[i].unsetCollision();
01255 
01256         for (size_t i = 0; i < capacity();) {
01257             Entry *src = &table[i];
01258 
01259             if (!src->isLive() || src->hasCollision()) {
01260                 ++i;
01261                 continue;
01262             }
01263 
01264             HashNumber keyHash = src->getKeyHash();
01265             HashNumber h1 = hash1(keyHash);
01266             DoubleHash dh = hash2(keyHash);
01267             Entry *tgt = &table[h1];
01268             while (true) {
01269                 if (!tgt->hasCollision()) {
01270                     src->swap(tgt);
01271                     tgt->setCollision();
01272                     break;
01273                 }
01274

01281         // they are on no collision path. We have the option of setting the

01380     template <class U>
01381     bool add(AddPtr &p, const U &rhs)
01382     {
01383         mozilla::ReentrancyGuard g(*this);
01384         JS_ASSERT(mutationCount == p.mutationCount);
01385         JS_ASSERT(table);
01386         JS_ASSERT(!p.found());
01387         JS_ASSERT(!(p.keyHash & sCollisionBit));
01388 
01389         // Changing an entry from removed to live does not affect whether we
01390         // are overloaded and can be handled separately.
01391         if (p.entry_->isRemoved()) {
01392             METER(stats.addOverRemoved++);
01393             removedCount--;
01394             p.keyHash |= sCollisionBit;
01395         } else {
01396             // Preserve the validity of |p.entry_|.
01397             RebuildStatus status = checkOverloaded();
01398             if (status == RehashFailed)
01399                 return false;
01400             if (status == Rehashed)
01401                 p.entry_ = &findFreeEntry(p.keyHash);
01402         }
01403 
01404         p.entry_->setLive(p.keyHash, rhs);
01405         entryCount++;
01406         mutationCount++;
01407         return true;
01408     }
