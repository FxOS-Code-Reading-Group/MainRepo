00009 /*
00010  * plstr.h
00011  *
00012  * This header file exports the API to the NSPR portable library or string-
00013  * handling functions.  
00014  * 
00015  * This API was not designed as an "optimal" or "ideal" string library; it 
00016  * was based on the good ol' unix string.3 functions, and was written to
00017  *
00018  *  1) replace the libc functions, for cross-platform consistency, 
00019  *  2) complete the API on platforms lacking common functions (e.g., 
00020  *     strcase*), and
00021  *  3) to implement some obvious "closure" functions that I've seen
00022  *     people hacking around in our code.
00023  *
00024  * Point number three largely means that most functions have an "strn"
00025  * limited-length version, and all comparison routines have a non-case-
00026  * sensitive version available.
00027  */

00076 /*
00077  * PL_strncpyz
00078  *
00079  * Copies the source string into the destination buffer, up to and including 
00080  * the trailing '\0' or up but not including the max'th character, whichever 
00081  * comes first.  It does not (can not) verify that the destination buffer is
00082  * large enough.  The destination string is always terminated with a '\0',
00083  * unlike the traditional libc implementation.  It returns the "dest" argument.
00084  *
00085  * NOTE: If you call this with a source "abcdefg" and a max of 5, the 
00086  * destination will end up with "abcd\0" (i.e., its strlen length will be 4)!
00087  *
00088  * This means you can do this:
00089  *
00090  *     char buffer[ SOME_SIZE ];
00091  *     PL_strncpyz(buffer, src, sizeof(buffer));
00092  *
00093  * and the result will be properly terminated.
00094  */
00095 
00096 PR_EXTERN(char *)
00097 PL_strncpyz(char *dest, const char *src, PRUint32 max);

00408  * PL_strtok_r
00409  *
00410  * Splits the string s1 into tokens, separated by one or more characters
00411  * from the separator string s2.  The argument lasts points to a
00412  * user-supplied char * pointer in which PL_strtok_r stores information
00413  * for it to continue scanning the same string.
00414  *
00415  * In the first call to PL_strtok_r, s1 points to a string and the value
00416  * of *lasts is ignored.  PL_strtok_r returns a pointer to the first
00417  * token, writes '\0' into the character following the first token, and
00418  * updates *lasts.
00419  *
00420  * In subsequent calls, s1 is null and lasts must stay unchanged from the
00421  * previous call.  The separator string s2 may be different from call to
00422  * call.  PL_strtok_r returns a pointer to the next token in s1.  When no
00423  * token remains in s1, PL_strtok_r returns null.
00424  */
