00029 //
00030 // XEvent is defined as a union of all more specific X*Events.
00031 // Luckily, as of xorg 1.6.0 / X protocol 11 rev 0, the only pointer
00032 // field contained in any of these specific X*Event structs is a
00033 // |Display*|.  So to simplify serializing these XEvents, we make the
00034 // 
00035 // ********** XXX ASSUMPTION XXX **********
00036 //
00037 // that the process to which the event is forwarded shares the same
00038 // display as the process on which the event originated.
00039 //
00040 // With this simplification, serialization becomes a simple memcpy to
00041 // the output stream.  Deserialization starts as just a memcpy from
00042 // the input stream, BUT we then have to write the correct |Display*|
00043 // into the right field of each X*Event that contains one.
00044 //
00045 
00046 namespace IPC {
00047 
00048 template <>
00049 struct ParamTraits<mozilla::plugins::NPRemoteEvent>     // synonym for XEvent
00050 {
00051     typedef mozilla::plugins::NPRemoteEvent paramType;
00052 
00053     static void Write(Message* aMsg, const paramType& aParam)
00054     {
00055         aMsg->WriteBytes(&aParam, sizeof(paramType));
00056     }
00057 
00058     static bool Read(const Message* aMsg, void** aIter, paramType* aResult)
00059     {
00060         const char* bytes = 0;
00061 
00062         if (!aMsg->ReadBytes(aIter, &bytes, sizeof(paramType))) {
00063             return false;
00064         }
00065 
00066         memcpy(aResult, bytes, sizeof(paramType));
00067 
00068 #ifdef MOZ_X11
00069         SetXDisplay(aResult->event);
00070 #endif
00071         return true;
00072     }
00073

00075     {
00076         // TODO
