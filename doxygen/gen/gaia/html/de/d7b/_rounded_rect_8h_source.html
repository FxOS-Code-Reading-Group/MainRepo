00018         // deflate the internal rect
00019         rect.x += aLeftWidth;
00020         rect.y += aTopWidth;
00021         rect.width = std::max(0., rect.width - aLeftWidth - aRightWidth);
00022         rect.height = std::max(0., rect.height - aTopWidth - aBottomWidth);
00023 
00024         corners.sizes[NS_CORNER_TOP_LEFT].width  = std::max(0., corners.sizes[NS_CORNER_TOP_LEFT].width - aLeftWidth);
00025         corners.sizes[NS_CORNER_TOP_LEFT].height = std::max(0., corners.sizes[NS_CORNER_TOP_LEFT].height - aTopWidth);
