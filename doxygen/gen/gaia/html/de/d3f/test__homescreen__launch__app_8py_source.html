00042     def test_launch_app(self):
00043         # We iterate through the homescreen pages until we find the app icon visible
00044         # It can be on different screens depending on what is packaged with the build
00045         while True:
00046             if self.is_element_present(*self._visible_icon_locator):
00047                 break
00048             if self._homescreen_has_more_pages():
00049                 self._go_to_next_page()
00050             else:
00051                 break
00052 
00053         # click icon and wait for h1 element displayed
00054         self.marionette.find_element(*self._visible_icon_locator).tap()
00055         self.marionette.switch_to_frame()
00056         iframe = self.marionette.find_element(*self._app_locator)
00057         self.marionette.switch_to_frame(iframe)
00058         self.wait_for_element_displayed(*self._header_locator, timeout=20)
00059         self.assertEqual(self.marionette.find_element(*self._header_locator).text, TITLE)

00062         self.marionette.execute_script('window.wrappedJSObject.GridManager.goToNextPage()')
00063         self.wait_for_condition(lambda m: m.find_element('tag name', 'body')
00064             .get_attribute('data-transitioning') != 'true')

00069             var pageHelper = window.wrappedJSObject.GridManager.pageHelper;
00070             return pageHelper.getCurrentPageNumber() < (pageHelper.getTotalPagesNumber() - 1);""")
