00300     mGLContext->fDrawArrays(LOCAL_GL_TRIANGLE_STRIP, 0, 4);
00301     mGLContext->fDisableVertexAttribArray(aVertAttribIndex);

00304       mGLContext->fDisableVertexAttribArray(aTexCoordAttribIndex);
