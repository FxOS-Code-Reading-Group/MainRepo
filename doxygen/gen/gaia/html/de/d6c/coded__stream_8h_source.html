00035 // This file contains the CodedInputStream and CodedOutputStream classes,
00036 // which wrap a ZeroCopyInputStream or ZeroCopyOutputStream, respectively,
00037 // and allow you to read or write individual pieces of data in various
00038 // formats.  In particular, these implement the varint encoding for
00039 // integers, a simple variable-length encoding in which smaller numbers
00040 // take fewer bytes.
00041 //
00042 // Typically these classes will only be used internally by the protocol
00043 // buffer library in order to encode and decode protocol buffers.  Clients
00044 // of the library only need to know about this class if they wish to write
00045 // custom message parsing or serialization procedures.
00046 //
00047 // CodedOutputStream example:
00048 //   // Write some data to "myfile".  First we write a 4-byte "magic number"
00049 //   // to identify the file type, then write a length-delimited string.  The
00050 //   // string is composed of a varint giving the length followed by the raw
00051 //   // bytes.
00052 //   int fd = open("myfile", O_WRONLY);
00053 //   ZeroCopyOutputStream* raw_output = new FileOutputStream(fd);
00054 //   CodedOutputStream* coded_output = new CodedOutputStream(raw_output);
00055 //
00056 //   int magic_number = 1234;
00057 //   char text[] = "Hello world!";
00058 //   coded_output->WriteLittleEndian32(magic_number);
00059 //   coded_output->WriteVarint32(strlen(text));
00060 //   coded_output->WriteRaw(text, strlen(text));
00061 //
00062 //   delete coded_output;
00063 //   delete raw_output;
00064 //   close(fd);
00065 //
00066 // CodedInputStream example:
00067 //   // Read a file created by the above code.
00068 //   int fd = open("myfile", O_RDONLY);
00069 //   ZeroCopyInputStream* raw_input = new FileInputStream(fd);
00070 //   CodedInputStream coded_input = new CodedInputStream(raw_input);
00071 //
00072 //   coded_input->ReadLittleEndian32(&magic_number);
00073 //   if (magic_number != 1234) {
00074 //     cerr << "File not in expected format." << endl;
00075 //     return;
00076 //   }
00077 //
00078 //   uint32 size;
00079 //   coded_input->ReadVarint32(&size);
00080 //
00081 //   char* text = new char[size + 1];
00082 //   coded_input->ReadRaw(buffer, size);
00083 //   text[size] = '\0';
00084 //
00085 //   delete coded_input;
00086 //   delete raw_input;
00087 //   close(fd);
00088 //
00089 //   cout << "Text is: " << text << endl;
00090 //   delete [] text;
00091 //
00092 // For those who are interested, varint encoding is defined as follows:
00093 //
00094 // The encoding operates on unsigned integers of up to 64 bits in length.
00095 // Each byte of the encoded value has the format:
00096 // * bits 0-6: Seven bits of the number being encoded.
00097 // * bit 7: Zero if this is the last byte in the encoding (in which
00098 //   case all remaining bits of the number are zero) or 1 if
00099 //   more bytes follow.
00100 // The first byte contains the least-significant 7 bits of the number, the
00101 // second byte (if present) contains the next-least-significant 7 bits,
00102 // and so on.  So, the binary number 1011000101011 would be encoded in two
00103 // bytes as "10101011 00101100".
00104 //
00105 // In theory, varint could be used to encode integers of any length.
00106 // However, for practicality we set a limit at 64 bits.  The maximum encoded
00107 // length of a number is thus 10 bytes.

00316   // Total Bytes Limit -----------------------------------------------
00317   // To prevent malicious users from sending excessively large messages
00318   // and causing integer overflows or memory exhaustion, CodedInputStream
00319   // imposes a hard limit on the total number of bytes it will read.
00320 
00321   // Sets the maximum number of bytes that this CodedInputStream will read
00322   // before refusing to continue.  To prevent integer overflows in the
00323   // protocol buffers implementation, as well as to prevent servers from
00324   // allocating enormous amounts of memory to hold parsed messages, the
00325   // maximum message length should be limited to the shortest length that
00326   // will not harm usability.  The theoretical shortest message that could
00327   // cause integer overflows is 512MB.  The default limit is 64MB.  Apps
00328   // should set shorter limits if possible.  If warning_threshold is not -1,
00329   // a warning will be printed to stderr after warning_threshold bytes are
00330   // read.  An error will always be printed to stderr if the limit is
00331   // reached.
00332   //
00333   // This is unrelated to PushLimit()/PopLimit().
00334   //
00335   // Hint:  If you are reading this because your program is printing a
00336   //   warning about dangerously large protocol messages, you may be
00337   //   confused about what to do next.  The best option is to change your
00338   //   design such that excessively large messages are not necessary.
00339   //   For example, try to design file formats to consist of many small
00340   //   messages rather than a single large one.  If this is infeasible,
00341   //   you will need to increase the limit.  Chances are, though, that
00342   //   your code never constructs a CodedInputStream on which the limit
00343   //   can be set.  You probably parse messages by calling things like
00344   //   Message::ParseFromString().  In this case, you will need to change
00345   //   your code to instead construct some sort of ZeroCopyInputStream
00346   //   (e.g. an ArrayInputStream), construct a CodedInputStream around
00347   //   that, then call Message::ParseFromCodedStream() instead.  Then
00348   //   you can adjust the limit.  Yes, it's more work, but you're doing
00349   //   something unusual.
00350   void SetTotalBytesLimit(int total_bytes_limit, int warning_threshold);

00368   // Extension Registry ----------------------------------------------
00369   // ADVANCED USAGE:  99.9% of people can ignore this section.
00370   //
00371   // By default, when parsing extensions, the parser looks for extension
00372   // definitions in the pool which owns the outer message's Descriptor.
00373   // However, you may call SetExtensionRegistry() to provide an alternative
00374   // pool instead.  This makes it possible, for example, to parse a message
00375   // using a generated class, but represent some extensions using
00376   // DynamicMessage.
00377 
00378   // Set the pool used to look up extensions.  Most users do not need to call
00379   // this as the correct pool will be chosen automatically.
00380   //
00381   // WARNING:  It is very easy to misuse this.  Carefully read the requirements
00382   //   below.  Do not use this unless you are sure you need it.  Almost no one
00383   //   does.
00384   //
00385   // Let's say you are parsing a message into message object m, and you want
00386   // to take advantage of SetExtensionRegistry().  You must follow these
00387   // requirements:
00388   //
00389   // The given DescriptorPool must contain m->GetDescriptor().  It is not
00390   // sufficient for it to simply contain a descriptor that has the same name
00391   // and content -- it must be the *exact object*.  In other words:
00392   //   assert(pool->FindMessageTypeByName(m->GetDescriptor()->full_name()) ==
00393   //          m->GetDescriptor());
00394   // There are two ways to satisfy this requirement:
00395   // 1) Use m->GetDescriptor()->pool() as the pool.  This is generally useless
00396   //    because this is the pool that would be used anyway if you didn't call
00397   //    SetExtensionRegistry() at all.
00398   // 2) Use a DescriptorPool which has m->GetDescriptor()->pool() as an
00399   //    "underlay".  Read the documentation for DescriptorPool for more
00400   //    information about underlays.
00401   //
00402   // You must also provide a MessageFactory.  This factory will be used to
00403   // construct Message objects representing extensions.  The factory's
00404   // GetPrototype() MUST return non-NULL for any Descriptor which can be found
00405   // through the provided pool.
00406   //
00407   // If the provided factory might return instances of protocol-compiler-
00408   // generated (i.e. compiled-in) types, or if the outer message object m is
00409   // a generated type, then the given factory MUST have this property:  If
00410   // GetPrototype() is given a Descriptor which resides in
00411   // DescriptorPool::generated_pool(), the factory MUST return the same
00412   // prototype which MessageFactory::generated_factory() would return.  That
00413   // is, given a descriptor for a generated type, the factory must return an
00414   // instance of the generated class (NOT DynamicMessage).  However, when
00415   // given a descriptor for a type that is NOT in generated_pool, the factory
00416   // is free to return any implementation.
00417   //
00418   // The reason for this requirement is that generated sub-objects may be
00419   // accessed via the standard (non-reflection) extension accessor methods,
00420   // and these methods will down-cast the object to the generated class type.
00421   // If the object is not actually of that type, the results would be undefined.
00422   // On the other hand, if an extension is not compiled in, then there is no
00423   // way the code could end up accessing it via the standard accessors -- the
00424   // only way to access the extension is via reflection.  When using reflection,
00425   // DynamicMessage and generated messages are indistinguishable, so it's fine
00426   // if these objects are represented using DynamicMessage.
00427   //
00428   // Using DynamicMessageFactory on which you have called
00429   // SetDelegateToGeneratedFactory(true) should be sufficient to satisfy the
00430   // above requirement.
00431   //
00432   // If either pool or factory is NULL, both must be NULL.
00433   //
00434   // Note that this feature is ignored when parsing "lite" messages as they do
00435   // not have descriptors.
00436   void SetExtensionRegistry(DescriptorPool* pool, MessageFactory* factory);
00437

00545 // Class which encodes and writes binary data which is composed of varint-
00546 // encoded integers and fixed-width pieces.  Wraps a ZeroCopyOutputStream.
00547 // Most users will not need to deal with CodedOutputStream.
00548 //
00549 // Most methods of CodedOutputStream which return a bool return false if an
00550 // underlying I/O error occurs.  Once such a failure occurs, the
00551 // CodedOutputStream is broken and is no longer useful. The Write* methods do
00552 // not return the stream status, but will invalidate the stream if an error
00553 // occurs. The client can probe HadError() to determine the status.
00554 //
00555 // Note that every method of CodedOutputStream which writes some data has
00556 // a corresponding static "ToArray" version. These versions write directly
00557 // to the provided buffer, returning a pointer past the last written byte.
00558 // They require that the buffer has sufficient capacity for the encoded data.
00559 // This allows an optimization where we check if an output stream has enough
00560 // space for an entire message before we start writing and, if there is, we
00561 // call only the ToArray methods to avoid doing bound checks for each
00562 // individual value.
00563 // i.e., in the example above:
00564 //
00565 //   CodedOutputStream coded_output = new CodedOutputStream(raw_output);
00566 //   int magic_number = 1234;
00567 //   char text[] = "Hello world!";
00568 //
00569 //   int coded_size = sizeof(magic_number) +
00570 //                    CodedOutputStream::VarintSize32(strlen(text)) +
00571 //                    strlen(text);
00572 //
00573 //   uint8* buffer =
00574 //       coded_output->GetDirectBufferForNBytesAndAdvance(coded_size);
00575 //   if (buffer != NULL) {
00576 //     // The output stream has enough space in the buffer: write directly to
00577 //     // the array.
00578 //     buffer = CodedOutputStream::WriteLittleEndian32ToArray(magic_number,
00579 //                                                            buffer);
00580 //     buffer = CodedOutputStream::WriteVarint32ToArray(strlen(text), buffer);
00581 //     buffer = CodedOutputStream::WriteRawToArray(text, strlen(text), buffer);
00582 //   } else {
00583 //     // Make bound-checked writes, which will ask the underlying stream for
00584 //     // more space as needed.
00585 //     coded_output->WriteLittleEndian32(magic_number);
00586 //     coded_output->WriteVarint32(strlen(text));
00587 //     coded_output->WriteRaw(text, strlen(text));
00588 //   }
00589 //
00590 //   delete coded_output;
