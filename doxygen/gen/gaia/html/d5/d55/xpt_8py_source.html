00177     @staticmethod
00178     def read(typelib, map, data_pool, offset):
00179         """
00180         Read a TypeDescriptor at |offset| from the mmaped file |map| with
00181         data pool offset |data_pool|. Returns (Type, next offset),
00182         where |next offset| is an offset suitable for reading the data
00183         following this TypeDescriptor.
00184         
00185         """
00186         start = data_pool + offset - 1
00187         (data,) = Type._prefixdescriptor.unpack_from(map, start)
00188         # first three bits are the flags
00189         flags = data & 0xE0
00190         flags = Type.decodeflags(flags)
00191         # last five bits is the tag
00192         tag = data & 0x1F
00193         offset += Type._prefixdescriptor.size
00194         t = None
00195         if tag <= Type.Tags.wchar_t_ptr or tag >= Type.Tags.UTF8String:
00196             t = SimpleType.get(data, tag, flags)
00197         elif tag == Type.Tags.Interface:
00198             t, offset = InterfaceType.read(typelib, map, data_pool, offset, flags)
00199         elif tag == Type.Tags.InterfaceIs:
00200             t, offset = InterfaceIsType.read(typelib, map, data_pool, offset, flags)
00201         elif tag == Type.Tags.Array:
00202             t, offset = ArrayType.read(typelib, map, data_pool, offset, flags)
00203         elif tag == Type.Tags.StringWithSize:
00204             t, offset = StringWithSizeType.read(typelib, map, data_pool, offset, flags)
00205         elif tag == Type.Tags.WideStringWithSize:
00206             t, offset = WideStringWithSizeType.read(typelib, map, data_pool, offset, flags)
00207         return t, offset

00528     @staticmethod
00529     def decodeflags(byte):
00530         """
00531         Given |byte|, an unsigned uint8 containing flag bits,

00545                 }

00548         """
00549         Encode the flags of this Param. Return a byte suitable for
00550         writing to a typelib file.
00551 
00552         """
00553         flags = 0
00554         if self.in_:
00555             flags |= 0x80
00556         if self.out:
00557             flags |= 0x40
00558         if self.retval:
00559             flags |= 0x20
00560         if self.shared:
00561             flags |= 0x10
00562         if self.dipper:
00563             flags |= 0x08
00564         if self.optional:
00565             flags |= 0x04
00566         return flags

00907     def read_descriptor(self, typelib, map, data_pool):
00908         offset = self._descriptor_offset
00909         if offset == 0:
00910             return
00911         start = data_pool + offset - 1
00912         parent, num_methods = Interface._descriptorstart.unpack_from(map, start)
00913         if parent > 0 and parent <= len(typelib.interfaces):
00914             self.parent = typelib.interfaces[parent - 1]
00915         # Read methods
00916         offset += Interface._descriptorstart.size
00917         for i in range(num_methods):
00918             m, offset = Method.read(typelib, map, data_pool, offset)
00919             self.methods.append(m)
00920         # Read constants
00921         start = data_pool + offset - 1
00922         (num_constants, ) = struct.unpack_from(">H", map, start)
00923         offset = offset + struct.calcsize(">H")
00924         for i in range(num_constants):
00925             c, offset = Constant.read(typelib, map, data_pool, offset)
00926             self.constants.append(c)
00927         # Read flags
00928         start = data_pool + offset - 1
00929         (flags, ) = struct.unpack_from(">B", map, start)
00930         offset = offset + struct.calcsize(">B")
00931         # only the first two bits are flags
00932         flags &= 0xE0
00933         if flags & 0x80:
00934             self.scriptable = True
00935         if flags & 0x40:
00936             self.function = True
00937         if flags & 0x20:
00938             self.builtinclass = True
00939         self.resolved = True

01004 class Typelib(object):
01005     """
01006     A typelib represents one entire typelib file and all the interfaces
01007     referenced within, whether defined entirely within the typelib or
01008     merely referenced by name or IID.
01009 
01010     Typelib objects may be instantiated directly and populated with data,
01011     or the static Typelib.read method may be called to read one from a file.
01012 
01013     """
01014     _header = struct.Struct(">16sBBHIII")
01015 
01016     def __init__(self, version=TYPELIB_VERSION, interfaces=[], annotations=[]):
01017         """
01018         Instantiate a new Typelib.
01019 
01020         """
01021         self.version = version
01022         self.interfaces = list(interfaces)
01023         self.annotations = list(annotations)
01024         self.filename = None
01025 
01026     @staticmethod
01027     def iid_to_string(iid):
01028         """
01029         Convert a 16-byte IID into a UUID string.
01030 
01031         """
01032         def hexify(s):
01033             return ''.join(["%02x" % ord(x) for x in s])
01034 
01035         return "%s-%s-%s-%s-%s" % (hexify(iid[:4]), hexify(iid[4:6]),
01036                                    hexify(iid[6:8]), hexify(iid[8:10]),
01037                                    hexify(iid[10:]))
01038 
01039     @staticmethod
01040     def string_to_iid(iid_str):
01041         """
01042         Convert a UUID string into a 16-byte IID.
01043 
01044         """
01045         s = iid_str.replace('-','')
01046         return ''.join([chr(int(s[i:i+2], 16)) for i in range(0, len(s), 2)])
01047 
01048     @staticmethod
01049     def read_string(map, data_pool, offset):
01050         if offset == 0:
01051             return ""
01052         sz = map.find('\x00', data_pool + offset - 1)
01053         if sz == -1:
01054             return ""
01055         return map[data_pool + offset - 1:sz]

01057     @staticmethod
01058     def read(input_file):
01059         """
01060         Read a typelib from |input_file| and return
01061         the constructed Typelib object. |input_file| can be a filename
01062         or a file-like object.
01063 
01064         """
01065         filename = ""
01066         data = None
01067         expected_size = None
01068         if isinstance(input_file, basestring):
01069             filename = input_file
01070             with open(input_file, "rb") as f:
01071                 st = os.fstat(f.fileno())
01072                 data = f.read(st.st_size)
01073                 expected_size = st.st_size
01074         else:
01075             data = input_file.read()
01076 
01077         (magic,
01078          major_ver,
01079          minor_ver,
01080          num_interfaces,
01081          file_length,
01082          interface_directory_offset,
01083          data_pool_offset) = Typelib._header.unpack_from(data)
01084         if magic != XPT_MAGIC:
01085             raise FileFormatError, "Bad magic: %s" % magic
01086         xpt = Typelib((major_ver, minor_ver))
01087         xpt.filename = filename
01088         if expected_size and file_length != expected_size:
01089             raise FileFormatError, "File is of wrong length, got %d bytes, expected %d" % (expected_size, file_length)
01090         #XXX: by spec this is a zero-based file offset. however,
01091         # the xpt_xdr code always subtracts 1 from data offsets
01092         # (because that's what you do in the data pool) so it
01093         # winds up accidentally treating this as 1-based.
01094         # Filed as: https://bugzilla.mozilla.org/show_bug.cgi?id=575343
01095         interface_directory_offset -= 1
01096         # make a half-hearted attempt to read Annotations,
01097         # since XPIDL doesn't produce any anyway.
01098         start = Typelib._header.size
01099         (anno, ) = struct.unpack_from(">B", data, start)
01100         islast = anno & 0x80
01101         tag = anno & 0x7F
01102         if tag == 0: # EmptyAnnotation
01103             xpt.annotations.append(None)
01104         # We don't bother handling PrivateAnnotations or anything
01105 
01106         for i in range(num_interfaces):
01107             # iid, name, namespace, interface_descriptor
01108             start = interface_directory_offset + i * Interface._direntry.size
01109             ide = Interface._direntry.unpack_from(data, start)
01110             iid = Typelib.iid_to_string(ide[0])
01111             name = Typelib.read_string(data, data_pool_offset, ide[1])
01112             namespace = Typelib.read_string(data, data_pool_offset, ide[2])
01113             iface = Interface(name, iid, namespace)
01114             iface._descriptor_offset = ide[3]
01115             iface.xpt_filename = xpt.filename
01116             xpt.interfaces.append(iface)
01117         for iface in xpt.interfaces:
01118             iface.read_descriptor(xpt, data, data_pool_offset)

01124     def _sanityCheck(self):
01125         """
01126         Check certain assumptions about data contained in this typelib.
01127         Sort the interfaces array by IID, check that all interfaces
01128         referenced by methods exist in the array.
01129 
01130         """
01131         self.interfaces.sort()
01132         for i in self.interfaces:
01133             if i.parent and i.parent not in self.interfaces:
01134                 raise DataError, "Interface %s has parent %s not present in typelib!" % (i.name, i.parent.name)
01135             for m in i.methods:
01136                 for n, p in enumerate(m.params):
01137                     if isinstance(p, InterfaceType) and \
01138                         p.iface not in self.interfaces:
01139                         raise DataError, "Interface method %s::%s, parameter %d references interface %s not present in typelib!" % (i.name, m.name, n, p.iface.name)
01140                 if isinstance(m.result, InterfaceType) and m.result.iface not in self.interfaces:
01141                     raise DataError, "Interface method %s::%s, result references interface %s not present in typelib!" % (i.name, m.name, m.result.iface.name)

01143     def writefd(self, fd):
01144         # write out space for a header + one empty annotation,
01145         # padded to 4-byte alignment.
01146         headersize = (Typelib._header.size + 1)
01147         if headersize % 4:
01148             headersize += 4 - headersize % 4
01149         fd.write("\x00" * headersize)
01150         # save this offset, it's the interface directory offset.
01151         interface_directory_offset = fd.tell()
01152         # write out space for an interface directory
01153         fd.write("\x00" * Interface._direntry.size * len(self.interfaces))
01154         # save this offset, it's the data pool offset.
01155         data_pool_offset = fd.tell()
01156         # write out all the interface descriptors to the data pool
01157         for i in self.interfaces:
01158             i.write_names(fd, data_pool_offset)
01159             i.write(self, fd, data_pool_offset)
01160         # now, seek back and write the header
01161         file_len = fd.tell()
01162         fd.seek(0)
01163         fd.write(Typelib._header.pack(XPT_MAGIC,
01164                                       TYPELIB_VERSION[0],
01165                                       TYPELIB_VERSION[1],
01166                                       len(self.interfaces),
01167                                       file_len,
01168                                       interface_directory_offset,
01169                                       data_pool_offset))
01170         # write an empty annotation
01171         fd.write(struct.pack(">B", 0x80))
01172         # now write the interface directory
01173         #XXX: bug-compatible with existing xpt lib, put it one byte
01174         # ahead of where it's supposed to be.
01175         fd.seek(interface_directory_offset - 1)
01176         for i in self.interfaces:
01177             i.write_directory_entry(fd)

01253 def xpt_link(inputs):
01254     """
01255     Link all of the xpt files in |inputs| together and return the result
01256     as a Typelib object. All entries in inputs may be filenames or
01257     file-like objects.
01258 
01259     """
01260     def read_input(i):
01261         if isinstance(i, Typelib):
01262             return i
01263         return Typelib.read(i)
01264 
01265     if not inputs:
01266         print >>sys.stderr, "Usage: xpt_link <destination file> <input files>"
01267         return None
01268     # This is the aggregate list of interfaces.
01269     interfaces = []
01270     # This will be a dict of replaced interface -> replaced with
01271     # containing interfaces that were replaced with interfaces from
01272     # another typelib, and the interface that replaced them.
01273     merged_interfaces = {}
01274     for f in inputs:
01275         t = read_input(f)
01276         interfaces.extend(t.interfaces)
01277     # Sort interfaces by name so we can merge adjacent duplicates
01278     interfaces.sort(key=operator.attrgetter('name'))
01279 
01280     Result = enum('Equal',     # Interfaces the same, doesn't matter
01281                   'NotEqual',  # Interfaces differ, keep both
01282                   'KeepFirst', # Replace second interface with first
01283                   'KeepSecond')# Replace first interface with second
01284         
01285     def compare(i, j):
01286         """
01287         Compare two interfaces, determine if they're equal or
01288         completely different, or should be merged (and indicate which
01289         one to keep in that case).
01290 
01291         """
01292         if i == j:
01293             # Arbitrary, just pick one
01294             return Result.Equal
01295         if i.name != j.name:
01296             if i.iid == j.iid and i.iid != Interface.UNRESOLVED_IID:
01297                 # Same IID but different names: raise an exception.
01298                 raise DataError, \
01299                     "Typelibs contain definitions of interface %s" \
01300                     " with different names (%s (%s) vs %s (%s))!" % \
01301                     (i.iid, i.name, i.xpt_filename, j.name, j.xpt_filename)
01302             # Otherwise just different interfaces.
01303             return Result.NotEqual
01304         # Interfaces have the same name, so either they need to be merged
01305         # or there's a data error. Sort out which one to keep
01306         if i.resolved != j.resolved:
01307             # prefer resolved interfaces over unresolved
01308             if j.resolved:
01309                 assert i.iid == j.iid or i.iid == Interface.UNRESOLVED_IID
01310                 # keep j
01311                 return Result.KeepSecond
01312             else:
01313                 assert i.iid == j.iid or j.iid == Interface.UNRESOLVED_IID
01314                 # replace j with i
01315                 return Result.KeepFirst
01316         elif i.iid != j.iid:
01317             # Prefer unresolved interfaces with valid IIDs
01318             if j.iid == Interface.UNRESOLVED_IID:
01319                 # replace j with i
01320                 assert not j.resolved
01321                 return Result.KeepFirst
01322             elif i.iid == Interface.UNRESOLVED_IID:
01323                 # keep j
01324                 assert not i.resolved
01325                 return Result.KeepSecond
01326             else:
01327                 # Same name but different IIDs: raise an exception.
01328                 raise DataError, \
01329                     "Typelibs contain definitions of interface %s" \
01330                                 " with different IIDs (%s (%s) vs %s (%s))!" % \
01331                                 (i.name, i.iid, i.xpt_filename, \
01332                                  j.iid, j.xpt_filename)
01333         raise DataError, "No idea what happened here: %s:%s (%s), %s:%s (%s)" % \
01334             (i.name, i.iid, i.xpt_filename, j.name, j.iid, j.xpt_filename)
01335     
01336     # Compare interfaces pairwise to find duplicates that should be merged.
01337     i = 1
01338     while i < len(interfaces):
01339         res = compare(interfaces[i-1], interfaces[i])
01340         if res == Result.NotEqual:
01341             i += 1
01342         elif res == Result.Equal:
01343             # Need to drop one but it doesn't matter which
01344             del interfaces[i]
01345         elif res == Result.KeepFirst:
01346             merged_interfaces[interfaces[i]] = interfaces[i-1]
01347             del interfaces[i]
01348         elif res == Result.KeepSecond:
01349             merged_interfaces[interfaces[i-1]] = interfaces[i]
01350             del interfaces[i-1]
01351     
01352     # Now fixup any merged interfaces
01353     def checkType(t):
01354         if isinstance(t, InterfaceType) and t.iface in merged_interfaces:
01355             t.iface = merged_interfaces[t.iface]
01356         elif isinstance(t, ArrayType) and \
01357              isinstance(t.element_type, InterfaceType) and \
01358              t.element_type.iface in merged_interfaces:
01359             t.element_type.iface = merged_interfaces[t.element_type.iface]
01360 
01361     for i in interfaces:
01362         # Replace parent references
01363         if i.parent in merged_interfaces:
01364             i.parent = merged_interfaces[i.parent]
01365         for m in i.methods:
01366             # Replace InterfaceType params and return values
01367             checkType(m.result.type)
01368             for p in m.params:
01369                 checkType(p.type)
01370     # Re-sort interfaces (by IID)
01371     interfaces.sort()
01372     return Typelib(interfaces=interfaces)
