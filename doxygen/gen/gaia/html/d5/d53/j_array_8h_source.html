00030 struct staticJArray {
00031   const T* arr;

00034   T& operator[] (L const index) { return ((T*)arr)[index]; }
00035   L binarySearch(T const elem) {
00036     L lo = 0;
00037     L hi = length - 1;
00038     while (lo <= hi) {
00039       L mid = (lo + hi) / 2;
00040       if (arr[mid] > elem) {
00041         hi = mid - 1;
00042       } else if (arr[mid] < elem) {
00043         lo = mid + 1;
00044       } else {
00045         return mid;
00046       }
00047     }
00048     return -1;
00049   }
00050 };
