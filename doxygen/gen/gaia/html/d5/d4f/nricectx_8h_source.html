00106   nsresult Init(const std::string& addr, uint16_t port) {
00107     PRStatus status = PR_StringToNetAddr(addr.c_str(), &addr_);
00108     if (status == PR_SUCCESS) {
00109       // Parseable as an address
00110       addr_.inet.port = PR_htons(port);
00111       port_ = port;
00112       has_addr_ = true;
00113       return NS_OK;
00114     }
00115     else if (addr.size() < 256) {
00116       // Apparently this is a hostname.
00117       host_ = addr;
00118       port_ = port;
00119       has_addr_ = false;
00120       return NS_OK;
00121     }
00122 
00123     return NS_ERROR_FAILURE;
00124   }
