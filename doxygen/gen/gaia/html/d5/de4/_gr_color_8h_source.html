00023 // shift amount to assign a component to a GrColor int
00024 // These shift values are chosen for compatibility with GL attrib arrays
00025 // ES doesn't allow BGRA vertex attrib order so if they were not in this order
00026 // we'd have to swizzle in shaders. Note the assumption that the cpu is little
00027 // endian.
00028 #define GrColor_SHIFT_R     0
00029 #define GrColor_SHIFT_G     8
00030 #define GrColor_SHIFT_B     16
00031 #define GrColor_SHIFT_A     24
00032 
00036 static inline GrColor GrColorPackRGBA(unsigned r, unsigned g,
00037                                       unsigned b, unsigned a) {
00038     GrAssert((uint8_t)r == r);
00039     GrAssert((uint8_t)g == g);
00040     GrAssert((uint8_t)b == b);
00041     GrAssert((uint8_t)a == a);
00042     return  (r << GrColor_SHIFT_R) |
00043             (g << GrColor_SHIFT_G) |
00044             (b << GrColor_SHIFT_B) |
00045             (a << GrColor_SHIFT_A);
00046 }

00062 static inline void GrColorToRGBAFloat(GrColor color, float rgba[4]) {
00063     static const float ONE_OVER_255 = 1.f / 255.f;
00064     rgba[0] = GrColorUnpackR(color) * ONE_OVER_255;

00066     rgba[2] = GrColorUnpackB(color) * ONE_OVER_255;
