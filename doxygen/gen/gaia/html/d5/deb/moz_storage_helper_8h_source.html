00036 class mozStorageTransactionBase
00037 {
00038 public:
00039   mozStorageTransactionBase(T* aConnection,
00040                             bool aCommitOnComplete,
00041                             int32_t aType = mozIStorageConnection::TRANSACTION_DEFERRED)
00042     : mConnection(aConnection),
00043       mHasTransaction(false),
00044       mCommitOnComplete(aCommitOnComplete),
00045       mCompleted(false)
00046   {
00047     // We won't try to get a transaction if one is already in progress.
00048     if (mConnection)
00049       mHasTransaction = NS_SUCCEEDED(mConnection->BeginTransactionAs(aType));
00050   }
00051   ~mozStorageTransactionBase()
00052   {
00053     if (mConnection && mHasTransaction && ! mCompleted) {
00054       if (mCommitOnComplete)
00055         mConnection->CommitTransaction();
00056       else
00057         mConnection->RollbackTransaction();
00058     }
00059   }
00060 
00066   nsresult Commit()
00067   {
00068     if (!mConnection || mCompleted)
00069       return NS_OK; // no connection, or already done
00070     mCompleted = true;
00071     if (! mHasTransaction)
00072       return NS_OK; // transaction not ours, ignore
00073     nsresult rv = mConnection->CommitTransaction();
00074     if (NS_SUCCEEDED(rv))
00075       mHasTransaction = false;
00076 
00077     return rv;
00078   }
00079 
00085   nsresult Rollback()
00086   {
00087     if (!mConnection || mCompleted)
00088       return NS_OK; // no connection, or already done
00089     mCompleted = true;
00090     if (! mHasTransaction)
00091       return NS_ERROR_FAILURE;
00092 
00093     // It is possible that a rollback will return busy, so we busy wait...
00094     nsresult rv = NS_OK;
00095     do {
00096       rv = mConnection->RollbackTransaction();
00097       if (rv == NS_ERROR_STORAGE_BUSY)
00098         (void)PR_Sleep(PR_INTERVAL_NO_WAIT);
00099     } while (rv == NS_ERROR_STORAGE_BUSY);
00100 
00101     if (NS_SUCCEEDED(rv))
00102       mHasTransaction = false;
00103 
00104     return rv;
00105   }
00106 
00112   bool HasTransaction()
00113   {
00114     return mHasTransaction;
00115   }
00116 
00121   void SetDefaultAction(bool aCommitOnComplete)
00122   {
00123     mCommitOnComplete = aCommitOnComplete;
00124   }
