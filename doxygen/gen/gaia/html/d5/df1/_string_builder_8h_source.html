00020    or C#. It uses a doubling allocation strategy
00021    to grow when out of capacity.

00023    This does not use nsTArray because nsTArray starts
00024    growing by multiples of page size after it is the
00025    size of one page. We want to keep doubling in size
00026    so that we can continue to append at high speed even
00027    for large strings.
00028 
00029    Eventually, this should be templated for wide characters.
00030 
00031  */

00037 public:
00038     StringBuilder() {
00039         mCapacity = 16;
00040         mLength = 0;
00041         mBuffer = static_cast<char*>(malloc(sizeof(char)*mCapacity));
00042         mBuffer[0] = '\0';
00043     }
00044 
00045     void Append(const char *s) {
00046         size_t newLength = strlen(s);
00047 
00048         EnsureCapacity(mLength + newLength+1);
00049 
00050         // copy the entire string including the null terminator
00051         memcpy(&mBuffer[mLength], s, newLength+1);
00052         mLength += newLength;
00053     }
00054

00063     size_t EnsureCapacity(size_t capacity) {
00064         if (capacity > mCapacity) {
00065             // make sure we at least double in size
00066             mCapacity = XPCOM_MAX(capacity, mCapacity*2);
00067             mBuffer = static_cast<char*>(realloc(mBuffer, mCapacity));
00068             mCapacity = moz_malloc_usable_size(mBuffer);
