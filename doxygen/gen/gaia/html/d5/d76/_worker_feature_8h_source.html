00031 enum Status
00032 {
00033   // Not yet scheduled.
00034   Pending = 0,
00035 
00036   // This status means that the close handler has not yet been scheduled.
00037   Running,
00038 
00039   // Inner script called close() on the worker global scope. Setting this
00040   // status causes the worker to clear its queue of events but does not abort
00041   // the currently running script. The close handler is also scheduled with
00042   // no expiration time.
00043   Closing,
00044 
00045   // Outer script called terminate() on the worker or the worker object was
00046   // garbage collected in its outer script. Setting this status causes the
00047   // worker to abort immediately, clear its queue of events, and schedules the
00048   // close handler with no expiration time.
00049   Terminating,
00050 
00051   // Either the user navigated away from the owning page or the owning page fell
00052   // out of bfcache. Setting this status causes the worker to abort immediately
00053   // and schedules the close handler with a short expiration time. Since the
00054   // page has gone away the worker may not post any messages.
00055   Canceling,
00056 
00057   // The application is shutting down. Setting this status causes the worker to
00058   // abort immediately and the close handler is never scheduled.
00059   Killing,
00060 
00061   // The close handler has run and the worker is effectively dead.
00062   Dead
00063 };
