00029 /* ANSI X9.57 defines DSA signatures as DER encoded data.  Our DSA1 code (and
00030  * most of the rest of the world) just generates 40 bytes of raw data.  These
00031  * functions convert between formats.
00032  */
00033 extern SECStatus DSAU_EncodeDerSig(SECItem *dest, SECItem *src);
00034 extern SECItem *DSAU_DecodeDerSig(const SECItem *item);
00035 
00036 /*
00037  * Unlike DSA1, raw DSA2 and ECDSA signatures do not have a fixed length.
00038  * Rather they contain two integers r and s whose length depends
00039  * on the size of q or the EC key used for signing.
00040  *
00041  * We can reuse the DSAU_EncodeDerSig interface to DER encode
00042  * raw ECDSA signature keeping in mind that the length of r 
00043  * is the same as that of s and exactly half of src->len.
00044  *
00045  * For decoding, we need to pass the length of the desired
00046  * raw signature (twice the key size) explicitly.
00047  */
00048 extern SECStatus DSAU_EncodeDerSigWithLen(SECItem *dest, SECItem *src, 
00049                       unsigned int len);
00050 extern SECItem *DSAU_DecodeDerSigToLen(const SECItem *item, unsigned int len);
