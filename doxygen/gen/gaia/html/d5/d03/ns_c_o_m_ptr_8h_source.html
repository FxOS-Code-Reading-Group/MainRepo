00239     /*
00240       An |nsCOMPtr_helper| transforms commonly called getters into typesafe forms
00241       that are more convenient to call, and more efficient to use with |nsCOMPtr|s.
00242       Good candidates for helpers are |QueryInterface()|, |CreateInstance()|, etc.
00243 
00244       Here are the rules for a helper:
00245         - it implements |operator()| to produce an interface pointer
00246         - (except for its name) |operator()| is a valid [XP]COM `getter'
00247         - the interface pointer that it returns is already |AddRef()|ed (as from any good getter)
00248         - it matches the type requested with the supplied |nsIID| argument
00249         - its constructor provides an optional |nsresult*| that |operator()| can fill
00250           in with an error when it is executed
00251           
00252       See |class nsGetInterface| for an example.
00253     */
00254   {
00255     public:
00256       virtual nsresult NS_FASTCALL operator()( const nsIID&, void** ) const = 0;
00257   };

00446       void
00447       assign_assuming_AddRef( nsISupports* newPtr )
00448         {
00449             /*
00450               |AddRef()|ing the new value (before entering this function) before
00451               |Release()|ing the old lets us safely ignore the self-assignment case.
00452               We must, however, be careful only to |Release()| _after_ doing the
00453               assignment, in case the |Release()| leads to our _own_ destruction,
00454               which would, in turn, cause an incorrect second |Release()| of our old
00455               pointer.  Thank <waterson@netscape.com> for discovering this.
00456             */
00457           nsISupports* oldPtr = mRawPtr;
00458           mRawPtr = newPtr;
00459           NSCAP_LOG_ASSIGNMENT(this, newPtr);
00460           NSCAP_LOG_RELEASE(this, oldPtr);
00461           if ( oldPtr )
00462             NSCAP_RELEASE(this, oldPtr);
00463         }
