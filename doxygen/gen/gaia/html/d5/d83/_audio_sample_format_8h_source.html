00119 inline void
00120 ConvertAudioSamplesWithScale(const int16_t* aFrom, int16_t* aTo, int aCount, float aScale)
00121 {
00122   if (aScale == 1.0f) {
00123     ConvertAudioSamples(aFrom, aTo, aCount);
00124     return;
00125   }
00126   if (0.0f <= aScale && aScale < 1.0f) {
