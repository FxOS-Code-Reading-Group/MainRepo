00066  * The PL_MAKE_MEM_ macros are modeled after the MOZ_MAKE_MEM_ macros in
00067  * Mozilla's mfbt/MemoryChecking.h. Only AddressSanitizer is supported now.
00068  *
00069  * Provides a common interface to the ASan (AddressSanitizer) and Valgrind

00071  * three macros are provided:
00072  *
00073  *   PL_MAKE_MEM_NOACCESS  - Mark memory as unsafe to access (e.g. freed)

00077  * With Valgrind in use, these directly map to the three respective Valgrind
00078  * macros. With ASan in use, the NOACCESS macro maps to poisoning the memory,
00079  * while the UNDEFINED/DEFINED macros unpoison memory.

00081  * With no memory checker available, all macros expand to the empty statement.

00137 #define PL_ARENA_ALLOCATE(p, pool, nb) \
00138     PR_BEGIN_MACRO \
00139         PLArena *_a = (pool)->current; \
00140         PRUint32 _nb = PL_ARENA_ALIGN(pool, nb); \
00141         PRUword _p = _a->avail; \
00142         PRUword _q = _p + _nb; \
00143         if (_q > _a->limit) { \
00144             _p = (PRUword)PL_ArenaAllocate(pool, _nb); \
00145         } else { \
00146             _a->avail = _q; \
00147         } \
00148         p = (void *)_p; \
00149         PL_MAKE_MEM_UNDEFINED(p, nb); \
00150         PL_ArenaCountAllocation(pool, nb); \
00151     PR_END_MACRO
00152 
00153 #define PL_ARENA_GROW(p, pool, size, incr) \
00154     PR_BEGIN_MACRO \
00155         PLArena *_a = (pool)->current; \
00156         PRUint32 _incr = PL_ARENA_ALIGN(pool, incr); \
00157         PRUword _p = _a->avail; \
00158         PRUword _q = _p + _incr; \
00159         if (_p == (PRUword)(p) + PL_ARENA_ALIGN(pool, size) && \
00160             _q <= _a->limit) { \
00161             PL_MAKE_MEM_UNDEFINED((unsigned char *)(p) + size, incr); \
00162             _a->avail = _q; \
00163             PL_ArenaCountInplaceGrowth(pool, size, incr); \
00164         } else { \
00165             p = PL_ArenaGrow(pool, p, size, incr); \
00166         } \
00167         PL_ArenaCountGrowth(pool, size, incr); \
00168     PR_END_MACRO

00193     PR_BEGIN_MACRO \
