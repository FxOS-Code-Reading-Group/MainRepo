00950

00952     var style = viewfinder.style;
00953     // Switch screen dimensions to landscape
00954     var screenWidth = document.body.clientHeight;
00955     var screenHeight = document.body.clientWidth;
00956     var pictureAspectRatio = this._pictureSize.height / this._pictureSize.width;
00957     var screenAspectRatio = screenHeight / screenWidth;
00958 
00959     // Previews should match the aspect ratio and not be smaller than the screen
00960     var validPreviews = camera.capabilities.previewSizes.filter(function(res) {
00961       var isLarger = res.height >= screenHeight && res.width >= screenWidth;
00962       var aspectRatio = res.height / res.width;
00963       var matchesRatio = Math.abs(aspectRatio - pictureAspectRatio) < 0.05;
00964       return matchesRatio && isLarger;
00965     });

00967     // We should always have a valid preview size, but just in case
00968     // we dont, pick the first provided.
00969     if (validPreviews.length) {
00970       // Pick the smallest valid preview
00971       this._previewConfig = validPreviews.sort(function(a, b) {
00972         return a.width * a.height - b.width * b.height;
00973       }).shift();
00974     } else {
00975       this._previewConfig = camera.capabilities.previewSizes[0];
00976     }

00979     var width, height;
00980     var translateX = 0;
00981 
00982     // The preview should be larger than the screen, shrink it so that as
00983     // much as possible is on screen.
00984     if (screenAspectRatio < pictureAspectRatio) {
00985       width = screenWidth;
00986       height = screenWidth * pictureAspectRatio;
00987     } else {
00988       width = screenHeight / pictureAspectRatio;

00990     }
00991 
00992     if (this._cameraNumber == 1) {
00993       /* backwards-facing camera */
00994       transform += ' scale(-1, 1)';

00999     // This translation goes after the rotation so the element is shifted up
01000     // (for back camera) - shifted up after it is rotated 90 degress clockwise.

01403   pickPictureSize: function camera_pickPictureSize(pictureSizes) {
01404     var targetSize = null;
01405     var targetFileSize = 0;
01406     if (this._pendingPick && this._pendingPick.source.data.maxFileSizeBytes) {
01407       // we use worse case of all compression method: gif, jpg, png
01408       targetFileSize = this._pendingPick.source.data.maxFileSizeBytes;
01409     }
01410     if (this._pendingPick && this._pendingPick.source.data.width &&
01411         this._pendingPick.source.data.height) {
01412       // if we have pendingPick with width and height, set it as target size.
01413       targetSize = {'width': this._pendingPick.source.data.width,
01414                     'height': this._pendingPick.source.data.height};
01415     }
01416 
01417     // CONFIG_MAX_IMAGE_PIXEL_SIZE is maximum image resolution for still photos
01418     // taken with camera. It's from config.js which is generated in build time,
01419     // 5 megapixels by default (see build/application-data.js). It should be
01420     // synced with Gallery app and update carefully.
01421     var maxRes = CONFIG_MAX_IMAGE_PIXEL_SIZE;
01422     var estimatedJpgSize = this.ESTIMATED_JPEG_FILE_SIZE;
01423     var size = pictureSizes.reduce(function(acc, size) {
01424       var mp = size.width * size.height;
01425       // we don't need the resolution larger than maxRes
01426       if (mp > maxRes) {
01427         return acc;
01428       }
01429       // We assume the relationship between MP to file size is linear.
01430       // This may be inaccurate on all cases.
01431       var estimatedFileSize = mp * estimatedJpgSize / maxRes;
01432       if (targetFileSize > 0 && estimatedFileSize > targetFileSize) {
01433         return acc;
01434       }
01435 
01436       if (targetSize) {
01437         // find a resolution both width and height are large than pick size
01438         if (size.width < targetSize.width || size.height < targetSize.height) {
01439           return acc;
01440         }
01441         // it's first pictureSize.
01442         if (!acc.width || acc.height) {
01443           return size;
01444         }
01445         // find large enough but as small as possible.
01446         return (mp < acc.width * acc.height) ? size : acc;
01447       } else {
01448         // no target size, find as large as possible.
01449         return (mp > acc.width * acc.height && mp <= maxRes) ? size : acc;
01450       }
01451     }, {width: 0, height: 0});
01452 
01453     if (size.width === 0 && size.height === 0) {
01454       return pictureSizes[0];
01455     } else {
01456       return size;
01457     }
01458   },
