00104 {
00105     unsigned int flags = 0;
00106     unsigned int mask = ~0;
00107     unsigned int reg_eax, reg_ebx, reg_ecx, reg_edx;
00108     char *env;
00109     (void)reg_ebx;
00110 
00111     /* See if the CPU capabilities are being overridden by the environment */
00112     env = getenv("VPX_SIMD_CAPS");
00113 
00114     if (env && *env)
00115         return (int)strtol(env, NULL, 0);
00116 
00117     env = getenv("VPX_SIMD_CAPS_MASK");
00118 
00119     if (env && *env)
00120         mask = strtol(env, NULL, 0);
00121 
00122     /* Ensure that the CPUID instruction supports extended features */
00123     cpuid(0, reg_eax, reg_ebx, reg_ecx, reg_edx);
00124 
00125     if (reg_eax < 1)
00126         return 0;
00127 
00128     /* Get the standard feature flags */
00129     cpuid(1, reg_eax, reg_ebx, reg_ecx, reg_edx);
00130 
00131     if (reg_edx & BIT(23)) flags |= HAS_MMX;
00132 
00133     if (reg_edx & BIT(25)) flags |= HAS_SSE; /* aka xmm */
00134 
00135     if (reg_edx & BIT(26)) flags |= HAS_SSE2; /* aka wmt */
00136 
00137     if (reg_ecx & BIT(0))  flags |= HAS_SSE3;
00138 
00139     if (reg_ecx & BIT(9))  flags |= HAS_SSSE3;
00140 
00141     if (reg_ecx & BIT(19)) flags |= HAS_SSE4_1;
00142 
00143     return flags & mask;
00144 }
