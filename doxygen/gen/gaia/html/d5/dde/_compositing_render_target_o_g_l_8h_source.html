00178   void InitializeImpl()
00179   {
00180     MOZ_ASSERT(mInitParams.mStatus == InitParams::READY);
00181 
00182     mGL->fBindFramebuffer(LOCAL_GL_FRAMEBUFFER, mFBO);
00183     mGL->fFramebufferTexture2D(LOCAL_GL_FRAMEBUFFER,

00189     // Making this call to fCheckFramebufferStatus prevents a crash on

00191     GLenum result = mGL->fCheckFramebufferStatus(LOCAL_GL_FRAMEBUFFER);
00192     if (result != LOCAL_GL_FRAMEBUFFER_COMPLETE) {
00193       nsAutoCString msg;
00194       msg.AppendPrintf("Framebuffer not complete -- error 0x%x, aFBOTextureTarget 0x%x, mFBO %d, mTextureHandle %d, aRect.width %d, aRect.height %d",
00195                        result, mInitParams.mFBOTextureTarget, mFBO, mTextureHandle, mInitParams.mSize.width, mInitParams.mSize.height);
00196       NS_ERROR(msg.get());

00200     mGL->fScissor(0, 0, mInitParams.mSize.width, mInitParams.mSize.height);
00201     if (mInitParams.mInit == INIT_MODE_CLEAR) {
00202       mGL->fClearColor(0.0, 0.0, 0.0, 0.0);
