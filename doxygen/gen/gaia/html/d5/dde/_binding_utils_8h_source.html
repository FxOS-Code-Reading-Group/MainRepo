00184 // Some callers don't want to set an exception when unwrapping fails
00185 // (for example, overload resolution uses unwrapping to tell what sort
00186 // of thing it's looking at).
00187 // U must be something that a T* can be assigned to (e.g. T* or an nsRefPtr<T>).
00188 template <prototypes::ID PrototypeID, class T, typename U>
00189 MOZ_ALWAYS_INLINE nsresult
00190 UnwrapObject(JSContext* cx, JSObject* obj, U& value)
00191 {
00192   /* First check to see whether we have a DOM object */
00193   const DOMClass* domClass = GetDOMClass(obj);
00194   if (!domClass) {
00195     /* Maybe we have a security wrapper or outer window? */
00196     if (!js::IsWrapper(obj)) {
00197       /* Not a DOM object, not a wrapper, just bail */
00198       return NS_ERROR_XPC_BAD_CONVERT_JS;
00199     }
00200 
00201     obj = js::CheckedUnwrap(obj, /* stopAtOuter = */ false);
00202     if (!obj) {
00203       return NS_ERROR_XPC_SECURITY_MANAGER_VETO;
00204     }
00205     MOZ_ASSERT(!js::IsWrapper(obj));
00206     domClass = GetDOMClass(obj);
00207     if (!domClass) {
00208       /* We don't have a DOM object */
00209       return NS_ERROR_XPC_BAD_CONVERT_JS;
00210     }
00211   }
00212 
00213   /* This object is a DOM object.  Double-check that it is safely
00214      castable to T by checking whether it claims to inherit from the
00215      class identified by protoID. */
00216   if (domClass->mInterfaceChain[PrototypeTraits<PrototypeID>::Depth] ==
00217       PrototypeID) {
00218     value = UnwrapDOMObject<T>(obj);
00219     return NS_OK;
00220   }
00221 
00222   /* It's the wrong sort of DOM object */
00223   return NS_ERROR_XPC_BAD_CONVERT_JS;
00224 }

01556                        StringificationBehavior undefinedBehavior,
01557                        FakeDependentString& result)
01558 {
01559   JSString *s;
01560   if (v.isString()) {
01561     s = v.toString();
01562   } else {
01563     StringificationBehavior behavior;
01564     if (v.isNull()) {
01565       behavior = nullBehavior;
01566     } else if (v.isUndefined()) {
01567       behavior = undefinedBehavior;
01568     } else {
01569       behavior = eStringify;

01592   }
