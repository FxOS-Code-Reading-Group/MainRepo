00043 static void nsObjCExceptionLog(NSException* aException)
00044 {
00045   NSLog(@"Mozilla has caught an Obj-C exception [%@: %@]",
00046         [aException name], [aException reason]);
00047 
00048 #if defined(MOZ_CRASHREPORTER) && defined(__cplusplus)
00049   // Attach exception info to the crash report.
00050   nsCOMPtr<nsICrashReporter> crashReporter =
00051     do_GetService("@mozilla.org/toolkit/crash-reporter;1");
00052   if (crashReporter)
00053     crashReporter->AppendObjCExceptionInfoToAppNotes(static_cast<void*>(aException));
00054 #endif
00055 
00056 #ifdef DEBUG
00057   @try {
00058     // Try to get stack information out of the exception. 10.5 returns the stack
00059     // info with the callStackReturnAddresses selector.
00060     NSArray *stackTrace = nil;
00061     if ([aException respondsToSelector:@selector(callStackReturnAddresses)]) {
00062       NSArray* addresses = (NSArray*)
00063         [aException performSelector:@selector(callStackReturnAddresses)];
00064       if ([addresses count])
00065         stackTrace = addresses;
00066     }
00067 
00068     // 10.4 doesn't respond to callStackReturnAddresses so we'll try to pull the
00069     // stack info out of the userInfo. It might not be there, sadly :(
00070     if (!stackTrace)
00071       stackTrace = [[aException userInfo] objectForKey:NSStackTraceKey];
00072 
00073     if (stackTrace) {
00074       // The command line should look like this:
00075       //   /usr/bin/atos -p <pid> -printHeader <stack frame addresses>
00076       NSMutableArray *args =
00077         [NSMutableArray arrayWithCapacity:[stackTrace count] + 3];
00078 
00079       [args addObject:@"-p"];
00080       int pid = [[NSProcessInfo processInfo] processIdentifier];
00081       [args addObject:[NSString stringWithFormat:@"%d", pid]];
00082 
00083       [args addObject:@"-printHeader"];
00084 
00085       unsigned int stackCount = [stackTrace count];
00086       unsigned int stackIndex = 0;
00087       for (; stackIndex < stackCount; stackIndex++)
00088         [args addObject:[[stackTrace objectAtIndex:stackIndex] stringValue]];
00089 
00090       NSPipe *outPipe = [NSPipe pipe];
00091 
00092       NSTask *task = [[NSTask alloc] init];
00093       [task setLaunchPath:@"/usr/bin/atos"];
00094       [task setArguments:args];
00095       [task setStandardOutput:outPipe];
00096       [task setStandardError:outPipe];
00097 
00098       NSLog(@"Generating stack trace for Obj-C exception...");
00099 
00100       // This will throw an exception if the atos tool cannot be found, and in
00101       // that case we'll just hit our @catch block below.
00102       [task launch];
00103 
00104       [task waitUntilExit];
00105       [task release];
00106 
00107       NSData *outData =
00108         [[outPipe fileHandleForReading] readDataToEndOfFile];
00109       NSString *outString =
00110         [[NSString alloc] initWithData:outData encoding:NSUTF8StringEncoding];
00111 
00112       NSLog(@"Stack trace:\n%@", outString);
00113 
00114       [outString release];
00115     }
00116     else {
00117       NSLog(@"<No stack information available for Obj-C exception>");
00118     }
00119   }
00120   @catch (NSException *exn) {
00121     NSLog(@"Failed to generate stack trace for Obj-C exception [%@: %@]",
00122           [exn name], [exn reason]);
00123   }
00124 #endif
00125 }
