00059     static inline bool SkScalarIsFinite(float x) {
00060         // We rely on the following behavior of infinities and nans
00061         // 0 * finite --> 0
00062         // 0 * infinity --> NaN
00063         // 0 * NaN --> NaN
00064         float prod = x * 0;
00065         // At this point, prod will either be NaN or 0
00066         // Therefore we can return (prod == prod) or (0 == prod).
00067         return prod == prod;
00068     }
