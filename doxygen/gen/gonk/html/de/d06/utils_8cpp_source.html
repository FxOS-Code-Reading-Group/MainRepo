00066 void strip(string& str)
00067 {
00068   string::iterator it;
00069   for (it = str.begin(); isspace(*it) && it != str.end(); ++it)
00070   {}
00071   str.erase(str.begin(), it);
00072 
00073   string::reverse_iterator rit;
00074   for (rit = str.rbegin(); isspace(*rit) && rit != str.rend(); ++rit)
00075   {}
00076   str.erase(rit.base(), str.end());
00077 
00078   // If the string is non-empty, its first and last chars must not be
00079   // whitespace.
00080   assert(str.begin() == str.end() ||
00081          (!isspace(*str.begin()) && !isspace(*str.rbegin())));
00082 }

00128 DIR* safe_opendir(const char* dir)
00129 {
00130   // opendir() calls open(), so it can fail with EINTR.  On the other hand,
00131   // opendir() can also fail due to malloc() returning null, in which case the
00132   // bionic implementation doesn't modify errno.  So to be totally safe, we
00133   // have to set errno to 0 before we invoke TEMP_FAILURE_RETRY.
00134   errno = 0;
00135   DIR* d;
00136   while (!(d = opendir(dir)) && errno == EINTR) {}
00137   return d;
00138 }
