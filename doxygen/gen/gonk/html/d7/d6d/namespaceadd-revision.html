00067 def parse_manifest(man_filename):
00068   """Parse a manifest XML file and return a 'Document' from it.
00069   Any <include name="another.xml"/> elements are replaced with the contents
00070   if the specified file."""
00071   doc = xml.dom.minidom.parse(man_filename)
00072   for inc_node in doc.getElementsByTagName("include"):
00073     inc_filename = inc_node.getAttribute('name')
00074     inc_directory = os.path.dirname(os.path.realpath(man_filename));
00075     inc_path = os.path.join(inc_directory, inc_filename)
00076     # parse the included xml
00077     inc_doc = parse_manifest(inc_path)
00078     # add all the elements found in the included manifest
00079     m = inc_doc.getElementsByTagName("manifest")[0]
00080     for c in m.childNodes:
00081       # prevent insertBefore() from deleting c from its parent because
00082       # that modifies m.childNodes and the loop gets screwed up
00083       c.parentNode = None
00084       # insert right before the <include .../> element
00085       inc_node.parentNode.insertBefore(c, inc_node)
00086     # remove the <include .../> element itself
00087     inc_node.parentNode.removeChild(inc_node)
00088   return doc
