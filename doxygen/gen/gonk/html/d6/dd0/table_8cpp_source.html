00119 {
00120   // Figure out how wide each column has to be.
00121   vector<size_t> col_widths;
00122   for (vector<TableRow>::const_iterator row = m_rows.begin();
00123        row != m_rows.end(); ++row) {
00124     if (col_widths.size() < row->size()) {
00125       col_widths.resize(row->size(), 0);
00126     }
00127 
00128     for (size_t i = 0; i < row->size(); i++) {
00129       col_widths[i] = max(col_widths[i], row->at(i).first.length());
00130     }
00131   }
00132 
00133   // Figure out the table's full width.
00134   int table_width = 0;
00135   for (size_t i = 0; i < col_widths.size(); i++) {
00136     table_width += col_widths[i] + 1;
00137   }
00138   table_width -= 1;
00139 
00140   // Print the multi-column header, if we have one.
00141   assert((m_multi_col_header_start == -1) == (m_multi_col_header_end == -1));
00142   assert(m_multi_col_header_start < (int) col_widths.size());
00143   assert(m_multi_col_header_end <= (int) col_widths.size());
00144 
00145   if (m_multi_col_header_start != -1) {
00146     print_spaces(indent);
00147 
00148     int i;
00149     for (i = 0; i < m_multi_col_header_start; i++) {
00150       print_spaces(col_widths[i] + 1);
00151     }
00152 
00153     putchar('|');
00154 
00155     // Figure out how many chars we have between the two |'s.
00156     int chars_between = 0;
00157     for (; i < m_multi_col_header_end - 1; i++) {
00158       chars_between += col_widths[i] + 1;
00159     }
00160     chars_between += col_widths[i++] - 2;
00161 
00162     int spaces_between = chars_between - m_multi_col_header_str.length();
00163     spaces_between = max(spaces_between, 0);
00164     print_spaces((spaces_between + 1)/ 2);
00165     fputs(m_multi_col_header_str.c_str(), stdout);
00166     print_spaces(spaces_between / 2);
00167 
00168     fputs("|\n", stdout);
00169   }
00170 
00171   // Now print each row.
00172   for (vector<TableRow>::const_iterator row = m_rows.begin();
00173        row != m_rows.end(); ++row) {
00174     print_spaces(indent);
00175 
00176     if (row->is_delimiter()) {
00177       for (int i = 0; i < table_width; i++) {
00178         putchar('-');
00179       }
00180       putchar('\n');
00181       continue;
00182     }
00183 
00184     for (size_t i = 0; i < row->size(); i++) {
00185       const cell_t& cell = row->at(i);
00186       const char* fmt_string = cell.second == ALIGN_RIGHT ? "%*s" : "%-*s";
00187       printf(fmt_string, col_widths[i], cell.first.c_str());
00188       if (i != row->size() - 1) {
00189         putchar(' ');
00190       }
00191     }
00192     putchar('\n');
