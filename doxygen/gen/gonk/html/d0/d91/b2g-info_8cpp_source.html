00074 string
00075 read_whole_file(const char* filename)
00076 {
00077   char buf[1024];
00078   int fd = TEMP_FAILURE_RETRY(open(filename, O_RDONLY));
00079   if (fd == -1) {
00080     return "";
00081   }
00082 
00083   ssize_t total_read = 0;
00084   ssize_t num_remaining = sizeof(buf) - 1;
00085   while (true) {
00086     // No more room in the buffer; we're done.
00087     if (num_remaining <= 0) {
00088       break;
00089     }
00090 
00091     ssize_t nread = TEMP_FAILURE_RETRY(read(fd, buf + total_read, num_remaining));
00092     if (nread == 0 || nread == -1) {
00093       break;
00094     }
00095 
00096     num_remaining -= nread;
00097     total_read += nread;
00098   }
00099 
00100   buf[total_read] = '\0';
00101   return buf;
00102 }

00104 void print_system_meminfo()
00105 {
00106   // We can't use sysinfo() here because iit doesn't tell us how much cached
00107   // memory we're using.  (On B2G, this is often upwards of 30mb.)
00108   //
00109   // Instead, we have to parse /proc/meminfo.
00110 
00111   FILE* meminfo = fopen("/proc/meminfo", "r");
00112   if (!meminfo) {
00113     perror("Couldn't open /proc/meminfo");
00114     return;
00115   }
00116 
00117   // These are all in kb.
00118   int total = -1;
00119   int free = -1;
00120   int buffers = -1;
00121   int cached = -1;
00122 
00123   char line[256];
00124   while(fgets(line, sizeof(line), meminfo)) {
00125     if (sscanf(line, "MemTotal: %d kB", &total) == 0 &&
00126         sscanf(line, "MemFree: %d kB", &free) == 0 &&
00127         sscanf(line, "Buffers: %d kB", &buffers) == 0 &&
00128         sscanf(line, "Cached: %d kB", &cached)) {
00129       // These four values should appear first in meminfo, so if this line
00130       // doesn't match any of them, we're done parsing.
00131       break;
00132     }
00133   }
00134 
00135   fclose(meminfo);
00136 
00137   if (total == -1 || free == -1 || buffers == -1 || cached == -1) {
00138     fprintf(stderr, "Unable to parse /proc/meminfo.\n");
00139     return;
00140   }
00141 
00142   int actually_used = total - free - buffers - cached;
00143 
00144   puts("System memory info:\n");
00145 
00146   Table t;
00147 
00148   t.start_row();
00149   t.add("Total");
00150   t.add_fmt("%0.1f MB", kb_to_mb(total));
00151 
00152   t.start_row();
00153   t.add("Used - cache");
00154   t.add_fmt("%0.1f MB", kb_to_mb(total - free - buffers - cached));
00155 
00156   t.start_row();
00157   t.add("B2G procs (PSS)");
00158 
00159   int b2g_mem_kb = 0;
00160   for (vector<Process*>::const_iterator it = ProcessList::singleton().b2g_processes().begin();
00161        it != ProcessList::singleton().b2g_processes().end(); ++it) {
00162     b2g_mem_kb += (*it)->pss_kb();
00163   }
00164   t.add_fmt("%0.1f MB", b2g_mem_kb / 1024.0);
00165 
00166   t.start_row();
00167   t.add("Non-B2G procs");
00168   t.add_fmt("%0.1f MB", kb_to_mb(total - free - buffers - cached - b2g_mem_kb));
00169 
00170   t.start_row();
00171   t.add("Free + cache");
00172   t.add_fmt("%0.1f MB", kb_to_mb(free + buffers + cached));
00173 
00174   t.start_row();
00175   t.add("Free");
00176   t.add_fmt("%0.1f MB", kb_to_mb(free));
00177 
00178   t.start_row();
00179   t.add("Cache");
00180   t.add_fmt("%0.1f MB", kb_to_mb(buffers + cached));
00181 
00182   t.print_with_indent(2);
00183 }
