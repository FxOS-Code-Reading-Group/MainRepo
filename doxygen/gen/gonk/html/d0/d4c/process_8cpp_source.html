00094     // to the console.
00095     if (errno != ENOENT) {
00096       perror("Unable to open /proc/<pid>/stat");
00097     }
00098     return;
00099   }

00102   char comm[32];

00106            "%d "   // pid
00107            "%17[^)]) "// comm
00108            "%*c "  // state
00109            "%d "   // ppid
00110            "%*d "  // pgrp
00111            "%*d "  // session
00112            "%*d "  // tty_nr

00126 
00127   fclose(stat_file);

00129   if (nread != 4) {

00131             filename, nread);
00132     return;
00133   }
00134 
00135   if (task_id() != pid2) {
00136     fprintf(stderr, "When reading %s, got pid %d, but expected pid %d.\n",

00140 
00141   // Okay, everything worked out.  Store the data we collected.

00280 Process::ensure_got_meminfo()
00281 {
00282   if (m_got_meminfo) {
00283     return;
00284   }
00285 
00286   // If anything goes wrong after this point (e.g. smaps doesn't exist), we
00287   // still want to say that we got meminfo; there's no point in trying again.
00288   m_got_meminfo = true;
00289 
00290   // Android has this pm_memusage interface to get the data we collect here.
00291   // But collecting the data from smaps isn't hard, and doing it this way
00292   // doesn't rely on any external code, which is nice.
00293   // 
00294   // Also, the vsize value I get out of procrank (which uses pm_memusage) is
00295   // way lower than what I get out of statm (which matches smaps).  I presume
00296   // that statm is correct here.
00297 
00298   char filename[128];
00299   snprintf(filename, sizeof(filename), "/proc/%d/smaps", pid());
00300   FILE *f = fopen(filename, "r");
00301   if (!f) {
00302     return;
00303   }
00304 
00305   m_vsize_kb = m_rss_kb = m_pss_kb = m_uss_kb = 0;
00306 
00307   char line[256];
00308   while(fgets(line, sizeof(line), f)) {
00309       int val = 0;
00310       if (sscanf(line, "Size: %d kB", &val) == 1) {
00311         m_vsize_kb += val;
00312       } else if (sscanf(line, "Rss: %d kB", &val) == 1) {
00313         m_rss_kb += val;
00314       } else if (sscanf(line, "Pss: %d kB", &val) == 1) {
00315         m_pss_kb += val;
00316       } else if (sscanf(line, "Private_Dirty: %d kB", &val) == 1 ||
00317                  sscanf(line, "Private_Clean: %d kB", &val) == 1) {
00318         m_uss_kb += val;
00319       }
00320   }
00321 
00322   fclose(f);
00323 }
